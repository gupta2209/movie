001package daikon.diff;
002
003import java.util.ArrayList;
004import java.util.Iterator;
005import java.util.List;
006import org.checkerframework.checker.nullness.qual.Nullable;
007import org.checkerframework.dataflow.qual.Pure;
008import org.plumelib.util.Pair;
009
010/**
011 * All nodes must subclass this class. The type parameter CONTENT is (half of) the type of the
012 * objects stored in this node: they are {@code Pair<CONTENT,CONTENT>}. The type parameter CHILD is
013 * the type of the children (and is ignored if there are no children).
014 */
015public abstract class Node<CONTENT extends @Nullable Object, CHILD> {
016
017  private List<CHILD> children = new ArrayList<>();
018  // Nonsensical for RootNode
019  private Pair<CONTENT, CONTENT> userObject;
020
021  protected Node(Pair<CONTENT, CONTENT> userObject) {
022    this.userObject = userObject;
023  }
024
025  protected Node(CONTENT left, CONTENT right) {
026    this.userObject = Pair.of(left, right);
027  }
028
029  public void add(CHILD newChild) {
030    children.add(newChild);
031  }
032
033  public Iterator<CHILD> children() {
034    return children.iterator();
035  }
036
037  public Pair<CONTENT, CONTENT> getUserObject() {
038    return userObject;
039  }
040
041  @Pure
042  public CONTENT getUserLeft() {
043    return userObject.a;
044  }
045
046  @Pure
047  public CONTENT getUserRight() {
048    return userObject.b;
049  }
050
051  public abstract void accept(Visitor v);
052}

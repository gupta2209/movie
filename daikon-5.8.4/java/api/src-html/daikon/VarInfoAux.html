115  /**
116   * Return an interned VarInfoAux that represents a given string. Elements are separated by commas,
117   * in the form:
118   *
119   * <p>x = a, "a key" = "a value"
120   *
121   * <p>Parse allow for quoted elements. White space to the left and right of keys and values do not
122   * matter, but inbetween does.
123   */
124  public static @Interned VarInfoAux parse(String inString) throws IOException {
125    Reader inStringReader = new StringReader(inString);
126    StreamTokenizer tok = new StreamTokenizer(inStringReader);
127    tok.resetSyntax();
128    tok.wordChars(0, Integer.MAX_VALUE);
129    tok.quoteChar('\"');
130    tok.whitespaceChars(' ', ' ');
131    tok.ordinaryChar('[');
132    tok.ordinaryChar(']');
133    tok.ordinaryChars(',', ',');
134    tok.ordinaryChars('=', '=');
135    Map<@Interned String, @Interned String> map = theDefault.map;
136
137    String key = "";
138    String value = "";
139    boolean seenEqual = false;
140    boolean insideVector = false;
141    for (int tokInfo = tok.nextToken();
142        tokInfo != StreamTokenizer.TT_EOF;
143        tokInfo = tok.nextToken()) {
144      @SuppressWarnings("interning") // initialization-checking pattern
145      boolean mapUnchanged = (map == theDefault.map);
146      if (mapUnchanged) {
147        // We use default values if none are specified.  We initialize
148        // here rather than above to save time when there are no tokens.
149
150        map = new HashMap<>(theDefault.map);
151      }
152
153      @Interned String token;
154      if (tok.ttype == StreamTokenizer.TT_WORD || tok.ttype == '\"') {
155        assert tok.sval != null
156            : "@AssumeAssertion(nullness): representation invariant of StreamTokenizer";
157        token = tok.sval.trim().intern();
158      } else {
159        token = ((char) tok.ttype + "").intern();
160      }
161
162      debug.fine("Token info: " + tokInfo + " " + token);
163
164      if (token == "[") { // interned
165        if (!seenEqual) throw new IOException("Aux option did not contain an '='");
166        if (insideVector) throw new IOException("Vectors cannot be nested in an aux option");
167        if (value.length() > 0) {
168          throw new IOException("Cannot mix scalar and vector values");
169        }
170
171        insideVector = true;
172        value = "";
173      } else if (token == "]") { // interned
174        if (!insideVector) throw new IOException("']' without preceding '['");
175        insideVector = false;
176      } else if (token == ",") { // interned
177        if (!seenEqual) throw new IOException("Aux option did not contain an '='");
178        if (insideVector) throw new IOException("',' cannot be used inside a vector");
179        map.put(key.intern(), value.intern());
180        key = "";
181        value = "";
182        seenEqual = false;
183      } else if (token == "=") { // interned
184        if (seenEqual) throw new IOException("Aux option contained more than one '='");
185        if (insideVector) throw new IOException("'=' cannot be used inside a vector");
186        seenEqual = true;
187      } else {
188        if (!seenEqual) {
189          key = (key + " " + token).trim();
190        } else if (insideVector) {
191          value = value + " \"" + token.trim() + "\"";
192        } else {
193          value = (value + " " + token).trim();
194        }
195      }
196    }
197
198    if (seenEqual) {
199      map.put(key.intern(), value.intern());
200    }
201
202    // Interning
203    VarInfoAux result = new VarInfoAux(map).intern();
204    assert interningMap != null
205        : "@AssumeAssertion(nullness):  application invariant:  postcondition of intern(), which was just called";
206    if (debug.isLoggable(Level.FINE)) {
207      debug.fine("New parse " + result);
208      debug.fine("Intern table size: " + interningMap.size());
209    }
210    return result;
211  }

227  /** Special handler for deserialization. */
228  private @Interned Object readResolve() throws ObjectStreamException {
229    isInterned = false;
230    Map<@Interned String, @Interned String> newMap = new HashMap<>();
231    for (String key : map.keySet()) {
232      newMap.put(key.intern(), map.get(key).intern());
233    }
234    map = newMap;
235    return this.intern();
236  }

305  // Two variables should not be put in the same equality set unless they have the same flags.  For
306  // example, suppose that variable "this" is known to be non-null, but it is initially equal to
307  // another variable.  Then NonZero will not be instantiated over "this", and when the equality set
308  // is broken, there will be no NonZero invariant to copy to the other variable.  We only need to
309  // check equality for every aux field that might affect methods such as instantiate_ok.
310  @SuppressWarnings("keyfor") // https://tinyurl.com/cfissue/877
311  @Pure
312  public boolean equals_for_instantiation(
313      @GuardSatisfied VarInfoAux this, @GuardSatisfied VarInfoAux o) {
314    return this.getValue(HAS_DUPLICATES).equals(o.getValue(HAS_DUPLICATES))
315        && this.getValue(HAS_ORDER).equals(o.getValue(HAS_ORDER))
316        && this.getValue(HAS_SIZE).equals(o.getValue(HAS_SIZE))
317        && this.getValue(HAS_NULL).equals(o.getValue(HAS_NULL))
318        && Objects.equals(this.getValueOrNull(MINIMUM_LENGTH), o.getValueOrNull(MINIMUM_LENGTH))
319        && Objects.equals(this.getValueOrNull(MAXIMUM_LENGTH), o.getValueOrNull(MAXIMUM_LENGTH))
320        && Objects.equals(this.getValueOrNull(MINIMUM_VALUE), o.getValueOrNull(MINIMUM_VALUE))
321        && Objects.equals(this.getValueOrNull(MAXIMUM_VALUE), o.getValueOrNull(MAXIMUM_VALUE))
322        && Objects.equals(this.getValueOrNull(VALID_VALUES), o.getValueOrNull(VALID_VALUES))
323        && this.getValue(IS_STRUCT).equals(o.getValue(IS_STRUCT))
324        && this.getValue(IS_NON_NULL).equals(o.getValue(IS_NON_NULL));
325  }

327  /**
328   * Returns canonical representation of this. Doesn't need to be called by outside classes because
329   * these are always interned.
330   */
331  @SuppressWarnings({"interning", "cast"}) // intern method
332  private @Interned VarInfoAux intern() {
333
334    for (@Interned String key : map.keySet()) {
335      assert key == key.intern();
336      assert map.get(key) == map.get(key).intern();
337    }
338
339    if (this.isInterned) {
340      return (@Interned VarInfoAux) this; // cast is redundant (except in JSR 308)
341    }
342
343    // Necessary because various static methods call intern(), possibly before static field
344    // interningMap's initializer would be executed.
345    if (interningMap == null) {
346      interningMap = new HashMap<>();
347    }
348
349    @Interned VarInfoAux result;
350    if (interningMap.containsKey(this)) {
351      result = interningMap.get(this);
352    } else {
353      // Intern values in map
354      interningMap.put(this, (@Interned VarInfoAux) this); // cast is redundant (except in JSR 308)
355      result = (@Interned VarInfoAux) this; // cast is redundant (except in JSR 308)
356      this.isInterned = true;
357    }
358    return result;
359  }

361  /**
362   * Returns the integer value associated with a key, assuming it is defined. It is recommended to
363   * check that it is defined first with {@link #hasValue(String)}.
364   *
365   * @throws RuntimeException if the key is not defined
366   * @throws NumberFormatException if the value of the key cannot be parsed as an integer
367   * @see #hasValue(String)
368   */
369  @Pure
370  public int getInt(@KeyFor("this.map") String key) {
371    if (!hasValue(key)) {
372      throw new RuntimeException(String.format("Key '%s' is not defined", key));
373    }
374    return Integer.parseInt(getValue(key));
375  }

377  /**
378   * Returns the string array associated with a key, assuming it is defined. It is recommended to
379   * check that it is defined first with {@link #hasValue(String)}.
380   *
381   * @throws RuntimeException if the key is not defined
382   * @see #hasValue(String)
383   */
384  public String[] getList(@KeyFor("this.map") String key) {
385    try {
386      if (!hasValue(key)) {
387        throw new RuntimeException(String.format("Key '%s' is not defined", key));
388      }
389      final String sValue = getValue(key);
390      StreamTokenizer tok = new StreamTokenizer(new StringReader(sValue));
391      tok.quoteChar('"');
392      tok.whitespaceChars(' ', ' ');
393      ArrayList<String> lValues = new ArrayList<>();
394
395      int tokInfo = tok.nextToken();
396      while (tokInfo != StreamTokenizer.TT_EOF) {
397        if (tok.ttype != '"') {
398          continue;
399        }
400        assert tok.sval != null
401            : "@AssumeAssertion(nullness)"; // tok.type == '"' guarantees not null
402        lValues.add(tok.sval.trim());
403        tokInfo = tok.nextToken();
404      }
405      return lValues.toArray(new String[] {});
406    } catch (IOException ex) {
407      throw new RuntimeException(String.format("Parsing for key '%s' failed", key), ex);
408    }
409  }

428  public boolean getFlag(@KeyFor("this.map") String key) {
429    assert map.containsKey(key);
430    Object value = map.get(key);
431    assert value == TRUE || value == FALSE;
432    return value.equals(TRUE);
433  }

435  /** Return a new VarInfoAux with the desired value set. Does not modify this. */
436  public @Interned VarInfoAux setValue(String key, String value) {

438    newMap.put(key.intern(), value.intern());
439    return new VarInfoAux(newMap).intern();

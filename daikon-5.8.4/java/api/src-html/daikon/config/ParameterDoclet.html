021/**
022 * ParameterDoclet is a JavaDoc doclet that collects information about the runtime configuration
023 * options for the Daikon tools. Refer to the {@code --config} command-line option in the Daikon
024 * manual for an introduction to the configuration system.
025 */
026@SuppressWarnings({
027  "deprecation", // in JDK 9+, ClassDoc, RootDoc, etc. are deprecated
028  "removal" // in JDK 11, ClassDoc, RootDoc, etc. are deprecated and marked for removal
029})
030public class ParameterDoclet {
031
032  /** Entry point for this doclet (invoked by javadoc). */
033  public static boolean start(RootDoc doc) throws IOException {
034    ParameterDoclet pd = new ParameterDoclet(doc);
035
036    pd.process();
037
038    String[][] options = doc.options();
039    for (int i = 0; i < options.length; i++) {
040      String[] optset = options[i];
041      String opt = optset[0];
042
043      if ("--texinfo".equals(opt)) {
044        String fname = optset[1];
045        System.out.println("Opening " + fname + " for output...");
046        PrintWriter outf = new PrintWriter(UtilPlume.bufferedFileWriter(fname));
047        pd.writeTexInfo(outf);
048        outf.close();
049      } else if ("--text".equals(opt)) {
050        String fname = optset[1];
051        System.out.println("Opening " + fname + " for output...");
052        PrintWriter outf = new PrintWriter(UtilPlume.bufferedFileWriter(fname));
053        pd.writeText(outf);
054        outf.close();
055      } else if ("--list".equals(opt)) {
056        String fname = optset[1];
057        System.out.println("Opening " + fname + " for output...");
058        PrintWriter outf = new PrintWriter(UtilPlume.bufferedFileWriter(fname));
059        pd.writeList(outf);
060        outf.close();
061      }
062    }
063
064    return true;
065  }

067  /**
068   * Invoked by javadoc to query whether an option is allowed.
069   *
070   * @return number of tokens used by one option
071   */
072  public static int optionLength(String opt) {
073    if ("--texinfo".equals(opt)) return 2; // == 1 tag + 1 argument
074
075    if ("--text".equals(opt)) return 2; // == 1 tag + 1 argument
076
077    if ("--list".equals(opt)) return 2; // == 1 tag + 1 argument
078
079    return 0; // unknown option
080  }

084  /**
085   * A document category, or a subsection of the "List of configuration options" section in the
086   * manual.
087   */
088  static class DocCategory {
089
090    /**
091     * If non-null, then include only variables whose fully-qualified name starts with this prefix.
092     */
093    public @Nullable String prefix;
094    /**
095     * If non-null, then include only variables with this simple name. Note that this is a field
096     * name that starts with "dkconfig_", not a configuration option name.
097     */
098    public @Nullable String fieldName;
099    /** Include all configuration options whose fully-qualified name is in the list. */
100    public List<String> fqNames;
101
102    /** The subsection name. */
103    public String description;
104    /** Text that appears in the subsection before the list of configuration options. */
105    public String blurb;
106    /** A map from a field name to its description. */
107    public Map<String, String> fields;
108
109    /**
110     * Create a new DocCategory.
111     *
112     * @param prefix if non-null, include only variables whose fully-qualified name starts with this
113     *     prefix
114     * @param simpleName if non-null, include only configuration options with this simple name
115     * @param fqConfigNames if non-null, include only configuration options whose fully-qualified
116     *     name is in this list
117     * @param description the subsection name
118     * @param blurbLines text that appears in the subsection before the list of configuration
119     *     options
120     */
121    public DocCategory(
122        @Nullable String prefix,
123        @Nullable String simpleName,
124        @Nullable List<String> fqConfigNames,
125        String description,
126        String[] blurbLines) {
127      this(
128          prefix,
129          simpleName,
130          fqConfigNames,
131          description,
132          String.join(System.lineSeparator(), blurbLines));
133    }
134
135    /**
136     * Create a new DocCategory.
137     *
138     * @param prefix if non-null, include only variables whose fully-qualified name starts with this
139     *     prefix
140     * @param simpleName if non-null, include only configuration options with this simple name
141     * @param fqConfigNames if non-null, include only configuration options whose fully-qualified
142     *     name is in this list
143     * @param description the subsection name
144     * @param blurb text that appears in the subsection before the list of configuration options
145     */
146    public DocCategory(
147        @Nullable String prefix,
148        @Nullable String simpleName,
149        @Nullable List<String> fqConfigNames,
150        String description,
151        String blurb) {
152      this.prefix = prefix;
153      if (simpleName == null) {
154        fieldName = null;
155      } else {
156        fieldName = Configuration.PREFIX + simpleName;
157      }
158      if (fqConfigNames == null) {
159        this.fqNames = Collections.emptyList();
160      } else {
161        this.fqNames = fqConfigNames;
162      }
163      this.description = description;
164      this.blurb = blurb;
165      fields = new HashMap<>();
166    }
167
168    /**
169     * Return true if the given variable (that represents a configuration option) should be included
170     * in this section.
171     *
172     * @param fullConfigName the fully-qualified name of a Daikon configuration variable (no
173     *     "dkconfig_")
174     * @param simpleFieldName the simple name of the variable (starts with "dkconfig_")
175     * @return true if the given field should be included in this section of the manual
176     */
177    public boolean matches(String fullConfigName, String simpleFieldName) {
178      return ((prefix == null || fullConfigName.startsWith(prefix))
179              && (fieldName == null || fieldName.equals(simpleFieldName)))
180          || fqNames.contains(fullConfigName);
181    }
182  }

301  /** Process a javadoc tree and call processField for each field found. */
302  public void process() {
303    ClassDoc[] clazzes = root.classes();
304    for (int i = 0; i < clazzes.length; i++) {
305      ClassDoc clazz = clazzes[i];
306      FieldDoc[] fields = clazz.fields();
307      for (int j = 0; j < fields.length; j++) {
308        processField(fields[j]);
309      }
310    }
311  }
312
313  /**
314   * Call Process(String, String, String) for each configuration field found. Intended to be
315   * overridden.
316   */
317  public void processField(FieldDoc field) {
318    String name = field.name();
319    if (name.startsWith(Configuration.PREFIX)) {
320      String fullname = field.qualifiedName();
321      int snip = fullname.indexOf(Configuration.PREFIX);
322      fullname =
323          fullname.substring(0, snip) + fullname.substring(snip + Configuration.PREFIX.length());
324      String desc = field.commentText();
325      process(fullname, name, desc);
326    }
327  }

329  public static String NO_DESCRIPTION = "(no description provided)";
330  public static String UNKNOWN_DEFAULT = "The default value is not known.";
331  public static Pattern endOfSentence;
332
333  static {
334    endOfSentence = Pattern.compile("[.?!>](\\))?$");
335  }
336
337  /** Add (name, desc) pair to the map field 'fields' for the appropriate category. */
338  public void process(String fullname, String name, String desc) {
339    // System.out.printf("%s - %s%n", fullname, name);
340
341    if ("".equals(desc.trim())) {
342      desc = NO_DESCRIPTION;
343    } else if (!endOfSentence.matcher(desc).find()) {
344      // Add period if there is no end-of-sentence delimiter
345      desc = desc + ".";
346    }
347
348    for (int i = 0; i < categories.length; i++) {
349      if (categories[i].matches(fullname, name)) {
350        categories[i].fields.put(fullname, desc);
351        break;
352      }
353    }
354  }

356  private String getDefaultString(String field) {
357    try {
358      int i = field.lastIndexOf('.');
359      @SuppressWarnings("signature") // application invariant
360      @ClassGetName String classname = field.substring(0, i);
361      String fieldname = field.substring(i + 1);
362      Class<?> c = ReflectionPlume.classForName(classname);
363      Field f = c.getField(Configuration.PREFIX + fieldname);
364      if (!Modifier.isStatic(f.getModifiers())) {
365        throw new Error("Field " + Configuration.PREFIX + fieldname + " should be static");
366      }
367      @SuppressWarnings("nullness") // the field is static
368      Object value = f.get(null);
369      return "The default value is `" + value + "'.";
370    } catch (Exception e) {
371      System.err.println(e);
372      return UNKNOWN_DEFAULT;
373    }
374  }

376  public void writeTexInfo(PrintWriter out) {
377    out.println("@c BEGIN AUTO-GENERATED CONFIG OPTIONS LISTING");
378    out.println();
379
380    out.println("@menu");
381    for (int c = 0; c < categories.length; c++) {
382      out.println("* " + categories[c].description + "::");
383    }
384    out.println("@end menu");
385    out.println();
386
387    String up = "List of configuration options";
388    for (int c = 0; c < categories.length; c++) {
389      String node = categories[c].description;
390      String next = (c + 1 < categories.length) ? categories[c + 1].description : "";
391      String prev = (c > 0) ? categories[c - 1].description : up;
392      out.println("@node " + node + ", " + next + ", " + prev + ", " + up);
393      out.println("@subsubsection " + node);
394      out.println();
395      out.println(categories[c].blurb);
396      out.println();
397      out.println("@table @option");
398      out.println();
399
400      for (
401      @KeyFor("categories[c].fields") String field : CollectionsPlume.sortedKeySet(categories[c].fields)) {
402        String desc = categories[c].fields.get(field);
403        String defaultString = getDefaultString(field);
404
405        // Simpler format for debugging
406        if (false) {
407          String value = defaultString.replaceFirst(".*`", "");
408          value = value.replaceFirst("'.*", "");
409          out.printf("@item %s %s%n%n", value, field);
410          continue;
411        }
412
413        // @item [field]
414        //  [desc]
415        out.println("@item " + field);
416        desc = HtmlToTexinfo.javadocHtmlToTexinfo(desc);
417        out.println(desc);
418        if (!desc.contains("The default value is")) {
419          out.println(defaultString);
420        }
421        out.println();
422      }
423      out.println("@end table");
424      out.println();
425    }
426
427    out.println("@c END AUTO-GENERATED CONFIG OPTIONS LISTING");
428    out.println();
429  }

431  public void writeText(PrintWriter out) {
432    for (int c = 0; c < categories.length; c++) {
433      out.println(categories[c].description);
434      out.println();
435
436      for (
437      @KeyFor("categories[c].fields") String field : CollectionsPlume.sortedKeySet(categories[c].fields)) {
438        String desc = categories[c].fields.get(field);
439        String defaultString = getDefaultString(field);
440
441        // [field]
442        //   [desc]
443        out.println("  " + field);
444        out.println("    " + desc);
445        if (!desc.contains("The default value is")) {
446          out.println("    " + defaultString);
447        }
448        out.println();
449      }
450    }
451  }
452
453  public void writeList(PrintWriter out) {
454    for (int c = 0; c < categories.length; c++) {
455      for (String field : CollectionsPlume.sortedKeySet(categories[c].fields)) {
456        out.println(field);
457      }
458    }

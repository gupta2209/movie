119/**
120 * All information about a single program point. A Ppt may also represent just part of the data: see
121 * PptConditional.
122 *
123 * <p>PptTopLevel doesn't do any direct computation, instead deferring that to its views that are
124 * slices and that actually contain the invariants.
125 *
126 * <p>The data layout is as follows:
127 *
128 * <ul>
129 *   <li>A PptMap is a collection of PptTopLevel objects.
130 *   <li>A PptTopLevel contains PptSlice objects, one for each set of variables at the program
131 *       point. For instance, if a PptTopLevel has variables a, b, and c, then it has three
132 *       PptSlice1 objects (one for a; one for b; and one for c), three PptSlice2 objects (one for
133 *       a,b; one for a,c; and one for b,c), and one PptSlice3 object (for a,b,c).
134 *   <li>A PptSlice object contains invariants. When a sample (a tuple of variable values) is fed to
135 *       a PptTopLevel, it in turn feeds it to all the slices, which feed it to all the invariants,
136 *       which act on it appropriately.
137 * </ul>
138 */

277  /**
278   * Iterator for all of the conditional ppts. Returns each PptConditional from each entry in
279   * splitters.
280   */
281  public class CondIterator implements java.util.Iterator<PptConditional> {
282
283    int splitter_index = 0;
284    int ppts_index = 0;
285
286    @Override
287    @SuppressWarnings(
288        "flowexpr.parse.error") // Checker Framework bug: splitters is a field in this class
289    @EnsuresNonNullIf(result = true, expression = "splitters")
290    public boolean hasNext(@GuardSatisfied CondIterator this) {
291      if (splitters == null) {
292        return false;
293      }
294      if (splitter_index >= splitters.size()) {
295        return false;
296      }
297      return true;
298    }
299
300    @Override
301    public PptConditional next(@GuardSatisfied CondIterator this) {
302
303      if (!hasNext()) {
304        throw new NoSuchElementException();
305      }
306
307      PptSplitter ppt_split = splitters.get(splitter_index);
308      PptConditional ppt = (PptConditional) ppt_split.ppts[ppts_index];
309
310      if (ppts_index < (ppt_split.ppts.length - 1)) {
311        ppts_index++;
312      } else {
313        splitter_index++;
314        ppts_index = 0;
315      }
316      return ppt;
317    }
318
319    @Override
320    public void remove(@GuardSatisfied CondIterator this) {
321      throw new UnsupportedOperationException("Remove unsupported in CondIterator");
322    }
323  }

445  @RequiresNonNull("var_infos")
446  @EnsuresNonNull({"mbtracker", "views", "value_sets"})
447  private void init_vars(@UnderInitialization(Ppt.class) PptTopLevel this) {
448
449    // debug_varinfo.log("initializing var_infos %s", Arrays.toString(var_infos));
450    // debug_varinfo.tb();
451
452    int val_idx = 0;
453    num_static_constant_vars = 0;
454    for (int i = 0; i < var_infos.length; i++) {
455      VarInfo vi = var_infos[i];
456      vi.varinfo_index = i;
457      if (vi.is_static_constant) {
458        vi.value_index = -1;
459        num_static_constant_vars++;
460      } else {
461        vi.value_index = val_idx;
462        val_idx++;
463      }
464    }
465    for (VarInfo vi : var_infos) {
466      assert (vi.value_index == -1) || !vi.is_static_constant;
467    }
468
469    views = new LinkedHashMap<>();
470
471    num_declvars = var_infos.length;
472    num_tracevars = val_idx;
473    num_orig_vars = 0;
474    assert num_static_constant_vars == num_declvars - num_tracevars;
475    assert num_tracevars == var_infos.length - num_static_constant_vars;
476    mbtracker = new ModBitTracker(num_tracevars);
477    /*NNC:@MonotonicNonNull*/ ValueSet[] new_value_sets = new ValueSet[num_tracevars];
478    for (VarInfo vi : var_infos) {
479      int value_index = vi.value_index;
480      if (value_index == -1) {
481        continue;
482      }
483      assert new_value_sets[value_index] == null;
484      new_value_sets[value_index] = ValueSet.factory(vi);
485    }
486    for (ValueSet vs : new_value_sets) {
487      assert vs != null;
488    }
489    new_value_sets = castNonNullDeep(new_value_sets); // https://tinyurl.com/cfissue/986
490    value_sets = new_value_sets;
491
492    for (VarInfo vi : var_infos) {
493      // TODO: This should not be necessary, since initialization is now complete
494      @SuppressWarnings({"nullness:assignment.type.incompatible"}) // initialization is now complete
495      @Initialized PptTopLevel initializedThis = this;
496      vi.ppt = initializedThis;
497    }
498
499    // Fix variable pointers so that they refer to the variables
500    // in this program point (they may have been cloned from a diff
501    // program point)
502    for (VarInfo vi : var_infos) {
503      vi.update_after_moving_to_new_ppt();
504    }
505
506    // Relate the variables to one another
507    for (VarInfo vi : var_infos) {
508      vi.relate_var();
509    }
510  }

559  /** Return the number of samples where vi1 and vi2 are both present (not missing). */
560  public int num_samples(VarInfo vi1, VarInfo vi2) {
561    if (vi1.is_static_constant) {
562      return num_samples(vi2);
563    }
564    if (vi2.is_static_constant) {
565      return num_samples(vi1);
566    }
567    BitSet b1 = mbtracker.get(vi1.value_index);
568    BitSet b2 = mbtracker.get(vi2.value_index);
569    int num_slice_samples = UtilPlume.intersectionCardinality(b1, b2);
570    return num_slice_samples;
571  }

573  /** Return the number of samples where vi1, vi2, and vi3 are all present (not missing). */
574  public int num_samples(VarInfo vi1, VarInfo vi2, VarInfo vi3) {
575    if (vi1.is_static_constant) {
576      return num_samples(vi2, vi3);
577    }
578    if (vi2.is_static_constant) {
579      return num_samples(vi1, vi3);
580    }
581    if (vi3.is_static_constant) {
582      return num_samples(vi1, vi2);
583    }
584    BitSet b1 = mbtracker.get(vi1.value_index);
585    BitSet b2 = mbtracker.get(vi2.value_index);
586    BitSet b3 = mbtracker.get(vi3.value_index);
587    int num_slice_samples = UtilPlume.intersectionCardinality(b1, b2, b3);
588    return num_slice_samples;
589  }

747  /**
748   * This routine creates derivations for one "pass"; that is, it adds some set of derived
749   * variables, according to the functions that are passed in. All the results involve at least one
750   * VarInfo object at an index i such that vi_index_min &le; i &lt; vi_index_limit (and possibly
751   * other VarInfos outside that range).
752   */
753  private Derivation[] derive(int vi_index_min, int vi_index_limit) {
754    boolean debug_bin_possible = false;
755
756    UnaryDerivationFactory[] unary = unaryDerivations;
757    BinaryDerivationFactory[] binary = binaryDerivations;
758    TernaryDerivationFactory[] ternary = ternaryDerivations;
759
760    // optimize track logging, otherwise it really takes a lot of time
761    if (Debug.logOn()) {
762      for (int di = 0; di < binary.length; di++) {
763        BinaryDerivationFactory d = binary[di];
764        if (Debug.class_match(d.getClass())) debug_bin_possible = true;
765      }
766    }
767
768    if (Global.debugDerive.isLoggable(Level.FINE)) {
769      Global.debugDerive.fine("Deriving one pass for ppt " + this.name);
770      Global.debugDerive.fine(
771          "vi_index_min="
772              + vi_index_min
773              + ", vi_index_limit="
774              + vi_index_limit
775              + ", unary.length="
776              + unary.length
777              + ", binary.length="
778              + binary.length
779              + ", ternary.length="
780              + ternary.length);
781    }
782
783    Collection<Derivation> result = new ArrayList<Derivation>();
784
785    Daikon.progress = "Creating derived variables for: " + ppt_name.toString() + " (unary)";
786    for (int i = vi_index_min; i < vi_index_limit; i++) {
787      VarInfo vi = var_infos[i];
788      if (Global.debugDerive.isLoggable(Level.FINE)) {
789        Global.debugDerive.fine("Unary: trying to derive from " + vi.name());
790      }
791      if (!worthDerivingFrom(vi)) {
792        if (Global.debugDerive.isLoggable(Level.FINE)) {
793          Global.debugDerive.fine("Unary: not worth deriving from " + vi.name());
794        }
795        continue;
796      }
797      for (int di = 0; di < unary.length; di++) {
798        UnaryDerivationFactory udf = unary[di];
799        UnaryDerivation[] uderivs = udf.instantiate(vi);
800        if (uderivs != null) {
801          for (int udi = 0; udi < uderivs.length; udi++) {
802            UnaryDerivation uderiv = uderivs[udi];
803            // System.out.printf("processing uderiv %s %s%n", uderiv,
804            //                   uderiv.getVarInfo());
805            if (!FileIO.var_included(uderiv.getVarInfo().name())) {
806              continue;
807            }
808            result.add(uderiv);
809          }
810        }
811      }
812    }
813
814    // I want to get all pairs of variables such that at least one of the
815    // variables is under consideration, but I want to generate each such
816    // pair only once.  This probably isn't the most efficient technique,
817    // but it's probably adequate and is not excessively complicated or
818    // excessively slow.
819    for (int i1 = 0; i1 < var_infos.length; i1++) {
820      VarInfo vi1 = var_infos[i1];
821      if (!worthDerivingFrom(vi1)) {
822        if (Global.debugDerive.isLoggable(Level.FINE)) {
823          Global.debugDerive.fine("Binary first VarInfo: not worth deriving from " + vi1.name());
824        }
825        continue;
826      }
827      // This guarantees that at least one of the variables is under
828      // consideration.
829      // target1 indicates whether the first variable is under consideration.
830      boolean target1 = (i1 >= vi_index_min) && (i1 < vi_index_limit);
831      int i2_min, i2_limit;
832      if (target1) {
833        i2_min = i1 + 1;
834        i2_limit = var_infos.length;
835      } else {
836        i2_min = Math.max(i1 + 1, vi_index_min);
837        i2_limit = vi_index_limit;
838      }
839      // if (Global.debugDerive.isLoggable(Level.FINE))
840      //   Global.debugDerive.fine ("i1=" + i1
841      //                      + ", i2_min=" + i2_min
842      //                      + ", i2_limit=" + i2_limit);
843      Daikon.progress =
844          "Creating derived variables for: "
845              + ppt_name.toString()
846              + " (binary, "
847              + vi1.name()
848              + ")";
849      for (int i2 = i2_min; i2 < i2_limit; i2++) {
850        VarInfo vi2 = var_infos[i2];
851        if (!worthDerivingFrom(vi2)) {
852          if (Global.debugDerive.isLoggable(Level.FINE)) {
853            Global.debugDerive.fine(
854                "Binary: not worth deriving from (" + vi1.name() + "," + vi2.name() + ")");
855          }
856          continue;
857        }
858        for (int di = 0; di < binary.length; di++) {
859          BinaryDerivationFactory d = binary[di];
860          if (debug_bin_possible && Debug.logOn()) {
861            Debug.log(d.getClass(), vi1.ppt, Debug.vis(vi1, vi2), "Trying Binary Derivation ");
862          }
863          BinaryDerivation[] bderivs = d.instantiate(vi1, vi2);
864          if (bderivs != null) {
865            for (int bdi = 0; bdi < bderivs.length; bdi++) {
866              BinaryDerivation bderiv = bderivs[bdi];
867              if (!FileIO.var_included(bderiv.getVarInfo().name())) {
868                continue;
869              }
870              result.add(bderiv);
871              if (Debug.logOn()) {
872                Debug.log(
873                    d.getClass(),
874                    vi1.ppt,
875                    Debug.vis(vi1, vi2),
876                    "Created Binary Derivation " + bderiv.getVarInfo().name());
877              }
878            }
879          }
880        }
881      }
882    }
883
884    // Ternary derivations follow the same pattern, one level deeper.
885    for (int i1 = 0; i1 < var_infos.length; i1++) {
886      VarInfo vi1 = var_infos[i1];
887      if (vi1.isDerived()) {
888        if (Global.debugDerive.isLoggable(Level.FINE)) {
889          Global.debugDerive.fine("Ternary first VarInfo: not worth deriving from " + vi1.name());
890        }
891        continue;
892      }
893      // This guarantees that at least one of the variables is under
894      // consideration.
895      // target1 indicates whether the first variable is under consideration.
896      boolean target1 = (i1 >= vi_index_min) && (i1 < vi_index_limit);
897      int i2_min, i2_limit;
898      if (target1) {
899        i2_min = i1 + 1;
900        i2_limit = var_infos.length;
901      } else {
902        i2_min = Math.max(i1 + 1, vi_index_min);
903        i2_limit = vi_index_limit;
904      }
905      // if (Global.debugDerive.isLoggable(Level.FINE))
906      //   Global.debugDerive.fine ("i1=" + i1
907      //                      + ", i2_min=" + i2_min
908      //                      + ", i2_limit=" + i2_limit);
909      Daikon.progress =
910          "Creating derived variables for: "
911              + ppt_name.toString()
912              + " (ternary, "
913              + vi1.name()
914              + ")";
915      for (int i2 = i2_min; i2 < i2_limit; i2++) {
916        VarInfo vi2 = var_infos[i2];
917        if (vi2.isDerived()
918            || !TernaryDerivationFactory.checkType(vi1, vi2)
919            || !TernaryDerivationFactory.checkComparability(vi1, vi2)) {
920          if (Global.debugDerive.isLoggable(Level.FINE)) {
921            Global.debugDerive.fine(
922                "Ternary 2nd: not worth deriving from (" + vi1.name() + "," + vi2.name() + ")");
923          }
924          continue;
925        }
926        boolean target2 = (i2 >= vi_index_min) && (i2 < vi_index_limit);
927        int i3_min, i3_limit;
928        if (target1 || target2) {
929          i3_min = i2 + 1;
930          i3_limit = var_infos.length;
931        } else {
932          i3_min = Math.max(i2 + 1, vi_index_min);
933          i3_limit = vi_index_limit;
934        }
935        for (int i3 = i3_min; i3 < i3_limit; i3++) {
936          VarInfo vi3 = var_infos[i3];
937          if (vi3.isDerived()) {
938            if (Global.debugDerive.isLoggable(Level.FINE)) {
939              Global.debugDerive.fine(
940                  "Ternary 3rd: not worth deriving from ("
941                      + vi1.name()
942                      + ","
943                      + vi2.name()
944                      + ")"
945                      + vi3.name()
946                      + ")");
947            }
948            continue;
949          }
950          for (int di = 0; di < ternary.length; di++) {
951            TernaryDerivationFactory d = ternary[di];
952            TernaryDerivation[] tderivs = d.instantiate(vi1, vi2, vi3);
953            if (tderivs != null) {
954              for (int tdi = 0; tdi < tderivs.length; tdi++) {
955                TernaryDerivation tderiv = tderivs[tdi];
956                if (!FileIO.var_included(tderiv.getVarInfo().name())) {
957                  continue;
958                }
959                result.add(tderiv);
960              }
961            } else {
962              if (Global.debugDerive.isLoggable(Level.FINE)) {
963                Global.debugDerive.fine(
964                    "Ternary instantiated but not used: "
965                        + vi1.name()
966                        + " "
967                        + vi2.name()
968                        + " "
969                        + vi3.name()
970                        + " ");
971              }
972            }
973          }
974        }
975      }
976    }
977
978    if (Global.debugDerive.isLoggable(Level.FINE)) {
979      Global.debugDerive.fine(
980          "Number of derived variables at program point " + this.name + ": " + result.size());
981      String derived_vars = "Derived:";
982      for (Iterator<Derivation> itor = result.iterator(); itor.hasNext(); ) {
983        derived_vars += " " + itor.next().getVarInfo().name();
984      }
985      Global.debugDerive.fine(derived_vars);
986    }
987    Derivation[] result_array = result.toArray(new Derivation[result.size()]);
988    return result_array;
989  }

2960    // expressible invariants, so that we can remove the least

2981    // partial ordering to determine background invariants, instead of

2999    // since in the current scheme, implications came from controlled

3010          continue;
3011        }

3014          // If format_simplify is not defined for this invariant, don't
3015          // confuse Simplify with the error message

3019        // was removed by Simplify, but what would the point be?  Also,
3020        // these "intermediate goals" might help out Simplify.

3023        // If this is the :::OBJECT ppt, also restate all of them in
3024        // orig terms, since the conditions also held upon entry.

3027        }

3034            "Warning: " + ppt_name + " background is contradictory, removing some parts");

4221  /** Builds a permutation from vis1 to vis2. The result is vis1[i] = vis2[permute[i]]. */
4222  public static int[] build_permute(VarInfo[] vis1, VarInfo[] vis2) {
4223
4224    int[] permute = new int[vis1.length];
4225    boolean[] matched = new boolean[vis1.length];
4226    Arrays.fill(matched, false);
4227
4228    for (int j = 0; j < vis1.length; j++) {
4229      for (int k = 0; k < vis2.length; k++) {
4230        if ((vis1[j] == vis2[k]) && !matched[k]) {
4231          permute[j] = k;
4232          matched[k] = true; // don't match the same one twice
4233          break;
4234        }
4235      }
4236    }
4237
4238    // Check results
4239    for (int j = 0; j < vis1.length; j++) {
4240      assert vis1[j] == vis2[permute[j]];
4241    }
4242
4243    return permute;
4244  }

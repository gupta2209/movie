088  /**
089   * Checks this suppression. Each suppressor is checked to see if it matches inv and if not,
090   * whether or not it is valid (true). The results are saved in each suppressor. The suppressor
091   * results are used later by {@link #invalidated()}.
092   *
093   * @param ppt program point in which to check suppression
094   * @param vis variables over which to check suppression
095   * @param inv falsified invariant (if any). Any suppressor that matches inv will be marked as
096   *     NIS.SuppressState.MATCH
097   * @return NIS.SuppressState.VALID if the suppression is valid, NIS.SuppressState.NONSENSICAL if
098   *     one or more suppressors were nonsensical and the rest were valid, NIS.SuppressState.INVALID
099   *     otherwise
100   */
101  public NIS.SuppressState check(PptTopLevel ppt, VarInfo[] vis, @Nullable Invariant inv) {
102
103    NIS.SuppressState status = NIS.SuppressState.VALID;
104    boolean set = false;
105    for (int i = 0; i < suppressors.length; i++) {
106      NISuppressor ssor = suppressors[i];
107      NIS.SuppressState st = ssor.check(ppt, vis, inv);
108
109      if (!set) {
110        if (st == NIS.SuppressState.NONSENSICAL) {
111          status = NIS.SuppressState.NONSENSICAL;
112        } else if (st != NIS.SuppressState.VALID) {
113          status = NIS.SuppressState.INVALID;
114          if (st == NIS.SuppressState.INVALID) {
115            return status;
116          }
117          // !valid in this case means invalid or match
118          // If invalid, then stop immediately
119          // otherwise, check state of the rest of the suppressors
120          // This check is needed because we are reviving the
121          // falsified method so match is now a valid status.
122          set = true;
123        }
124      }
125    }
126    return status;
127  }

129  /**
130   * Determines whether or not the falsified invariant previously passed to {@link
131   * #check(PptTopLevel,VarInfo[],Invariant)} was the first suppressor to be falsified in this
132   * suppression. If the falsified invariant is not involved in this suppression, then it can't have
133   * been invalidated.
134   */
135  public boolean invalidated() {
136
137    // We return true when every suppressor except the falsified
138    // one is valid and at least one suppressor matches the falsified
139    // invariant.  Note that match can be true on more than one
140    // suppressor due to reflexive (x, x, x) invariants.  In this
141    // code, the suppressor should never be nonsensical, since we should
142    // have never looked at a slice with nonsensical variables.
143    boolean inv_match = false;
144    for (int i = 0; i < suppressors.length; i++) {
145      NISuppressor ssor = suppressors[i];
146      assert ssor.state != NIS.SuppressState.NONSENSICAL;
147      if (ssor.state == NIS.SuppressState.MATCH) {
148        inv_match = true;
149      } else if (ssor.state != NIS.SuppressState.VALID) {
150        return false;
151      }
152    }
153    return inv_match;
154  }

156  /**
157   * Finds all of the invariants that are suppressed by this suppression.
158   *
159   * @param suppressed_invs any invariants that are suppressed by the antecedent invariants in
160   *     {@code ants} using this suppression are added to this set
161   * @param ants antecedents organized by class
162   */
163  public void find_suppressed_invs(Set<NIS.SupInv> suppressed_invs, NIS.Antecedents ants) {
164
165    // debug = suppressee.sup_class.getName().indexOf("NonZero") != -1;
166    if (debug) {
167      System.out.println("In find_suppressed_invs for " + this);
168    }
169
170    // Get the antecedents that match our suppressors.  Return if there are
171    // no antecedents for a particular suppressor.
172    List<Invariant>[] antecedents = antecedents_for_suppressors(ants);
173    if (antecedents == null) {
174      return;
175    }
176
177    // Recursively check each combination of possible antecedents that
178    // match our suppressors for suppressions
179    VarInfo vis[] = new VarInfo[suppressee.var_count];
180    find_suppressed_invs(suppressed_invs, antecedents, vis, 0);
181
182    if (debug) System.out.println("  suppressed invariants: " + suppressed_invs);
183  }

185  /**
186   * Finds invariants that have become unsuppressed (one or more of their antecedent invariants is
187   * falsified). The invariant may still be suppressed by a different suppression.
188   *
189   * @param unsuppressed_invs any invariants that are suppressed by the antecedent invariants in
190   *     ants using this suppression are added to this set if one or more of the antecedents are
191   *     falsified
192   * @param ants antecedents organized by class
193   */
194  public void find_unsuppressed_invs(Set<NIS.SupInv> unsuppressed_invs, NIS.Antecedents ants) {
195
196    // debug = suppressee.sup_class.getName().indexOf("SeqIntLessEqual") != -1;
197
198    // Get the antecedents that match our suppressors.  Return if there are
199    // no antecedents for a particular suppressor.
200    List<Invariant>[] antecedents = antecedents_for_suppressors(ants);
201    if (antecedents == null) {
202      return;
203    }
204
205    int total_false_cnt = 0;
206    for (int i = 0; i < antecedents.length; i++) {
207      List<Invariant> a = antecedents[i];
208      int false_cnt = 0;
209      for (Invariant inv : a) {
210        if (inv.is_false()) false_cnt++;
211      }
212
213      total_false_cnt += false_cnt;
214    }
215
216    if (total_false_cnt == 0) {
217      return;
218    }
219
220    // Recursively check each combination of possible antecedents that
221    // match our suppressors for suppressions
222    VarInfo vis[] = new VarInfo[suppressee.var_count];
223    // int old_size = unsuppressed_invs.size();
224    Invariant[] cinvs = new Invariant[antecedents.length];
225    find_unsuppressed_invs(unsuppressed_invs, antecedents, vis, 0, false, cinvs);
226    if (debug) System.out.println("  unsuppressed invariants: " + unsuppressed_invs);
227  }

229  /**
230   * Recursively finds suppressed invariants. The cross product of antecedents for each suppressor
231   * are examined and each valid combination will yield an entry in suppressed_invs.
232   *
233   * @param unsuppressed_invs this set is updated with any invariants that used to be suppressed,
234   *     but no longer are
235   * @param antecedents array of antecedents per suppressor
236   * @param vis current variables for the suppressed invariant As antecedents are chosen, their
237   *     variables are placed into vis
238   * @param idx current index into suppressors and antecedents
239   * @see #find_unsuppressed_invs (Set, List, VarInfo[], int, boolean)
240   * @see #consider_inv (Invariant, NISuppressor, VarInfo[])
241   */
242  private void find_suppressed_invs(
243      Set<NIS.SupInv> unsuppressed_invs, List<Invariant> antecedents[], VarInfo vis[], int idx) {
244
245    // Loop through each antecedent that matches the current suppressor
246    NISuppressor s = suppressors[idx];
247    for (Invariant inv : antecedents[idx]) {
248      PptTopLevel ppt = inv.ppt.parent;
249      assert ppt.equality_view != null : "@AssumeAssertion(nullness): need to check justification";
250
251      // See if this antecedent can be used with the ones we have found so far
252      VarInfo[] cvis = consider_inv(inv, s, vis);
253      if (cvis == null) {
254        continue;
255      }
256
257      // If this is the last suppressor
258      if ((idx + 1) == suppressors.length) {
259
260        // Create descriptions of the suppressed invariants
261        List<NIS.SupInv> new_invs = suppressee.find_all(cvis, ppt, null);
262        unsuppressed_invs.addAll(new_invs);
263
264        // Check to insure that none of the invariants already exists
265        if (Debug.dkconfig_internal_check) {
266          for (NIS.SupInv supinv : new_invs) {
267            Invariant cinv = supinv.already_exists();
268            if (cinv != null) {
269              @SuppressWarnings("nullness")
270              @NonNull NISuppressionSet ss = cinv.get_ni_suppressions();
271              // this is apparently called for side effect (debugging output)
272              ss.suppressed(cinv.ppt);
273              throw new Error(
274                  "inv "
275                      + cinv.repr()
276                      + " of class "
277                      + supinv.suppressee
278                      + " already exists in ppt "
279                      + ppt.name
280                      + " suppressionset = "
281                      + ss
282                      + " suppression = "
283                      + this
284                      + " last antecedent = "
285                      + inv.format());
286            }
287          }
288        }
289      } else {
290        // Recursively process the next suppressor
291        find_suppressed_invs(unsuppressed_invs, antecedents, cvis, idx + 1);
292      }
293    }
294  }

296  /**
297   * Recursively finds unsuppressed invariants. The cross product of antecedents for each suppressor
298   * is examined and each valid combination with at least one falsified antecedent will yield an
299   * entry in unsuppressed_invs.
300   *
301   * @param unsuppressed_invs this set is updated with any invariants that were suppressed, but one
302   *     of the suppressors is falsified (thus, the invariant is no longer suppressed)
303   * @param antecedents array of antecedents per suppressor
304   * @param vis current variables for the suppressed invariant As antecedents are chosen, their
305   *     variables are placed into vis
306   * @param idx current index into suppressors and antecedents
307   * @param false_antecedents true if a false antecedent has been found
308   * @param cinvs the invariants associated with the current set of antecedents. Used only for debug
309   *     printing. May be side-effected by having cinvs[idx] set to null.
310   * @see find_unsuppressed_invs (Set, List, VarInfo[], int)
311   * @see #consider_inv (Invariant, NISuppressor, VarInfo[])
312   */
313  private void find_unsuppressed_invs(
314      Set<NIS.SupInv> unsuppressed_invs,
315      List<Invariant> antecedents[],
316      VarInfo vis[],
317      int idx,
318      boolean false_antecedents,
319      @Nullable Invariant[] cinvs) {
320
321    boolean all_true_at_end = ((idx + 1) == suppressors.length) && !false_antecedents;
322
323    // Loop through each antecedent that matches the current suppressor
324    NISuppressor s = suppressors[idx];
325    for (Invariant inv : antecedents[idx]) {
326      PptTopLevel ppt = inv.ppt.parent;
327      assert ppt.equality_view != null : "@AssumeAssertion(nullness): need to check justification";
328      cinvs[idx] = inv;
329
330      // If this is the last suppressor, no previous antecedents were
331      // false, and this antecedent is not false either, we can stop
332      // checking.  The antecedent lists are sorted so that the false
333      // ones are first.  There is no need to look at antecedents that
334      // are all true.
335      if (all_true_at_end && !inv.is_false()) {
336        cinvs[idx] = null;
337        return;
338      }
339      // See if this antecedent can be used with the ones we have found so far
340      VarInfo[] cvis = consider_inv(inv, s, vis);
341      if (cvis == null) {
342        continue;
343      }
344
345      // If this is the last suppressor
346      if ((idx + 1) == suppressors.length) {
347
348        // JHP: this check can be removed if the earlier check for all
349        // true antecedents is included.
350        if (!false_antecedents && !inv.is_false()) {
351          if (debug) {
352            System.out.printf("Skipping %s, no false antecedents%n", Arrays.toString(cvis));
353          }
354          continue;
355        }
356
357        // Create descriptions of the suppressed invariants
358        List<NIS.SupInv> new_invs = suppressee.find_all(cvis, ppt, cinvs);
359        if (debug) System.out.printf("created %s new invariants", new_invs);
360        unsuppressed_invs.addAll(new_invs);
361
362        // Check to insure that none of the invariants already exists
363        if (Debug.dkconfig_internal_check) {
364          for (NIS.SupInv supinv : new_invs) {
365            Invariant cinv = supinv.already_exists();
366            if (cinv != null) {
367              throw new Error(
368                  "inv "
369                      + cinv.format()
370                      + " of class "
371                      + supinv.suppressee
372                      + " already exists in ppt "
373                      + ppt.name);
374            }
375          }
376        }
377      } else {
378        // Recursively process the next suppressor
379        find_unsuppressed_invs(
380            unsuppressed_invs,
381            antecedents,
382            cvis,
383            idx + 1,
384            false_antecedents || inv.is_false(),
385            cinvs);
386      }
387    }
388    cinvs[idx] = null;
389  }

391  /**
392   * Determine if the specified invariant can be used as part of this suppression. The invariant
393   * must match suppressor and its variables must match up with any antecedents that have been
394   * previously processed. As invariants are processed by this method, their variables are added to
395   * the slots in vis that correspond to their suppressor.
396   *
397   * <p>For example, consider the invariant 'result = arg1 * arg2', the suppression '(result=arg1) ^
398   * (arg2=1)' and the invariants 'x = y' and 'q = 1'. If the varinfo_index of 'q' is less than 'x'
399   * then it can't be used (because it would form an invalid permutation. Note that this set of
400   * antecedents will match a different suppression for multiply that has a different argument
401   * permutation. More complex suppressions may refer to the same variable more than once. In those
402   * cases, the antecedent invariants must also be over the same variables.
403   *
404   * @param inv the invariant to attempt to add to the suppression
405   * @param supor the suppressor we are trying to match
406   * @param vis the current variables (if any) that have already been determined by previous
407   *     antecedents
408   * @return a new VarInfo[] containing the variables of inv, or null if inv does not match in some
409   *     way
410   */
411  private VarInfo @Nullable [] consider_inv(Invariant inv, NISuppressor supor, VarInfo[] vis) {
412
413    // Make sure this invariant really matches this suppressor.  We know
414    // the class already matches, but if the invariant has a swap variable
415    // it must match as well
416    if (!supor.match(inv)) {
417      return null;
418    }
419
420    // Assign the variables from this invariant into vis.  If a variable
421    // is already there and doesn't match this variable, then this
422    // antecedent can't be used.
423    VarInfo v1 = inv.ppt.var_infos[0];
424    if ((vis[supor.v1_index] != null) && (vis[supor.v1_index] != v1)) {
425      return null;
426    }
427    if ((supor.v2_index != -1)
428        && (vis[supor.v2_index] != null)
429        && (vis[supor.v2_index] != inv.ppt.var_infos[1])) {
430      return null;
431    }
432    VarInfo cvis[] = vis.clone();
433    cvis[supor.v1_index] = v1;
434    if (supor.v2_index != -1) {
435      cvis[supor.v2_index] = inv.ppt.var_infos[1];
436    }
437    if (debug) {
438      System.out.printf(
439          "Placed antecedent '%s' into cvis %s%n", inv.format(), Arrays.toString(cvis));
440    }
441
442    // Make sure the resulting variables are in the proper order and are
443    // compatible
444    if (!vis_order_ok(cvis) || !vis_compatible(cvis)) {
445      if (debug) System.out.println("Skipping, cvis has bad order or is incompatible");
446      return null;
447    }
448
449    return cvis;
450  }

452  /**
453   * Builds an array of lists of antecedents that corresponds to each suppressor in this
454   * suppression. Returns null if the list is empty for any suppressor (because that means there
455   * can't be any suppressions based on these antecedents).
456   */
457  List<Invariant> @Nullable [] antecedents_for_suppressors(NIS.Antecedents ants) {
458
459    @SuppressWarnings({"unchecked", "rawtypes"})
460    /*NNC:@MonotonicNonNull*/ List<Invariant> antecedents[] =
461        (List<Invariant>[]) new List[suppressors.length];
462
463    // Find the list of antecedents that matches each suppressor.  If any
464    // suppressor doesn't have any matching antecedents, there can't be
465    // any invariants that are suppressed by this suppression.
466    for (int i = 0; i < suppressors.length; i++) {
467      NISuppressor s = suppressors[i];
468      List<Invariant> alist = ants.get(s.get_inv_class());
469      if (alist == null) {
470        return null;
471      }
472      antecedents[i] = alist;
473    }
474
475    if (debug) {
476      System.out.println(
477          suppressee.sup_class.getName() + " " + antecedents_for_suppression(antecedents));
478    }
479
480    antecedents = castNonNullDeep(antecedents); // https://tinyurl.com/cfissue/986
481
482    return antecedents;
483  }

485  /**
486   * Determines whether the order of the variables in vis a valid permutations (i.e., their
487   * varinfo_index's are ordered). Null elements are ignored (and an all-null list is OK).
488   */
489  private boolean vis_order_ok(VarInfo[] vis) {
490
491    VarInfo prev = vis[0];
492    for (int i = 1; i < vis.length; i++) {
493      if ((prev != null) && (vis[i] != null)) {
494        if (vis[i].varinfo_index < prev.varinfo_index) {
495          return false;
496        }
497      }
498      if (vis[i] != null) prev = vis[i];
499    }
500    return true;
501  }

503  /**
504   * Determines if the non-null entries in vis are comparable. Returns true if they are, false if
505   * they are not.
506   *
507   * <p>JHP: This should really be part of is_slice_ok.
508   */
509  public static boolean vis_compatible(VarInfo[] vis) {
510
511    // Unary vis are always compatble
512    if (vis.length == 1) {
513      return true;
514    }
515
516    // Check binary
517    if (vis.length == 2) {
518      if ((vis[0] == null) || (vis[1] == null)) {
519        return true;
520      }
521
522      if (vis[0].rep_type.isArray() == vis[1].rep_type.isArray()) {
523        return (vis[0].compatible(vis[1]));
524      } else if (vis[0].rep_type.isArray()) {
525        return (vis[0].eltsCompatible(vis[1]));
526      } else {
527        return (vis[1].eltsCompatible(vis[0]));
528      }
529    }
530
531    // Check ternary
532    if ((vis[1] != null) && (vis[2] != null)) {
533      if (!vis[1].compatible(vis[2])) {
534        return false;
535      }
536    }
537
538    if ((vis[0] != null) && (vis[2] != null)) {
539      if (!vis[0].compatible(vis[2])) {
540        return false;
541      }
542    }
543
544    return true;
545  }

547  public List<NISuppression> recurse_definition(NISuppressionSet ss) {
548
549    NISuppressee sse = ss.get_suppressee();
550    List<NISuppression> new_suppressions = new ArrayList<>();
551
552    // Create a list of all of our suppressors that don't match the suppressee
553    // of ss
554    List<NISuppressor> old_sors = new ArrayList<>();
555    NISuppressor match = null;
556    for (int i = 0; i < suppressors.length; i++) {
557      if (suppressors[i].match(sse)) {
558        match = suppressors[i];
559      } else {
560        old_sors.add(suppressors[i]);
561      }
562    }
563
564    // If we didn't match any suppressor there is nothing to do
565    if (match == null) {
566      return new_suppressions;
567    }
568
569    // Right now this only works if we match exactly one suppressor
570    assert (old_sors.size() + 1) == suppressors.length;
571
572    // Create one new suppression for each suppression in ss.  The suppressee
573    // of ss is replaced by one of the suppressions of ss.  Each suppressor
574    // in ss have its variable indices modified to match the original
575    // suppressor.
576    for (int i = 0; i < ss.suppression_set.length; i++) {
577      NISuppression s = ss.suppression_set[i];
578      List<NISuppressor> sors = new ArrayList<>(old_sors);
579      for (int j = 0; j < s.suppressors.length; j++) {
580        sors.add(s.suppressors[j].translate(match));
581      }
582      new_suppressions.add(new NISuppression(sors, suppressee));
583    }
584
585    return new_suppressions;
586  }

588  /** Clears the suppressor state in each suppressor. */
589  public void clear_state() {
590    for (int i = 0; i < suppressors.length; i++) {
591      suppressors[i].clear_state();
592    }

595  /** Returns {@code "suppressor && suppressor ... ==> suppressee"}. */
596  @SideEffectFree
597  @Override
598  public String toString(@GuardSatisfied NISuppression this) {
599    String suppressorsString =
600        (suppressors.length == 1)
601            ? suppressors[0].toString()
602            : "(" + UtilPlume.join(" && ", suppressors) + ")";
603    return suppressorsString + " ==> " + suppressee;
604  }

606  /** Returns a string describing each of the antecedents for each suppressor. */
607  public String antecedents_for_suppression(List<Invariant> antecedents[]) {
608
609    String sep = Global.lineSep;
610
611    String out = "suppression " + this + sep;
612    for (int i = 0; i < antecedents.length; i++) {
613      out += "antecedents for suppressor " + i + sep;
614      for (Invariant inv : antecedents[i]) {
615        out += "    " + inv.format() + (inv.is_false() ? " [false]" : " t") + sep;
616      }
617    }
618    return out;
619  }

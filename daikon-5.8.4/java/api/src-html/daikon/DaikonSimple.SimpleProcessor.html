023/**
024 * DaikonSimple reads a declaration file and trace file and outputs a list of likely invariants
025 * using the simple incremental algorithm. Its methods parallel those of Daikon but oftentimes
026 * certain checks are eliminated from DaikonSimple's methods because there is less filtering of
027 * invariants and variables.
028 *
029 * <p>DaikonSimple was written to check the implementation of the optimizations in Daikon.
030 * DaikonSimple does not use any optimizations, and its processing will produce a complete set of
031 * true invariants. Daikon does have flags to "turn off" some of its optimizations but there are
032 * some optimizations are built into the way Daikon processes the samples (e.g. variable hierarchy
033 * and bottom up processing). In addition, we want to check the optimizations, so we don't want to
034 * bypass them. In Daikon, code was written to "undo" the optimizations, so we could recover the
035 * invariants that were previously filtered out or not created (see Daikon.dkconfig_undo_opts flag).
036 * By comparing the output from the two, we can find problems with the optimization implementation
037 * by tracking the cause of the differences.
038 */
039@SuppressWarnings("nullness") // not actively maintained
040public class DaikonSimple {
041

079  /**
080   * This does the work of {@link #main}, but it never calls System.exit, so it is appropriate to be
081   * called programmatically.
082   *
083   * <p>Difference from {@link daikon.Daikon#mainHelper(String[])}Helper: turn off optimization
084   * flags (equality, dynamic constants, NIS suppression).
085   */
086  public static void mainHelper(final String[] args) throws IOException, FileNotFoundException {
087
088    // set up logging information
089    daikon.LogHelper.setupLogs(daikon.LogHelper.INFO);
090
091    // No optimizations used in the simple incremental algorithm so
092    // optimizations are turned off.
093    Daikon.use_equality_optimization = false;
094    DynamicConstants.dkconfig_use_dynamic_constant_optimization = false;
095    NIS.dkconfig_enabled = false;
096
097    // The flag tells FileIO and Daikon to use DaikonSimple
098    // specific methods (e.g. FileIO.read_declaration_file).
099    // When FileIO reads and processes
100    // samples, it must use the SimpleProcessor rather than the
101    // default Processor.
102    Daikon.using_DaikonSimple = true;
103
104    // Read command line options
105    Daikon.FileOptions files = Daikon.read_options(args, usage);
106    // DaikonSimple does not supply nor use the spinfo_files and map_files
107    Set<File> decls_files = files.decls;
108    Set<String> dtrace_files = files.dtrace;
109
110    if ((decls_files.size() == 0) && (dtrace_files.size() == 0)) {
111      throw new Daikon.UserError("No .decls or .dtrace files specified");
112    }
113
114    // Create the list of all invariant types
115    Daikon.setup_proto_invs();
116
117    // Create the program points for enter and numbered exits and
118    // initializes the points (adding orig and derived variables)
119    all_ppts = FileIO.read_declaration_files(decls_files);
120
121    // Create the combined exits (and add orig and derived vars)
122    // Daikon.create_combined_exits(all_ppts);
123
124    // Read and process the data trace files
125    SimpleProcessor processor = new SimpleProcessor();
126    FileIO.read_data_trace_files(dtrace_files, all_ppts, processor, true);
127
128    // System.exit(0);
129
130    // Print out the invariants for each program point (sort first)
131    for (PptTopLevel ppt : all_ppts.pptIterable()) {
132
133      // We do not need to print out program points that have not seen
134      // any samples.
135      if (ppt.num_samples() == 0) {
136        continue;
137      }
138      List<Invariant> invs = PrintInvariants.sort_invariant_list(ppt.invariants_vector());
139      List<Invariant> filtered_invs = Daikon.filter_invs(invs);
140      // The dkconfig_quiet printing is used for creating diffs between
141      // DaikonSimple
142      // and Daikon's output. The second kind of printing is used for
143      // debugging. Since the names of the program points are the same for both
144      // Daikon and DaikonSimple, diffing the two output will result in
145      // only differences in the invariants, but we can not see at which program
146      // points these differing invariants appear. Using the second kind of
147      // printing,
148      // Daikon's output does not have the '+' in the program point name, so in
149      // addition
150      // to the invariants showing up in the diff, we will also see the program
151      // point
152      // names.
153
154      if (Daikon.dkconfig_quiet) {
155        System.out.println("====================================================");
156        System.out.println(ppt.name());
157      } else {
158        System.out.println("===================================================+");
159        System.out.println(ppt.name() + " +");
160      }
161
162      // Sometimes the program points actually differ in number of
163      // samples seen due to differences in how Daikon and DaikonSimple
164      // see the variable hierarchy.
165      System.out.println(ppt.num_samples());
166
167      for (Invariant inv : filtered_invs) {
168        System.out.println(inv.getClass());
169        System.out.println(inv);
170      }
171    }
172  }

174  /**
175   * Install views and the invariants. Duplicated from PptTopLevel's version because DaikonSimple
176   * needs to use its own version of slice checking code.
177   *
178   * <p>Difference from PptTopLevel's version:
179   *
180   * <ol>
181   *   <li>canonical (leader of equality set) check of variables is turned off because every
182   *       variable is in its own equality set
183   *   <li>debugging information turned off because DaikonSimple's code is more contained
184   *   <li>less constraints on the slices
185   * </ol>
186   *
187   * @see daikon.PptTopLevel#instantiate_views_and_invariants()
188   */
189
190  // Note that some slightly inefficient code has been added to aid
191  // in debugging. When creating binary and ternary views and debugging
192  // is on, the outer loops will not terminate prematurely on inappropriate
193  // (i.e., non-canonical) variables. This allows explicit debug statements
194  // for each possible combination, simplifying determining why certain
195  // slices were not created.
196  //
197  // Note that '///*' indicates code duplicated from PptTopLevel's
198  // version but commented out because DaikonSimple does not need
199  // to perform these checks
200  public static void instantiate_views_and_invariants(PptTopLevel ppt) {
201
202    // used only for debugging
203    int old_num_vars = ppt.var_infos.length;
204
205    // / 1. all unary views
206
207    // Unary slices/invariants.
208    // Currently, there are no constraints on the unary
209    // slices. Since we are trying to create all of the invariants, the
210    // variables does not have to be a leader and can be a constant.
211    // Note that the always missing check is only applicable when the
212    // dynamic constants optimization is turned on (so we do not do the
213    // check here).
214
215    List<PptSlice> unary_views = new ArrayList<>(ppt.var_infos.length);
216    for (VarInfo vi : ppt.var_infos) {
217
218      // /* if (!is_slice_ok(vi))
219      // /* continue;
220
221      PptSlice1 slice1 = new PptSlice1(ppt, vi);
222      slice1.instantiate_invariants();
223
224      unary_views.add(slice1);
225    }
226    ppt.addViews(unary_views);
227    unary_views = null;
228
229    // / 2. all binary views
230
231    // Binary slices/invariants.
232    List<PptSlice> binary_views = new ArrayList<>();
233    for (int i1 = 0; i1 < ppt.var_infos.length; i1++) {
234      VarInfo var1 = ppt.var_infos[i1];
235
236      // Variables can be constant and missing in DaikonSimple invariants
237      // /* if (!is_var_ok_binary(var1))
238      // /* continue;
239
240      for (int i2 = i1; i2 < ppt.var_infos.length; i2++) {
241        VarInfo var2 = ppt.var_infos[i2];
242
243        // Variables can be constant and missing in DaikonSimple invariants
244        // /* if (!is_var_ok_binary(var2))
245        // /* continue;
246
247        if (!(var1.compatible(var2)
248            || (var1.type.isArray() && var1.eltsCompatible(var2))
249            || (var2.type.isArray() && var2.eltsCompatible(var1)))) {
250          continue;
251        }
252
253        PptSlice2 slice2 = new PptSlice2(ppt, var1, var2);
254        slice2.instantiate_invariants();
255
256        binary_views.add(slice2);
257      }
258    }
259    ppt.addViews(binary_views);
260    binary_views = null;
261
262    // 3. all ternary views
263    List<PptSlice> ternary_views = new ArrayList<>();
264    for (int i1 = 0; i1 < ppt.var_infos.length; i1++) {
265      VarInfo var1 = ppt.var_infos[i1];
266
267      if (!is_var_ok(var1)) {
268        continue;
269      }
270
271      for (int i2 = i1; i2 < ppt.var_infos.length; i2++) {
272        VarInfo var2 = ppt.var_infos[i2];
273
274        if (!is_var_ok(var2)) {
275          continue;
276        }
277
278        for (int i3 = i2; i3 < ppt.var_infos.length; i3++) {
279          VarInfo var3 = ppt.var_infos[i3];
280
281          if (!is_var_ok(var3)) {
282            continue;
283          }
284
285          if (!is_slice_ok(var1, var2, var3)) {
286            continue;
287          }
288          PptSlice3 slice3 = new PptSlice3(ppt, var1, var2, var3);
289          slice3.instantiate_invariants();
290          ternary_views.add(slice3);
291        }
292      }
293    }
294
295    ppt.addViews(ternary_views);
296
297    // This method didn't add any new variables.
298    assert old_num_vars == ppt.var_infos.length;
299    ppt.repCheck();
300  }

302  // This method is exclusively for checking variables participating
303  // in ternary invariants. The variable must be integer or float, and
304  // can not be an array.
305  @Pure
306  public static boolean is_var_ok(VarInfo var) {
307
308    return (var.file_rep_type.isIntegral() || var.file_rep_type.isFloat())
309        && !var.rep_type.isArray();
310  }

312  /**
313   * Returns whether or not the specified binary slice should be created. The slice should not be
314   * created if the vars are not compatible.
315   *
316   * <p>Since we are trying to create all of the invariants, the variables does not have to be a
317   * leader and can be a constant. Note that the always missing check is only applicable when the
318   * dynamic constants optimization is turned on (so we do not do the check here).
319   *
320   * @see daikon.PptTopLevel#is_slice_ok(VarInfo, VarInfo)
321   */
322  @Pure
323  public static boolean is_slice_ok(VarInfo v1, VarInfo v2) {
324
325    return v1.compatible(v2);
326  }

328  /**
329   * Returns whether or not the specified ternary slice should be created. The slice should not be
330   * created if any of the following are true
331   *
332   * <ul>
333   *   <li>Any var is an array
334   *   <li>Any of the vars are not compatible with the others
335   *   <li>Any var is not (integral or float)
336   * </ul>
337   *
338   * Since we are trying to create all of the invariants, the variables does not have to be a leader
339   * and can be a constant. Note that the always missing check is only applicable when the dynamic
340   * constants optimization is turned on (so we do not do the check here). In addition, we do want
341   * to create the reflexive ones and partially reflexive invariants.
342   *
343   * @see daikon.PptTopLevel#is_slice_ok(VarInfo, VarInfo, VarInfo)
344   */
345  @Pure
346  public static boolean is_slice_ok(VarInfo v1, VarInfo v2, VarInfo v3) {
347
348    // Vars must be compatible
349    return (v1.compatible(v2) && v1.compatible(v3) && v2.compatible(v3));
350  }

352  /**
353   * The Call class helps the SimpleProcessor keep track of matching enter and exit program points
354   * and also object program points. Each Call object represents one entry in the dtrace file, i.e.
355   * enter, exit, object entry.
356   */
357  static final class Call {
358
359    public PptTopLevel ppt;
360
361    public ValueTuple vt;
362
363    public Call(PptTopLevel ppt, ValueTuple vt) {
364
365      this.ppt = ppt;
366      this.vt = vt;
367    }
368  }

370  /** The SimpleProcessor class processes each sample in the dtrace file. */
371  public static class SimpleProcessor extends FileIO.Processor {
372    PptMap all_ppts;
373
374    /** {@code nonce -> List<Call,Call>} */
375    // The first Call is the enter entry and the second is the object entry
376    Map<Integer, List<Call>> call_map = new LinkedHashMap<>();
377
378    // Flag for whether there are out of order entries in the
379    // dtrace file. For unterminated calls (enter but
380    // not exit entry in the dtrace file), because DaikonSimple had
381    // processed each entry separately (not bottom up like Daikon),
382    // DaikonSimple applied the enter and object call before seeing the
383    // exit call, which is not consistent with Daikon. Daikon does not
384    // process unterminated method calls.
385
386    // The method of holding the enter and object calls until finding
387    // a matching exit call assumes:
388    // - enter always comes before exit
389    // - first entry in dtrace is an enter
390    // - order in dtrace is enter, exit, object [for constructors] or
391    // enter, object, exit, object [for methods] but not necessarily
392    // sequential
393    boolean wait = false;
394
395    // pointer to last nonce so we can associate the object entry
396    // with the right enter entry
397    Integer last_nonce = -1;
398
399    /**
400     * Creates a ValueTuple for the receiver using the vt of the original. The method copies over
401     * the values of variables shared by both program points and sets the rest of the variables in
402     * the receiver's ValueTuple as missing. Also, adds the orig and derived variables to the
403     * receiver and returns the newly created ValueTuple.
404     */
405    private static ValueTuple copySample(
406        PptTopLevel receiver, PptTopLevel original, ValueTuple vt, int nonce) {
407
408      // Make the vt for the receiver ppt
409      //      Object[] values = new Object[receiver.num_tracevars];
410      //      int[] mods = new int[receiver.num_tracevars];
411      @Nullable @Interned Object[] values =
412          new @Interned Object[receiver.var_infos.length - receiver.num_static_constant_vars];
413      int[] mods = new int[receiver.var_infos.length - receiver.num_static_constant_vars];
414
415      // Build the vt for the receiver ppt by looking through the current
416      // vt and filling in the gaps.
417      int k = 0;
418      for (VarInfo var : receiver.var_infos) {
419        if (var.is_static_constant) {
420          continue;
421        }
422        boolean found = false;
423        for (VarInfo var2 : original.var_infos) {
424          if (var.name().equals(var2.name())) {
425            values[k] = vt.getValueOrNull(var2);
426            mods[k] = vt.getModified(var2);
427            found = true;
428            break;
429          }
430        }
431
432        if (!found) {
433          values[k] = null;
434          mods[k] = 2;
435        }
436        k++;
437      }
438
439      ValueTuple receiver_vt = new ValueTuple(values, mods);
440
441      FileIO.compute_orig_variables(receiver, receiver_vt.vals, receiver_vt.mods, nonce);
442      FileIO.compute_derived_variables(receiver, receiver_vt.vals, receiver_vt.mods);
443
444      return receiver_vt;
445    }

447    /**
448     * Process the sample by checking it against each existing invariant at the program point and
449     * removing the invariant from the list of possibles if any invariant is falsified.
450     */
451    @Override
452    public void process_sample(
453        PptMap all_ppts, PptTopLevel ppt, ValueTuple vt, @Nullable Integer nonce) {
454      this.all_ppts = all_ppts;
455
456      // Add samples to orig and derived variables
457      FileIO.compute_orig_variables(ppt, vt.vals, vt.mods, nonce);
458      FileIO.compute_derived_variables(ppt, vt.vals, vt.mods);
459
460      // Intern the sample
461      vt = new ValueTuple(vt.vals, vt.mods);
462
463      // DaikonSimple must make the object program point manually because
464      // the new Chicory produced dtrace files do not contain object ppts
465      // in the dtrace part of the file (the program point is declared).
466
467      // Make the object ppt
468      PptName ppt_name = ppt.ppt_name;
469
470      PptTopLevel object_ppt = null;
471      PptTopLevel class_ppt = null;
472      ValueTuple object_vt = null;
473      ValueTuple class_vt = null;
474
475      if ((ppt_name.isEnterPoint() && !ppt_name.isConstructor()) || ppt_name.isExitPoint()) {
476        object_ppt = all_ppts.get(ppt_name.makeObject());
477        class_ppt = all_ppts.get(ppt_name.makeClassStatic());
478      }
479
480      // C programs do not have object ppts
481      // check whether the ppt is a static or instance method
482      // that decides whether the sample is copied over to the object and/or
483      // class ppt
484      if (object_ppt != null) {
485
486        // the check assumes that static fields are not stored first in the
487        // object ppt
488        if (ppt.find_var_by_name(object_ppt.var_infos[0].name()) != null) {
489          // object and class ppt should be created
490          object_vt = copySample(object_ppt, ppt, vt, nonce);
491
492          if (class_ppt != null) {
493            class_vt = copySample(class_ppt, ppt, vt, nonce);
494          }
495
496        } else {
497          // only class ppt should be created
498          if (class_ppt != null) {
499            class_vt = copySample(class_ppt, ppt, vt, nonce);
500          }
501
502          object_vt = null;
503          object_ppt = null;
504        }
505      }
506
507      // If this is an enter point, just remember it for later
508      if (ppt_name.isEnterPoint()) {
509        assert nonce != null;
510        assert call_map.get(nonce) == null;
511        List<Call> value = new ArrayList<>();
512        value.add(new Call(ppt, vt));
513
514        if (object_ppt != null) {
515          value.add(new Call(object_ppt, object_vt));
516        }
517
518        if (class_ppt != null) {
519          value.add(new Call(class_ppt, class_vt));
520        }
521
522        call_map.put(nonce, value);
523        last_nonce = nonce;
524        wait = true;
525        return;
526      }
527
528      // If this is an exit point, process the saved enter (and sometimes
529      // object) point
530      if (ppt_name.isExitPoint()) {
531        assert nonce != null;
532        List<Call> value = call_map.remove(nonce);
533
534        add(ppt, vt, nonce);
535
536        for (Call ec : value) {
537          add(ec.ppt, ec.vt, nonce);
538        }
539        wait = false;
540      }
541
542      if (object_ppt != null) add(object_ppt, object_vt, nonce); // apply object vt
543
544      if (class_ppt != null) add(class_ppt, class_vt, nonce);
545    }

547    /**
548     * The method iterates through all of the invariants in the ppt and manually adds the sample to
549     * the invariant and removing the invariant if it is falsified.
550     */
551    private void add(PptTopLevel ppt, ValueTuple vt, int nonce) {
552
553      // if this is a numbered exit, apply to the combined exit as well
554      if (ppt.ppt_name.isNumberedExitPoint()) {
555
556        // Daikon.create_combined_exits(all_ppts);
557        PptTopLevel parent = all_ppts.get(ppt.ppt_name.makeExit());
558        if (parent != null) {
559          parent.get_missingOutOfBounds(ppt, vt);
560          add(parent, vt, nonce);
561
562        } else {
563          // make parent and apply
564
565          // this is a hack. it should probably filter out orig and derived
566          // vars instead of taking the first n.
567          int len = ppt.num_tracevars + ppt.num_static_constant_vars;
568          VarInfo[] exit_vars = new VarInfo[len];
569          for (int j = 0; j < len; j++) {
570            @SuppressWarnings("interning") // newly created, about to be used in a program point
571            @Interned VarInfo exit_var = new VarInfo(ppt.var_infos[j]);
572            exit_vars[j] = exit_var;
573            exit_vars[j].varinfo_index = ppt.var_infos[j].varinfo_index;
574            exit_vars[j].value_index = ppt.var_infos[j].value_index;
575            exit_vars[j].equalitySet = null;
576          }
577
578          parent = new PptTopLevel(ppt.ppt_name.makeExit().getName(), exit_vars);
579          Daikon.init_ppt(parent, all_ppts);
580          all_ppts.add(parent);
581          parent.get_missingOutOfBounds(ppt, vt);
582          add(parent, vt, nonce);
583        }
584      }
585
586      // If the point has no variables, skip it
587      if (ppt.var_infos.length == 0) {
588        // The sample should be skipped but Daikon does not do this so
589        // DaikonSimple will not do this to be consistent.
590        // The better idea is for Daikon to assert that these valuetuples are
591        // empty and then skip the sample.
592        assert vt.size() == 0;
593        return;
594      }
595
596      // Instantiate slices and invariants if this is the first sample
597      if (ppt.num_samples() == 0) {
598        instantiate_views_and_invariants(ppt);
599      }
600
601      // manually inc the sample number because DaikonSimple does not
602      // use any of PptTopLevel's add methods which increase the sample
603      // number
604      ppt.incSampleNumber();
605
606      // Loop through each slice
607      for (PptSlice slice : ppt.views_iterable()) {
608        Iterator<Invariant> k = slice.invs.iterator();
609        boolean missing = false;
610
611        for (VarInfo v : slice.var_infos) {
612          // If any var has encountered out of array bounds values,
613          // stop all invariants in this slice. The presumption here is that
614          // an index out of bounds implies that the derived variable (eg a[i])
615          // doesn't really make any sense (essentially that i is not a valid
616          // index for a). Invariants on the derived variable are thus not
617          // relevant.
618          // If any variables are out of bounds, remove the invariants
619          if (v.missingOutOfBounds()) {
620            while (k.hasNext()) {
621              k.next();
622              k.remove();
623            }
624            missing = true;
625            break;
626          }
627
628          // If any variables are missing, skip this slice
629          if (v.isMissing(vt)) {
630            missing = true;
631            break;
632          }
633        }
634
635        // keep a list of the falsified invariants
636        if (!missing) {
637          while (k.hasNext()) {
638
639            Invariant inv = k.next();
640            for (VarInfo vi : inv.ppt.var_infos) {
641              assert vt.getValue(vi) != null : vi;
642            }
643            if (inv.ppt instanceof PptSlice2) assert inv.ppt.var_infos.length == 2;
644            InvariantStatus status = inv.add_sample(vt, 1);
645            if (status == InvariantStatus.FALSIFIED) {
646              k.remove();
647            }
648          }
649        }
650
651        // update num_samples and num_values of a slice manually
652        // because DaikonSimple does not call any of PptTopLevel's
653        // add methods
654        for (int j = 0; j < vt.vals.length; j++) {
655          if (!vt.isMissing(j)) {
656            ValueSet vs = ppt.value_sets[j];
657            vs.add(vt.vals[j]);
658          }
659        }
660        ppt.mbtracker.add(vt, 1);
661      }
662    }
663  }

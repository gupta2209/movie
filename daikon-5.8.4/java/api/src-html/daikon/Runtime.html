018/**
019 * The Runtime class provides methods for printing values to a Daikon data trace file. Daikon can
020 * process the data trace information, either while the target program is running or after-the-fact,
021 * to produce likely invariants.
022 *
023 * <p>The Daikon front end for Java, named Chicory, modifies the target program by inserting calls
024 * to the printing routines of the Runtime class. Neither Chicory nor Daikon calls the methods of
025 * the Runtime class; only the target program (as instrumented by Chicory) does, in order to create
026 * input to Daikon.
027 */

101  /**
102   * Convert a classname from JVML format to Java format. For example, convert "[Ljava/lang/Object;"
103   * to "java.lang.Object[]".
104   *
105   * <p>If the argument is not a field descriptor, returns it as is. This enables this method to be
106   * used on the output of {@link Class#getName()}.
107   */
108  @SuppressWarnings("signature") // conversion function
109  public static @BinaryName String fieldDescriptorToBinaryName(@FieldDescriptor String classname) {
110    int dims = 0;
111    while (classname.startsWith("[")) {
112      dims++;
113      classname = classname.substring(1);
114    }
115    String result;
116    if (classname.startsWith("L") && classname.endsWith(";")) {
117      result = classname.substring(1, classname.length() - 1);
118      result = result.replace('/', '.');
119    } else {
120      result = primitiveClassesFromJvm.get(classname);
121      if (result == null) {
122        // If the input is not a field descriptor, return it.
123        result = classname;
124      }
125    }
126    for (int i = 0; i < dims; i++) {
127      result += "[]";
128    }
129    return result;
130  }

162  // Ensures that no more dtrace output will occur.  May terminate Java.
163  @SuppressWarnings("StaticGuardedByInstance")
164  public static void noMoreOutput() {
165    // The incrementRecords method (which calls this) is called inside a
166    // synchronized block, but re-synchronize just to be sure, or in case
167    // this is called from elsewhere.
168
169    // Runtime.dtrace should be effectively final in that it refers
170    // to the same value throughout the execution of the synchronized
171    // block below (including the lock acquisition).
172    // Unfortunately, the Lock Checker cannot verify this,
173    // so a final local variable is used to satisfy the Lock Checker's
174    // requirement that all variables used as locks be final or
175    // effectively final.  If a bug exists whereby Runtime.dtrace
176    // is not effectively final, this would unfortunately mask that error.
177    final @GuardedBy("<self>") PrintStream dtrace = Runtime.dtrace;
178
179    synchronized (dtrace) {
180      // The shutdown hook is synchronized on this, so close it up
181      // ourselves, lest the call to System.exit cause deadlock.
182      dtrace.println();
183      dtrace.println("# EOF (added by no_more_output)");
184      dtrace.close();
185
186      // Don't set dtrace to null, because if we continue running, there will
187      // be many attempts to synchronize on it.  (Is that a performance
188      // bottleneck, if we continue running?)
189      // daikon.Runtime.dtrace = null;
190      dtrace_closed = true;
191
192      if (dtraceLimitTerminate) {
193        // System.err.println("Printed " + printedRecords + " records.  Exiting.");
194        // System.exit(1);
195        throw new Daikon.NormalTermination("Printed " + printedRecords + " records.  Exiting.");
196      } else {
197        // By default, no special output if the system continues to run.
198        // System.err.println("Printed " + printedRecords + " records.  No more Daikon output.");
199        // prevent any future output
200        no_dtrace = true;
201        ps_count++;
202      }
203    }
204  }

228  @EnsuresNonNull("dtrace")
229  public static void setDtrace(String filename, boolean append) {
230    // System.out.printf("entered daikon.Runtime.setDtrace(%s, %b)%n", filename, append);
231    if (no_dtrace) {
232      throw new Error("setDtrace called when no_dtrace was specified");
233    }
234    try {
235      File file = new File(filename);
236      File parent = file.getParentFile();
237      if (parent != null) parent.mkdirs();
238      OutputStream os = new FileOutputStream(filename, append);
239      if (filename.endsWith(".gz")) {
240        if (append) {
241          throw new Error(
242              "DTRACEAPPEND environment variable is set."
243                  + lineSep
244                  + "Cannot append to gzipped dtrace file "
245                  + filename);
246        }
247        os = new GZIPOutputStream(os);
248      }
249      dtraceLimit = Integer.getInteger("DTRACELIMIT", Integer.MAX_VALUE).intValue();
250      dtraceLimitTerminate = Boolean.getBoolean("DTRACELIMITTERMINATE");
251      // 8192 is the buffer size in BufferedReader
252      BufferedOutputStream bos = new BufferedOutputStream(os, 8192);
253      dtrace = new PrintStream(bos);
254    } catch (Exception e) {
255      e.printStackTrace();
256      throw new Error(e);
257    }
258    if (supportsAddShutdownHook()) {
259      addShutdownHook();
260    } else {
261      System.err.println("Warning: .dtrace file may be incomplete if program is aborted");
262    }
263    // System.out.printf("exited daikon.Runtime.setDtrace(%s, %b)%n", filename, append);
264  }

266  /**
267   * If the current data trace file is not yet set (and no_dtrace is not true), then set it. The
268   * value of the DTRACEFILE environment variable is used; if that environment variable is not set,
269   * then the argument to this method is used instead.
270   *
271   * @param default_filename the filename for the trace file; used only if the DTRACEFILE
272   *     environment variable is not set
273   */
274  public static void setDtraceMaybe(String default_filename) {
275    // System.out.println("setDtraceMaybe(" + default_filename + "); old = " + dtrace);
276    if ((dtrace == null) && !no_dtrace) {
277      // Jeremy used "daikon.dtrace.filename".
278      String filename = System.getProperty("DTRACEFILE", default_filename);
279      boolean append = System.getProperty("DTRACEAPPEND") != null;
280      setDtrace(filename, append);
281    }
282  }

284  private static boolean supportsAddShutdownHook() {
285    try {
286      Class<java.lang.Runtime> rt = java.lang.Runtime.class;
287      rt.getMethod("addShutdownHook", new Class<?>[] {java.lang.Thread.class});
288      return true;
289    } catch (Exception e) {
290      return false;
291    }
292  }

294  // Add a shutdown hook to close the PrintStream when the program
295  // exits
296  private static void addShutdownHook() {
297    java.lang.Runtime.getRuntime()
298        .addShutdownHook(
299            new Thread() {
300              @SuppressWarnings("lock") // non-final field
301              @Override
302              public void run() {
303                if (!dtrace_closed) {
304
305                  // When the program being instrumented exits, the buffers
306                  // of the "dtrace" (PrintStream) object are not flushed,
307                  // so we miss the tail of the file.
308
309                  synchronized (daikon.Runtime.dtrace) {
310                    dtrace.println();
311                    // This lets us know we didn't lose any data.
312                    dtrace.println("# EOF (added by daikon.Runtime.addShutdownHook)");
313                    dtrace.close();
314                  }
315                }
316              }
317            });
318  }

418  // Lifted directly from plume/UtilPlume.java, where it is called
419  // escapeJava(), but repeated here to make this class self-contained.
420  /** Quote \, ", \n, and \r characters in the target; return a new string. */
421  public static String quote(String orig) {
422    StringBuilder sb = new StringBuilder();
423    // The previous escape (or escaped) character was seen right before
424    // this position.  Alternately:  from this character forward, the string
425    // should be copied out verbatim (until the next escaped character).
426    int post_esc = 0;
427    int orig_len = orig.length();
428    for (int i = 0; i < orig_len; i++) {
429      char c = orig.charAt(i);
430      switch (c) {
431        case '\"':
432        case '\\':
433          if (post_esc < i) {
434            sb.append(orig.substring(post_esc, i));
435          }
436          sb.append('\\');
437          post_esc = i;
438          break;
439        case '\n': // not lineSep
440          if (post_esc < i) {
441            sb.append(orig.substring(post_esc, i));
442          }
443          sb.append("\\n"); // not lineSep
444          post_esc = i + 1;
445          break;
446        case '\r':
447          if (post_esc < i) {
448            sb.append(orig.substring(post_esc, i));
449          }
450          sb.append("\\r");
451          post_esc = i + 1;
452          break;
453        default:
454          // Do nothing; i gets incremented.
455      }
456    }
457    if (sb.length() == 0) {
458      return orig;
459    }
460    sb.append(orig.substring(post_esc));
461    return sb.toString();
462  }

464  // The overhead of this is too high to call in quote(String)
465  public static String quote(Character ch) {
466    char c = ch.charValue();
467    switch (c) {
468      case '\"':
469        return "\\\"";
470      case '\\':
471        return "\\\\";
472      case '\n': // not lineSep
473        return "\\n"; // not lineSep
474      case '\r':
475        return "\\r";
476      default:
477        return new String(new char[] {c});
478    }
479  }

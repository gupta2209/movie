092  /**
093   * See dkconfig_replace_prestate.
094   *
095   * <p>Return the variable name corresponding to expr. Create a new varname and an expr &rarr;
096   * varname mapping if there is not already one.
097   */
098  public static String addPrestateExpression(String expr) {
099    if (expr == null) {
100      throw new IllegalArgumentException(expr);
101    }
102    if (exprToVar.containsKey(expr)) {
103      return exprToVar.get(expr);
104    }
105    String v = "v" + Integer.toString(varNameCounter++);
106    exprToVar.put(expr, v);
107    return v;
108  }

310  public static void mainHelper(String[] args)

326          new LongOpt(Daikon.wrap_xml_SWITCH, LongOpt.NO_ARGUMENT, null, 0),
327          new LongOpt(PrintInvariants.output_SWITCH, LongOpt.REQUIRED_ARGUMENT, null, 0),
328          new LongOpt(
329              PrintInvariants.print_csharp_metadata_SWITCH, LongOpt.OPTIONAL_ARGUMENT, null, 0),
330        };
331    Getopt g = new Getopt("daikon.PrintInvariants", args, "h", longopts);
332    int c;
333    while ((c = g.getopt()) != -1) {
334      switch (c) {
335        case 0:
336          // got a long option
337          String option_name = longopts[g.getLongind()].getName();
338          if (Daikon.help_SWITCH.equals(option_name)) {
339            System.out.println(usage);
340            throw new Daikon.NormalTermination();
341          } else if (Daikon.ppt_regexp_SWITCH.equals(option_name)) {
342            if (ppt_regexp != null) {
343              throw new Error(
344                  "multiple --"
345                      + Daikon.ppt_regexp_SWITCH
346                      + " regular expressions supplied on command line");
347            }
348            String regexp_string = Daikon.getOptarg(g);
349            if (!RegexUtil.isRegex(regexp_string)) {
350              throw new Daikon.UserError(
351                  "Bad regexp "
352                      + regexp_string
353                      + " for "
354                      + Daikon.ppt_regexp_SWITCH
355                      + ": "
356                      + RegexUtil.regexError(regexp_string));
357            }
358            regexp_string =
359                RegexUtil.asRegex(regexp_string); // @SuppressWarnings("regex") // flow-sensitivity
360            ppt_regexp = Pattern.compile(regexp_string);
361          } else if (Daikon.disc_reason_SWITCH.equals(option_name)) {
362            try {
363              PrintInvariants.discReasonSetup(Daikon.getOptarg(g));
364            } catch (IllegalArgumentException e) {
365              assert e.getMessage() != null
366                  : "@AssumeAssertion(nullness):  application invariant:  if discReasonSetup throws IllegalArgumentException, its message is non-null";
367              throw new Daikon.UserError(e.getMessage());
368            }
369          } else if (Daikon.suppress_redundant_SWITCH.equals(option_name)) {
370            Daikon.suppress_redundant_invariants_with_simplify = true;
371          } else if (Daikon.format_SWITCH.equals(option_name)) {
372            String format_name = Daikon.getOptarg(g);
373            Daikon.output_format = OutputFormat.get(format_name);
374            if (Daikon.output_format == null) {
375              throw new Daikon.UserError("Unknown output format:  --format " + format_name);
376            }
377          } else if (PrintInvariants.print_csharp_metadata_SWITCH.equals(option_name)) {
378            PrintInvariants.print_csharp_metadata = true;
379          } else if (Daikon.output_num_samples_SWITCH.equals(option_name)) {
380            Daikon.output_num_samples = true;
381          } else if (Daikon.config_SWITCH.equals(option_name)) {
382            String config_file = Daikon.getOptarg(g);
383            try {
384              InputStream stream = new FileInputStream(config_file);
385              Configuration.getInstance().apply(stream);
386            } catch (IOException e) {
387              throw new RuntimeException("Could not open config file " + config_file);
388            }
389            break;
390          } else if (PrintInvariants.output_SWITCH.equals(option_name)) {
391            String output_file = Daikon.getOptarg(g);
392            try {
393              out_stream = new FileOutputStream(output_file);
394            } catch (IOException e) {
395              out_stream = null;
396              throw new RuntimeException("Could not create output file " + output_file);
397            }
398            break;
399          } else if (Daikon.config_option_SWITCH.equals(option_name)) {
400            String item = Daikon.getOptarg(g);
401            daikon.config.Configuration.getInstance().apply(item);
402            break;
403          } else if (Daikon.debugAll_SWITCH.equals(option_name)) {
404            Global.debugAll = true;
405          } else if (Daikon.debug_SWITCH.equals(option_name)) {
406            LogHelper.setLevel(Daikon.getOptarg(g), LogHelper.FINE);
407          } else if (Daikon.track_SWITCH.equals(option_name)) {
408            LogHelper.setLevel("daikon.Debug", LogHelper.FINE);
409            String error = Debug.add_track(Daikon.getOptarg(g));
410            if (error != null) {
411              throw new Daikon.UserError(
412                  "Error parsing track argument '" + Daikon.getOptarg(g) + "' - " + error);
413            }
414          } else if (Daikon.wrap_xml_SWITCH.equals(option_name)) {
415            wrap_xml = true;
416          } else {
417            throw new RuntimeException("Unknown long option received: " + option_name);
418          }
419          break;
420        case 'h':
421          System.out.println(usage);
422          throw new Daikon.NormalTermination();
423        case '?':
424          break; // getopt() already printed an error
425        default:
426          System.out.println("getopt() returned " + c);
427          break;
428      }
429    }
430
431    // Set up debug traces; note this comes after reading command line options.
432    LogHelper.setupLogs(Global.debugAll ? LogHelper.FINE : LogHelper.INFO);
433
434    validateGuardNulls();
435
436    // The index of the first non-option argument -- the name of the file
437    int fileIndex = g.getOptind();
438    if (args.length - fileIndex != 1) {
439      System.out.println(usage);
440      throw new Daikon.UserError("Wrong number of arguments (expected 1)");
441    }
442
443    // Read in the invariants
444    String filename = args[fileIndex];
445    PptMap ppts =
446        FileIO.read_serialized_pptmap(
447            new File(filename), true // use saved config
448            );
449    // Setup the list of prototype invariants and initialize NIS suppressions

451    Daikon.setup_NISuppression();
452
453    // Make sure ppts' rep invariants hold
454    ppts.repCheck();
455
456    // If requested, just print the number of true invariants
457    if (dkconfig_true_inv_cnt) {
458      print_true_inv_cnt(ppts);
459      return;
460    }
461
462    validateGuardNulls();
463    //     if ((Daikon.dkconfig_guardNulls == "always") // interned
464    //         || (Daikon.dkconfig_guardNulls == "missing")) { // interned
465    //       Daikon.guardInvariants(ppts);
466    //     }
467
468    // Debug print the hierarchy is a more readable manner
469    if (debug.isLoggable(Level.FINE)) {
470      debug.fine("Printing PPT Hierarchy");
471      for (PptTopLevel my_ppt : ppts.pptIterable()) {
472        if (my_ppt.parents.size() == 0) my_ppt.debug_print_tree(debug, 0, null);
473      }
474    }
475
476    print_invariants(ppts);
477
478    // Close the output stream if --output was specified.
479    if (out_stream != null) {
480      out_stream.flush();
481      out_stream.close();
482    }
483  }

490  /**
491   * Prints out all the discardCodes and discardStrings of the Invariants that will not be printed
492   * if the {@code --disc_reason} command-line option is used.
493   */
494  public static void print_reasons(PptMap ppts) {
495    if (!print_discarded_invariants || Daikon.no_text_output) {
496      return;
497    }
498
499    System.out.println();
500    System.out.println("DISCARDED INVARIANTS:");
501    // DiscReasonMap.debug(discPpt);
502
503    // Makes things faster if a ppt is specified
504    if (discPpt != null) {
505      PptTopLevel ppt = ppts.get(discPpt);
506      if (ppt == null) {
507        System.out.println("No such ppt found: " + discPpt);
508      } else {
509        String toPrint = "";
510        toPrint += print_reasons_from_ppt(ppt, ppts);
511
512        StringTokenizer st = new StringTokenizer(toPrint, lineSep);
513        if (st.countTokens() > 2) {
514          System.out.print(toPrint);
515        } else {
516          String matching = "";
517          if (discVars != null || discClass != null) matching = " matching ";
518          System.out.println("No" + matching + "discarded Invariants found in " + ppt.name());
519        }
520      }
521      return;
522    }
523
524    // Uses the custom comparator to get the Ppt objects in sorted order
525    Comparator<PptTopLevel> comparator = new Ppt.NameComparator();
526    TreeSet<PptTopLevel> ppts_sorted = new TreeSet<>(comparator);
527    ppts_sorted.addAll(ppts.asCollection());
528
529    // Iterate over the PptTopLevels in ppts
530    for (PptTopLevel ppt : ppts_sorted) {
531      StringBuilder toPrint = new StringBuilder();
532      toPrint.append(print_reasons_from_ppt(ppt, ppts));
533
534      // A little hack so that PptTopLevels without discarded Invariants of
535      // interest don't get their names printed
536      StringTokenizer st = new StringTokenizer(toPrint.toString(), lineSep);
537      if (st.countTokens() > 2) {
538        System.out.print(toPrint.toString());
539      }
540    }
541  }

543  /** Validate guardNulls config option. */
544  public static void validateGuardNulls() {
545    Daikon.dkconfig_guardNulls = Daikon.dkconfig_guardNulls.intern();
546    // Complicated default!
547    if (Daikon.dkconfig_guardNulls == "default") { // interned
548      if (Daikon.output_format == OutputFormat.JML
549          || Daikon.output_format == OutputFormat.ESCJAVA) {
550        Daikon.dkconfig_guardNulls = "missing";
551      } else {
552        Daikon.dkconfig_guardNulls = "never";
553      }
554    }
555    if (!((Daikon.dkconfig_guardNulls == "always") // interned
556        || (Daikon.dkconfig_guardNulls == "never") // interned
557        || (Daikon.dkconfig_guardNulls == "missing")) // interned
558    ) {
559      throw new Error(
560          "Bad guardNulls config option \""
561              + Daikon.dkconfig_guardNulls
562              + "\", should be one of \"always\", \"never\", or \"missing\"");
563    }
564  }

566  /** Add discard reasons for invariants that are filtered out. */
567  private static void add_filter_reasons(PptTopLevel ppt) {
568    InvariantFilters fi = InvariantFilters.defaultFilters();
569    for (Iterator<Invariant> fullInvItor = ppt.invariants_iterator(); fullInvItor.hasNext(); ) {
570      Invariant nextInv = fullInvItor.next();
571      InvariantFilter varFilter = fi.shouldKeepVarFilters(nextInv);
572      if (varFilter != null) {
573        DiscReasonMap.put(nextInv, DiscardCode.findCode(varFilter), varFilter.getDescription());
574      } else {
575        InvariantFilter propFilter = fi.shouldKeepPropFilters(nextInv);
576        if (propFilter != null) {
577          DiscardInfo di;
578          if (propFilter instanceof ObviousFilter) {
579            di = nextInv.isObvious();
580            assert di != null : "@AssumeAssertion(nullness)";
581            if (Invariant.logOn()) {

632  /**
633   * Method used to setup fields if the {@code --disc_reason} command-line option is used if
634   * (arg==null) then show all discarded Invariants, otherwise just show the ones specified in arg,
635   * where arg = <em>class-name</em>&lt;<em>var1</em>,<em>var2</em>,...&gt;@<em>ppt.name</em> e.g.:
636   * OneOf&lt;x&gt;@foo():::ENTER would only show OneOf Invariants that involve x at the program
637   * point foo:::ENTER (any of the 3 params can be omitted, e.g. OneOf@foo:::ENTER)
638   *
639   * @throws IllegalArgumentException if arg is not of the proper syntax
640   */
641  public static void discReasonSetup(String arg) {
642    print_discarded_invariants = true;
643    usage =
644        "Usage: <class-name><<var1>,<var2>,,,,>@<ppt.name()>"
645            + lineSep
646            + "or use --disc_reason \"all\" to show all discarded Invariants"
647            + lineSep
648            + "e.g.: OneOf<x>@foo():::ENTER"
649            + lineSep;
650
651    // Will print all discarded Invariants in this case
652    if (arg == null || arg.length() == 0 || arg.equals("all")) {
653      return;
654    }
655
656    // User wishes to specify a classname for the discarded Invariants of
657    // interest
658    char firstChar = arg.charAt(0);
659    // This temp is used later as a way of "falling through" the cases
660    String temp = arg;
661    if (firstChar != '@' && firstChar != '<') {
662      StringTokenizer splitArg = new StringTokenizer(arg, "@<");
663      discClass = splitArg.nextToken();
664      if ((arg.indexOf('<') != -1)
665          && (arg.indexOf('@') != -1)
666          && (arg.indexOf('@') < arg.indexOf('<')))
667        temp = arg.substring(arg.indexOf('@')); // in case the pptname has a < in it
668      else if (arg.indexOf('<') != -1) temp = arg.substring(arg.indexOf('<'));
669      else if (arg.indexOf('@') != -1) temp = arg.substring(arg.indexOf('@'));
670      else {
671        return;
672      }
673    }
674    firstChar = temp.charAt(0);
675
676    // User wants to specify the variable names of interest
677    if (firstChar == '<') {
678      if (temp.length() < 2) throw new IllegalArgumentException("Missing '>'" + lineSep + usage);
679      if (temp.indexOf('>', 1) == -1) {
680        throw new IllegalArgumentException("Missing '>'" + lineSep + usage);
681      }
682      StringTokenizer parenTokens = new StringTokenizer(temp, "<>");
683      if ((temp.indexOf('@') == -1 && parenTokens.countTokens() > 0)
684          || (temp.indexOf('@') > -1 && parenTokens.countTokens() > 2))
685        throw new IllegalArgumentException("Too many brackets" + lineSep + usage);
686      StringTokenizer vars = new StringTokenizer(parenTokens.nextToken(), ",");
687      if (vars.hasMoreTokens()) {
688        discVars = vars.nextToken();
689        while (vars.hasMoreTokens()) discVars += "," + vars.nextToken();
690        // Get rid of *all* spaces since we know varnames can't have them
691        discVars = discVars.replaceAll(" ", "");
692      }
693      if (temp.endsWith(">")) {
694        return;
695      } else if (temp.charAt(temp.indexOf('>') + 1) != '@') {
696        throw new IllegalArgumentException("Must have '@' after '>'" + lineSep + usage);
697      } else {
698        temp = temp.substring(temp.indexOf('>') + 1);
699      }
700    }
701
702    // If it made it this far, the first char of temp has to be '@'
703    assert temp.charAt(0) == '@';
704    if (temp.length() == 1) {
705      throw new IllegalArgumentException("Must provide ppt name after '@'" + lineSep + usage);
706    }
707    discPpt = temp.substring(1);
708  }

711  // output format of V2.  In V2, combined exit points are printed after
712  // the original exit points (rather than before as they are following
713  // the PptMap sort order).
714  //
715  // Also, V2 only prints out a single ppt when there is only one
716  // exit point.  This seems correct.  Probably a better solution to
717  // this would be to not create the combined exit point at all when there
718  // is only a single exit.  Its done here instead so as not to futz with

723  @RequiresNonNull("FileIO.new_decl_format")

725
726    if (out_stream == null) {
727      out_stream = System.out;
728    }

730        new PrintWriter(new BufferedWriter(new OutputStreamWriter(out_stream, UTF_8)), true);
731    if (wrap_xml) {
732      pw.println("<INVARIANTS>");
733    }
734
735    PptTopLevel combined_exit = null;
736    boolean enable_exit_swap = true; // !Daikon.dkconfig_df_bottom_up;
737
738    if (Daikon.no_text_output) {
739      return;
740    }
741
742    // Retrieve Ppt objects in sorted order.  Put them in an array list
743    // so that it is easier to look behind and ahead.
744    PptTopLevel[] ppts = new PptTopLevel[all_ppts.size()];
745    int ii = 0;
746    for (PptTopLevel ppt : all_ppts.pptIterable()) {
747      ppts[ii++] = ppt;
748      // System.out.printf("considering ppt %s%n", ppts[ii-1].name());
749    }
750
751    for (int i = 0; i < ppts.length; i++) {
752      PptTopLevel ppt = ppts[i];
753
754      if (debug.isLoggable(Level.FINE)) {
755        debug.fine("Looking at point " + ppt.name());
756      }
757
758      // If this point is not an exit point, print out any retained combined
759      // exit point
760      if (enable_exit_swap && !ppt.ppt_name.isExitPoint()) {
761        if (combined_exit != null) {
762          print_invariants_maybe(combined_exit, pw, all_ppts);
763        }
764        combined_exit = null;
765      }
766
767      // Just cache the combined exit point for now, print it after the
768      // EXITnn points.
769      if (enable_exit_swap && ppt.ppt_name.isCombinedExitPoint()) {
770        combined_exit = ppt;
771        continue;
772      }
773
774      // If there is only one exit point, just show the combined one (since
775      // the EXITnn point will be empty)  This is accomplished by skipping this
776      // point if it is an EXITnn point and the previous point was a combined
777      // exit point and the next one is not an EXITnn point.  But don't skip
778      // any conditional ppts attached to the skipped ppt.
779      if (enable_exit_swap && (i > 0) && ppt.ppt_name.isExitPoint()) {
780        if (ppts[i - 1].ppt_name.isCombinedExitPoint()) {
781          if (((i + 1) >= ppts.length) || !ppts[i + 1].ppt_name.isExitPoint()) {
782            //             if (Daikon.dkconfig_output_conditionals
783            //                 && Daikon.output_format == OutputFormat.DAIKON) {
784            //               for (PptConditional pcond : ppt.cond_iterable()) {
785            //                 print_invariants_maybe(pcond, pw, all_ppts);
786            //               }
787            //             }
788            continue;
789          }
790        }
791      }
792
793      if (false) {
794        VarInfo v = ppt.find_var_by_name("size(/map.tiles[])");
795        System.out.printf("Found variable %s%n", v);
796        if (v != null) {
797          List<Invariant> invs = ppt.find_assignment_inv(v);
798          System.out.printf("assignment invs = %s%n", invs);
799        }
800      }
801
802      print_invariants_maybe(ppt, pw, all_ppts);
803    }
804
805    // print a last remaining combined exit point (if any)
806    if (enable_exit_swap && combined_exit != null) {
807      print_invariants_maybe(combined_exit, pw, all_ppts);
808    }
809
810    if (wrap_xml) {
811      pw.println("</INVARIANTS>");
812    }
813
814    pw.flush();
815  }

947  /** Prints all variables that were modified if the format is ESCJAVA or DBCJAVA. */
948  public static void print_modified_vars(PptTopLevel ppt, PrintWriter out) {
949
950    debugPrintModified.fine("Doing print_modified_vars for: " + ppt.name());
951
952    List<VarInfo> modified_vars = new ArrayList<>();
953    List<VarInfo> reassigned_parameters = new ArrayList<>();
954    List<VarInfo> unmodified_vars = new ArrayList<>();
955
956    // Loop through each variable at this ppt
957    for (VarInfo vi : ppt.var_infos) {
958
959      // Skip any orig variables
960      if (vi.isPrestate()) {
961        debugPrintModified.fine("  skipping " + vi.name() + ": is prestate");
962        continue;
963      }
964      debugPrintModified.fine("  Considering var: " + vi.name());
965
966      // Get the orig version of this variable.  If none is found then this
967      // isn't a variable about which it makes sense to consider modifiability.
968      VarInfo vi_orig = ppt.find_var_by_name(vi.prestate_name());
969      if (vi_orig == null) {
970        debugPrintModified.fine("  skipping " + vi.name() + ": no orig variable");
971        continue;
972      }
973
974      // TODO: When we can get information from the decl file that
975      // indicates if a variable is 'final', we should add such a test
976      // here.  For now we use the equality invariant between the
977      // variable and its orig variable to determine if it has been
978      // modified.
979
980      if (ppt.is_equal(vi, vi_orig)) {
981        debugPrintModified.fine("  " + vi.name() + " = " + vi_orig.name());
982        unmodified_vars.add(vi);
983      } else { // variables are not equal
984        if (vi.isParam()) {
985          reassigned_parameters.add(vi);
986        } else {
987          modified_vars.add(vi);
988        }
989      }
990    }
991
992    if (Daikon.output_num_samples
993        || (Daikon.output_format == OutputFormat.ESCJAVA)
994        || (Daikon.output_format == OutputFormat.DBCJAVA)) {
995      if (modified_vars.size() > 0) {
996        out.print("      Modified variables:");
997        for (VarInfo vi : modified_vars) {
998          out.print(" " + vi.old_var_name());
999        }
1000        out.println();
1001      }
1002      if (reassigned_parameters.size() > 0) {
1003        // out.print("      Reassigned parameters:");
1004        out.print("      Modified primitive arguments:");
1005        for (VarInfo vi : reassigned_parameters) {
1006          out.print(" " + vi.old_var_name());
1007        }
1008        out.println();
1009      }
1010      if (unmodified_vars.size() > 0) {
1011        out.print("      Unmodified variables:");
1012        for (VarInfo vi : unmodified_vars) {
1013          out.print(" " + vi.old_var_name());
1014        }
1015        out.println();
1016      }
1017    }
1018
1019    // Remove non-variables from the assignable output
1020    if (Daikon.output_format == OutputFormat.ESCJAVA || Daikon.output_format == OutputFormat.JML) {
1021      List<VarInfo> mods = new ArrayList<>();
1022      for (VarInfo vi : modified_vars) {
1023        if (!vi.is_assignable_var()) {
1024          continue;
1025        }
1026        mods.add(vi);
1027      }
1028
1029      // Print out the modifies/assignable list
1030      if (mods.size() > 0) {
1031        if (Daikon.output_format == OutputFormat.ESCJAVA) {
1032          out.print("modifies ");
1033        } else {
1034          out.print("assignable ");
1035        }
1036        int inserted = 0;
1037        for (VarInfo vi : mods) {
1038          String name = vi.old_var_name();
1039          if (!name.equals("this")) {
1040            if (inserted > 0) {
1041              out.print(", ");
1042            }
1043            if (name.endsWith("[]")) {
1044              name = name.substring(0, name.length() - 1) + "*]";
1045            }
1046            out.print(name);
1047            inserted++;
1048          }
1049        }
1050        out.println();
1051      }
1052    }

1201  /**
1202   * Parses the variables from varInfo.
1203   *
1204   * @param varInfo the Daikon variable representation to parse
1205   * @param sort true to parse as a grouping variable, false to parse as a filtering variable
1206   * @return the parsed variable string
1207   */
1208  public static String parse_csharp_invariant_variable(VarInfo varInfo, boolean sort) {
1209    // Do not ever want to sort by old value.
1210    if (varInfo.postState != null) {
1211      return parse_csharp_invariant_variable(varInfo.postState, sort);
1212    }
1213
1214    // Do not ever want to sort by function.
1215    if (varInfo.var_kind == VarInfo.VarKind.FUNCTION
1216        && !varInfo.var_flags.contains(VarFlags.IS_PROPERTY)) {
1217      // The assertion says that currently, Celeriac only visits a static
1218      // method when the method has a single parameter with the same type
1219      // as the declaring class. For example: string.NullOrEmpty(string
1220      // arg). In these cases, Celeriac will output arg as being the
1221      // enclosing var. The DECLs would look something like:
1222      // var fieldA
1223      // var string.NullOrEmpty(fieldA)
1224      //   parent fieldA
1225      assert (varInfo.enclosing_var != null)
1226          : "@AssumeAssertion(nullness)"; // application invariant; see comment above
1227      return parse_csharp_invariant_variable(varInfo.enclosing_var, sort);
1228    }
1229
1230    if (!sort) {
1231      String r = varInfo.name_using(OutputFormat.CSHARPCONTRACT);
1232      int a = r.indexOf("[");
1233      int b = r.indexOf("]");
1234      if (a != 1 && b != 1 && a < b) {
1235        String middle = r.substring(a + 1, b);
1236        if (middle.equals("..")) {
1237          r = r.substring(0, a + 1) + ".." + r.substring(b, r.length());
1238        } else {
1239          r = r.substring(0, a + 1) + "i" + r.substring(b, r.length());
1240        }
1241      }
1242
1243      return r;
1244    }
1245
1246    if ((varInfo.var_kind == VarInfo.VarKind.FIELD || varInfo.var_kind == VarInfo.VarKind.FUNCTION)
1247        && varInfo.enclosing_var != null
1248        && !varInfo.enclosing_var.csharp_name().equals("this")) {
1249
1250      return parse_csharp_invariant_variable(varInfo.enclosing_var, sort);
1251    } else {
1252      return varInfo.csharp_name();
1253    }
1254  }

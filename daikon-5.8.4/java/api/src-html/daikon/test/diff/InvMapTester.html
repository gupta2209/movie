001package daikon.test.diff;
002
003import static org.junit.Assert.assertFalse;
004import static org.junit.Assert.assertSame;
005
006import daikon.*;
007import daikon.diff.*;
008import daikon.inv.Invariant;
009import java.util.ArrayList;
010import java.util.Iterator;
011import java.util.List;
012import junit.framework.*;
013import org.checkerframework.checker.nullness.qual.MonotonicNonNull;
014import org.checkerframework.checker.nullness.qual.RequiresNonNull;
015import org.junit.Before;
016import org.junit.BeforeClass;
017import org.junit.Test;
018
019public class InvMapTester {
020
021  private @MonotonicNonNull InvMap map; // initialized by setUp()
022  private PptTopLevel pptA = new PptTopLevel("A:::OBJECT", new VarInfo[0]);
023  private PptTopLevel pptB = new PptTopLevel("A:::OBJECT", new VarInfo[0]);
024  private PptTopLevel pptC = new PptTopLevel("A:::OBJECT", new VarInfo[0]);
025  private List<Invariant> invsA = new ArrayList<>();
026  private List<Invariant> invsB = new ArrayList<>();
027  private List<Invariant> invsC = new ArrayList<>();
028
029  @BeforeClass
030  public static void setUpClass() {
031    daikon.LogHelper.setupLogs(LogHelper.INFO);
032    FileIO.new_decl_format = true;
033  }
034
035  @Before
036  public void setUp() {
037    map = new InvMap();
038  }
039
040  @RequiresNonNull("map")
041  // implicit flow: setUp was called by JUnit";
042  @Test
043  public void testABC() {
044    map.put(pptA, invsA);
045    map.put(pptB, invsB);
046    map.put(pptC, invsC);
047
048    // Test the keys
049    Iterator<PptTopLevel> i = map.pptIterator();
050    assertSame(i.next(), pptA);
051    assertSame(i.next(), pptB);
052    assertSame(i.next(), pptC);
053    assertFalse(i.hasNext());
054
055    // Test the values
056    assertSame(map.get(pptA), invsA);
057    assertSame(map.get(pptB), invsB);
058    assertSame(map.get(pptC), invsC);
059  }
060
061  @RequiresNonNull("map")
062  // implicit flow: setUp was called by JUnit";
063  @Test
064  public void testCAB() {
065    map.put(pptC, invsC);
066    map.put(pptA, invsA);
067    map.put(pptB, invsB);
068
069    // Test the keys
070    Iterator<PptTopLevel> i = map.pptIterator();
071    assertSame(i.next(), pptC);
072    assertSame(i.next(), pptA);
073    assertSame(i.next(), pptB);
074    assertFalse(i.hasNext());
075
076    // Test the values
077    assertSame(map.get(pptA), invsA);
078    assertSame(map.get(pptB), invsB);
079    assertSame(map.get(pptC), invsC);
080  }
081}

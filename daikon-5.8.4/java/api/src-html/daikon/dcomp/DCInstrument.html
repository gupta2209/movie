194  /** Class that defines a method (by its name and argument types) */
195  static class MethodDef {
196    String name;
197    Type[] arg_types;
198
199    MethodDef(String name, Type[] arg_types) {
200      this.name = name;
201      this.arg_types = arg_types;
202    }
203
204    @EnsuresNonNullIf(result = true, expression = "#1")
205    boolean equals(@GuardSatisfied MethodDef this, String name, Type[] arg_types) {
206      if (!name.equals(this.name)) {
207        return false;
208      }
209      if (this.arg_types.length != arg_types.length) {
210        return false;
211      }
212      for (int ii = 0; ii < arg_types.length; ii++) {
213        if (!arg_types[ii].equals(this.arg_types[ii])) {
214          return false;
215        }
216      }
217      return true;
218    }
219
220    @EnsuresNonNullIf(result = true, expression = "#1")
221    @Pure
222    @Override
223    public boolean equals(@GuardSatisfied MethodDef this, @GuardSatisfied @Nullable Object obj) {
224      if (!(obj instanceof MethodDef)) {
225        return false;
226      }
227      MethodDef md = (MethodDef) obj;
228      return equals(md.name, md.arg_types);
229    }
230
231    @Pure
232    @Override
233    public int hashCode(@GuardSatisfied MethodDef this) {
234      int code = name.hashCode();
235      for (Type arg : arg_types) {
236        code += arg.hashCode();
237      }
238      return code;
239    }
240  }

294  /**
295   * Instruments the original class to perform dynamic comparabilty and returns the new class
296   * definition.
297   *
298   * @return the modified JavaClass
299   */
300  public JavaClass instrument() {
301
302    String classname = gen.getClassName();
303
304    // Don't instrument annotations.  They aren't executed and adding
305    // the marker argument causes subtle errors
306    if ((gen.getModifiers() & Const.ACC_ANNOTATION) != 0) {
307      debug_transform.log("Not instrumenting annotation %s%n", classname);
308      return gen.getJavaClass().copy();
309    }
310
311    if (retransforming) {
312      debug_transform.log("Re-Instrumenting class %s%n", classname);
313      remove_dummy_instrumentation();
314    }
315
316    debug_transform.log("Instrumenting class %s%n", classname);
317    debug_transform.indent();
318
319    // Create the ClassInfo for this class and its list of methods
320    ClassInfo class_info = new ClassInfo(classname, loader);
321    boolean track_class = false;
322
323    if (!retransforming) {
324      // Handle object methods for this class
325      handle_object(gen);
326
327      // Have all top-level classes implement our interface
328      if (gen.getSuperclassName().equals("java.lang.Object")) {
329        // Add equals method if it doesn't already exist. This ensures
330        // that an instrumented version, equals(Object, DCompMarker),
331        // will be created in this class.
332        Method eq = gen.containsMethod("equals", "(Ljava/lang/Object;)Z");
333        if (eq == null) {
334          debug_transform.log("Added equals method%n");
335          add_equals_method(gen);
336        }
337
338        // Add DCompInstrumented interface and the required
339        // equals_dcomp_instrumented method.
340        add_dcomp_interface(gen);
341      }
342    }
343
344    // Process each method
345    for (Method m : gen.getMethods()) {
346
347      tag_frame_local = null;
348      try {
349        // Note whether we want to track the daikon variables in this method
350        boolean track = should_track(classname, methodEntryName(classname, m));
351        // If any one method is tracked, then the class is tracked.
352        if (track) {
353          track_class = true;
354        }
355
356        // If we are tracking variables, make sure the class is public
357        if (track && !gen.isPublic()) {
358          gen.isPrivate(false);
359          gen.isProtected(false);
360          gen.isPublic(true);
361        }
362
363        debug_transform.log("  Processing method %s, track=%b%n", m, track);
364        debug_transform.indent();
365
366        MethodGen mg = new MethodGen(m, classname, pool);
367        mgen = mg; // copy to global
368
369        InstructionList il = mg.getInstructionList();
370        boolean has_code = (il != null);
371        if (has_code) {
372          set_current_stack_map_table(mg, gen.getMajor());
373          if (retransforming) {
374            // If we are retransforming a JDK class, then
375            // the JVM has removed any StackMapTable.
376            needStackMap = false;
377          }
378          build_unitialized_NEW_map(il);
379        }
380
381        fix_local_variable_table(mg);
382
383        // If the method is native
384        if (mg.isNative()) {
385
386          // Create Java code that cleans up the tag stack and calls the real native method.
387          fix_native(gen, mg);
388          has_code = true;
389          set_current_stack_map_table(mg, gen.getMajor());
390
391          // Add the DCompMarker argument to distinguish our version
392          add_dcomp_arg(mg);
393
394        } else { // normal method
395
396          // Add the DCompMarker argument to distinguish our version
397          add_dcomp_arg(mg);
398
399          // Create a MethodInfo that describes this method's arguments
400          // and exit line numbers (information not available via reflection)
401          // and add it to the list for this class.
402          MethodInfo mi = null;
403          if (track && has_code) {
404            mi = create_method_info(class_info, mg);
405            class_info.method_infos.add(mi);
406            DCRuntime.methods.add(mi);
407          }
408
409          // Instrument the method
410          if (has_code) {
411            // Create the local to store the tag frame for this method
412            tag_frame_local = create_tag_frame_local(mg);
413            build_exception_handler(mg);
414            instrument_method(mg);
415            if (track) {
416              add_enter(mg, mi, DCRuntime.methods.size() - 1);
417              add_exit(mg, mi, DCRuntime.methods.size() - 1);
418            }
419            install_exception_handler(mg);
420          }
421        }
422
423        if (has_code) {
424          update_uninitialized_NEW_offsets(mg.getInstructionList());
425          create_new_stack_map_attribute(mg);
426          mg.setMaxLocals();
427          mg.setMaxStack();
428        } else {
429          mg.removeCodeAttributes();
430          mg.removeLocalVariables();
431        }
432
433        remove_local_variable_type_table(mg);
434
435        // We do not want to copy the @HotSpotIntrinsicCandidate annotations from
436        // the normal methods to our instrumented methods.  It doesn't cause an
437        // execution problem but will produce a massive number of warnings.
438        // Unfortunately, BCEL does not have the methods necessary to do this,
439        // so we hit it with a huge hammer and remove all the annotations.
440        // This annotation was added post Java 8.
441        if (BcelUtil.javaVersion > 8) {
442          mg.removeAnnotationEntries();
443        }
444
445        if (!BcelUtil.isMain(mg) && !BcelUtil.isClinit(mg)) {
446          // doubling
447          try {
448            if (has_code) {
449              il = mg.getInstructionList();
450              InstructionHandle end = il.getEnd();
451              int length = end.getPosition() + end.getInstruction().getLength();
452              if (length >= Const.MAX_CODE_SIZE) {
453                throw new ClassGenException(
454                    "Code array too big: must be smaller than " + Const.MAX_CODE_SIZE + " bytes.");
455              }
456            }
457            gen.addMethod(mg.getMethod());
458          } catch (Exception e) {
459            String s = e.getMessage();
460            if (s == null) throw e;
461            if (s.startsWith("Branch target offset too large")
462                || s.startsWith("Code array too big")) {
463              System.out.printf(
464                  "DynComp warning: ClassFile: %s - method %s is too large to instrument and is being skipped.%n",
465                  classname, mg.getName());
466              // Build a dummy instrumented method that has DCompMarker
467              // argument and no instrumentation.
468              // first, restore unmodified method
469              mg = new MethodGen(m, classname, pool);
470              // Add the DCompMarker argument
471              add_dcomp_arg(mg);
472              remove_local_variable_type_table(mg);
473              // try again
474              gen.addMethod(mg.getMethod());
475            } else {
476              throw e;
477            }
478          }
479        } else {
480          // replacing
481          gen.replaceMethod(m, mg.getMethod());
482          if (BcelUtil.isMain(mg)) gen.addMethod(create_dcomp_stub(mg).getMethod());
483        }
484        debug_transform.exdent();
485      } catch (Throwable t) {
486        if (debug_instrument.enabled) t.printStackTrace();
487        throw new Error("Unexpected error processing " + classname + "." + m.getName(), t);
488      }
489    }
490
491    // Add tag accessor methods for each primitive in the class
492    create_tag_accessors(gen);
493
494    if (!retransforming) {
495      // Keep track of when the class is initialized (so we don't look
496      // for fields in uninitialized classes)
497      track_class_init();
498    }
499    debug_transform.exdent();
500
501    // The code that builds the list of daikon variables for each ppt
502    // needs to know what classes are instrumented.  Its looks in the
503    // Chicory runtime for this information.
504    if (track_class) {
505      debug_transform.log("DCInstrument adding %s to all class list%n", class_info);
506      synchronized (daikon.chicory.SharedData.all_classes) {
507        daikon.chicory.SharedData.all_classes.add(class_info);
508      }
509    }
510    debug_transform.log("Instrumentation complete: %s%n", classname);
511
512    return (gen.getJavaClass().copy());
513  }

697  /**
698   * General Java Runtime instrumentation strategy:
699   *
700   * <p>It is a bit of a misnomer, but the Daikon code and documentation uses the term JDK to refer
701   * to the Java Runtime Environment class libraries. In Java 8 and earlier, they were usually found
702   * in {@code <your java installation>/jre/lib/rt.jar}. For these versions of Java, we
703   * pre-instrumented the entire rt.jar.
704   *
705   * <p>In Java 9 and later, the Java Runtime classes have been divided into modules that are
706   * usually found in: {@code <your java installation>/jmods/*.jmod}.
707   *
708   * <p>With the conversion to modules for Java 9 and beyond, we have elected to pre-instrument only
709   * java.base.jmod and instrument all other Java Runtime (aka JDK) classes dynamically as they are
710   * loaded.
711   *
712   * <p>Post Java 8 there are increased security checks when loading JDK classes. In particular, the
713   * core classes contained in the java.base module may not reference anything outside of java.base.
714   * This means we cannot pre-instrument classes in the same manner as was done for Java 8 as this
715   * would introduce references to the DynComp runtime (DCRuntime.java).
716   *
717   * <p>As an additional complication, the {@code retransformClasses} method of the Java
718   * Instrumentation Interface does not allow adding additional methods or changing method
719   * signatures. Hence, we have adopted the following strategy: We will pre-instrument java.base in
720   * order to add the instrumented versions of the JDK methods. However, these instrumented methods
721   * are dummies that are missing external references to DCRuntime and will never be executed. At
722   * DynComp runtime we reinstrument these methods, discarding the bodies of the dummy versions and
723   * replacing them with the correctly instrumented versions.
724   *
725   * <p>A simpler approach would be to just instrument JDK classes as they are loaded. However, a
726   * significant number of JDK classes (>800) are loaded prior to DynComp receiving control.
727   * Experiments have shown that not having these classes instrumented can lead to DynComp producing
728   * comparability sets quite different than those provided by the Java 8 version of Daikon.
729   */
730
731  /**
732   * Instruments a JDK class to perform dynamic comparability and returns the new class definition.
733   * A second version of each method in the class is created which is instrumented for
734   * comparability.
735   *
736   * @return the modified JavaClass
737   */

739
740    String classname = gen.getClassName();
741
742    // Don't instrument annotations.  They aren't executed and adding
743    // the marker argument causes subtle errors
744    if ((gen.getModifiers() & Const.ACC_ANNOTATION) != 0) {
745      debug_transform.log("Not instrumenting annotation %s%n", classname);
746      return gen.getJavaClass().copy();

754        return gen.getJavaClass().copy();
755      }
756      initial_jdk_instrument = true;
757    }
758
759    debug_transform.log("Instrumenting class(JDK) %s%n", classname);
760    debug_transform.indent();
761
762    // Handle object methods for this class
763    handle_object(gen);
764
765    // Have all top-level classes implement our interface
766    if (gen.getSuperclassName().equals("java.lang.Object")) {
767      // Add equals method if it doesn't already exist. This ensures
768      // that an instrumented version, equals(Object, DCompMarker),
769      // will be created in this class.
770      Method eq = gen.containsMethod("equals", "(Ljava/lang/Object;)Z");

776      // equals_dcomp_instrumented method.

778    }
779
780    // Process each method
781    for (Method m : gen.getMethods()) {
782
783      tag_frame_local = null;
784      try {
785        // Don't modify class initialization methods.  They can't affect
786        // user comparability and there isn't any way to get a second
787        // copy of them.
788        if (BcelUtil.isClinit(m)) {
789          continue;
790        }
791
792        debug_transform.log("  Processing method %s%n", m);
793        debug_transform.indent();
794
795        MethodGen mg = new MethodGen(m, classname, pool);
796        mgen = mg; // copy to global
797
798        InstructionList il = mg.getInstructionList();
799        boolean has_code = (il != null);
800        if (has_code) {
801          set_current_stack_map_table(mg, gen.getMajor());
802          build_unitialized_NEW_map(il);
803        }
804
805        fix_local_variable_table(mg);
806
807        // If the method is native
808        if (mg.isNative()) {
809
810          // Create Java code that cleans up the tag stack and calls the real native method.
811          fix_native(gen, mg);
812          has_code = true;
813          set_current_stack_map_table(mg, gen.getMajor());
814
815          // Add the DCompMarker argument to distinguish our version
816          add_dcomp_arg(mg);
817
818        } else { // normal method
819
820          // Add the DCompMarker argument to distinguish our version
821          add_dcomp_arg(mg);
822
823          if (initial_jdk_instrument) {
824            // Do not make any other changes to method at this time.
825            // Retransformation during DynComp will do instrumentation.
826          } else {
827            // Instrument the method
828            if (has_code) {
829              // Create the local to store the tag frame for this method
830              tag_frame_local = create_tag_frame_local(mg);
831              build_exception_handler(mg);
832              instrument_method(mg);
833              install_exception_handler(mg);
834            }
835          }
836        }
837
838        if (has_code) {
839          update_uninitialized_NEW_offsets(mg.getInstructionList());
840          create_new_stack_map_attribute(mg);
841          mg.setMaxLocals();
842          mg.setMaxStack();
843        } else {
844          mg.removeCodeAttributes();
845          mg.removeLocalVariables();
846        }
847
848        remove_local_variable_type_table(mg);
849
850        if (BcelUtil.javaVersion > 8) {
851          // see comments in instrument();
852          mg.removeAnnotationEntries();
853        }
854
855        if (initial_jdk_instrument && has_code) {
856          // mark this method as needing to be reinstrumented at runtime
857          mg.addLocalVariable(instrumentation_marker_variable, Type.INT, 0, null, null);
858        }
859        gen.addMethod(mg.getMethod());
860
861        debug_transform.exdent();
862      } catch (Throwable t) {
863        if (debug_instrument.enabled) t.printStackTrace();
864        skip_method(mgen);
865        if (quit_if_error) {
866          throw new Error("Unexpected error processing " + classname + "." + m.getName(), t);
867        } else {
868          System.out.printf("Unexpected error processing %s.%s: %s%n", classname, m.getName(), t);
869          System.out.printf("Method is NOT instrumented.%n");
870        }
871      }
872    }
873
874    // Add tag accessor methods for each primitive in the class
875    create_tag_accessors(gen);
876
877    // We don't need to track class initialization in the JDK because
878    // that is only used when printing comparability which is only done
879    // for client classes
880    // track_class_init();
881
882    debug_transform.exdent();
883    debug_transform.log("Instrumentation complete: %s%n", classname);
884
885    return (gen.getJavaClass().copy());
886  }

1044  /**
1045   * Instrument the specified method for dynamic comparability.
1046   *
1047   * @param mg MethodGen for the method to be instrumented
1048   */
1049  public void instrument_method(MethodGen mg) {
1050
1051    // Because the tag_frame_local is active for the entire method
1052    // and its creation will change the state of the locals layout,
1053    // we need to insert the code to initialize it now so that the
1054    // stack anaylsis we are about to do is correct for potential
1055    // code replacements we might make later.
1056    InstructionHandle orig_start = mg.getInstructionList().getStart();
1057    add_create_tag_frame(mg);
1058    // Calculate the operand stack value(s) for revised code.
1059    mg.setMaxStack();
1060    // Calculate stack types information
1061    StackTypes stack_types = bcel_calc_stack_types(mg);
1062    if (stack_types == null) {
1063      skip_method(mg);
1064      return;
1065    }
1066
1067    InstructionList il = mg.getInstructionList();
1068    OperandStack stack = null;
1069
1070    // Prior to adding support for Stack Maps, the position field
1071    // of each InstructionHandle was not updated until the modified
1072    // method was written out.  Hence, it could be used as the
1073    // index into stack_types.  To support StackMaps we need to
1074    // update the position field as we modify the code bytes.  So
1075    // we need a mapping from InstructionHandle to orignal offset.
1076    // I beleive we always visit the InstructionHandle nodes of
1077    // the method's InstructionList in order - hence, we will use
1078    // a simple array for now.  If this turns out to not be the
1079    // case we will need to use a hash map.
1080
1081    int[] handle_offsets = new int[il.getLength()];
1082    InstructionHandle ih = orig_start;
1083    int index = 0;
1084    // Loop through each instruction, building up offset map.
1085    while (ih != null) {
1086      handle_offsets[index++] = ih.getPosition();
1087
1088      if (debug_instrument.enabled) {
1089        debug_instrument.log("inst: %s %n", ih);
1090        for (InstructionTargeter it : ih.getTargeters()) {
1091          debug_instrument.log("targeter: %s %n", it);
1092        }
1093      }
1094
1095      ih = ih.getNext();
1096    }
1097
1098    index = 0;
1099    // Loop through each instruction, making substitutions
1100    for (ih = orig_start; ih != null; ) {
1101      debug_instrument.log("instrumenting instruction %s%n", ih);
1102      InstructionList new_il = null;
1103
1104      // Remember the next instruction to process
1105      InstructionHandle next_ih = ih.getNext();
1106
1107      // Get the stack information
1108      stack = stack_types.get(handle_offsets[index++]);
1109
1110      // Get the translation for this instruction (if any)
1111      new_il = xform_inst(mg, ih, stack);
1112
1113      // If this instruction was modified, replace it with the new
1114      // instruction list. If this instruction was the target of any
1115      // jumps or line numbers, replace them with the first
1116      // instruction in the new list.
1117      replace_instructions(mg, il, ih, new_il);
1118
1119      ih = next_ih;
1120    }
1121  }

1234  /** Adds a try/catch block around the entire method. */
1235  public void install_exception_handler(MethodGen mg) {
1236
1237    if (global_catch_il == null) {
1238      return;
1239    }
1240
1241    InstructionList cur_il = mg.getInstructionList();
1242    InstructionHandle start = global_exception_handler.getStartPC();
1243    InstructionHandle end = global_exception_handler.getEndPC();
1244    InstructionHandle exc = cur_il.append(global_catch_il);
1245    cur_il.setPositions();
1246    mg.addExceptionHandler(start, end, exc, throwable);
1247    // discard temporary handler
1248    global_catch_il = null;
1249    global_exception_handler = null;
1250
1251    if (!needStackMap) {
1252      return;
1253    }
1254
1255    int exc_offset = exc.getPosition();
1256
1257    debug_instrument.log(
1258        "New ExceptionHandler: %x %x %x %n", start.getPosition(), end.getPosition(), exc_offset);
1259
1260    // This is a trick to get running_offset set to
1261    // value of last stack map entry.
1262    update_stack_map_offset(exc_offset, 0);
1263    int map_offset = exc_offset - running_offset - 1;
1264
1265    // Get the argument types for this method
1266    Type[] arg_types = mg.getArgumentTypes();
1267
1268    int arg_index = (mg.isStatic() ? 0 : 1);
1269    StackMapType[] arg_map_types = new StackMapType[arg_types.length + arg_index];
1270    if (!mg.isStatic()) {
1271      arg_map_types[0] =
1272          new StackMapType(
1273              Const.ITEM_Object, pool.addClass(mg.getClassName()), pool.getConstantPool());
1274    }
1275    for (int ii = 0; ii < arg_types.length; ii++) {
1276      arg_map_types[arg_index++] = generate_StackMapType_from_Type(arg_types[ii]);
1277    }
1278
1279    StackMapEntry map_entry;
1280    StackMapType stack_map_type =
1281        new StackMapType(
1282            Const.ITEM_Object, pool.addClass(throwable.getClassName()), pool.getConstantPool());
1283    StackMapType[] stack_map_types = {stack_map_type};
1284    map_entry =
1285        new StackMapEntry(
1286            Const.FULL_FRAME, map_offset, arg_map_types, stack_map_types, pool.getConstantPool());
1287
1288    int orig_size = stack_map_table.length;
1289    StackMapEntry[] new_stack_map_table = new StackMapEntry[orig_size + 1];
1290    System.arraycopy(stack_map_table, 0, new_stack_map_table, 0, orig_size);
1291    new_stack_map_table[orig_size] = map_entry;
1292    stack_map_table = new_stack_map_table;

1314  /**
1315   * Adds the code to create the tag frame to the beginning of the method. This needs to be before
1316   * the call to DCRuntime.enter (since it passed to that method).
1317   */
1318  public void add_create_tag_frame(MethodGen mg) {
1319
1320    InstructionList nl = create_tag_frame(mg, tag_frame_local);
1321
1322    // We add a temporary NOP at the end of the create_tag_frame
1323    // code that we will replace with runtime initization code
1324    // later.  We do this so that any existing stack map at
1325    // instruction offset 0 is not replaced by the one we are
1326    // about to add for initializing the tag_frame variable.
1327    insertion_placeholder = nl.append(new NOP());
1328
1329    byte[] code = nl.getByteCode();
1330    // -1 because of the NOP we inserted.
1331    int len_code = code.length - 1;
1332
1333    insert_at_method_start(mg, nl);
1334
1335    if (!needStackMap) {
1336      return;
1337    }
1338
1339    // For Java 7 and beyond the StackMapTable is part of the
1340    // verification process.  We need to create and or update it to
1341    // account for instrumentation code we have inserted as well as
1342    // adjustments for the new 'tag_frame' local.
1343
1344    // Get existing StackMapTable (if present)
1345    if (stack_map_table.length > 0) {
1346      // Each stack map frame specifies (explicity or implicitly) an
1347      // offset_delta that is used to calculate the actual bytecode
1348      // offset at which the frame applies.  This is caluclated by
1349      // by adding offset_delta + 1 to the bytecode offset of the
1350      // previous frame, unless the previous frame is the initial
1351      // frame of the method, in which case the bytecode offset is
1352      // offset_delta. (From the Java Virual Machine Specification,
1353      // Java SE 7 Edition, section 4.7.4)
1354
1355      // Since we are inserting a new stack map frame at the
1356      // beginning of the stack map table, we need to adjust the
1357      // offset_delta of the original first stack map frame due to
1358      // the fact that it will no longer be the first entry.  We
1359      // must subtract (len_code + 1).
1360      // (We don't have to worry about the special case of the
1361      // original first entry having an offset of 0 because of the
1362      // NOP we inserted above.
1363
1364      stack_map_table[0].updateByteCodeOffset(-(len_code + 1));
1365    }
1366
1367    int new_table_length = stack_map_table.length + 1;
1368    StackMapEntry[] new_stack_map_table = new StackMapEntry[new_table_length];
1369
1370    // Insert a new StackMapEntry at the beginning of the table
1371    // that adds the tag_frame variable.
1372    StackMapType tag_frame_type = generate_StackMapType_from_Type(object_arr);
1373    StackMapType[] stack_map_type_arr = {tag_frame_type};
1374    new_stack_map_table[0] =
1375        new StackMapEntry(
1376            Const.APPEND_FRAME, len_code, stack_map_type_arr, null, pool.getConstantPool());
1377
1378    // We can just copy the rest of the stack frames over as the FULL_FRAME
1379    // ones were already updated when the tag_frame variable was allocated.
1380    for (int i = 0; i < stack_map_table.length; i++) {
1381      new_stack_map_table[i + 1] = stack_map_table[i];
1382    }
1383    stack_map_table = new_stack_map_table;
1384    // print_stack_map_table ("add_create_tag_frame");
1385  }

1455  /**
1456   * Pushes the object, method info index, parameters, and return value on the stack and calls the
1457   * specified Method (normally enter or exit) in DCRuntime. The parameters are passed as an array
1458   * of objects.
1459   */
1460  InstructionList call_enter_exit(
1461      MethodGen mg, int method_info_index, String method_name, int line) {
1462
1463    InstructionList il = new InstructionList();
1464    Type[] arg_types = mg.getArgumentTypes();
1465
1466    // Push the tag frame
1467    il.append(InstructionFactory.createLoad(object_arr, tag_frame_local.getIndex()));
1468
1469    // Push the object.  Null if this is a static method or a constructor
1470    if (mg.isStatic() || (method_name.equals("enter") && BcelUtil.isConstructor(mg))) {
1471      il.append(new ACONST_NULL());
1472    } else { // must be an instance method
1473      il.append(InstructionFactory.createLoad(Type.OBJECT, 0));
1474    }
1475
1476    // Determine the offset of the first parameter
1477    int param_offset = 1;
1478    if (mg.isStatic()) {
1479      param_offset = 0;
1480    }
1481
1482    // Push the MethodInfo index
1483    il.append(ifact.createConstant(method_info_index));
1484
1485    // Create an array of objects with elements for each parameter
1486    il.append(ifact.createConstant(arg_types.length));
1487    il.append(ifact.createNewArray(Type.OBJECT, (short) 1));
1488
1489    // Put each argument into the array
1490    int param_index = param_offset;
1491    for (int ii = 0; ii < arg_types.length; ii++) {
1492      il.append(InstructionFactory.createDup(object_arr.getSize()));
1493      il.append(ifact.createConstant(ii));
1494      Type at = arg_types[ii];
1495      if (at instanceof BasicType) {
1496        il.append(new ACONST_NULL());
1497        // il.append (create_wrapper (c, at, param_index));
1498      } else { // must be reference of some sort
1499        il.append(InstructionFactory.createLoad(Type.OBJECT, param_index));
1500      }
1501      il.append(InstructionFactory.createArrayStore(Type.OBJECT));
1502      param_index += at.getSize();
1503    }
1504
1505    // If this is an exit, push the return value and line number.
1506    // The return value
1507    // is stored in the local "return__$trace2_val"  If the return
1508    // value is a primitive, wrap it in the appropriate runtime wrapper
1509    if (method_name.equals("exit")) {
1510      Type ret_type = mg.getReturnType();
1511      if (ret_type == Type.VOID) {
1512        il.append(new ACONST_NULL());
1513      } else {
1514        LocalVariableGen return_local = get_return_local(mg, ret_type);
1515        if (ret_type instanceof BasicType) {
1516          il.append(new ACONST_NULL());
1517          // il.append (create_wrapper (c, ret_type, return_local.getIndex()));
1518        } else {
1519          il.append(InstructionFactory.createLoad(Type.OBJECT, return_local.getIndex()));
1520        }
1521      }
1522
1523      // push line number
1524      il.append(ifact.createConstant(line));
1525    }
1526
1527    // Call the specified method
1528    Type[] method_args;
1529    if (method_name.equals("exit")) {
1530      method_args =
1531          new Type[] {object_arr, Type.OBJECT, Type.INT, object_arr, Type.OBJECT, Type.INT};
1532    } else {
1533      method_args = new Type[] {object_arr, Type.OBJECT, Type.INT, object_arr};
1534    }
1535    il.append(
1536        ifact.createInvoke(
1537            DCRuntime.class.getName(), method_name, Type.VOID, method_args, Const.INVOKESTATIC));
1538
1539    return il;
1540  }

1637  /**
1638   * Transforms instructions to track comparability. Returns a list of instructions that replaces
1639   * the specified instruction. Returns null if the instruction should not be replaced.
1640   *
1641   * @param mg method being instrumented
1642   * @param ih handle of Instruction to translate
1643   * @param stack current contents of the stack
1644   */
1645  @Nullable InstructionList xform_inst(MethodGen mg, InstructionHandle ih, OperandStack stack) {
1646
1647    Instruction inst = ih.getInstruction();
1648
1649    switch (inst.getOpcode()) {
1650
1651        // Replace the object comparison instructions with a call to
1652        // DCRuntime.object_eq or DCRuntime.object_ne.  Those methods
1653        // return a boolean which is used in a ifeq/ifne instruction
1654      case Const.IF_ACMPEQ:
1655        return (object_comparison((BranchInstruction) inst, "object_eq", Const.IFNE));
1656      case Const.IF_ACMPNE:
1657        return (object_comparison((BranchInstruction) inst, "object_ne", Const.IFNE));
1658
1659        // These instructions compare the integer on the top of the stack
1660        // to zero.  Nothing is made comparable by this, so we need only
1661        // discard the tag on the top of the stack.
1662      case Const.IFEQ:
1663      case Const.IFNE:
1664      case Const.IFLT:
1665      case Const.IFGE:
1666      case Const.IFGT:
1667      case Const.IFLE:
1668        {
1669          return discard_tag_code(inst, 1);
1670        }
1671
1672        // Instanceof pushes either 0 or 1 on the stack depending on whether
1673        // the object on top of stack is of the specified type.  We push a
1674        // tag for a constant, since nothing is made comparable by this.
1675      case Const.INSTANCEOF:

1677
1678        // Duplicates the item on the top of stack.  If the value on the
1679        // top of the stack is a primitive, we need to do the same on the
1680        // tag stack.  Otherwise, we need do nothing.
1681      case Const.DUP:
1682        {
1683          return dup_tag(inst, stack);
1684        }

1687        // values down in the stack.  If the value at the top of the stack
1688        // is not a primitive, there is nothing to do here.  If the second
1689        // value is not a primitive, then we need only to insert the duped
1690        // value down 1 on the tag stack (which contains only primitives)

1903        // might argue that the key should be made comparable to each value in
1904        // the jump table.  But the tags for those values are not available.
1905        // And since they are all constants, its not clear how interesting it
1906        // would be anyway.
1907      case Const.LOOKUPSWITCH:
1908      case Const.TABLESWITCH:
1909        return discard_tag_code(inst, 1);
1910
1911        // Make the integer argument to ANEWARRAY comparable to the new
1912        // array's index.
1913      case Const.ANEWARRAY:
1914      case Const.NEWARRAY:
1915        {
1916          return new_array(inst);
1917        }
1918
1919        // If the new array has 2 dimensions, make the integer arguments
1920        // comparable to the corresponding indices of the new array.
1921        // For any other number of dimensions, discard the tags for the
1922        // arguments.
1923      case Const.MULTIANEWARRAY:
1924        {

1926        }
1927
1928        // Mark the array and its index as comparable.  Also for primitives,
1929        // push the tag of the array element on the tag stack

1932      case Const.CALOAD:
1933      case Const.DALOAD:
1934      case Const.FALOAD:
1935      case Const.IALOAD:
1936      case Const.LALOAD:
1937      case Const.SALOAD:

1943        // the tag for the value on the top of the stack in the tag storage

1958      case Const.DASTORE:

1960      case Const.FASTORE:
1961        return array_store(inst, "fastore", Type.FLOAT);
1962      case Const.IASTORE:

1970        // to handle the tag stack
1971      case Const.ARETURN:
1972      case Const.DRETURN:
1973      case Const.FRETURN:
1974      case Const.IRETURN:
1975      case Const.LRETURN:
1976      case Const.RETURN:

1982        // to call the instrumented version (with the DCompMarker argument).
1983        // Calls to uninstrumented code (rare) discard primitive arguments

1985      case Const.INVOKESTATIC:
1986      case Const.INVOKEVIRTUAL:
1987      case Const.INVOKESPECIAL:
1988      case Const.INVOKEINTERFACE:
1989      case Const.INVOKEDYNAMIC:
1990        return handle_invoke((InvokeInstruction) inst);
1991
1992        // Throws an exception.  This clears the operand stack of the current
1993        // frame.  We need to clear the tag stack as well.
1994      case Const.ATHROW:
1995        return build_il(dcr_call("throw_op", Type.VOID, Type.NO_ARGS), inst);
1996
1997        // Opcodes that don't need any modifications.  Here for reference
1998      case Const.ACONST_NULL:
1999      case Const.ALOAD:
2000      case Const.ALOAD_0:
2001      case Const.ALOAD_1:
2002      case Const.ALOAD_2:
2003      case Const.ALOAD_3:
2004      case Const.ASTORE:
2005      case Const.ASTORE_0:
2006      case Const.ASTORE_1:
2007      case Const.ASTORE_2:
2008      case Const.ASTORE_3:
2009      case Const.CHECKCAST:
2010      case Const.D2F: // double to float
2011      case Const.D2I: // double to integer
2012      case Const.D2L: // double to long
2013      case Const.DNEG: // Negate double on top of stack
2014      case Const.F2D: // float to double
2015      case Const.F2I: // float to integer
2016      case Const.F2L: // float to long
2017      case Const.FNEG: // Negate float on top of stack
2018      case Const.GOTO:
2019      case Const.GOTO_W:
2020      case Const.I2B: // integer to byte
2021      case Const.I2C: // integer to char
2022      case Const.I2D: // integer to double
2023      case Const.I2F: // integer to float
2024      case Const.I2L: // integer to long
2025      case Const.I2S: // integer to short
2026      case Const.IFNONNULL:
2027      case Const.IFNULL:

2029      case Const.INEG: // negate integer on top of stack

2200    Type ret_type = invoke.getReturnType(pool);

2210    } else {

2212      callee_instrumented = callee_instrumented(classname);

2271      }
2272    }
2273

2279    //                     invoke, method_name, ignore_toString);

2282    // replacement, a static method in DCRuntime

2286

2291                "daikon.dcomp.DCRuntime",
2292                "dcomp_super_equals",
2293                ret_type,
2294                new_arg_types,
2295                Const.INVOKESTATIC));
2296      } else {
2297        // just a regular equals(Object) call
2298        il.append(
2299            ifact.createInvoke(
2300                "daikon.dcomp.DCRuntime",
2301                "dcomp_equals",
2302                ret_type,
2303                new_arg_types,
2304                Const.INVOKESTATIC));
2305      }
2306
2307    } else if (is_object_clone(method_name, ret_type, arg_types)

2311
2312    } else if (callee_instrumented) {
2313
2314      // Add the DCompMarker argument so that it calls the instrumented version
2315      il.append(new ACONST_NULL());

2321              method_name,
2322              ret_type,
2323              new_arg_types,

2464

2466
2467    Type[] arg_types = invoke.getArgumentTypes(pool);
2468    Type[] new_arg_types = BcelUtil.postpendToArray(arg_types, dcomp_marker);
2469    String method_name = invoke.getMethodName(pool);
2470    Type ret_type = invoke.getReturnType(pool);
2471    String classname = invoke.getClassName(pool);
2472    String dcr_method_name = "uninstrumented_" + method_name + dcr_suffix;
2473

2475    String cap_method_name = method_name.substring(0, 1).toUpperCase() + method_name.substring(1);

2478        new ObjectType(Signatures.addPackage(dcomp_prefix, "DComp" + cap_method_name));

3695   * class that contains the method to be loaded.
3696   *
3697   * @param invoke instruction to check
3698   * @return true if the invoke calls a native method
3699   */
3700  @Pure
3701  boolean is_native(InvokeInstruction invoke) {
3702
3703    // Get the class of the method
3704    ClassLoader loader = getClass().getClassLoader();
3705    Class<?> clazz;
3706    try {
3707      clazz = Class.forName(invoke.getClassName(pool), false, loader);
3708    } catch (Exception e) {
3709      throw new Error("can't get class " + invoke.getClassName(pool), e);
3710    }
3711
3712    // Get the arguments to the method
3713    Type[] arg_types = invoke.getArgumentTypes(pool);
3714    Class<?>[] arg_classes = new Class<?>[arg_types.length];
3715    for (int ii = 0; ii < arg_types.length; ii++) {
3716      arg_classes[ii] = type_to_class(arg_types[ii], loader);
3717    }
3718
3719    // Find the method and determine if its native
3720    int modifiers = 0;
3721    String method_name = invoke.getMethodName(pool);
3722    String classes = clazz.getName();
3723    try {
3724      if (method_name.equals("<init>")) {
3725        Constructor<?> c = clazz.getDeclaredConstructor(arg_classes);
3726        modifiers = c.getModifiers();
3727      } else if (clazz.isInterface()) {
3728        return false; // presume interfaces aren't native...
3729      } else {
3730
3731        java.lang.reflect.Method m = null;
3732        while (m == null) {
3733          try {
3734            m = clazz.getDeclaredMethod(method_name, arg_classes);
3735            modifiers = m.getModifiers();
3736          } catch (NoSuchMethodException e) {
3737            clazz = clazz.getSuperclass();
3738            classes += ", " + clazz.getName();
3739          }
3740        }
3741      }
3742    } catch (Exception e) {
3743      throw new Error(
3744          "can't find method "
3745              + method_name
3746              + " "
3747              + Arrays.toString(arg_classes)
3748              + " "
3749              + classes
3750              + " "
3751              + invoke.toString(pool.getConstantPool()),
3752          e);
3753    }
3754
3755    return (Modifier.isNative(modifiers));
3756  }

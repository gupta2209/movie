031 * the javaagent switch on the target program and if requested starts Daikon on the result.

168  /**
169   * Entry point of Chicory.
170   *
171   * @param args see usage for argument descriptions
172   */
173  public static void main(String[] args) {
174
175    // Parse our arguments
176    Options options = new Options(synopsis, Chicory.class);
177    options.setParseAfterArg(false);
178    String[] target_args = options.parse(true, args);
179    check_args(options, target_args);
180
181    // Turn on basic logging if the debug was selected
182    basic.enabled = debug;
183    basic.log("target_args = %s%n", Arrays.toString(target_args));
184
185    // Start the target.  Pass the same options to the premain as
186    // were passed here.
187
188    Chicory chicory = new Chicory();
189    chicory.start_target(options.getOptionsString(), target_args);
190  }

192  /**
193   * Check the command-line arguments for legality. If not legal, prints a message and exits the
194   * JVM.
195   *
196   * @param options set of legal options to Chicory
197   * @param target_args arguments being passed to the target program
198   */
199  public static void check_args(Options options, String[] target_args) {
200    if (nesting_depth < 0) {
201      System.out.printf("nesting depth (%d) must not be negative%n", nesting_depth);
202      options.printUsage();
203      System.exit(1);
204    }
205    if (target_args.length == 0) {
206      System.out.println("target program must be specified");
207      options.printUsage();
208      System.exit(1);
209    }
210  }

227  /**
228   * Starts the target program with the java agent setup to do the transforms. All java agent
229   * arguments are passed to it. Our classpath is passed to the new JVM.
230   */
231  void start_target(String premain_args, String[] target_args) {
232
233    // Default the trace file name to the <target-program-name>.dtrace.gz
234    if (dtrace_file == null) {
235      String target_class = target_args[0].replaceFirst(".*[/.]", "");
236      dtrace_file = new File(String.format("%s.dtrace.gz", target_class));
237      premain_args += " --dtrace-file=" + dtrace_file;
238    }
239
240    // Get the current classpath
241    String cp = System.getProperty("java.class.path");
242    basic.log("classpath = '%s'%n", cp);
243    if (cp == null) {
244      cp = ".";
245    }
246
247    // The the separator for items in the class path
248    String path_separator = System.getProperty("path.separator");
249    basic.log("path_separator = %s%n", path_separator);
250    if (!RegexUtil.isRegex(path_separator)) {
251      throw new Daikon.UserError(
252          "Bad regexp "
253              + path_separator
254              + " for path.separator: "
255              + RegexUtil.regexError(path_separator));
256    }
257
258    // Look for ChicoryPremain.jar along the classpath
259    if (premain == null) {
260      String[] cpath = cp.split(path_separator);
261      for (String path : cpath) {
262        File poss_premain = new File(path, "ChicoryPremain.jar");
263        if (poss_premain.canRead()) {
264          premain = poss_premain;
265          break;
266        }
267      }
268    }
269
270    // If not on the classpath look in ${DAIKONDIR}/java
271    String daikon_dir = System.getenv("DAIKONDIR");
272    if (premain == null) {
273      if (daikon_dir != null) {
274        String file_separator = System.getProperty("file.separator");
275        File poss_premain = new File(daikon_dir + file_separator + "java", "ChicoryPremain.jar");
276        if (poss_premain.canRead()) {
277          premain = poss_premain;
278        }
279      }
280    }
281
282    // If not found, try the daikon.jar file itself
283    if (premain == null) {
284      for (String path : cp.split(path_separator)) {
285        File poss_premain = new File(path);
286        if (poss_premain.getName().equals("daikon.jar")) {
287          if (poss_premain.canRead()) {
288            premain = poss_premain;
289          }
290        }
291      }
292    }
293
294    // If we didn't find a premain, give up
295    if (premain == null) {
296      System.err.printf("Can't find ChicoryPremain.jar or daikon.jar on the classpath");
297      if (daikon_dir == null) {
298        System.err.printf(" and $DAIKONDIR is not set.%n");
299      } else {
300        System.err.printf(" or in $DAIKONDIR/java .%n");
301      }
302      System.err.printf("It should be found in the directory where Daikon was installed.%n");
303      System.err.printf("Use the --premain switch to specify its location,%n");
304      System.err.printf("or change your classpath to include it.%n");
305      System.exit(1);
306    }
307
308    String dtraceLim, terminate;
309    dtraceLim = System.getProperty(traceLimString);
310    terminate = System.getProperty(traceLimTermString);
311
312    // Run Daikon if we're in online mode
313    StreamRedirectThread daikon_err = null;
314    StreamRedirectThread daikon_out = null;
315    if (daikon_online) {
316      runDaikon();
317
318      StreamRedirectThread tmp_daikon_err =
319          new StreamRedirectThread("stderr", daikon_proc.getErrorStream(), System.err);
320      daikon_err = tmp_daikon_err;
321      daikon_err.start();
322
323      @NonNull InputStream daikonStdOut = daikon_proc.getInputStream();
324      // daikonReader escapes, so it is not closed in this method.
325      BufferedReader daikonReader = new BufferedReader(new InputStreamReader(daikonStdOut, UTF_8));
326
327      // Examine up to 100 lines of Daikon output, looking for
328      // the "DaikonChicoryOnlinePort=" line.  Note that if file progress
329      // is turned on in Daikon, it may be preceded by a timestamp.
330      for (int i = 0; i < 100; i++) {
331        String line;
332        try {
333          line = daikonReader.readLine();
334        } catch (IOException e1) {
335          System.out.printf("Exception reading output from Daikon: %s%n", e1);
336          line = null;
337        }
338
339        if (line == null) {
340          throw new RuntimeException("Did not receive socket port from Daikon!");
341        } else {
342          System.out.println(line);
343
344          if (line.contains("DaikonChicoryOnlinePort=")) {
345            String portStr = line.replaceFirst(".*DaikonChicoryOnlinePort=", "");
346            daikon_port = Integer.decode(portStr);
347            System.out.println("GOT PORT STRING " + daikon_port);
348            break;
349          }
350        }
351      }
352
353      if (daikon_port == -1) {
354        throw new RuntimeException("After 100 lines of output, Daikon port not received");
355      }
356
357      // continue reading daikon output in separate thread
358      daikon_out = new StreamRedirectThread("stdout", daikonStdOut, System.out);
359      daikon_out.start();
360    }
361
362    // Build the command line to execute the target with the javaagent
363    List<String> cmdlist = new ArrayList<>();
364    cmdlist.add("java");
365
366    if (RemoteDebug) {
367      // -Xdebug -Xrunjdwp:server=y,transport=dt_socket,address=4142,suspend=n
368      cmdlist.add("-Xdebug -Xrunjdwp:server=n,transport=dt_socket,address=8000,suspend=y");
369      // cmdlist.add("-Xdebug -Xnoagent
370      // -Xrunjdwp:transport=dt_socket,server=n,suspend=n,address=8000 -Djava.compiler=NONE");
371    }
372
373    cmdlist.add("-cp");
374    cmdlist.add(cp);
375    cmdlist.add("-ea");
376    cmdlist.add("-Xmx" + heap_size);
377    // cmdlist.add ("-verbose");
378
379    if (dtraceLim != null) cmdlist.add("-D" + traceLimString + "=" + dtraceLim);
380    if (terminate != null) {
381      cmdlist.add("-D" + traceLimTermString + "=" + terminate);
382    }
383
384    // Specify the port to use to talk to Daikon if in online mode
385    if (daikon_online) {
386      assert daikon_port != -1 : daikon_port;
387      premain_args += " --daikon-port " + daikon_port;
388    }
389
390    cmdlist.add(String.format("-javaagent:%s=%s", premain, premain_args));
391
392    for (String target_arg : target_args) {
393      cmdlist.add(target_arg);
394    }
395    if (verbose) {
396      System.out.printf("%nExecuting target program: %s%n", args_to_string(cmdlist));
397    }
398    String[] cmdline = cmdlist.toArray(new String[cmdlist.size()]);
399
400    // Execute the command, sending all output to our streams
401    java.lang.Runtime rt = java.lang.Runtime.getRuntime();
402    Process chicory_proc = null;
403    try {
404      chicory_proc = rt.exec(cmdline);
405    } catch (Exception e) {
406      System.out.printf("Exception '%s' while executing '%s'%n", e, cmdline);
407      System.exit(1);
408    }
409
410    StreamRedirectThread stdin_thread =
411        new StreamRedirectThread("stdin", System.in, chicory_proc.getOutputStream(), false);
412    stdin_thread.start();
413
414    int targetResult = redirect_wait(chicory_proc);
415
416    if (daikon) {
417      // Terminate if target didn't end properly
418      if (targetResult != 0) {
419        System.out.printf(
420            "Warning: Did not run Daikon because target exited with %d status%n", targetResult);
421        System.exit(targetResult);
422      }
423
424      runDaikon();
425      int daikonResult = waitForDaikon();
426      System.exit(daikonResult);
427    } else if (daikon_online) {
428      assert daikon_proc != null
429          : "@AssumeAssertion(nullness): conditional: just tested daikon_online, and ran runDaikon() earlier in this method";
430      if (targetResult != 0) {
431        System.out.printf("Warning: Target exited with %d status%n", targetResult);
432      }
433
434      // Wait for the process to terminate and return the results
435      int daikonResult = 0; // initialized to nonsense value to suppress compiler warning
436      while (true) {
437        try {
438          daikonResult = daikon_proc.waitFor();
439          break;
440        } catch (InterruptedException e) {
441          System.out.printf("unexpected interrupt %s while waiting for target to finish", e);
442        }
443      }
444
445      // Make sure all output is forwarded before we finish
446      try {
447        assert daikon_err != null
448            : "@AssumeAssertion(nullness): dependent: because daikon_online is true";
449        assert daikon_out != null
450            : "@AssumeAssertion(nullness): dependent: because daikon_online is true";
451        daikon_err.join();
452        daikon_out.join();
453      } catch (InterruptedException e) {
454        System.out.printf("unexpected interrupt %s while waiting for threads to join", e);
455      }
456
457      if (daikonResult != 0) {
458        System.out.printf("Warning: Daikon exited with %d status%n", daikonResult);
459      }
460      System.exit(daikonResult);
461    } else {
462      // No daikon command specified, so just exit
463      if (targetResult != 0) {
464        System.out.printf("Warning: Target exited with %d status%n", targetResult);
465      }
466      System.exit(targetResult);
467    }
468  }

470  /** Runs daikon either online or on the generated trace file. */
471  @EnsuresNonNull("daikon_proc")
472  public void runDaikon() {
473
474    java.lang.Runtime rt = java.lang.Runtime.getRuntime();
475
476    // Get the current classpath
477    String cp = System.getProperty("java.class.path");
478    if (cp == null) {
479      cp = ".";
480    }
481
482    String cmdstr;
483    if (daikon_online) {
484      cmdstr =
485          String.format("java -Xmx%s -cp %s -ea daikon.Daikon %s +", heap_size, cp, daikon_args);
486    } else {
487      cmdstr =
488          String.format(
489              "java -Xmx%s -cp %s -ea daikon.Daikon %s %s/%s",
490              heap_size, cp, daikon_args, output_dir, dtrace_file);
491    }
492
493    // System.out.println("daikon command is " + daikon_cmd);
494    // System.out.println("daikon command cmdstr " + cmdstr);
495
496    if (verbose) {
497      System.out.printf("%nExecuting daikon: %s%n", cmdstr);
498    }
499
500    try {
501      daikon_proc = rt.exec(cmdstr);
502    } catch (Exception e) {
503      System.out.printf("Exception '%s' while executing '%s'%n", e, cmdstr);
504      System.exit(1);
505    }
506  }

508  /** Wait for daikon to complete and return its exit status. */
509  @RequiresNonNull("daikon_proc")
510  private int waitForDaikon() {
511    int result = redirect_wait(daikon_proc);
512    return result;
513  }
514
515  /** Wait for stream redirect threads to complete and return its exit status. */
516  public int redirect_wait(Process p) {
517
518    // Create the redirect theads and start them
519    StreamRedirectThread err_thread =
520        new StreamRedirectThread("stderr", p.getErrorStream(), System.err);
521
522    StreamRedirectThread out_thread =
523        new StreamRedirectThread("stdout", p.getInputStream(), System.out);
524
525    err_thread.start();
526    out_thread.start();
527
528    // Wait for the process to terminate and return the results
529    int result = -1;
530    while (true) {
531      try {
532        result = p.waitFor();
533        break;
534      } catch (InterruptedException e) {
535        System.out.printf("unexpected interrupt %s while waiting for target to finish", e);
536      }
537    }
538
539    // Make sure all output is forwarded before we finish
540    try {
541      err_thread.join();
542      out_thread.join();
543    } catch (InterruptedException e) {
544      System.out.printf("unexpected interrupt %s while waiting for threads to join", e);
545    }
546
547    return result;
548  }

550  public PrintWriter openFileInDirectory(String fileName, String dirName) {
551    PrintWriter outFile = null;
552    try {
553      if (dirName != null) {
554        File directory = new File(dirName);
555
556        // make the output directory if non-existent
557        if (!directory.exists()) directory.mkdir();
558      }
559
560      outFile = new PrintWriter(new File(dirName, fileName), UTF_8.name());
561    } catch (IOException e) {
562      if (outFile == null) {
563        throw new RuntimeException("This can't happen.");
564      }
565      outFile.close();
566
567      throw new Error("File creation of file " + fileName + " failed", e);
568    }
569    return outFile;
570  }

581  /** Convert a list of arguments into a command-line string. Only used for debugging output. */
582  public String args_to_string(List<String> args) {
583    String str = "";
584    for (String arg : args) {
585      if (arg.indexOf(" ") != -1) {
586        str = "'" + str + "'";
587      }
588      str += arg + " ";
589    }
590    return (str.trim());
591  }

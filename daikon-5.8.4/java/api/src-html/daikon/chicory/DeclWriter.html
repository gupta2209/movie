023/**
024 * DeclWriter writes the {@code .decls} file to a stream. It uses traversal pattern trees (see
025 * {@link DaikonVariableInfo}) for each program point. These are also used by the {@link
026 * DTraceWriter}.
027 */
028@SuppressWarnings("nullness") // to do
029public class DeclWriter extends DaikonWriter {
030  // Notes:
031  //
032  //  Class.getName() returns JVM names (eg, [Ljava.lang.String;)
033
034  /** Header string before each new method entry or exit point. */
035  public static final String declareHeader = "DECLARE";
036
037  boolean debug = false;
038
039  // If the --comparability-file option is active, there might be
040  // variables for which DynComp saw no interactions and did not
041  // generate a comparability value. In order to reduce the number
042  // of useless invariants (when running Daikon on the output of
043  // Chicory), we generate a unique, dummy comparability value in
044  // these cases.  We start at the maximum integer and decrement
045  // for each use.  At the start of each method, object or class
046  // we reset the value back - though it's hard to imagine a case
047  // where there would be a collision with DynComp's values which
048  // increment up from an initial value of 1.
049  private static int initial_compare_value = Integer.MAX_VALUE;
050  private static int unique_compare_value;
051
052  /** Stream to write to. */
053  private PrintStream outFile;
054
055  /**
056   * Enable parent relations other than methods to their class objects. Turned off for now to match
057   * previous behavior.
058   */
059  private static boolean enable_object_user = false;

062   * Constructs a DeclWriter, preparing it to receive messages.
063   *
064   * @param writer stream to write to
065   */
066  public DeclWriter(PrintStream writer) {
067    super();
068    outFile = writer;
069    debug = Chicory.debug_decl_print;

072  /**
073   * Prints header information to the decls file. Should be called once before emitting any other
074   * declarations.
075   *
076   * @param className name of the top-level class (used only for printing comments)
077   */
078  public void printHeaderInfo(String className) {
079    outFile.println("// Declarations for " + className);
080    outFile.println("// Declarations written " + LocalDateTime.now(ZoneId.systemDefault()));
081    outFile.println();
082
083    // Determine comparability string
084    String comparability = "none";
085    if (Runtime.comp_info != null) {
086      comparability = "implicit";
087    }
088
089    if (Chicory.new_decl_format) {
090      outFile.printf("decl-version 2.0%n%n");
091      outFile.printf("var-comparability %s%n%n", comparability);
092    } else {
093      outFile.printf("VarComparability%n%s%n%n", Runtime.comp_info != null ? "implicit" : "none");
094
095      outFile.println("ListImplementors");
096      outFile.println("java.util.List");
097      outFile.println();
098    }
099  }

101  /**
102   * Returns the correctly formulated ":::OBJECT" name of the class (i.e., the program point name)
103   *
104   * @param type the ClassType type
105   * @return the correctly formulated String
106   */
107  public static String classObjectName(Class<?> type) {
108    return (type.getName() + ":::OBJECT");
109  }

111  /**
112   * Prints declarations for all the methods in the indicated class. This method is called in
113   * Runtime to print decls info for a class.
114   *
115   * @param cinfo class whose declarations should be printed
116   */
117  public void printDeclClass(ClassInfo cinfo, @Nullable DeclReader comp_info) {
118    if (Chicory.new_decl_format) {
119      print_decl_class(cinfo, comp_info);
120      return;
121    }
122
123    // Print all methods and constructors
124    for (MethodInfo mi : cinfo.get_method_infos()) {
125      Member member = mi.member;
126
127      // Don't want to instrument these types of methods
128      if (!shouldInstrumentMethod(member)) {
129        continue;
130      }
131
132      // Gset the root of the method's traversal pattern
133      RootInfo enterRoot = mi.traversalEnter;
134      assert enterRoot != null : "Traversal pattern not initialized at method " + mi.method_name;
135
136      printMethod(enterRoot, methodEntryName(member), comp_info);
137
138      // Print exit program point for EACH exit location in the method
139      // (that was encountered during this execution of the program).
140      Set<Integer> theExits = new HashSet<>(mi.exit_locations);
141      assert theExits.size() > 0 : mi;
142      for (Integer exitLoc : theExits) {
143        // Get the root of the method's traversal pattern
144        RootInfo exitRoot = mi.traversalExit;
145        assert enterRoot != null : "Traversal pattern not initialized at method " + mi.method_name;
146
147        printMethod(exitRoot, methodExitName(member, exitLoc.intValue()), comp_info);
148      }
149    }
150
151    printClassPpt(cinfo, cinfo.class_name + ":::CLASS", comp_info);
152    printObjectPpt(cinfo, classObjectName(cinfo.clazz), comp_info);
153  }

155  /**
156   * Prints a method's program point. This includes the declare header ("DECLARE"), the program
157   * point name, and the variable information.
158   *
159   * <p>This method uses variable information from the traversal tree.
160   *
161   * @param root the root of the traversal tree
162   * @param name the program point name
163   * @param comp_info comparability information
164   */
165  private void printMethod(RootInfo root, String name, DeclReader comp_info) {
166    outFile.println(declareHeader);
167    outFile.println(name);
168
169    if (debug) {
170      System.out.println("printMethod " + name);
171    }
172
173    for (DaikonVariableInfo childOfRoot : root) {
174      traverseDecl(childOfRoot, ((comp_info == null) ? null : comp_info.find_ppt(name)));
175    }
176
177    outFile.println();
178  }

180  /**
181   * Prints the .decls information for a single DaikonVariableInfo object, and recurses on its
182   * children. If the current variable has comparability defined in decl_ppt, that comparability is
183   * used. Otherwise -1 is used if there is comparability information available and the information
184   * in the variable is used if it is not.
185   */
186  private void traverseDecl(DaikonVariableInfo curInfo, DeclReader.@Nullable DeclPpt decl_ppt) {
187    if (curInfo.declShouldPrint()) {
188
189      if (!(curInfo instanceof StaticObjInfo)) {
190        outFile.println(curInfo.getName());
191        outFile.println(curInfo.getTypeName());
192        outFile.println(curInfo.getRepTypeName());
193        String comp_str = curInfo.getCompareString();
194        if (decl_ppt != null) {
195          comp_str = "-1";
196          DeclReader.DeclVarInfo varinfo = decl_ppt.find_var(curInfo.getName());
197          if (varinfo != null) comp_str = varinfo.get_comparability();
198        }
199        outFile.println(comp_str);
200      }
201    }
202
203    // Go through all of the current node's children
204    // and recurse
205    for (DaikonVariableInfo child : curInfo) {
206      traverseDecl(child, decl_ppt);
207    }
208  }

210  /**
211   * Prints the object program point. This contains the "this" object and the class's fields.
212   *
213   * @param cinfo information about the class whose program point to print
214   * @param name the program point name
215   * @param comp_info a declaration file
216   */
217  private void printObjectPpt(ClassInfo cinfo, String name, DeclReader comp_info) {
218    outFile.println(declareHeader);
219    outFile.println(name);
220
221    RootInfo root = RootInfo.getObjectPpt(cinfo, Runtime.nesting_depth);
222    for (DaikonVariableInfo childOfRoot : root) {
223      traverseDecl(childOfRoot, ((comp_info == null) ? null : comp_info.find_ppt(name)));
224    }
225
226    outFile.println();
227  }

229  /**
230   * Prints the class program point. This contains only the static variables. If there are no static
231   * variables to print, this method does nothing.
232   */
233  private void printClassPpt(ClassInfo cinfo, String name, DeclReader comp_info) {
234    if (num_class_vars(cinfo) == 0) {
235      return;
236    }
237
238    boolean printedHeader = false;
239    RootInfo root = RootInfo.getClassPpt(cinfo, Runtime.nesting_depth);
240
241    for (DaikonVariableInfo childOfRoot : root) {
242      // If we are here, there is at least 1 child
243      if (!printedHeader) {
244        outFile.println(declareHeader);
245        outFile.println(name);
246        printedHeader = true;
247      }
248
249      // Should just print out static fields
250      traverseDecl(childOfRoot, ((comp_info == null) ? null : comp_info.find_ppt(name)));
251    }
252
253    if (printedHeader) outFile.println();
254  }

256  /**
257   * Prints declarations for all the methods in the indicated class. This method is called in
258   * Runtime to print decls info for a class.
259   *
260   * @param cinfo class whose declarations should be printed
261   * @param comp_info comparability information
262   */
263  public void print_decl_class(ClassInfo cinfo, @Nullable DeclReader comp_info) {
264
265    if (debug) {
266      System.out.println("Enter print_decl_class: " + cinfo);
267    }
268
269    // Print all methods and constructors
270    for (MethodInfo mi : cinfo.get_method_infos()) {
271
272      Member member = mi.member;
273
274      // Don't want to instrument these types of methods
275      if (!shouldInstrumentMethod(member)) {
276        continue;
277      }
278
279      // Gset the root of the method's traversal pattern
280      RootInfo enterRoot = mi.traversalEnter;
281      assert enterRoot != null : "Traversal pattern not initialized at method " + mi.method_name;
282
283      String entryName =
284          (member != null
285              ? methodEntryName(member)
286              : mi.class_info.class_name + ".<clinit>" + FileIO.enter_tag);
287      print_method(mi, enterRoot, entryName, PptType.ENTER, comp_info);
288
289      // Print exit program point for EACH exit location in the method
290      // Note that there may not be any exits.  They may get filtered out,
291      // or some methods don't have an exit (only a throw)
292      Set<Integer> theExits = new HashSet<>(mi.exit_locations);
293      for (Integer exitLoc : theExits) {
294        // Get the root of the method's traversal pattern
295        RootInfo exitRoot = mi.traversalExit;
296        assert enterRoot != null : "Traversal pattern not initialized at method " + mi.method_name;
297
298        String exitName =
299            (member != null
300                ? methodExitName(member, exitLoc)
301                : mi.class_info.class_name + ".<clinit>" + FileIO.exit_tag + exitLoc);
302        print_method(mi, exitRoot, exitName, PptType.SUBEXIT, comp_info);
303      }
304    }
305
306    print_class_ppt(cinfo, cinfo.class_name + ":::CLASS", comp_info);
307    print_object_ppt(cinfo, classObjectName(cinfo.clazz), comp_info);
308
309    if (debug) System.out.println("Exit print_decl_class");
310  }

312  /**
313   * Prints a method's program point. This includes the ppt declaration, all of the ppt records, and
314   * records for each variable.
315   *
316   * <p>This method uses variable information from the traversal tree.
317   *
318   * @param mi the method information for the method
319   * @param root the root of the traversal tree
320   * @param name the program point name
321   * @param ppt_type the type of the program point (enter, exit, etc)
322   * @param comp_info comparability information
323   */
324  private void print_method(
325      MethodInfo mi, RootInfo root, String name, PptType ppt_type, @Nullable DeclReader comp_info) {
326
327    if (debug) {
328      System.out.println("Enter print_method: " + name);
329    }
330
331    // reset dummy comparability value
332    unique_compare_value = initial_compare_value;
333
334    outFile.println("ppt " + escape(name));
335
336    outFile.println("ppt-type " + ppt_type.name().toLowerCase());
337
338    // Look for and print any hierarchy relations
339    List<VarRelation> relations = new ArrayList<>();
340    for (DaikonVariableInfo child : root) {
341      find_relations(null, mi.is_static(), null, child, relations);
342    }
343    for (VarRelation relation : relations) {
344      outFile.println("parent parent " + relation.parent_ppt_name + " " + relation.id);
345    }
346
347    // Print each variable
348    for (DaikonVariableInfo childOfRoot : root) {
349      if (debug) System.out.println("method var: " + childOfRoot.getName());
350      traverse_decl(
351          null,
352          mi.is_static(),
353          null,
354          childOfRoot,
355          null,
356          relations,
357          ((comp_info == null) ? null : comp_info.find_ppt(name)));
358    }
359
360    outFile.println();
361
362    if (debug) System.out.println("Exit print_method ");
363  }

366   * Prints the class program point. This contains only the static variables. If there are no static
367   * variables to print, this method does nothing.
368   *
369   * @param cinfo class whose static declarations should be printed
370   * @param name the program point name
371   * @param comp_info comparability information
372   */
373  private void print_class_ppt(ClassInfo cinfo, String name, DeclReader comp_info) {
374
375    if (debug) {
376      System.out.println("Enter print_class_ppt: " + cinfo);
377    }
378
379    // reset dummy comparability value
380    unique_compare_value = initial_compare_value;
381
382    if (num_class_vars(cinfo) > 0) {
383
384      outFile.println("ppt " + escape(name));
385      outFile.println("ppt-type class");
386
387      // Print out the static fields
388      for (DaikonVariableInfo childOfRoot : RootInfo.getClassPpt(cinfo, Runtime.nesting_depth)) {
389        if (debug) System.out.println("class var: " + childOfRoot.getName());
390        traverse_decl(
391            null,
392            false,
393            null,
394            childOfRoot,
395            null,
396            null,
397            ((comp_info == null) ? null : comp_info.find_ppt(name)));
398      }
399
400      outFile.println();
401    }
402
403    if (debug) System.out.println("Exit print_class_ppt");
404  }

406  /**
407   * Prints the object program point. This contains the "this" object and the class' fields.
408   *
409   * @param cinfo class whose object program point should be printed
410   * @param name the program point name
411   * @param comp_info comparability information
412   */
413  private void print_object_ppt(ClassInfo cinfo, String name, DeclReader comp_info) {
414
415    if (debug) {
416      System.out.println("Enter print_object_ppt: " + cinfo);
417    }
418
419    // reset dummy comparability value
420    unique_compare_value = initial_compare_value;
421
422    outFile.println("ppt " + escape(name));
423    outFile.println("ppt-type object");
424    RootInfo root = RootInfo.getObjectPpt(cinfo, Runtime.nesting_depth);
425
426    // If there are any static variables, add the relation to
427    // the class ppt
428    List<VarRelation> relations = new ArrayList<>();
429    if (num_class_vars(cinfo) > 0) {
430      VarRelation relation = new VarRelation(cinfo.class_name + ":::CLASS", "parent");
431      relation.id = 1;
432      relations.add(relation);
433    }
434
435    // Look for and print any object-user relations
436    for (DaikonVariableInfo child : root) {
437      find_relations(cinfo, false, null, child, relations);
438    }
439    for (VarRelation relation : relations) {
440      outFile.println(
441          "parent " + relation.type + " " + relation.parent_ppt_name + " " + relation.id);
442    }
443
444    // Write out the variables
445    for (DaikonVariableInfo childOfRoot : root) {
446      if (debug) System.out.println("object var: " + childOfRoot.getName());
447      traverse_decl(
448          cinfo,
449          false,
450          null,
451          childOfRoot,
452          null,
453          relations,
454          ((comp_info == null) ? null : comp_info.find_ppt(name)));
455    }
456
457    outFile.println();
458
459    if (debug) System.out.println("Exit print_object_ppt");
460  }

462  /**
463   * Object program points are constructed for invariants about an object. We define an object
464   * invariant as one that is true at the entrance and exit of each public method and also each time
465   * an instance of the object is available to another method (eg, when it is passed as a parameter
466   * or available as a static). Daikon implements object invariants by merging the invariants from
467   * each public method and each user of the object. We refer to the relationship between variables
468   * at these program points as a Program point / variable hierarchy. This relationship must be
469   * defined in the declaration record. The VarRelation class tracks one relation.
470   */
471  private static class VarRelation {
472    /** Name of the program point for the parent. */
473    String parent_ppt_name;
474    /** Prefix of the variable name that is not part of the parent name. */
475    String local_prefix;
476    /** Prefix of the parent that replaces the local prefix. Normally 'this'. */
477    String parent_prefix;
478    /** Top level variable for the relation. */
479    String local_variable;
480    /** Type of the relation (parent, user, etc) */
481    String type;
482    /** Number that identifies this relation within this ppt. */
483    int id;
484
485    static SimpleLog debug = new SimpleLog(false);
486
487    /** Create a VarRelation. */
488    public VarRelation(
489        String parent_ppt_name,
490        String type,
491        String local_prefix,
492        String parent_prefix,
493        String local_variable) {
494      this.parent_ppt_name = parent_ppt_name;
495      this.type = type;
496      this.local_prefix = local_prefix;
497      this.parent_prefix = parent_prefix;
498      this.local_variable = local_variable;
499      debug.log("Created %s", this);
500    }
501
502    /** Create a var relation with the matching names. */
503    public VarRelation(String parent_ppt_name, String type) {
504      this(parent_ppt_name, type, null, null, null);
505    }
506
507    @SideEffectFree
508    @Override
509    public String toString(@GuardSatisfied VarRelation this) {
510      return String.format(
511          "VarRelation %s (%s->%s) %s [%s]",
512          parent_ppt_name, local_prefix, parent_prefix, local_variable, type);
513    }
514
515    /**
516     * Returns whether or not this relation is from a static variable in an object ppt to its
517     * matching variable at the class level.
518     */
519    @Pure
520    public boolean is_class_relation() {
521      return (parent_ppt_name.endsWith(":::CLASS"));
522    }
523
524    /**
525     * Returns the string defining the relation for the specified variable The format is
526     * parent-ppt-name id parent-variable-name. If the variable is static, it always has the same
527     * name in the parent (since fully specified names are used for static variables).
528     */
529    public String relation_str(DaikonVariableInfo var) {
530      String out = parent_ppt_name + " " + id;
531      if (!var.isStatic() && (local_prefix != null) && !local_prefix.equals(parent_prefix)) {
532        out += " " + var.getName().replaceFirst(Pattern.quote(local_prefix), parent_prefix);
533      }
534      return out;
535    }
536
537    /** Two VarRelations are equal if the refer to the same program point and local variable. */
538    @Override
539    @EnsuresNonNullIf(result = true, expression = "#1")
540    @Pure
541    public boolean equals(@GuardSatisfied VarRelation this, @GuardSatisfied @Nullable Object o) {
542      if (!(o instanceof VarRelation) || (o == null)) {
543        return false;
544      }
545      VarRelation vr = (VarRelation) o;
546      return (vr.parent_ppt_name.equals(parent_ppt_name)
547          && (((vr.local_variable == null) && local_variable == null)
548              || ((vr.local_variable != null) && vr.local_variable.equals(local_variable))));
549    }
550
551    @Override
552    @Pure
553    public int hashCode(@GuardSatisfied VarRelation this) {
554      return (parent_ppt_name.hashCode()
555          + ((local_variable == null) ? 0 : local_variable.hashCode()));
556    }
557  }

559  /**
560   * Prints the .decls information for a single DaikonVariableInfo object, and recurses on its
561   * children. If the current variable has comparability defined in compare_ppt, that comparability
562   * is used. Otherwise -1 is used if there is comparability information available and the
563   * information in the variable is used if it is not.
564   */
565  private void traverse_decl(
566      ClassInfo cinfo,
567      boolean is_static_method,
568      DaikonVariableInfo parent,
569      DaikonVariableInfo var,
570      VarRelation relation,
571      List<VarRelation> relations,
572      DeclReader.DeclPpt compare_ppt) {
573
574    if (debug) {
575      System.out.println("Enter traverse_decl: " + cinfo + ", " + var + ", " + parent);
576    }
577
578    if (!var.declShouldPrint()) {
579      // don't do anything
580    } else if (!(var instanceof StaticObjInfo)) {
581
582      // Write out the variable and its name
583      outFile.println("variable " + escape(var.getName()));
584
585      // Write out the kind of variable and its relative name
586      VarKind kind = var.get_var_kind();
587      String relative_name = var.get_relative_name();
588      outFile.print("  var-kind " + out_name(kind));
589      if (relative_name != null) {
590        outFile.print(" " + relative_name);
591      }
592      outFile.println();
593
594      // Write out the enclosing variable.
595      // If we are in an inner class, we need to special case the
596      // 'hidden' field that holds the outer class 'this' pointer.
597      // If the field name ends with ".this", it can only be this
598      // special case and we need to not output the enclosing-var.
599      if ((parent != null)
600          && !var.isStatic()
601          && !((relative_name != null) && relative_name.endsWith(".this"))) {
602        if (debug) System.out.println("traverse var parent: " + parent.getName());
603        outFile.println("  enclosing-var " + escape(parent.getName()));
604      }
605
606      // If this variable has multiple value, indicate it is an array
607      if (var.isArray()) {
608        outFile.println("  array 1");
609      }
610
611      // Write out the declared and representation types
612      outFile.println("  dec-type " + escape(var.getTypeNameOnly()));
613      outFile.println("  rep-type " + escape(var.getRepTypeNameOnly()));
614
615      // Write out the constant value (if present)
616      String const_val = var.get_const_val();
617      if (const_val != null) {
618        outFile.println("  constant " + const_val);
619      }
620
621      // Write out the arguments used to create (if present) the variable if it is a function
622      String function_args = var.get_function_args();
623      if (function_args != null) {
624        outFile.println("  function-args " + function_args);
625      }
626
627      // Write out the variable flags if any are set
628      EnumSet<VarFlags> var_flags = var.get_var_flags();
629      if (var_flags.size() > 0) {
630        outFile.print("  flags");
631        for (Enum<?> e : var_flags) {
632          outFile.print(" " + out_name(e));
633        }
634        outFile.println();
635      }
636
637      // Determine comparability and write it out
638      // Currently, the value returned by getCompareString() is always 22.
639      String comp_str = var.getCompareString();
640      if (compare_ppt != null) {
641        comp_str = "-1";
642        DeclReader.DeclVarInfo varinfo = compare_ppt.find_var(var.getName());
643        if (varinfo != null) comp_str = varinfo.get_comparability();
644      } else {
645        // Check to see if DynComp data is present.
646        if (Runtime.comp_info != null) {
647          // There is no comparability value for this variable as DynComp
648          // saw no interactions. In order to reduce the number of useless
649          // invariants, we will generate a unique, dummy comparability value.
650          comp_str = Integer.toString(unique_compare_value--);
651          if (var.isArray()) {
652            // Should output n index values to match number of dimensions.
653            // However, that value is hard to obtain at this point, so just
654            // always do one.  May cause some multi-dimension variables to
655            // be put into incorrect comparability set.  This is certainly
656            // no worse than previous algorithm. (markro)
657            comp_str = comp_str + "[" + Integer.toString(unique_compare_value--) + "]";
658          }
659        }
660      }
661      outFile.println("  comparability " + comp_str);
662
663      // Determine if there is a ppt for variables of this type
664      // If found this should match one of the previously found relations
665      // for this ppt.
666      // Once a relation has been found, we don't look for recursive
667      // relationships
668      if ((relation == null) && (relations != null)) {
669        relation = find_relation(cinfo, is_static_method, parent, var);
670        if (relation != null) {
671          // System.out.printf("Found relation %s, variable %s%n", relation,
672          //                   var);
673          int index = relations.indexOf(relation);
674          assert (index != -1) : "Relation " + relation + " not found in " + relations;
675          relation = relations.get(index);
676        }
677      }
678
679      // Put out the variable relation (if one exists).
680      if (relation != null) {
681        outFile.println("  parent " + relation.relation_str(var));
682      }
683
684    } else { // this is the dummy root for class statics
685      if ((relations != null) && (relations.size() > 0)) {
686        relation = find_relation(cinfo, true, parent, var);
687        if (relation != null) {
688          int index = relations.indexOf(relation);
689          assert (index != -1) : "Relation " + relation + " not found in " + relations;
690          relation = relations.get(index);
691          // System.out.printf("Found class relation %s for cinfo %s%n",
692          //                   relation, cinfo);
693        } else {
694          System.out.printf("No class relation found for cinfo %s%n", cinfo);
695        }
696      }
697    }
698
699    // Go through all of the current node's children
700    // and recurse
701    for (DaikonVariableInfo child : var) {
702      if (debug) System.out.println("traverse var: " + child.getName());
703      traverse_decl(cinfo, is_static_method, var, child, relation, relations, compare_ppt);
704    }
705
706    if (debug) System.out.println("Exit traverse_decl");
707  }

717  /**
718   * Looks to see if there is a class that we are instrumenting that matches the type of this
719   * variable. If so, returns a VarRelation that describes the hierarchy relationship between this
720   * variable (and its field) and the variables within the object ppt for the class. If this is an
721   * object ppt (ci != null), then each top level static variable has a relation to the class ppt.
722   *
723   * @param cinfo class of the object ppt. Null if this is not an object ppt.
724   * @param is_static_method true if this ppt is a static method enter
725   * @param parent parent of var in the variable tree
726   * @param var variable whose relation is desired
727   */
728  private @Nullable VarRelation find_relation(
729      @Nullable ClassInfo cinfo,
730      boolean is_static_method,
731      DaikonVariableInfo parent,
732      DaikonVariableInfo var) {
733
734    // Look for object->class static relationship.  This starts on each
735    // static variable under 'this' (the static variables of a class are
736    // placed in the CLASS ppt).
737    if (cinfo != null && var.isStatic() && (parent instanceof ThisObjInfo)) {
738      return new VarRelation(cinfo.class_name + ":::CLASS", "parent");
739    }
740
741    // Only hashcodes have object ppts
742    if (!var.getRepTypeNameOnly().equals("hashcode")) {
743      return null;
744    }
745
746    // Get the type (class) of this variable
747    String decl_type = var.getTypeNameOnly();
748    // System.out.printf("Looking for hierarchy type %s%n", decl_type);
749
750    // If this ppt is the object ppt for this type, don't create a relation
751    // to it.
752    if ((cinfo != null) && cinfo.class_name.equals(decl_type)) {
753      return null;
754    }
755
756    // Look to see if we are instrumenting this class.  If we are, then
757    // there should be an object ppt for this class.  If this is a static
758    // method and the relation is over the dummy static variable, it
759    // relates directly to the class ppt, otherwise to the object
760    // ppt.  Note that a relation to the class ppt is returned only if there
761    // are static variables.
762    for (ClassInfo ci : SharedData.all_classes) {
763      if (ci.class_name.equals(decl_type)) {
764        // System.out.printf("*Found match for %s : %s%n", decl_type, ci);
765        String ppt_marker = ":::OBJECT";
766        if (is_static_method && (var instanceof StaticObjInfo)) {
767          // System.out.printf("num_class_vars for classinfo %s%n", ci);
768          if (num_class_vars(ci) == 0) {
769            return null;
770          }
771          ppt_marker = ":::CLASS";
772        }
773        if (!enable_object_user && !var.getName().equals("this")) {
774          return null;
775        }
776        return new VarRelation(
777            decl_type + ppt_marker, "parent", var.getName(), "this", var.getName());
778      }
779    }
780
781    return null;
782  }

784  /**
785   * Looks for all of the object-user ppt/variable hiearchy relations beginning at var. Once a
786   * relation is found, no more relations are looked for under that variable. In most cases, it
787   * would be expected that only one relation will be found (either var is a class with a
788   * corresponding object ppt or it is not). However, depending on what classes are being
789   * instrumented, it might be possible for a class not to have an object ppt while multiple
790   * children do have object ppts.
791   *
792   * <p>Any relations that are found are added to the relations list.
793   */
794  private void find_relations(
795      ClassInfo ci,
796      boolean is_static_method,
797      DaikonVariableInfo parent,
798      DaikonVariableInfo var,
799      List<VarRelation> relations) {
800
801    // If there is a new relation for this variable add it to the list and
802    // return it.  Note that each static variable in an object ppt will
803    // have a relation to the matching static variable in class ppt.  Only
804    // one of these should go in the list of relations.
805    VarRelation relation = find_relation(ci, is_static_method, parent, var);
806    if (relation != null) {
807      if ((relations.size() == 0)
808          || (relations.get(0).is_class_relation() && relation.is_class_relation())) {
809        relations.add(relation);
810        relation.id = relations.size();
811        return;
812      }
813    }
814
815    // Look for a relation in each child.
816    for (DaikonVariableInfo child : var) {
817      find_relations(ci, is_static_method, parent, child, relations);
818    }
819  }

821  /**
822   * Returns the number of variables in the CLASS program point. The CLASS ppt contains all of the
823   * static variables in the class (if any).
824   */
825  private int num_class_vars(ClassInfo cinfo) {
826
827    RootInfo class_root = RootInfo.getClassPpt(cinfo, Runtime.nesting_depth);
828    assert class_root.children.size() == 1;
829    DaikonVariableInfo static_root = class_root.children.get(0);
830    return static_root.children.size();
831  }

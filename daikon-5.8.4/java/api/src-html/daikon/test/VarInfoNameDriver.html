034  private static final Map<String, Handler> handlers = new HashMap<>();
035
036  public static void run(InputStream commands, PrintStream output) {
037    try {
038      run_helper(commands, output);
039    } catch (IOException e) {
040      throw new RuntimeException(e.toString());
041    }
042  }
043
044  private static void run_helper(InputStream _commands, PrintStream output) throws IOException {
045    BufferedReader commands = new BufferedReader(new InputStreamReader(_commands, UTF_8));
046    Map<String, VarInfoName> variables = new HashMap<>();
047
048    String command;
049    while ((command = commands.readLine()) != null) {
050      if (command.startsWith(";")) {
051        output.println(command);
052        continue;
053      }
054
055      // tokenize arguments
056      StringTokenizer tok = new StringTokenizer(command);
057      Deque<String> list = new ArrayDeque<String>();
058      while (tok.hasMoreTokens()) {
059        list.add(tok.nextToken());
060      }
061
062      // ignore blank lines
063      if (list.size() == 0) {
064        continue;
065      }
066
067      output.println("; " + command);
068
069      // call the handler
070      String method = list.removeFirst();
071      String[] args = list.toArray(new String[list.size()]);
072      Handler handler = handlers.get(method);
073      if (handler == null) {
074        throw new UnsupportedOperationException("Unknown method: " + method);
075      }
076      handler.handle(variables, args, output);
077    }
078  }

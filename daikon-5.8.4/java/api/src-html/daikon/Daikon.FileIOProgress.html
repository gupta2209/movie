668  public static void handleDaikonTerminationException(DaikonTerminationException e) {
669    if (e instanceof NormalTermination) {
670      System.out.println();
671      if (e.getMessage() != null) {
672        System.out.println(e.getMessage());
673      }
674      System.exit(0);
675    } else if (e instanceof UserError) {
676      System.err.println();
677      System.err.println(e.getMessage());
678      if (Debug.dkconfig_show_stack_trace) {
679        e.printStackTrace();
680      }
681      System.exit(1);
682    } else if (e instanceof BugInDaikon) {
683      System.err.println();
684      System.err.println("Bug in Daikon.  Please report.");
685      e.printStackTrace(System.err);
686      System.err.println("Bug in Daikon.  Please report.");
687      System.exit(2);
688    } else {
689      // This caes should never be executed.
690      System.err.println();
691      System.err.println("Bug in Daikon.  Please report.");
692      e.printStackTrace(System.err);
693      System.err.println("Bug in Daikon.  Please report.");
694      System.exit(2);
695    }
696  }

703  public static void mainHelper(final String[] args) {

711    Set<File> spinfo_files = files.spinfo;
712    Set<File> map_files = files.map;

714      System.out.println("No .decls or .dtrace files specified");
715      throw new Daikon.UserError("No .decls or .dtrace files specified");
716    }
717
718    // Never disable splitting for csharp format.
719    if (Daikon.dkconfig_undo_opts && Daikon.output_format != OutputFormat.CSHARPCONTRACT) {
720      PptSplitter.dkconfig_disable_splitting = true;
721    }
722
723    if (Daikon.dkconfig_quiet) {
724      Daikon.dkconfig_progress_delay = -1;
725    }
726    if (System.console() == null) {
727      // not connected to a terminal
728      Daikon.dkconfig_progress_delay = -1;
729    }
730
731    // Set up debug traces; note this comes after reading command line options.
732    LogHelper.setupLogs(Global.debugAll ? LogHelper.FINE : LogHelper.INFO);
733
734    if (!noversion_output) {
735      if (!Daikon.dkconfig_quiet) System.out.println(release_string);
736    }
737
738    // figure out which algorithm to use in NIS to process suppressions
739    if (NIS.dkconfig_suppression_processor == SuppressionProcessor.HYBRID) {
740      NIS.hybrid_method = true;
741    } else {
742      if (NIS.dkconfig_suppression_processor == SuppressionProcessor.ANTECEDENT) {
743        NIS.antecedent_method = true;
744        NIS.hybrid_method = false;
745      } else {
746        assert (NIS.dkconfig_suppression_processor == SuppressionProcessor.FALSIFIED);
747        NIS.antecedent_method = false;
748        NIS.hybrid_method = false;
749      }
750    }
751
752    // Create the list of all invariant types
753    setup_proto_invs();
754
755    if (PrintInvariants.print_discarded_invariants) {
756      DiscReasonMap.initialize();
757    }
758
759    fileio_progress = new FileIOProgress();
760    fileio_progress.start();
761
762    // Load declarations and splitters
763    load_spinfo_files(spinfo_files);
764    all_ppts = load_decls_files(decls_files);
765    load_map_files(map_files);
766
767    all_ppts.trimToSize();
768
769    // Only for assertion checks
770    isInferencing = true;
771
772    // Infer invariants
773    process_data(all_ppts, dtrace_files);
774    isInferencing = false;
775    if (Debug.logOn()) {
776      Debug.check(all_ppts, "After process data");
777    }
778
779    // If requested, just calculate the total number of invariants possible
780    if (dkconfig_calc_possible_invs) {
781      fileio_progress.shouldStop = true;
782      int total_invs = 0;
783      // Can't use new for syntax because the default iterator for all_ppts
784      // is not the one I want here.
785      for (PptTopLevel ppt : all_ppts.ppt_all_iterable()) {
786        System.out.printf("Processing %s with %d variables", ppt.name(), ppt.var_infos.length);
787        int inv_cnt = 0;
788        if (ppt.var_infos.length > 1600) {
789          System.out.println("Skipping, too many variables!");
790        } else {
791          inv_cnt = ppt.getInvariants().size();
792          System.out.println(" " + inv_cnt + " invariants");
793          total_invs += inv_cnt;
794        }
795      }
796      System.out.println(total_invs + " invariants total");
797      return;
798    }
799
800    if (suppress_redundant_invariants_with_simplify) {
801      suppressWithSimplify(all_ppts);
802    }
803
804    // Check that PptMap created was correct
805    all_ppts.repCheck();
806
807    // Remove undesired invariants, if requested
808    if (omit_from_output) {
809      processOmissions(all_ppts);
810    }
811
812    // Write serialized output - must be done before guarding invariants
813    if (inv_file != null) {
814      try {
815        FileIO.write_serialized_pptmap(all_ppts, inv_file);
816      } catch (IOException e) {
817        throw new RuntimeException("Error while writing .inv file: " + inv_file, e);
818      }
819    }
820
821    //     if ((Daikon.dkconfig_guardNulls == "always") // interned
822    //         || (Daikon.dkconfig_guardNulls == "missing")) { // interned
823    //       // This side-effects the PptMap, but it has already been saved
824    //       // to disk and is now being used only for printing.
825    //       guardInvariants(all_ppts);
826    //     }
827
828    // Debug print information about the variables
829    if (false) {
830      for (PptTopLevel ppt : all_ppts.all_ppts()) {
831        System.out.printf("Dumping variables for ppt %s%n", ppt.name());
832        for (VarInfo vi : ppt.var_infos) {
833          System.out.printf("  vi %s%n", vi);
834          System.out.printf("    file_rep_type = %s%n", vi.file_rep_type);
835          System.out.printf("    type = %s%n", vi.type);
836        }
837      }
838    }
839
840    // print out the invariants for each program point
841    if (Daikon.dkconfig_undo_opts) {
842      // Print out the invariants for each program point (sort first)
843      for (PptTopLevel ppt : all_ppts.pptIterable()) {
844
845        // We do not need to print out program points that have not seen
846        // any samples.
847        if (ppt.num_samples() == 0) {
848          continue;
849        }
850        List<Invariant> invs = PrintInvariants.sort_invariant_list(ppt.invariants_vector());
851        List<Invariant> filtered_invs = filter_invs(invs);
852
853        // Debugging output.  Sometimes the program points actually differ in number of samples seen
854        // due to differences in how Daikon and DaikonSimple see the variable hierarchy.
855        if (false) {
856          System.out.println("====================================================");
857          System.out.println(ppt.name());
858          System.out.println(ppt.num_samples());
859
860          for (Invariant inv : filtered_invs) {
861            System.out.println(inv.getClass());
862            System.out.println(inv);
863          }
864        }
865      }
866
867      // exit the program
868      if (false) {
869        return;
870      }
871    }
872
873    // Display invariants
874    if (output_num_samples) {
875      System.out.println("The --output_num_samples debugging flag is on.");
876      System.out.println("Some of the debugging output may only make sense to Daikon programmers.");
877    }
878
879    // If they want to see discarded invariants, they probably don't
880    // want to see the true ones.
881    if (!PrintInvariants.print_discarded_invariants) {
882      PrintInvariants.print_invariants(all_ppts);
883    } else {
884      PrintInvariants.print_reasons(all_ppts);
885    }
886
887    if (output_num_samples) {
888      Global.output_statistics();
889    }
890    if (dkconfig_print_sample_totals) {
891      System.out.println(FileIO.samples_processed + " samples processed");
892    }
893
894    // print statistics concerning what invariants are printed
895    if (debugStats.isLoggable(Level.FINE)) {
896      for (PptTopLevel ppt : all_ppts.ppt_all_iterable()) {
897        PrintInvariants.print_filter_stats(debugStats, ppt, all_ppts);
898      }
899    }
900
901    // Done
902    if (!Daikon.dkconfig_quiet) {
903      System.out.println("Exiting Daikon.");
904    }
905  }

980    HashSet<File> map_files = new LinkedHashSet<>();
981
982    LongOpt[] longopts =
983        new LongOpt[] {
984          // Control output
985          new LongOpt(help_SWITCH, LongOpt.NO_ARGUMENT, null, 0),

1003          new LongOpt(ppt_omit_regexp_SWITCH, LongOpt.REQUIRED_ARGUMENT, null, 0),

1027          if (help_SWITCH.equals(option_name)) {
1028            System.out.println(usage);
1029            throw new Daikon.NormalTermination();

1036              throw new Daikon.UserError("Unknown output format:  --format " + format_name);
1037            }

1039            show_progress = true;

1046            output_num_samples = true;

1050              for (String filename : new EntryReader(files_from_filename)) {
1051                // Ignore blank lines in file.
1052                if (filename.equals("")) {
1053                  continue;
1054                }
1055                // This code is duplicated below outside the options loop.
1056                // These aren't "endsWith()" because there might be a suffix
1057                // on the end (eg, a date, or ".gz").

1060                  throw new Daikon.UserError("File " + filename + " not found.");
1061                }

1065                  dtrace_files.add(filename);

1069                  map_files.add(file);
1070                } else {

1072                }

1095            }

1099            try {
1100              String list_type_string = getOptarg(g);
1101              ProglangType.list_implementors.add(list_type_string);
1102            } catch (Exception e) {
1103              throw new Daikon.UserError("Problem parsing " + list_type_SWITCH + " option: " + e);
1104            }
1105            break;
1106          } else if (user_defined_invariant_SWITCH.equals(option_name)) {
1107            try {
1108              String user_defined_invariant_string = getOptarg(g);
1109              Matcher m = classGetNamePattern.matcher(user_defined_invariant_string);
1110              if (!m.matches()) {
1111                throw new Daikon.UserError(
1112                    "Bad argument "
1113                        + user_defined_invariant_string
1114                        + " for "
1115                        + ppt_regexp_SWITCH
1116                        + ": not in the format required by Class.getName(String)");
1117              }
1118              @SuppressWarnings("signature") // Regex match guarantees the format of Class.getName()
1119              @ClassGetName String cname = user_defined_invariant_string;
1120              userDefinedInvariants.add(cname);
1121            } catch (Exception e) {
1122              throw new Daikon.UserError(
1123                  "Problem parsing " + user_defined_invariant_SWITCH + " option: " + e);
1124            }
1125            break;
1126          } else if (disable_all_invariants_SWITCH.equals(option_name)) {

1147            {
1148              ClassLoader cl = Thread.currentThread().getContextClassLoader();
1149              if (cl == null) {
1150                throw new Daikon.BugInDaikon(

1152              }
1153              try {
1154                classesAsObject = f.get(cl);
1155              } catch (IllegalAccessException e) {
1156                throw new Daikon.BugInDaikon("Field ClassLoader.classes was not made accessible");
1157              }

1160              "unchecked", // type of ClassLoader.classes field is known
1161              "nullness" // ClassLoader.classes is non-null

1171                  if (field.getType() != Boolean.TYPE) {
1172                    throw new Daikon.BugInDaikon(

1174                            + invType
1175                            + ".dkconfig_enabled has type "
1176                            + field.getType()
1177                            + " instead of boolean");
1178                  } else {
1179                    setStaticField(field, false);
1180                    // System.out.println(
1181                    //     "Set field "
1182                    //         + invType
1183                    //         + ".dkconfig_enabled to false");
1184                  }
1185                } catch (NoSuchFieldException e) {
1186                  // System.out.println(
1187                  //     "Class " + invType + " does not have a dkconfig_enabled field");
1188                } catch (IllegalAccessException e) {
1189                  throw new Daikon.BugInDaikon(
1190                      "IllegalAccessException for field " + invType + ".dkconfig_enabled");
1191                }
1192              }
1193            }
1194          } else if (no_dataflow_hierarchy_SWITCH.equals(option_name)) {
1195            use_dataflow_hierarchy = false;
1196          } else if (suppress_redundant_SWITCH.equals(option_name)) {
1197            suppress_redundant_invariants_with_simplify = true;
1198          }
1199
1200          // Process only part of the trace file
1201          else if (ppt_regexp_SWITCH.equals(option_name)) {
1202            if (ppt_regexp != null) {
1203              throw new Daikon.UserError(
1204                  "multiple --"
1205                      + ppt_regexp_SWITCH
1206                      + " regular expressions supplied on command line");
1207            }
1208            String regexp_string = getOptarg(g);
1209            if (!RegexUtil.isRegex(regexp_string)) {
1210              throw new Daikon.UserError(
1211                  "Bad regexp "
1212                      + regexp_string
1213                      + " for "
1214                      + ppt_regexp_SWITCH
1215                      + ": "
1216                      + RegexUtil.regexError(regexp_string));
1217            }
1218            ppt_regexp = Pattern.compile(regexp_string);
1219            break;
1220          } else if (ppt_omit_regexp_SWITCH.equals(option_name)) {
1221            if (ppt_omit_regexp != null) {
1222              throw new Daikon.UserError(
1223                  "multiple --"
1224                      + ppt_omit_regexp_SWITCH
1225                      + " regular expressions supplied on command line");
1226            }
1227            String regexp_string = getOptarg(g);
1228            if (!RegexUtil.isRegex(regexp_string)) {
1229              throw new Daikon.UserError(
1230                  "Bad regexp "
1231                      + regexp_string
1232                      + " for "
1233                      + ppt_omit_regexp_SWITCH
1234                      + ": "
1235                      + RegexUtil.regexError(regexp_string));
1236            }
1237            ppt_omit_regexp = Pattern.compile(regexp_string);
1238            break;
1239          } else if (var_regexp_SWITCH.equals(option_name)) {

1243                      + var_regexp_SWITCH
1244                      + " regular expressions supplied on command line");

1254                      + RegexUtil.regexError(regexp_string));

1258          } else if (var_omit_regexp_SWITCH.equals(option_name)) {
1259            if (var_omit_regexp != null) {
1260              throw new Daikon.UserError(
1261                  "multiple --"
1262                      + var_omit_regexp_SWITCH
1263                      + " regular expressions supplied on command line");
1264            }
1265            String regexp_string = getOptarg(g);
1266            if (!RegexUtil.isRegex(regexp_string)) {
1267              throw new Daikon.UserError(

1269                      + regexp_string
1270                      + " for "
1271                      + var_omit_regexp_SWITCH
1272                      + ": "
1273                      + RegexUtil.regexError(regexp_string));

1275            var_omit_regexp = Pattern.compile(regexp_string);
1276            break;
1277          } else if (server_SWITCH.equals(option_name)) {

1279            server_dir = new File(input_dir);

1283            }
1284            break;

1286            // Configuration options
1287
1288          } else if (config_SWITCH.equals(option_name)) {

1306            break;
1307          } else if (debugAll_SWITCH.equals(option_name)) {

1326          } else {
1327            throw new Daikon.UserError("Unknown option " + option_name + " on command line");
1328          }
1329          break;
1330        case 'h':

1333        case 'o':
1334          String inv_filename = getOptarg(g);
1335
1336          if (inv_file != null) {

1340                    + " "
1341                    + inv_filename);
1342          }

1346          if (!UtilPlume.canCreateAndWrite(inv_file)) {
1347            throw new Daikon.UserError("Cannot write to serialization output file " + inv_file);
1348          }
1349          break;
1350          //
1351        case '?':
1352          // break; // getopt() already printed an error
1353          System.out.println(usage);
1354          throw new Daikon.NormalTermination();
1355          //
1356        default:
1357          throw new Daikon.BugInDaikon("getopt() returned " + c);
1358      }
1359    }
1360
1361    // This code is duplicated above within the switch processing.
1362    // First check that all the file names are OK, so we don't do lots of
1363    // processing only to bail out at the end.
1364    for (int i = g.getOptind(); i < args.length; i++) {
1365      String filename = args[i];
1366      if (filename.equals("-") || filename.equals("+")) {
1367        dtrace_files.add(filename);
1368        continue;
1369      }
1370
1371      File file = new File(filename);
1372      if (!file.exists()) {
1373        throw new Daikon.UserError("File " + file + " not found.");
1374      }
1375      filename = file.toString();
1376
1377      // These aren't "endsWith()" because there might be a suffix on the end
1378      // (eg, a date or ".gz").
1379      if (filename.indexOf(".decls") != -1) {
1380        decl_files.add(file);
1381      } else if (filename.indexOf(".dtrace") != -1) {
1382        dtrace_files.add(filename);
1383        // Always output an invariant file by default, even if none is
1384        // specified on the command line.
1385        if (inv_file == null) {
1386          String basename;
1387          // This puts the .inv file in the current directory.
1388          basename = new File(filename).getName();
1389          // This puts the .inv file in the same directory as the .dtrace file.
1390          // basename = filename;
1391          int base_end = basename.indexOf(".dtrace");
1392          String inv_filename = basename.substring(0, base_end) + ".inv.gz";
1393
1394          inv_file = new File(inv_filename);
1395          if (!UtilPlume.canCreateAndWrite(inv_file)) {
1396            throw new Daikon.UserError("Cannot write to file " + inv_file);
1397          }
1398        }
1399      } else if (filename.indexOf(".spinfo") != -1) {
1400        spinfo_files.add(file);
1401      } else if (filename.indexOf(".map") != -1) {
1402        map_files.add(file);
1403      } else {
1404        throw new Daikon.UserError("Unrecognized file type: " + file);
1405      }
1406    }
1407
1408    // Set the fuzzy float comparison ratio.  This needs to be done after
1409    // any configuration options (which may set the ratio) are processed.
1410    Global.fuzzy.setRelativeRatio(Invariant.dkconfig_fuzzy_ratio);
1411
1412    // Setup ppt_max_name based on the specified percentage of ppts to process
1413    if (dkconfig_ppt_perc != 100) {
1414      ppt_max_name = setup_ppt_perc(decl_files, dkconfig_ppt_perc);
1415      System.out.println("Max ppt name = " + ppt_max_name);
1416    }
1417
1418    // Validate guardNulls option
1419    PrintInvariants.validateGuardNulls();
1420
1421    return new FileOptions(decl_files, dtrace_files, spinfo_files, map_files);
1422  }

1664
1665    // Binary sequence-scalar invariants
1666    {
1667      // Comparison of scalar to each array element (SeqIntComparison.java.jpp)
1668      proto_invs.add(SeqIntEqual.get_proto());
1669      proto_invs.add(SeqIntLessThan.get_proto());
1670      proto_invs.add(SeqIntGreaterThan.get_proto());
1671      proto_invs.add(SeqIntLessEqual.get_proto());
1672      proto_invs.add(SeqIntGreaterEqual.get_proto());
1673      proto_invs.add(SeqFloatEqual.get_proto());
1674      proto_invs.add(SeqFloatLessThan.get_proto());
1675      proto_invs.add(SeqFloatGreaterThan.get_proto());
1676      proto_invs.add(SeqFloatLessEqual.get_proto());
1677      proto_invs.add(SeqFloatGreaterEqual.get_proto());
1678
1679      // Scalar is an element of the array (Member.java.jpp)
1680      proto_invs.add(Member.get_proto());
1681      proto_invs.add(MemberFloat.get_proto());
1682      proto_invs.add(MemberString.get_proto());
1683    }
1684
1685    // Ternary invariants
1686    {
1687      // FunctionBinary (FunctionBinary.java.jpp)
1688      proto_invs.addAll(FunctionBinary.get_proto_all());
1689      proto_invs.addAll(FunctionBinaryFloat.get_proto_all());
1690
1691      // LinearTernary (LinearTernary.java.jpp)
1692      proto_invs.add(LinearTernary.get_proto());
1693      proto_invs.add(LinearTernaryFloat.get_proto());
1694    }
1695
1696    // User-defined invariants
1697    for (String invariantClassName : userDefinedInvariants) {
1698      Class<?> invClass;
1699      try {
1700        invClass = Class.forName(invariantClassName);
1701      } catch (ClassNotFoundException e) {
1702        throw new Daikon.UserError(
1703            "Cannot load class "
1704                + invariantClassName
1705                + " in "
1706                + user_defined_invariant_SWITCH
1707                + " command-line argument; is it on the classpath?");
1708      }
1709      Method get_proto_method;
1710      try {
1711        get_proto_method = invClass.getMethod("get_proto");
1712      } catch (NoSuchMethodException e) {
1713        throw new Daikon.UserError(
1714            "No get_proto() method in user-defined invariant class " + invariantClassName);
1715      } catch (SecurityException e) {
1716        throw new Daikon.UserError(
1717            e,
1718            "SecurityException while looking up get_proto() method in user-defined invariant class "
1719                + invariantClassName);
1720      }
1721      Invariant inv;
1722      try {
1723        @SuppressWarnings("nullness") // null argument is OK because get_proto_method is static
1724        Object inv_as_object = get_proto_method.invoke(null);
1725        if (inv_as_object == null) {
1726          throw new Daikon.UserError(
1727              invariantClassName
1728                  + ".get_proto() returned null but should have returned an Invariant");
1729        }
1730        if (!(inv_as_object instanceof Invariant)) {
1731          Class<?> cls = inv_as_object.getClass();
1732          throw new Daikon.UserError(
1733              invariantClassName
1734                  + ".get_proto() returned object of the wrong type.  It should have been a subclass of invariant, but was "
1735                  + cls
1736                  + ": "
1737                  + inv_as_object);
1738        }
1739        inv = (Invariant) inv_as_object;
1740      } catch (Exception e) {
1741        throw new Daikon.UserError(
1742            e, "Exception while invoking " + invariantClassName + ".get_proto()");
1743      }
1744      proto_invs.add(inv);
1745    }
1746
1747    // Remove any elements that are not enabled
1748    for (Iterator<@Prototype Invariant> i = proto_invs.iterator(); i.hasNext(); ) {
1749      @Prototype Invariant inv = i.next();
1750      assert inv != null;
1751      if (!inv.enabled()) i.remove();
1752    }
1753  }

1879  // The function filters out the reflexive invs in binary slices,
1880  // reflexive and partially reflexive invs in ternary slices
1881  // and also filters out the invariants that have not seen enough
1882  // samples in ternary slices.
1883  static List<Invariant> filter_invs(List<Invariant> invs) {
1884    List<Invariant> new_list = new ArrayList<>();
1885
1886    for (Invariant inv : invs) {
1887      VarInfo[] vars = inv.ppt.var_infos;
1888
1889      // This check is the most non-intrusive way to filter out the invs
1890      // Filter out reflexive invariants in the binary invs
1891      if (!((inv.ppt instanceof PptSlice2) && vars[0] == vars[1])) {
1892
1893        // Filter out the reflexive and partially reflexive invs in the
1894        // ternary slices
1895        if (!((inv.ppt instanceof PptSlice3)
1896            && (vars[0] == vars[1] || vars[1] == vars[2] || vars[0] == vars[2]))) {
1897          if (inv.ppt.num_values() != 0) {
1898
1899            // filters out "warning: too few samples for
1900            // daikon.inv.ternary.threeScalar.LinearTernary invariant"
1901            if (inv.isActive()) {
1902              new_list.add(inv);
1903            }
1904          }
1905        }
1906      }
1907    }
1908
1909    return new_list;
1910  }

1912  /**
1913   * Add orig() variables to the given EXIT/EXITnn point. Does nothing if exit_ppt is not an
1914   * EXIT/EXITnn.
1915   */
1916  private static void create_orig_vars(PptTopLevel exit_ppt, PptMap ppts) {
1917    if (!exit_ppt.ppt_name.isExitPoint()) {
1918      if (VarInfo.assertionsEnabled()) {
1919        for (VarInfo vi : exit_ppt.var_infos) {
1920          try {
1921            vi.checkRep();
1922          } catch (Throwable e) {
1923            System.err.println();
1924            System.err.println("Error with variable " + vi + " at ppt " + exit_ppt);
1925            throw e;
1926          }
1927        }
1928      }
1929      return;
1930    }
1931
1932    if (debugInit.isLoggable(Level.FINE)) {
1933      debugInit.fine("Doing create and relate orig vars for: " + exit_ppt.name());
1934    }
1935
1936    PptTopLevel entry_ppt = ppts.get(exit_ppt.ppt_name.makeEnter());
1937    if (entry_ppt == null) {
1938      throw new Daikon.UserError("exit found with no corresponding entry: " + exit_ppt.name());
1939    }
1940
1941    // Add "orig(...)" (prestate) variables to the program point.
1942    // Don't bother to include the constants.  Walk through
1943    // entry_ppt's vars.  For each non-constant, put it on the
1944    // new_vis worklist after fixing its comparability information.
1945    exit_ppt.num_orig_vars = entry_ppt.num_tracevars;
1946    VarInfo[] new_vis = new VarInfo[exit_ppt.num_orig_vars];
1947    {
1948      VarInfo[] entry_ppt_vis = entry_ppt.var_infos;
1949      int new_vis_index = 0;
1950      for (int k = 0; k < entry_ppt.num_declvars; k++) {
1951        VarInfo vi = entry_ppt_vis[k];
1952        assert !vi.isDerived() : "Derived when making orig(): " + vi.name();
1953        if (vi.isStaticConstant()) {
1954          continue;
1955        }
1956        VarInfo origvar = VarInfo.origVarInfo(vi);
1957        // Fix comparability
1958        VarInfo postvar = exit_ppt.find_var_by_name(vi.name());
1959        if (postvar == null) {
1960          System.out.printf(
1961              "Looking for postvar, can't find var %s in exit of ppt %s%n", vi, exit_ppt.name());
1962          for (VarInfo cvi : entry_ppt.var_infos) {
1963            System.out.printf("  entry var = %s%n", cvi);
1964          }
1965          for (VarInfo cvi : exit_ppt.var_infos) {
1966            System.out.printf("  exit var = %s%n", cvi);
1967          }
1968          throw new RuntimeException("this can't happen: postvar is null");
1969        }
1970        origvar.postState = postvar;
1971        origvar.comparability = postvar.comparability.makeAlias();
1972
1973        // add parents for override relations
1974        // exit_ppt.parents has not been loaded at this point
1975        for (VarParent pi : postvar.parents) {
1976          PptTopLevel parentppt = ppts.get(pi.parent_ppt);
1977          if (parentppt != null) {
1978            if (!parentppt.is_object() && !parentppt.is_class()) {
1979              VarInfo pvi =
1980                  parentppt.find_var_by_name(
1981                      pi.parent_variable == null ? postvar.name() : pi.parent_variable);
1982              if (pvi != null) {
1983                origvar.parents.add(
1984                    new VarParent(pi.parent_ppt, pi.parent_relation_id, pvi.prestate_name()));
1985              }
1986            }
1987          }
1988        }
1989
1990        // Add to new_vis
1991        new_vis[new_vis_index] = origvar;
1992        new_vis_index++;
1993        // System.out.printf("adding origvar %s to ppt %s%n", origvar.name(),
1994        //                   exit_ppt.name());
1995      }
1996      assert new_vis_index == exit_ppt.num_orig_vars;
1997    }
1998    exit_ppt.addVarInfos(new_vis);
1999
2000    if (VarInfo.assertionsEnabled()) {
2001      for (VarInfo vi : exit_ppt.var_infos) {
2002        try {
2003          vi.checkRep();
2004        } catch (Throwable e) {
2005          System.err.println();
2006          System.err.println("Error with variable " + vi + " at ppt " + exit_ppt);
2007          throw e;
2008        }
2009      }
2010    }
2011  }

2583  /**
2584   * Returns the ppt name, max_ppt, that corresponds to the specified percentage of ppts (presuming
2585   * that only those ppts &le; max_ppt will be processed).
2586   */
2587  private static @Nullable String setup_ppt_perc(Collection<File> decl_files, int ppt_perc) {
2588
2589    // Make sure the percentage is valid
2590    if ((ppt_perc < 1) || (ppt_perc > 100)) {
2591      // The number should already have been checked, so don't use UserError.
2592      throw new BugInDaikon("ppt_perc of " + ppt_perc + " is out of range 1..100");
2593    }
2594    if (ppt_perc == 100) {
2595      return null;
2596    }
2597
2598    // Keep track of all of the ppts in a set ordered by the ppt name
2599    Set<String> ppts = new TreeSet<>();
2600
2601    // Read all of the ppt names out of the decl files
2602    try {
2603      for (File file : decl_files) {
2604
2605        // Open the file
2606        LineNumberReader fp = UtilPlume.lineNumberFileReader(file);
2607
2608        // Read each ppt name from the file
2609        for (String line = fp.readLine(); line != null; line = fp.readLine()) {
2610          if (line.equals("") || FileIO.isComment(line)) {
2611            continue;
2612          }
2613          if (!line.equals("DECLARE")) {
2614            continue;
2615          }
2616          // Just read "DECLARE", so next line has ppt name.
2617          String ppt_name = fp.readLine();
2618          if (ppt_name == null) {
2619            throw new Daikon.UserError("File " + file + " terminated prematurely");
2620          }
2621          ppts.add(ppt_name);
2622        }
2623
2624        fp.close();
2625      }
2626    } catch (IOException e) {
2627      e.printStackTrace();
2628      throw new Error(e);
2629    }
2630
2631    // Determine the ppt_name that matches the specified percentage.  Always
2632    // return the last exit point from the method (so we don't get half the
2633    // exits from a method or enters without exits, etc)
2634    int ppt_cnt = (ppts.size() * ppt_perc) / 100;
2635    if (ppt_cnt == 0) {
2636      throw new Daikon.UserError(
2637          "ppt_perc of " + ppt_perc + "% results in processing 0 out of " + ppts.size() + " ppts");
2638    }
2639    for (Iterator<String> i = ppts.iterator(); i.hasNext(); ) {
2640      String ppt_name = i.next();
2641      if (--ppt_cnt <= 0) {
2642        String last_ppt_name = ppt_name;
2643        while (i.hasNext()) {
2644          ppt_name = i.next();
2645          if ((last_ppt_name.indexOf("EXIT") != -1) && (ppt_name.indexOf("EXIT") == -1)) {
2646            return last_ppt_name;
2647          }
2648          last_ppt_name = ppt_name;
2649        }
2650        return ppt_name;
2651      }
2652    }
2653    // Execution should not reach this line
2654    throw new Error("ppt_cnt " + ppt_cnt + " ppts.size " + ppts.size());
2655  }

2657  /**
2658   * Undoes the invariants suppressed for the dynamic constant, suppression and equality set
2659   * optimizations (should yield the same invariants as the simple incremental algorithm.
2660   */
2661  @SuppressWarnings("flowexpr.parse.error") // private field
2662  @RequiresNonNull({"NIS.all_suppressions", "NIS.suppressor_map"})
2663  public static void undoOpts(PptMap all_ppts) {
2664
2665    // undo suppressions
2666    for (PptTopLevel ppt : all_ppts.ppt_all_iterable()) {
2667      NIS.create_suppressed_invs(ppt);
2668    }
2669
2670    // undo equality sets
2671    for (PptTopLevel ppt : all_ppts.ppt_all_iterable()) {
2672      PptSliceEquality sliceEquality = ppt.equality_view;
2673
2674      // some program points have no equality sets?
2675      if (sliceEquality == null) {
2676        // System.out.println(ppt.name);
2677        continue;
2678      }
2679
2680      // get the new leaders
2681      List<Equality> allNewInvs = new ArrayList<>();
2682
2683      for (Invariant eq_as_inv : sliceEquality.invs) {
2684        Equality eq = (Equality) eq_as_inv;
2685        VarInfo leader = eq.leader();
2686        List<VarInfo> vars = new ArrayList<>();
2687
2688        for (VarInfo var : eq.getVars()) {
2689          if (!var.equals(leader)) {
2690            vars.add(var);
2691          }
2692        }
2693
2694        if (vars.size() > 0) {
2695
2696          // Create new equality sets for all of the non-equal vars
2697          List<Equality> newInvs = sliceEquality.createEqualityInvs(vars, eq);
2698
2699          // Create new slices and invariants for each new leader
2700          // copyInvsFromLeader(sliceEquality, leader, vars);
2701          sliceEquality.copyInvsFromLeader(leader, vars);
2702
2703          // Keep track of all of the new invariants created.
2704          // Add all of the new equality sets to our list
2705          allNewInvs.addAll(newInvs);
2706        }
2707      }
2708
2709      sliceEquality.invs.addAll(allNewInvs);
2710    }

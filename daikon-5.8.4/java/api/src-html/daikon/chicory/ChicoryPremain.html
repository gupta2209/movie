059  /**
060   * This method is the entry point of the java agent. Its main purpose is to set up the transformer
061   * so that when classes from the target app are loaded, they are first transformed.
062   *
063   * <p>This method also sets up some other initialization tasks: it connects to Daikon over a port
064   * if necessary, or reads in a purity analysis.
065   */
066  public static void premain(String agentArgs, Instrumentation inst) throws IOException {
067
068    // System.out.format ("In premain, agentargs ='%s', " +
069    //                   "Instrumentation = '%s'%n", agentArgs, inst);
070
071    // Parse our arguments using Chicory's argument parser
072    Options options = new Options(Chicory.synopsis, Chicory.class, ChicoryPremain.class);
073    String[] target_args = options.parse(true, Options.tokenize(agentArgs));
074    if (target_args.length > 0) {
075      System.err.printf("Unexpected ChicoryPremain arguments %s%n", Arrays.toString(target_args));
076      System.exit(1);
077    }
078
079    debug = Chicory.debug;
080    if (debug_runtime) {
081      Runtime.debug = true;
082    }
083
084    // Open the dtrace file
085    if (Chicory.daikon_online) {
086      Runtime.setDtraceOnlineMode(daikon_port);
087    } else if (Chicory.dtrace_file == null) {
088      File trace_file_path = new File(Chicory.output_dir, "dtrace.gz");
089      Runtime.setDtraceMaybe(trace_file_path.toString());
090    } else {
091      File trace_file_path = new File(Chicory.output_dir, Chicory.dtrace_file.getPath());
092      Runtime.setDtraceMaybe(trace_file_path.toString());
093    }
094
095    // Setup argument fields in Runtime
096    Runtime.nesting_depth = Chicory.nesting_depth;
097    Runtime.ppt_omit_pattern = Chicory.ppt_omit_pattern;
098    Runtime.ppt_select_pattern = Chicory.ppt_select_pattern;
099    Runtime.sample_start = Chicory.sample_start;
100    DaikonVariableInfo.std_visibility = Chicory.std_visibility;
101    DaikonVariableInfo.debug_vars.enabled = Chicory.debug_decl_print;
102    if (Chicory.comparability_file != null) {
103      Runtime.comp_info = new DeclReader();
104      try {
105        castNonNull(Runtime.comp_info).read(castNonNull(Chicory.comparability_file));
106      } catch (FileNotFoundException e) {
107        System.err.printf("%nCould not find comparability file: %s%n", Chicory.comparability_file);
108        Runtime.chicoryLoaderInstantiationError = true;
109        System.exit(1);
110      }
111      if (debug) {
112        System.out.printf("Read comparability from %s%n", Chicory.comparability_file);
113        // Runtime.comp_info.dump();
114      }
115    }
116
117    if (Chicory.doPurity()) {
118      System.err.println("Executing a purity analysis is currently disabled");
119      System.exit(1);
120    } else if (Chicory.get_purity_file() != null) {
121      readPurityFile(Chicory.get_purity_file(), Chicory.config_dir);
122      doPurity = true;
123    }
124
125    initializeDeclAndDTraceWriters();
126
127    // Setup the transformer
128    Object transformer;
129    // use a special classloader to ensure correct version of BCEL is used
130    ClassLoader loader = new ChicoryLoader();
131    try {
132      transformer =
133          loader.loadClass("daikon.chicory.Instrument").getDeclaredConstructor().newInstance();
134      // @SuppressWarnings("unchecked")
135      // Class<Instrument> c = (Class<Instrument>) transformer.getClass();
136      // System.out.printf("Classloader of tranformer = %s%n",
137      //                    c.getClassLoader());
138    } catch (Exception e) {
139      throw new RuntimeException("Unexpected error loading Instrument", e);
140    }
141
142    // Instrument transformer = new Instrument();
143    inst.addTransformer((ClassFileTransformer) transformer);
144  }

146  /** Set up the declaration and dtrace writer. */
147  // Runtime.dtrace is @GuardedBy("<self>") because in the Runtime class,
148  // the printing of final lines and then closing of dtrace only happens
149  // when the monitor of dtrace is held in order for the closing of the
150  // trace to happen only once.  See Runtime.noMoreOutput() and
151  // Runtime.addShutdownHook() for more details.  DeclWriter and DTraceWriter
152  // never perform this operation (print final lines and close) on the
153  // value of dtrace passed in, therefore they do not need to make use
154  // of synchronization and their references to dtrace do not need to
155  // be annotated with @GuardedBy("<self>").
156  @SuppressWarnings("lock:argument.type.incompatible")
157  private static void initializeDeclAndDTraceWriters() {
158    // The include/exclude filter are implemented in the transform,
159    // so they don't need to be handled here.
160    // (It looks like these can be called even if Runtime.dtrace is null...)
161    Runtime.decl_writer = new DeclWriter(Runtime.dtrace);
162    Runtime.dtrace_writer = new DTraceWriter(Runtime.dtrace);
163  }

165  /**
166   * Reads purity file. Each line should contain exactly one method. Care must be taken to supply
167   * the correct format.
168   *
169   * <p>From the Sun JDK API:
170   *
171   * <p>"The string is formatted as the method access modifiers, if any, followed by the method
172   * return type, followed by a space, followed by the class declaring the method, followed by a
173   * period, followed by the method name, followed by a parenthesized, comma-separated list of the
174   * method's formal parameter types. If the method throws checked exceptions, the parameter list is
175   * followed by a space, followed by the word throws followed by a comma-separated list of the
176   * thrown exception types. For example:
177   *
178   * <p>public boolean java.lang.Object.equals(java.lang.Object)
179   *
180   * <p>The access modifiers are placed in canonical order as specified by "The Java Language
181   * Specification". This is public, protected or private first, and then other modifiers in the
182   * following order: abstract, static, final, synchronized native.
183   */
184  private static void readPurityFile(File purityFileName, @Nullable File pathLoc) {
185    pureMethods = new HashSet<String>();
186    File purityFile = new File(pathLoc, purityFileName.getPath());
187
188    BufferedReader reader;
189    try {
190      reader = UtilPlume.bufferedFileReader(purityFile);
191    } catch (FileNotFoundException e) {
192      System.err.printf(
193          "%nCould not find purity file %s = %s%n", purityFileName, purityFile.getAbsolutePath());
194      Runtime.chicoryLoaderInstantiationError = true;
195      System.exit(1);
196      throw new Error("Unreachable control flow");
197    } catch (IOException e) {
198      throw new Error(
199          "Problem reading purity file " + purityFileName + " = " + purityFile.getAbsolutePath(),
200          e);
201    }
202
203    if (Chicory.verbose) {
204      System.out.printf("Reading '%s' for pure methods %n", purityFileName);
205    }
206
207    String line = null;
208    do {
209      try {
210        line = reader.readLine();
211      } catch (IOException e) {
212        throw new Error(
213            "Error reading file " + purityFileName + " = " + purityFile.getAbsolutePath(), e);
214      }
215
216      if (line != null) {
217        pureMethods.add(line.trim());
218        // System.out.printf("Adding '%s' to list of pure methods%n",
219        //                   line);
220      }
221    } while (line != null);
222
223    try {
224      reader.close();
225    } catch (IOException e) {
226      System.err.println("Error while closing " + purityFileName + " after reading.");
227      System.exit(1);
228    }
229
230    // System.out.printf("leaving purify file%n");
231
232  }

234  /** Return true iff Chicory has run a purity analysis or read a {@code *.pure} file. */
235  @SuppressWarnings("nullness") // dependent:  pureMethods is non-null if doPurity is true
236  // @EnsuresNonNullIf(result=true, expression="ChicoryPremain.pureMethods")
237  @EnsuresNonNullIf(result = true, expression = "pureMethods")
238  public static boolean shouldDoPurity() {
239    return doPurity;
240  }

242  /**
243   * Checks if member is one of the pure methods found in a purity analysis or supplied from a
244   * {@code *.pure} file.
245   *
246   * @return true iff member is a pure method
247   */
248  // @RequiresNonNull("ChicoryPremain.pureMethods")
249  @RequiresNonNull("pureMethods")
250  public static boolean isMethodPure(Member member) {
251    assert shouldDoPurity() : "Can't query for purity if no purity analysis was executed";
252
253    // TODO just use Set.contains(member.toString()) ?
254    for (String methName : pureMethods) {
255      if (methName.equals(member.toString())) {
256        return true;
257      }
258    }
259
260    return false;
261  }

270  /**
271   * Classloader for the BCEL code. Using this classloader guarantees that we get the correct
272   * version of BCEL and not a possible incompatible version from elsewhere on the user's classpath.
273   * We also load daikon.chicory.Instrument via this (since that class is the user of all of the
274   * BCEL classes). All references to BCEL must be within that class (so that all references to BCEL
275   * will get resolved by this classloader).
276   *
277   * <p>There are several versions of BCEL that have been released:
278   *
279   * <ul>
280   *   <li>the original 5.2 version
281   *   <li>an interim 6.0 version
282   *   <li>the offical 6.0 release version
283   *   <li>the offical 6.1 release version
284   *   <li>the PLSE 6.1 release version (includes LocalVariableGen fix)
285   *   <li>the offical 6.2 release version (includes LocalVariableGen fix)
286   *   <li>the offical 6.3 release version
287   *   <li>the offical 6.3.1 release version
288   *   <li>the offical 6.4.1 release version
289   *   <li>the PLSE 6.4.1.1 release version (includes JDK 11 support)
290   * </ul>
291   *
292   * <p>Note that both Chicory and DynComp use the ChicoryLoader to load BCEL and to verify that the
293   * version loaded is acceptable. However, the official 6.1 release version is sufficient for
294   * Chicory while DynComp requires the latest PLSE 6.4.1.1 version. Hence, this loader only checks
295   * for the official 6.1 release version (or newer). After loading BCEL, DynComp will make an
296   * additional check to verify that the 6.4.1.1 version has been loaded.
297   *
298   * <p>There are two classes present in 6.1 and subsequent releases that are not in previous
299   * versions. Thus, we can identify version 6.1 (and later) of BCEL by the presence of the class:
300   * org.apache.bcel.classfile.ConstantModule.class.
301   *
302   * <p>Earlier versions of Chicory inspected all version of BCEL found on the path and selected the
303   * correct one, if present. We now (9/15/16) simplify this to say the first BCEL found must be the
304   * correct one. This allows us to use the normal loader for all of the classes.
305   */
306  public static class ChicoryLoader extends ClassLoader {
307
308    public static final SimpleLog debug = new SimpleLog(Chicory.verbose);
309
310    @SuppressWarnings("StaticAssignmentInConstructor") // sets static variable only if aborting
311    public ChicoryLoader() throws IOException {
312
313      String bcel_classname = "org.apache.bcel.Constants";
314      String plse_marker_classname = "org.apache.bcel.classfile.ConstantModule";
315
316      List<URL> bcel_urls = get_resource_list(bcel_classname);
317      List<URL> plse_urls = get_resource_list(plse_marker_classname);
318
319      if (plse_urls.size() == 0) {
320        System.err.printf(
321            "%nBCEL 6.1 or newer must be on the classpath.  Normally it is found in daikon.jar .%n");
322        Runtime.chicoryLoaderInstantiationError = true;
323        System.exit(1);
324      }
325      if (bcel_urls.size() < plse_urls.size()) {
326        System.err.printf("%nCorrupted BCEL library, bcel %s, plse %s%n", bcel_urls, plse_urls);
327        Runtime.chicoryLoaderInstantiationError = true;
328        System.exit(1);
329      }
330
331      // No need to do anything if only our versions of bcel are present
332      if (bcel_urls.size() == plse_urls.size()) {
333        return;
334      }
335
336      URL bcel = bcel_urls.get(0);
337      URL plse = plse_urls.get(0);
338      if (!plse.getProtocol().equals("jar")) {
339        System.err.printf("%nDaikon BCEL must be in jar file.  Found at %s%n", plse);
340        Runtime.chicoryLoaderInstantiationError = true;
341        System.exit(1);
342      }
343      if (!same_location(bcel, plse)) {
344        System.err.printf(
345            "%nDaikon BCEL (%s) is not first BCEL on the classpath (%s).%n", plse, bcel);
346        Runtime.chicoryLoaderInstantiationError = true;
347        System.exit(1);
348      } else {
349        JarFile bcel_jar = new JarFile(extract_jar_path(plse));
350        debug.log("Daikon BCEL found in jar %s%n", bcel_jar.getName());
351      }
352    }
353
354    /**
355     * Returns whether or not the two URL represent the same location for org.apache.bcel. Two
356     * locations match if they refer to the same jar file or the same directory in the filesystem.
357     */
358    private static boolean same_location(URL url1, URL url2) {
359      if (!url1.getProtocol().equals(url2.getProtocol())) {
360        return false;
361      }
362
363      if (url1.getProtocol().equals("jar")) {
364        // System.out.printf("url1 = %s, file=%s, path=%s, protocol=%s, %s%n",
365        //                  url1, url1.getFile(), url1.getPath(),
366        //                  url1.getProtocol(), url1.getClass());
367        // System.out.printf("url2 = %s, file=%s, path=%s, protocol=%s, %s%n",
368        //                    url2, url2.getFile(), url2.getPath(),
369        //                    url2.getProtocol(), url1.getClass());
370        String jar1 = extract_jar_path(url1);
371        String jar2 = extract_jar_path(url2);
372        return (jar1.equals(jar2));
373      } else if (url1.getProtocol().equals("file")) {
374        String loc1 = url1.getFile().replaceFirst("org\\.apache\\.bcel\\..*$", "");
375        String loc2 = url2.getFile().replaceFirst("org\\.apache\\.bcel\\..*$", "");
376        return (loc1.equals(loc2));
377      } else {
378        throw new Error("unexpected protocol " + url1.getProtocol());
379      }
380    }
381
382    /**
383     * Returns the pathname of a jar file specified in the URL. The protocol must be 'jar'. Only
384     * file jars are supported.
385     */
386    private static String extract_jar_path(URL url) {
387      assert url.getProtocol().equals("jar") : url.toString();
388
389      // Remove the preceeding 'file:' and trailing '!filename'
390      String path = url.getFile();
391      path = path.replaceFirst("^[^:]*:", "");
392      path = path.replaceFirst("![^!]*$", "");
393
394      return path;
395    }
396
397    /**
398     * Get all of the URLs that match the specified name in the classpath. The name should be in
399     * normal classname format (eg, org.apache.bcel.Const). An empty list is returned if no names
400     * match.
401     */
402    @SuppressWarnings("JdkObsolete") // ClassLoader.getSystemResources returns an Enumeration
403    static List<URL> get_resource_list(String classname) throws IOException {
404
405      String name = classname_to_resource_name(classname);
406      Enumeration<URL> enum_urls = ClassLoader.getSystemResources(name);
407      List<URL> urls = new ArrayList<>();
408      while (enum_urls.hasMoreElements()) {
409        urls.add(enum_urls.nextElement());
410      }
411      return urls;
412    }
413
414    /**
415     * Changes a class name in the normal format (eg, org.apache.bcel.Const) to that used to lookup
416     * resources (eg, org/apache/bcel/Const.class).
417     */
418    private static String classname_to_resource_name(String name) {
419      return (name.replace(".", "/") + ".class");
420    }

423    protected Class<?> loadClass(@BinaryName String name, boolean resolve)
424        throws java.lang.ClassNotFoundException {
425
426      return super.loadClass(name, resolve);
427    }

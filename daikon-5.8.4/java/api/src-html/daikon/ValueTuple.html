017/**
018 * This data structure holds a tuple of values for a particular program point. VarInfo objects can
019 * use this to get the values of the variables they represent.
020 *
021 * <p>It has two fields: vals and mods. While the arrays and their elements are interned, the
022 * ValueTuple objects themselves are not interned.
023 */
024public final class ValueTuple implements Cloneable {
025

029  // These arrays are interned, and so are their elements.
030  // Each element is null only if it is missing (according to the mods array).
031  public @Nullable @Interned Object @Interned [] vals;
032
033  // Could consider putting the mods array in the first slot of "vals", to
034  // avoid the Object overhead of a pair of val and mods.
035
036  /**
037   * Modification bit per value, possibly packed into fewer ints than the vals field. Don't use a
038   * single int because that won't scale to (say) more than 32 values.
039   */
040  public int @Interned [] mods;
041
042  // Right now there are only three meaningful values for a mod:
043  /** Not modified. */
044  public static final int UNMODIFIED = 0;
045  /** Modified. */
046  public static final int MODIFIED = 1;
047  /**
048   * Missing value because the expression doesn't make sense: x.a when x is null. Data trace files
049   * can contain this modbit.
050   */
051  public static final int MISSING_NONSENSICAL = 2;
052  /**
053   * Missing value because of data flow: this.x.x isn't available from a ppt. Data trace files must
054   * not contain this modbit.
055   */
056  public static final int MISSING_FLOW = 3;
057  /** Maximum mod bit value. Always set to 1+ last modbit value. */
058  public static final int MODBIT_VALUES = 4;
059  // Out of the range of MODBIT_VALUES because this won't appear in the
060  // tables; it gets converted to UNMODIFIED or MODIFIED, depending on
061  // whether this is the first sample.  (Not sure whether that is the right
062  // strategy in the long term; it does let me avoid changing code in the
063  // short term.)
064  public static final int STATIC_CONSTANT = 22;

158  // A tuplemod is summary modification information about the whole tuple
159  // rather than about specific elements of the tuple.
160  // There are two potentially useful abstractions for mod bits over an
161  // entire tuple in aggregate:
162  //  * return missing if any is missing (good for slices;
163  //    indicates that we can't use that value)
164  //  * return missing is all are missing (good for non-slices;
165  //    the number that are guaranteed to be missing in slices
166  // The same thing can be argued about "unmodified", actually.
167  // So there are 8 states corresponding to 3 booleans:
168  // modified, unmodified, missing
169  //  * has modified, has unmodified, has missing
170  //  * has modified, has unmodified, no missing
171  //  * has modified, no unmodified, has missing
172  //  * has modified, no unmodified, no missing
173  //    ie, all modified
174  //  * no modified, has unmodified, has missing
175  //  * no modified, has unmodified, no missing
176  //    ie, all unmodified
177  //  * no modified, no unmodified, has missing
178  //    ie, all missing; probably impossible
179  //  * no modified, no unmodified, no missing
180  //    impossible
181
182  public static final int TUPLEMOD_VALUES = MathPlume.pow(2, MODBIT_VALUES);
183  public static final int UNMODIFIED_BITVAL = MathPlume.pow(2, UNMODIFIED);
184  public static final int MODIFIED_BITVAL = MathPlume.pow(2, MODIFIED);
185  public static final int MISSING_NONSENSICAL_BITVAL = MathPlume.pow(2, MISSING_NONSENSICAL);
186  public static final int MISSING_FLOW_BITVAL = MathPlume.pow(2, MISSING_FLOW);
187  // Various slices of the 8 (=TUPLEMOD_VALUES) possible tuplemod values.
188  // The arrays are filled up in a static block below.
189  // (As of 1/9/2000, tuplemod_modified_not_missing is used only in
190  // num_mod_samples(), and tuplemod_not_missing is not used.)
191  public static final int[] tuplemod_not_missing = new int[TUPLEMOD_VALUES / 2];
192  public static final int[] tuplemod_modified_not_missing = new int[TUPLEMOD_VALUES / 4];
193
194  static {
195    int i1 = 0, i2 = 0;
196    for (int tm = 0; tm < TUPLEMOD_VALUES; tm++) {
197      if (!tuplemodHasMissingFlow(tm) && !tuplemodHasMissingNonsensical(tm)) {
198        tuplemod_not_missing[i1] = tm;
199        i1++;
200      }
201      if (tuplemodHasModified(tm)
202          && !tuplemodHasMissingFlow(tm)
203          && !tuplemodHasMissingNonsensical(tm)) {
204        tuplemod_modified_not_missing[i2] = tm;
205        i2++;
206      }
207    }
208  }
209
210  static int make_tuplemod(
211      boolean unmodified, boolean modified, boolean missingNonsensical, boolean missingFlow) {
212    int result = 0;
213    if (unmodified) result += UNMODIFIED_BITVAL;
214    if (modified) result += MODIFIED_BITVAL;
215    if (missingNonsensical) result += MISSING_NONSENSICAL_BITVAL;
216    if (missingFlow) result += MISSING_FLOW_BITVAL;
217    return result;
218  }
219
220  static boolean tuplemodHasModified(int tuplemod) {
221    return ((tuplemod & MODIFIED_BITVAL) != 0);
222  }
223
224  static boolean tuplemodHasUnmodified(int tuplemod) {
225    return ((tuplemod & UNMODIFIED_BITVAL) != 0);
226  }
227
228  static boolean tuplemodHasMissingNonsensical(int tuplemod) {
229    return ((tuplemod & MISSING_NONSENSICAL_BITVAL) != 0);
230  }
231
232  static boolean tuplemodHasMissingFlow(int tuplemod) {
233    return ((tuplemod & MISSING_FLOW_BITVAL) != 0);
234  }

236  /**
237   * In output, M=modified, U=unmodified, X=missing. Capital letters indicate the specified modbit
238   * does occur, lowercase letters indicate it does not occur.
239   */
240  static String tuplemodToStringBrief(int tuplemod) {
241    return ((tuplemodHasModified(tuplemod) ? "M" : "m")
242        + (tuplemodHasUnmodified(tuplemod) ? "U" : "u")
243        + (tuplemodHasMissingNonsensical(tuplemod) ? "X" : "x")
244        + (tuplemodHasMissingFlow(tuplemod) ? "F" : "f"));
245  }

247  static int tupleMod(int[] mods) {
248    boolean[] has_modbit_val = new boolean[MODBIT_VALUES];
249    // Extraneous, as the array is initialized to all zeroes.
250    Arrays.fill(has_modbit_val, false);
251    for (int i = 0; i < mods.length; i++) {
252      has_modbit_val[mods[i]] = true;
253    }
254    return make_tuplemod(
255        has_modbit_val[UNMODIFIED],
256        has_modbit_val[MODIFIED],
257        has_modbit_val[MISSING_NONSENSICAL],
258        has_modbit_val[MISSING_FLOW]);
259  }
260
261  int tupleMod() {
262    return ValueTuple.tupleMod(mods);
263  }

319  public void checkRep(@UnknownInitialization(ValueTuple.class) ValueTuple this) {
320    assert vals.length == mods.length;
321    for (int i = 0; i < vals.length; i++) {
322      assert 0 <= mods[i] && mods[i] < MODBIT_VALUES
323          : String.format("mods: %s i:%d mods[i]: %s%n", Arrays.toString(mods), i, mods[i]);
324      assert (isMissing(i) ? vals[i] == null : true);
325    }
326  }

328  /** Default constructor that interns its argument. */
329  public ValueTuple(@Nullable @Interned Object[] vals, int[] mods) {
330    this.vals = Intern.intern(vals);
331    this.mods = Intern.intern(mods);
332    checkRep();
333  }

335  /**
336   * Private constructor that doesn't perform interning.
337   *
338   * @param vals the values
339   * @param mods the modbits
340   * @param check if true, require vals and mods to be interned
341   */
342  @SuppressWarnings("interning") // interning constructor
343  private ValueTuple(@Nullable Object[] vals, int[] mods, boolean check) {
344    assert !check || Intern.isInterned(vals);
345    assert !check || Intern.isInterned(mods);
346    this.vals = vals;
347    this.mods = mods;
348    checkRep();
349  }

384  @EnsuresNonNullIf(result = true, expression = "#1")
385  @Pure
386  @Override
387  public boolean equals(@GuardSatisfied ValueTuple this, @GuardSatisfied @Nullable Object obj) {
388    if (!(obj instanceof ValueTuple)) {
389      return false;
390    }
391    ValueTuple other = (ValueTuple) obj;
392    return (vals == other.vals) && (mods == other.mods);
393  }

423  /**
424   * Return the values of this tuple ("missing" is used for each missing value). If vis is non-null,
425   * the values are annotated with the VarInfo name that would be associated with the value.
426   */
427  @SideEffectFree
428  public String toString(@GuardSatisfied ValueTuple this, VarInfo @Nullable [] vis) {
429    StringBuilder sb = new StringBuilder("[");
430    assert vals.length == mods.length;
431    assert vis == null || vals.length == vis.length;
432    for (int i = 0; i < vals.length; i++) {
433      if (i > 0) sb.append("; ");
434      if (vis != null) {
435        sb.append(vis[i].name() + "=");
436      }
437      Object val = vals[i];
438      int mod = mods[i];
439      switch (mod) {
440        case UNMODIFIED:
441        case MODIFIED:
442          if (val instanceof String) {
443            sb.append("\"" + val + "\"");
444          } else if (val instanceof long[]) {
445            sb.append(Arrays.toString((long[]) val));
446          } else if (val instanceof int[]) {
447            // shouldn't reach this case -- should be long[], not int[]
448            // sb.append(Arrays.toString((int[])val));
449            throw new Error("should be long[], not int[]");
450          } else if (val instanceof double[]) {
451            sb.append(Arrays.toString((double[]) val));
452          } else if (val instanceof String[]) {
453            sb.append(Arrays.toString((String[]) val));
454          } else {
455            sb.append(val);
456          }
457          if (mod == UNMODIFIED) {
458            sb.append("(U)");
459          }
460          break;
461        case MISSING_NONSENSICAL:
462          sb.append("(missing)");
463          break;
464        case MISSING_FLOW:
465          sb.append("(missing-flow)");
466          break;
467        default:
468          throw new Error("bad modbit " + mod);
469      }
470    }
471    sb.append("]");
472    return sb.toString();
473  }

475  public static String valsToString(@Nullable Object[] vals) {
476    StringBuilder sb = new StringBuilder("[");
477    for (int i = 0; i < vals.length; i++) {
478      if (i > 0) sb.append(", ");
479      sb.append(valToString(vals[i]));
480    }
481    sb.append("]");
482    return sb.toString();
483  }
484
485  public static String valToString(@Nullable Object val) {
486    if (val == null) {
487      return "null";
488    }
489    if (val instanceof long[]) {
490      return (Arrays.toString((long[]) val));
491    } else if (val instanceof int[]) {
492      // shouldn't reach this case -- should be long[], not int[]
493      return (Arrays.toString((int[]) val));
494    } else {
495      return (val.toString());
496    }
497  }

499  /**
500   * Return a new ValueTuple consisting of the elements of this one with indices listed in indices.
501   */
502  public ValueTuple slice(int[] indices) {
503    int new_len = indices.length;
504    @Nullable @Interned Object[] new_vals = new @Nullable @Interned Object[new_len];
505    int[] new_mods = new int[new_len];
506    for (int i = 0; i < new_len; i++) {
507      new_vals[i] = vals[indices[i]];
508      new_mods[i] = mods[indices[i]];
509    }
510    return new ValueTuple(new_vals, new_mods);
511  }

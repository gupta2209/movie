031/**
032 * Each DaikonVariableInfo object is a node in the tree structure of the variables in the target
033 * application. In general, the variable a will be the parent of the variables a.b and a.c in the
034 * tree, where b and c are fields in a's class. There is such a tree structure associated with every
035 * program point.
036 *
037 * <p>Each node can have any non-negative number of child nodes. DaikonVariableInfo is an abstract
038 * class. Its subtypes are designed to represent specific types of variables, such as arguments,
039 * arrays, etc.
040 *
041 * <p>The tree structure is built in the DeclWriter and traversed in the DTraceWriter.
042 *
043 * <p>This architecture makes it possible to avoid the issue of "traversal pattern duplication" in
044 * which both the DeclWriter and DTraceWriter must traverse the target application's variables
045 * identically.
046 */
047public abstract class DaikonVariableInfo
048    implements Iterable<DaikonVariableInfo>, Comparable<DaikonVariableInfo> {

142  /**
143   * Constructs a DaikonVariableInfo object.
144   *
145   * @param theName the variable's name
146   * @param arr true iff the variable is an array
147   */
148  protected DaikonVariableInfo(String theName, String typeName, String repTypeName, boolean arr) {
149    // Intern the names because there will be many of the
150    // same variable names at different program points within
151    // the same class.
152    name = theName.intern();
153    this.typeName = typeName.intern();
154    this.repTypeName = repTypeName.intern();
155
156    debug_vars.log(
157        "Construct DaikonVariableInfo: %s : %s : %s", this.getClass().getName(), name, typeName);
158
159    children = new ArrayList<DaikonVariableInfo>();
160    isArray = arr;
161
162    if ((theName != null) && (theName.contains("[..]") || theName.contains("[]")) && !isArray) {
163      debug_array.log("%s is not an array", theName);
164      debug_array.logStackTrace();
165    }
166  }

168  /** Returns the name of this variable. */
169  public @Nullable String getName(@GuardSatisfied DaikonVariableInfo this) {
170    if (name == null) {
171      return null;
172    }
173
174    if (Chicory.new_decl_format) {
175      return name.replaceFirst("\\[\\]", "[..]");
176    } else {
177      return name;
178    }
179  }

181  /**
182   * Add a child to this node. Should only be called while the tree is being constructed.
183   *
184   * @param info the child object, must be non-null. The child's fields name, typeName, repTypeName,
185   *     and compareInfoString should also be non-null.
186   */
187  protected void addChild(DaikonVariableInfo info) {
188    assert info != null : "info cannot be null in DaikonVariableInfo.addChild()";
189    assert info.name != null : "Child's name should not be null";
190    assert info.typeName != null : "Child's type name should not be null";
191    assert info.repTypeName != null : "Child's representation type name should not be null";
192    assert info.compareInfoString != null : "Child's comparability information should not be null";
193
194    debug_vars.log("Adding %s to %s", info, this);
195    children.add(info);
196  }

205  /**
206   * Returns a string representation of this node and its descandants.

208   * @return a string representation of this node and its descandants
209   */
210  public String treeString() {
211    return getStringBuilder(new StringBuilder("--")).toString();

214  /**
215   * Return a StringBuilder that contains the name of this node and all ancestors of this node.
216   * Longer indentations correspond to deeper levels in the tree.
217   *
218   * @param offset the offset to begin each line with
219   * @return StringBuilder that contains all children of this node
220   */
221  private StringBuilder getStringBuilder(StringBuilder offset) {
222    StringBuilder theBuf = new StringBuilder();
223
224    theBuf.append(
225        offset + name + " [" + System.identityHashCode(this) + "]" + DaikonWriter.lineSep);
226
227    StringBuilder childOffset = new StringBuilder(offset);
228    childOffset.append("--");
229    for (DaikonVariableInfo info : children) {
230      theBuf.append(info.getStringBuilder(childOffset));
231    }
232
233    return theBuf;
234  }

240   * @return an iterator over all the node's children

242  @Override
243  public Iterator<DaikonVariableInfo> iterator() {
244    return Collections.unmodifiableList(children).iterator();

247  /** Returns the complete tree of variables as a list. */
248  public List<DaikonVariableInfo> tree_as_list() {
249    List<DaikonVariableInfo> list = new ArrayList<>();
250    list.add(this);
251    for (DaikonVariableInfo dv : children) {
252      list.addAll(dv.tree_as_list());
253    }
254    return list;

286  /** Gets the value of an object and concatenates the associated "modified" integer. */
287  protected String getValueStringOfObjectWithMod(Object theValue, boolean hashArray) {
288    String retString = getValueStringOfObject(theValue, hashArray) + DaikonWriter.lineSep;
289
290    if (theValue instanceof NonsensicalObject) {
291      retString += "2";
292    } else {
293      retString += "1";

299  /**
300   * Gets the value, but with no endline. If hashArray is true, it prints the "hash code" of the
301   * array and not its separate values.
302   */
303  private String getValueStringOfObject(Object theValue, boolean hashArray) {
304    if (theValue == null) {
305      return "null";
306    }
307
308    Class<?> type = theValue.getClass();
309
310    assert !type.isPrimitive() : "Objects cannot be primitive";
311
312    if (theValue instanceof Runtime.PrimitiveWrapper) {
313      return getPrimitiveValueString(theValue);
314    } else if (!hashArray && type.isArray()) {
315      // show the full array
316      return getValueStringOfArray(theValue);
317    } else if (theValue instanceof NonsensicalObject) {
318      return "nonsensical";
319    } else {
320      // basically, show the hashcode of theValue
321      return getObjectHashCode(theValue);
322    }
323  }
324
325  /** Get value string for a primitive (wrapped) object. */
326  private String getPrimitiveValueString(Object obj) {
327    assert (obj instanceof Runtime.PrimitiveWrapper)
328        : "Objects passed to showPrimitive must implement PrimitiveWrapper"
329            + DaikonWriter.lineSep
330            + "This object is type: "
331            + obj.getClass().getName();
332
333    // use wrapper classes toString methods to print value
334    return (obj.toString());
335  }

337  /** Gets a string representation of the values in an array. */
338  private String getValueStringOfArray(Object array) {
339    List<Object> theList = DTraceWriter.getListFromArray(array);
340    return getValueStringOfList(theList);

343  /** Gets the Object's unique ID as a string. In other words, a "hash code". */
344  private String getObjectHashCode(Object theObject) {
345    if (theObject == null) {
346      return "null";
347    } else if (theObject instanceof NonsensicalObject) {
348      return "nonsensical";
349    } else {
350      return Integer.toString(System.identityHashCode(theObject));
351    }
352  }

355   * Gets the list of values (as a string) from getValueStringOfList and concatenates the "modified"
356   * value.
357   */
358  private String getValueStringOfListWithMod(List<Object> theValues) {
359    String retString = getValueStringOfList(theValues) + DaikonWriter.lineSep;
360
361    if (theValues instanceof NonsensicalList) {
362      retString += "2";
363    } else {
364      retString += "1";

370  /**
371   * Returns a string representation of the values of a list of values as if it were an array.
372   *
373   * @param theValues the values to print out
374   */
375  protected String getValueStringOfList(List<Object> theValues) {
376    if (theValues == null) {
377      return "null";
378    }
379
380    if (theValues instanceof NonsensicalList) {
381      return "nonsensical";
382    }
383
384    StringBuilder buf = new StringBuilder();
385
386    buf.append("[");
387    for (Iterator<Object> iter = theValues.iterator(); iter.hasNext(); ) {
388      Object elementVal = iter.next();
389
390      // hash arrays...
391      // don't want to print arrays within arrays
392      buf.append(getValueStringOfObject(elementVal, true));
393
394      // put space between elements in array
395      if (iter.hasNext()) buf.append(" ");
396    }
397    buf.append("]");
398
399    return buf.toString();
400  }

406  /**
407   * Add the parameters of the given method to this node.
408   *
409   * @param cinfo the method's class
410   * @param method the method
411   * @param argnames the method's arguments
412   * @param depth the remaining depth to print variables to
413   */
414  protected void addParameters(ClassInfo cinfo, Member method, List<String> argnames, int depth) {
415    debug_vars.log("enter addParameters%n");
416
417    Class<?>[] parameterTypes =
418        (method instanceof Constructor<?>)
419            ? ((Constructor<?>) method).getParameterTypes()
420            : ((Method) method).getParameterTypes();
421    assert argnames.size() == parameterTypes.length;
422
423    int param_offset = 0;
424    for (int i = 0; i < parameterTypes.length; i++) {
425      Class<?> type = parameterTypes[i];
426      String name = argnames.get(i);
427      if (type.getName().equals("daikon.dcomp.DCompMarker")
428          || type.getName().equals("java.lang.DCompMarker")) {
429        continue;
430      }
431      debug_vars.log("processing parameter '%s'%n", name);
432      debug_vars.indent();
433      DaikonVariableInfo theChild =
434          addParamDeclVar(cinfo, type, name, /*offset=*/ "", depth, i, param_offset);
435      param_offset++;
436      if ((type == Double.TYPE) || (type == Long.TYPE)) {
437        param_offset++;
438      }
439      assert cinfo.clazz != null : "@AssumeAssertion(nullness): need to check justification";
440      theChild.addChildNodes(cinfo, type, name, /*offset=*/ "", depth);
441      debug_vars.exdent();
442    }
443    debug_vars.log("exit addParameters%n");
444  }

446  /**
447   * Adds class variables (i.e., the fields) for the given type and attach new nodes as children of
448   * this node.
449   *
450   * @param type the class whose fields should all be added to this node
451   * @param offset the prefix for variables -- that is, the expression whose fields are being
452   *     printed
453   * @param depth the remaining depth to print variables to
454   */
455  @RequiresNonNull("#1.clazz")
456  protected void addClassVars(
457      ClassInfo cinfo, boolean dontPrintInstanceVars, Class<?> type, String offset, int depth) {
458
459    debug_vars.log("addClassVars: %s : %s : %s: [%s]%n", this, cinfo, type, offset);
460
461    boolean topLevelCall = offset.equals(""); // true if at first level of recursion
462
463    DaikonVariableInfo thisInfo; // DaikonVariableInfo corresponding to the "this" object
464    if (!dontPrintInstanceVars && topLevelCall) {
465      // "this" variable; must must be at the first level of recursion (not lower) to print it
466      thisInfo = new ThisObjInfo(type);
467      addChild(thisInfo);
468
469      // .class variable
470      if (shouldAddRuntimeClass(type)) {
471        DaikonVariableInfo thisClass =
472            new DaikonClassInfo(
473                "this" + class_suffix, classClassName, stringClassName, "this", false);
474        thisInfo.addChild(thisClass);
475      }
476    } else if (topLevelCall) {
477      // Create a non-printing root for static variables.
478      thisInfo = new StaticObjInfo(type);
479      addChild(thisInfo);
480    } else {
481      thisInfo = this;
482    }
483
484    // Get the fields
485    // System.out.printf("getting fields for %s%n", type);
486
487    // We need to get fields of superclass(es) as well.
488    List<Field> fields = new ArrayList<>();
489    Class<?> c = type;
490    while (c != null && c != Object.class) {
491      fields.addAll(Arrays.asList(c.getDeclaredFields()));
492      c = c.getSuperclass();
493    }
494
495    // if (fields.length > 50)
496    //    System.out.printf("%d fields in %s%n", fields.length, type);
497
498    debug_vars.log(
499        "%s: [%s] %d dontPrintInstanceVars = %b, inArray = %b%n",
500        type, offset, fields.size(), dontPrintInstanceVars, isArray);
501
502    for (Field classField : fields) {
503      boolean is_static = Modifier.isStatic(classField.getModifiers());
504
505      debug_vars.log("considering field %s -> %s%n", offset, classField);
506
507      // Skip some fields
508
509      // In the future, perhaps skip some synthetic fields.  So far, the
510      // only ones we have seen are 'this$0' (a field in an inner class
511      // that contains a pointer to the instance of the outer class),
512      // 'this$1' for an inner inner class, and so on.  These variables
513      // expose the outer class fields of an inner class to Daikon.
514
515      if (!is_static && dontPrintInstanceVars) {
516        debug_vars.log("--field (!static && dontPrintInstanceVars) %s%n", classField);
517        continue;
518      }
519
520      // Skip variables that match the ignore pattern
521      if (Chicory.omit_var != null) {
522        String fullname = offset + "." + classField.getName();
523        if (is_static) {
524          fullname = classField.getDeclaringClass().getName() + "." + classField.getName();
525        }
526        Matcher m = Chicory.omit_var.matcher(fullname);
527        if (m.find()) {
528          // System.out.printf("VAR %s matches omit pattern %s%n", fullname, Chicory.omit_var);
529          continue;
530        }
531      }
532
533      // Don't print arrays of the same static field
534      if (is_static && isArray) {
535        debug_vars.log("--field static and inArray%n");
536        continue;
537      }
538
539      // Skip any statics that have been already included
540      if (is_static) {
541        String full_name = classField.getDeclaringClass().getName() + "." + classField.getName();
542        if (ppt_statics.contains(full_name) && (depth <= 0)) {
543          debug_vars.log("already included static %s (no children)", full_name);
544
545          continue;
546        }
547      }
548
549      if (!isFieldVisible(cinfo.clazz, classField)) {
550        debug_vars.log("--field not visible%n");
551        continue;
552      }
553
554      // ... end of code to skip some fields.
555
556      Class<?> fieldType = classField.getType();
557
558      StringBuilder buf = new StringBuilder();
559      DaikonVariableInfo newChild = thisInfo.addDeclVar(classField, offset, buf);
560
561      debug_vars.log("--Created DaikonVariable %s%n", newChild);
562      debug_vars.indent();
563
564      String newOffset = buf.toString();
565      newChild.addChildNodes(cinfo, fieldType, classField.getName(), newOffset, depth);
566      debug_vars.exdent();
567    }
568
569    // If appropriate, print out decls information for pure methods
570    // and add to the tree.
571    // Check dontPrintInstanceVars is basically checking if the program point method
572    // (not the pure method) is static.  If it is, don't continue because we can't
573    // call instance methods (all pure methods we consider are instance methods)
574    // from static methods.
575    if (ChicoryPremain.shouldDoPurity() && !dontPrintInstanceVars) {
576      ClassInfo typeInfo;
577
578      try {
579        typeInfo = Runtime.getClassInfoFromClass(type);
580      } catch (RuntimeException e) {
581        // Could not find the class... no further purity analysis
582        typeInfo = null;
583      }
584
585      if (typeInfo != null) {
586        // Pure methods with no parameters
587        for (MethodInfo meth : typeInfo.method_infos) {
588          if (meth.isPure() && meth.arg_names.length == 0) {
589            StringBuilder buf = new StringBuilder();
590            DaikonVariableInfo newChild =
591                thisInfo.addPureMethodDecl(
592                    cinfo, meth, new DaikonVariableInfo[] {}, offset, depth, buf);
593            String newOffset = buf.toString();
594            debug_vars.log("Pure method");
595            debug_vars.indent();
596            assert meth.member != null
597                : "@AssumeAssertion(nullness): member of method_infos have .member field"; // dependent type
598            newChild.addChildNodes(
599                cinfo,
600                ((Method) meth.member).getReturnType(),
601                meth.member.getName(),
602                newOffset,
603                depth);
604            debug_vars.exdent();
605          }
606        }
607
608        // List containing all class variables, excluding pure methods with parameters
609        List<DaikonVariableInfo> siblings = new ArrayList<>(thisInfo.children);
610
611        // Pure methods with one parameter
612        for (MethodInfo meth : typeInfo.method_infos) {
613          if (meth.isPure() && meth.arg_names.length == 1) {
614            for (DaikonVariableInfo sib : siblings) {
615              String sibType = sib.getTypeNameOnly();
616              Class<?> sibClass;
617
618              // Get class type of the class variable
619              try {
620                sibClass =
621                    ReflectionPlume.classForName(Signatures.binaryNameToClassGetName(sibType));
622              } catch (ClassNotFoundException e) {
623                throw new Error(e);
624              }
625
626              // Add node if the class variable can be used as the pure method's parameter
627              if (ReflectionPlume.isSubtype(sibClass, meth.arg_types[0])) {
628                DaikonVariableInfo[] arg = {sib};
629                StringBuilder buf = new StringBuilder();
630                DaikonVariableInfo newChild =
631                    thisInfo.addPureMethodDecl(cinfo, meth, arg, offset, depth, buf);
632                String newOffset = buf.toString();
633                debug_vars.log("Pure method");
634                debug_vars.indent();
635                assert meth.member != null
636                    : "@AssumeAssertion(nullness): member of method_infos have .member field"; // fix with dependent type
637                newChild.addChildNodes(
638                    cinfo,
639                    ((Method) meth.member).getReturnType(),
640                    meth.member.getName(),
641                    newOffset,
642                    depth);
643                debug_vars.exdent();
644              }
645            }
646          }
647        }
648      }
649    }
650    debug_vars.log("exit addClassVars%n");
651  }

653  /**
654   * Adds the decl info for a single parameter as a child of this node. Also adds "derived"
655   * variables such as the runtime .class variable.
656   *
657   * @return the newly created DaikonVariableInfo object, whose parent is this
658   */
659  protected DaikonVariableInfo addParamDeclVar(
660      ClassInfo cinfo,
661      Class<?> type,
662      String name,
663      String offset,
664      int depth,
665      int argNum,
666      int param_offset) {
667    debug_vars.log("enter addParamDeclVar%n");
668    // add this variable to the tree as a child of curNode
669    DaikonVariableInfo newChild = new ParameterInfo(offset + name, argNum, type, param_offset);
670
671    addChild(newChild);
672
673    boolean ignore = newChild.check_for_dup_names();
674    if (!ignore) {
675      newChild.checkForDerivedVariables(type, name, offset);
676    }
677
678    debug_vars.log("exit addParamDeclVar%n");
679    return newChild;

753  /**
754   * Adds the decl info for a single class variable (a field) as a child of this node. Also adds
755   * "derived" variables such as the runtime .class variable.
756   *
757   * @return the newly created DaikonVariableInfo object, whose parent is this
758   */
759  @SuppressWarnings("deprecation") // in Java 9+, use canAccess instead of isAccessible
760  protected DaikonVariableInfo addDeclVar(Field field, String offset, StringBuilder buf) {
761    debug_vars.log("enter addDeclVar(field):%n");
762    debug_vars.log("  field: %s, offset: %s%n", field, offset);
763    String arr_str = "";
764    if (isArray) {
765      arr_str = "[]";
766    }
767
768    // Temporarily make the field accessible.
769    boolean changedAccess = false;
770    if (!field.isAccessible()) {
771      changedAccess = true;
772      field.setAccessible(true);
773    }
774
775    int modifiers = field.getModifiers();
776    if (Modifier.isStatic(modifiers)) {
777      offset = field.getDeclaringClass().getName() + ".";
778    } else if (offset.length() == 0) { // instance fld, 1st recursion step
779      offset = "this.";
780    }
781
782    Class<?> type = field.getType();
783    String type_name = stdClassName(type) + arr_str + appendAuxInfo(field);
784
785    String theName = field.getName();
786
787    // Convert the internal reflection name for an outer class
788    // 'this' field to the Java language format.
789    if (theName.startsWith("this$")) {
790      offset = "";
791      theName = type.getName() + ".this";
792      if (!type_name.contains("#")) {
793        type_name += " # isNonNull=true";
794      } else {
795        type_name += ", isNonNull=true";
796      }
797    }
798
799    DaikonVariableInfo newField =
800        new FieldInfo(
801            offset + theName, field, type_name, getRepName(type, false) + arr_str, isArray);
802    boolean ignore = newField.check_for_dup_names();
803
804    if (DaikonWriter.isStaticConstField(field) && !isArray) {
805      ClassInfo cinfo = Runtime.getClassInfoFromClass(field.getDeclaringClass());
806      String value = null;
807      boolean isPrimitive = true;
808
809      if (cinfo != null) {
810        value = cinfo.staticMap.get(theName);
811
812        if (DaikonVariableInfo.dkconfig_constant_infer) {
813          if (value == null) {
814            isPrimitive = false;
815            String className = field.getDeclaringClass().getName();
816            // If the class has already been statically initialized, get its hash
817            if (Runtime.isInitialized(className)) {
818              try {
819                @SuppressWarnings("nullness") // the field is static, so null is OK as argument
820                Object fieldValue = field.get(null);
821                value = Integer.toString(System.identityHashCode(fieldValue));
822              } catch (Exception e) {
823                throw new BugInDaikon("Problem with field " + field);
824              }
825            }
826          }
827        }
828      }
829
830      // System.out.printf("static final value = %s%n", value);
831
832      // in this case, we don't want to print this variable to
833      // the dtrace file
834      if (value != null) {
835        newField.repTypeName += " = " + value;
836        newField.const_val = value;
837        newField.dtraceShouldPrint = false;
838        if (DaikonVariableInfo.dkconfig_constant_infer && isPrimitive) {
839          newField.dtraceShouldPrintChildren = false;
840        }
841      }
842      // else
843      // {
844      // don't print anything
845      // because this field wasn't declared with an actual "hardcoded" constant
846      // }
847
848    }
849
850    addChild(newField);
851
852    if (!ignore) {
853      newField.checkForDerivedVariables(type, theName, offset);
854    }
855
856    buf.append(offset);
857
858    if (changedAccess) {
859      field.setAccessible(false);
860    }
861
862    debug_vars.log("exit addDeclVar(field)%n");
863    return newField;
864  }

875  /**
876   * Given a type, gets the representation type to be used in Daikon. For example, the
877   * representation type of a class object is "hashcode."
878   *
879   * @param type the type of the variable
880   * @param asArray whether the variable is being output as an array (true) or as a pointer (false)
881   * @return the representation type as a string
882   */
883  public static String getRepName(Class<?> type, boolean asArray) {
884    if (type == null) {
885      return "hashcode";
886    } else if (type.isPrimitive()) {
887      if (type.equals(Double.TYPE)) {
888        return "double";
889      } else if (type.equals(Float.TYPE)) {
890        return "double";
891      } else if (type.equals(Boolean.TYPE)) {
892        return "boolean";
893      } else {
894        return "int";
895      }
896    } else if (type.getName().equals("java.lang.String")) {
897      // if we are printing the actual array, the rep type is "java.lang.String"
898      if (true) {
899        return "hashcode";
900      }
901      if (asArray) {
902        return "java.lang.String";
903      }
904      // otherwise, it is just a hashcode
905      else {
906        return "hashcode";
907      }
908    } else {
909      return "hashcode";
910    }
911  }

913  /**
914   * Determines if type needs a corresponding .class runtime class variable.
915   *
916   * <p>The .class variable is printed for interfaces, abstract classes, and Object. For these
917   * types, the run-time class is always (or, for Object, usually) different than the declared type.
918   * An alternate, and possibly more useful, heuristic would be to print the .class variable for any
919   * type that has subtypes.
920   *
921   * @param type the variable's type
922   */
923  protected static boolean shouldAddRuntimeClass(Class<?> type) {
924    // For some reason, abstracts seems to be set on arrays
925    // and primitives.  This is a temporary fix to get things
926    // close.
927    if (type.isPrimitive()) {
928      return false;
929    }
930    if (type.isArray()) {
931      Class<?> eltType = type.getComponentType();
932      assert eltType != null; // because type is an array
933      return !eltType.isPrimitive();
934    }
935
936    if (type.getName().equals("java.lang.Object")) {
937      // Objects
938      // System.out.println ("type is object " + type);
939      return true;
940    } else if (Modifier.isAbstract(type.getModifiers())) {
941      // System.out.printf("Type [%s] is abstract %Xh %Xh %s%n", type,
942      //                   type.getModifiers(), Modifier.ABSTRACT,
943      //                   Modifier.toString (type.getModifiers()));
944      return true;
945    } else if (type.isInterface()) {
946      // System.out.println ("type is interface " + type);
947      return true;
948    } else if (type.isArray()) {
949      // Arrays of non-primitive types
950      // System.out.println ("type is array " + type);
951      Class<?> eltType = type.getComponentType();
952      assert eltType != null; // because type is an array
953      return !eltType.isPrimitive();
954    } else {
955      return false;
956    }
957  }

959  /**
960   * Returns whether or not the specified field is visible from the Class current. All fields within
961   * instrumented classes are considered visible from everywhere (to match dfej behavior).
962   */
963  public static boolean isFieldVisible(Class<?> current, Field field) {
964    Class<?> fclass = field.getDeclaringClass();
965    int modifiers = field.getModifiers();
966
967    // If the field is within the current class, it is always visible
968    if (current.equals(fclass)) {
969      return true;
970    }
971
972    if (!std_visibility) {
973      // If the field is in any instrumented class it is always visible
974      synchronized (SharedData.all_classes) {
975        for (ClassInfo ci : SharedData.all_classes) {
976          // System.out.printf("comparing %s vs %s%n", ci.class_name,
977          // fclass.getName());
978          if (ci.class_name.equals(fclass.getName())) {
979            return true;
980          }
981        }
982      }
983    }
984
985    // Otherwise we consider the variable not to be visible, even
986    // though it is.  This mimics dfej behavior
987    if (!std_visibility) {
988      return false;
989    }
990
991    // Everything in the same class is visible
992    if (current == fclass) {
993      return true;
994    }
995
996    // If the field is in the same package, it's visible if it is
997    // not private or protected.
998    if (current.getPackage() != null && current.getPackage().equals(fclass.getPackage())) {
999      return !(Modifier.isPrivate(modifiers) || Modifier.isProtected(modifiers));
1000    }
1001
1002    // The field must be in an unrelated class, it must be marked
1003    // public to be visible
1004    return (Modifier.isPublic(modifiers));
1005  }

1147  /**
1148   * Explores the tree one level deeper (see {@link DaikonVariableInfo}). This method adds child
1149   * nodes to this node.
1150   *
1151   * <p>For example: "recurse" on a hashcode array object to print the actual array of values or
1152   * recurse on hashcode variable to print its fields. Also accounts for derived variables (.class,
1153   * .tostring) and "recurses" on arrays (that is, adds a variable to print out the arrays's
1154   * elements as opposed to just the hashcode of the array).
1155   *
1156   * @param theName the name of the variable currently being examined, such as "ballCount"
1157   * @param offset the representation of the variables we have previously examined. For examples,
1158   *     offset could be "this." in which case offset + name would be "this.ballCount.".
1159   */
1160  @RequiresNonNull("#1.clazz")
1161  protected void addChildNodes(
1162      ClassInfo cinfo, Class<?> type, String theName, String offset, int depthRemaining) {
1163
1164    debug_vars.log("enter addChildNodes:%n");
1165    debug_vars.log("  name: %s, offset: %s%n", theName, offset);
1166
1167    if (type.isPrimitive()) {
1168      return;
1169    }
1170
1171    // Convert the internal reflection name for an outer class
1172    // 'this' field to the Java language format.
1173    if (theName.startsWith("this$")) {
1174      theName = type.getName() + ".this";
1175      offset = "";
1176    }
1177
1178    if (type.isArray()) {
1179      // don't go into more than one dimension of a multi-dimensional array
1180      if (isArray) {
1181        return;
1182      }
1183
1184      Class<?> eltType = type.getComponentType();
1185      assert eltType != null; // because type is an array
1186      if (eltType.isPrimitive()) {
1187
1188        DaikonVariableInfo newChild = new ArrayInfo(offset + theName + "[]", eltType);
1189
1190        newChild.check_for_dup_names();
1191
1192        addChild(newChild);
1193      }
1194      // multi-dimensional arrays (not currently used)
1195      else if (eltType.isArray()) {
1196        DaikonVariableInfo newChild = new ArrayInfo(offset + theName + "[]", eltType);
1197
1198        newChild.check_for_dup_names();
1199
1200        addChild(newChild);
1201
1202        debug_vars.log("Array variable");
1203        debug_vars.indent();
1204        newChild.addChildNodes(cinfo, eltType, "", offset + theName + "[]", depthRemaining);
1205        debug_vars.exdent();
1206      }
1207      // array is 1-dimensional and element type is a regular class
1208      else {
1209        DaikonVariableInfo newChild = new ArrayInfo(offset + theName + "[]", eltType);
1210
1211        boolean ignore = newChild.check_for_dup_names();
1212
1213        addChild(newChild);
1214
1215        // Print out the class of each element in the array.
1216        // The offset will only be equal to ""
1217        // if we are examining a local variable (parameter).
1218        if (!ignore) {
1219          if (!theName.equals("return") && !offset.equals("")) {
1220            newChild.checkForRuntimeClass(type, theName + "[]", offset);
1221          }
1222
1223          newChild.checkForString(eltType, theName + "[]", offset);
1224        }
1225        newChild.addClassVars(cinfo, false, eltType, offset + theName + "[].", depthRemaining - 1);
1226      }
1227    }
1228    // regular old class type
1229    else {
1230      debug_vars.log("**Depth Remaining = %d%n", depthRemaining);
1231
1232      if (depthRemaining <= 0) {
1233        // don't recurse any more!
1234        return;
1235      }
1236      if (!systemClass(type)) {
1237        addClassVars(cinfo, false, type, offset + theName + ".", depthRemaining - 1);
1238      }
1239    }
1240    debug_vars.log("exit addChildNodes%n");
1241  }

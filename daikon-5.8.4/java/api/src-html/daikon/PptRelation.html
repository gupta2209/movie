024/**
025 * Class that builds and describes relations in the ppt hierarchy. Building the relationship is
026 * specific to each type of parent/child relationship (eg, method to object, exit to combined exit,
027 * etc). The use of the relationship is general.
028 *
029 * <p>The basic function of the class is to translate from a variable in the parent to the
030 * equivalent variable in the child and vice-versa. For example, in the ENTER &rarr; EXIT
031 * relationship, the parent is the ENTER ppt and the child is the EXIT ppt. Each variable in the
032 * ENTER ppt is connected to the corresponding orig variable in the EXIT ppt.
033 */
034public class PptRelation implements Serializable {
035
036  // We are Serializable, so we specify a version to allow changes to
037  // method signatures without breaking serialization.  If you add or
038  // remove fields, you should change this number to the current date.
039  static final long serialVersionUID = 20030819L;
040
041  /**
042   * The different ppt/variable hierarchy relationships. Parent and User relations are specified in
043   * the declaration record of the ppt. ENTER_EXIT, EXIT_EXITNN, and PPT_COND are automatically
044   * constructed. MERGE_CHILD is not used by Daikon.
045   */
046  public enum PptRelationType {
047    /** Acyclic relationship to a parent, eg, method to its object. */
048    PARENT,
049    /** Possibly cyclic relationship, eg. nested object instances */
050    USER,
051    /** Entrance of method to exit of method. */
052    ENTER_EXIT,
053    /** Combined exit to numbered exit of a method. */
054    EXIT_EXITNN,
055    /** Relation between the same ppt in two different PptMaps. */
056    MERGE_CHILD,
057    /** Relation from a program point to its conditional ppts. */
058    PPT_PPTCOND
059  };

100  /**
101   * Create a relation between the specified parent and child. The actual variable relations are
102   * filled in by the caller. Note that this creates the connection between this relation and the
103   * parent/child. As a side effect, the constructed PptRelation is stored in both the parent and
104   * the child.
105   */
106  private PptRelation(PptTopLevel parent, PptTopLevel child, PptRelationType rel_type) {
107
108    this.parent = parent;
109    this.child = child;
110    parent_to_child_map = new LinkedHashMap<>();
111    child_to_parent_map = new LinkedHashMap<>();
112    // rel_type is one of the above relationship types because this is a
113    // private constructor, called only within this file.
114    relationship = rel_type;
115    connect();
116  }

118  /** Adds this relation to its child's parent list and its parent's children list. */
119  @SuppressWarnings({
120    "nullness:argument.type.incompatible" // won't be used until initialization is finished
121  })
122  private void connect(@UnderInitialization(PptRelation.class) PptRelation this) {
123    assert !child.parents.contains(this);
124    assert !parent.children.contains(this);
125    child.parents.add(this);
126    parent.children.add(this);
127  }

141  /** Return a string containing all of the parent&rarr;child var relations. */
142  public String parent_to_child_var_string() {
143
144    StringBuilder var_str = new StringBuilder();
145    for (VarInfo pv : parent_to_child_map.keySet()) {
146      VarInfo cv = parent_to_child_map.get(pv);
147      if (var_str.length() > 0) var_str.append(", ");
148      var_str.append(pv.name() + "->" + cv.name());
149    }
150
151    return var_str.toString();
152  }

154  /**
155   * Relates all of the variables with the same name in parent and child. Returns true if each
156   * non-static parent variable was related to a child variable.
157   */
158  public boolean relate_same_name() {
159
160    boolean relate_all = true;
161    for (VarInfo vp : parent.var_infos) {
162      boolean relate_var = relate(vp, vp.name());
163      if (!relate_var && !vp.isStaticConstant()) {
164        // System.out.printf("no relation for '%s' from %s-%s with vars %s%n",
165        //                   vp.name(), parent.name(), child.name(),
166        //                   child.varNames());
167        relate_all = false;
168      }
169    }
170
171    return relate_all;
172  }

183   * of the PPt tree (so that extra relations don't result in duplicative information).
184   *
185   * <p>Somewhat arbitrarily, Object&rarr;User and Enter&rarr;Exit are not considered primary while

207  /**
208   * Like parentVar(VarInfo), but if no parent is found, tries every variable in the equality set
209   * and returns null only if none of them has a parent.
210   */
211  public @Nullable VarInfo parentVarAnyInEquality(VarInfo childVar) {
212    VarInfo result = parentVar(childVar);
213    if (result != null) {
214      return result;
215    }
216    if (childVar.equalitySet == null) {
217      return null;
218    }
219    for (VarInfo v : childVar.equalitySet.getVars()) {
220      result = parentVar(v);
221      if (result != null) {
222        return result;
223      }
224    }
225    return null;
226  }

241  /**
242   * Returns a map of VarInfo.Pair with an entry for each pair of equal variables in all of the
243   * equality sets of the child. The variables are the corresponding parent variables and not the
244   * child variables themselves. The map is from the pair to itself, which allows the pair to be
245   * looked up (which is not possible with a set).
246   */
247  public Map<VarInfo.Pair, VarInfo.Pair> get_child_equalities_as_parent() {
248
249    debug.fine(
250        "get_child_equalities for "
251            + child.name()
252            + " for parent "
253            + parent.name()
254            + " "
255            + relationship);
256    Map<VarInfo.Pair, VarInfo.Pair> emap = new LinkedHashMap<>();
257
258    if (child.equality_view == null) {
259      throw new Error(
260          "child.equality_view == null for child ppt: "
261              + child.name()
262              + " samples = "
263              + child.num_samples());
264    }
265    if (child.equality_view.invs == null) {
266      throw new Error(
267          "child.equality_view.invs == null for child ppt: "
268              + child.name()
269              + " samples = "
270              + child.num_samples()
271              + "children = "
272              + child.children);
273    }
274
275    // Loop through each equality set in the child
276    for (Invariant inv : child.equality_view.invs) {
277      Equality e = (Equality) inv;
278      debug.fine("-- processing equality set " + e);
279      Set<VarInfo> eqset = e.getVars();
280      VarInfo[] varr = eqset.toArray(new VarInfo[eqset.size()]);
281
282      // Build each combination of variables in the equality set and produce
283      // a pair for each.  Skip any variables that do not have corresponding
284      // variables in the parent.
285      for (int j = 0; j < varr.length; j++) {
286        VarInfo v1 = parentVar(varr[j]);
287        if (v1 == null) {
288          debug.fine("-- -- " + varr[j].name() + " not in parent (skip)");
289          continue;
290        }
291        for (int k = j + 1; k < varr.length; k++) {
292          VarInfo v2 = parentVar(varr[k]);
293          if (v2 == null) {
294            debug.fine("-- -- " + varr[k].name() + " not in parent (skip)");
295            continue;
296          }
297          VarInfo.Pair parent_pair = new VarInfo.Pair(v1, v2, e.numSamples());
298          emap.put(parent_pair, parent_pair);
299          if (debug.isLoggable(Level.FINE)) {
300            debug.fine(
301                "-- -- "
302                    + varr[j].name()
303                    + ", "
304                    + varr[k].name()
305                    + " in child yield "
306                    + parent_pair
307                    + " in parent");
308          }
309        }
310      }
311    }
312    return emap;
313  }

315  /**
316   * Relates parent_var to a variable in child that matches name.
317   *
318   * @param parent_var the parent variable being matched
319   * @param viname the name to look for in child variables
320   * @return true if there was a matching variable, false otherwise
321   */
322  private boolean relate(VarInfo parent_var, String viname) {
323
324    for (VarInfo vc : child.var_infos) {
325      if (viname.equals(vc.name())) {
326        child_to_parent_map.put(vc, parent_var);
327        parent_to_child_map.put(parent_var, vc);
328        return true;
329      }
330    }
331    return false;
332  }

371  /**
372   * Creates a USER or PARENT relation from child to parent. The variable relationships are
373   * specified in the declaration record and stored in the VarInfo for each variable.
374   * RuntimeException will be thrown if any of the parent variables cannot be found.
375   */
376  public static PptRelation newParentRelation(
377      ParentRelation pr, PptTopLevel parent, PptTopLevel child) {
378
379    assert pr != null && parent != null && child != null;
380    // System.out.printf("Parent Relation %s[%d] to %s%n", pr.parent_ppt_name,
381    //                   pr.id, child.name());
382
383    PptRelation rel = new PptRelation(parent, child, pr.rel_type);
384    for (VarInfo vc : child.var_infos) {
385      for (VarParent pi : vc.parents) {
386        // System.out.printf("--child variable %s, ppt %s[%d], parent_var %s%n",
387        //                    vc.name(), pi.parent_ppt, pi.parent_relation_id,
388        //                    pi.parent_variable);
389        if (pi.parent_relation_id != pr.id) {
390          continue;
391        }
392
393        // Get the name of the parent variable.  Its the same as this one if
394        // not specified.  For now, remove the array placeholder (..) since
395        // VarInfoName doesn't support it.
396        String parent_name = pi.parent_variable;
397        if (parent_name == null) {
398          parent_name = vc.name();
399        }
400        // parent_name = parent_name.replace ("[..]", "[]");
401
402        // System.out.printf("---parent name %s%n", parent_name);
403        VarInfo vp = parent.find_var_by_name(parent_name);
404        if (vp == null) {
405          throw new RuntimeException(
406              String.format(
407                  "Can't find parent variable '%s' in ppt '%s', "
408                      + "with vars %s specified by var '%s' in ppt '%s'",
409                  parent_name, pi.parent_ppt, parent.var_names(), vc.name(), child.name()));
410        }
411        rel.child_to_parent_map.put(vc, vp);
412        rel.parent_to_child_map.put(vp, vc);
413      }
414    }
415    return rel;
416  }

419   * Returns a relation in the ppt hierarchy from an object (parent) to a user (child) of that
420   * objects (eg, from the object B to the method A.foo (B arg)).
421   *
422   * @param parent ppt of the object definition
423   * @param child ppt of a user of parent's object
424   * @param arg variable of type object found in child
425   */
426  public static PptRelation newObjectUserRel(PptTopLevel parent, PptTopLevel child, VarInfo arg) {
427
428    assert (parent != null) && (child != null);
429
430    PptRelation rel = new PptRelation(parent, child, PptRelationType.USER);
431
432    // Connect each each field in arg between parent and child.  Do this
433    // by substituting args name for this in the parent and then looking
434    // for a name match in the child
435    for (VarInfo vp : parent.var_infos) {
436      rel.relate(vp, vp.replace_this(arg));
437    }
438    return rel;
439  }

441  /**
442   * Returns a relation in the ppt hierarchy from enter points to exit points over orig variables.
443   */
444  public static PptRelation newEnterExitRel(PptTopLevel parent, PptTopLevel child) {
445
446    assert (parent != null) && (child != null);
447
448    PptRelation rel = new PptRelation(parent, child, PptRelationType.ENTER_EXIT);
449
450    // Look for orig versions of each non-derived parent variable in the child
451    // Note that static constants don't have orig versions (since they are
452    // known to be the same), so we connect to the post version instead.
453    for (VarInfo vp : parent.var_infos) {
454      if (vp.derived != null) {
455        continue;
456      }
457      if (vp.isStaticConstant()) {
458        @SuppressWarnings("UnusedVariable") // see comment below
459        boolean found = rel.relate(vp, vp.name());
460        // Static constants are not always placed at each level in hierarchy
461        // (due to mutually recursive constants that contain one another as
462        // fields).
463        // assert found;
464      } else {
465        // VarInfoName orig_name = vp.name.applyPrestate().intern();
466        boolean found = rel.relate(vp, vp.prestate_name());
467        assert found
468            : String.format(
469                "vp %s orig_name %s parent %s child %s with vars %s",
470                vp, vp.prestate_name(), parent.name(), child.name(), child.var_names());
471      }
472    }
473
474    // Look for orig versions of derived variables in the child.  This is
475    // done by finding the base of each derived variable and looking for
476    // a child variable with the same bases and the same equation.  This
477    // is necessary because derivations are done AFTER orig variables so
478    // applying the prestate name (as done above) won't work (the resulting
479    // variable is really the same but the name is constructed differently).
480
481    // Loop through each derived parent (ENTER) variable
482    for (VarInfo vp : parent.var_infos) {
483      if (vp.derived == null) {
484        continue;
485      }
486
487      // Get a child version of each of the bases of the derivation
488      VarInfo[] vp_bases = vp.derived.getBases();
489      // TODO: Is this "@Nullable" annotation correct?  (That is, can the
490      // element value actually be null?)
491      @Nullable VarInfo[] child_vp_bases = new VarInfo[vp_bases.length];
492      for (int j = 0; j < vp_bases.length; j++) {
493        child_vp_bases[j] = rel.childVar(vp_bases[j]);
494      }
495
496      // Loop through the child (exit) looking for a matching derived variable
497      for (VarInfo vc : child.var_infos) {
498        if (vc.derived == null) {
499          continue;
500        }
501        if (vc.derived.isSameFormula(vp.derived)) {
502          assert vc.derived != null;
503          VarInfo[] vc_bases = vc.derived.getBases();
504          if (Arrays.equals(child_vp_bases, vc_bases)) {
505            rel.child_to_parent_map.put(vc, vp);
506            rel.parent_to_child_map.put(vp, vc);
507            break;
508          }
509        }
510      }
511    }
512
513    // Make sure every non-static ENTER variable was found in the EXIT point
514    boolean all_found = true;
515    for (VarInfo vp : parent.var_infos) {
516      if (vp.isStaticConstant()) {
517        continue;
518      }
519      if (!rel.parent_to_child_map.containsKey(vp)) {
520        if (all_found) {
521          System.out.printf(
522              "missing variables in newEnterExitRel:%n  parent = %s%n  child = %s%nparent varinfos missing from parent_to_child_map:%n",
523              parent.name(), child.name());
524          all_found = false;
525        }
526        System.out.printf("   %s%n", vp.name());
527      }
528    }
529    if (!all_found) {
530      System.out.printf("rel.parent_to_child_map:%n");
531      for (Map.Entry<VarInfo, VarInfo> entry : rel.parent_to_child_map.entrySet()) {
532        System.out.printf("    %s => %s%n", entry.getKey(), entry.getValue());
533      }
534      System.out.printf("child.var_infos:%n");
535      for (VarInfo vc : child.var_infos) {
536        System.out.println("    " + vc.name());
537      }
538      System.out.printf(
539          "End of diagnostics for newEnterExitRel(%s, %s)%n", parent.name(), child.name());
540      // throw new Error("Missing orig variable in EXIT");
541    }
542    return rel;
543  }

564      rel.parent_to_child_map.put(vp, vc);
565    }

589  /**
590   * Returns a an artificial relation in the Program point hierarchy between the same ppt in two
591   * different PptMaps. Used to merge invariants between different data sets. The parent and the
592   * child should have exactly the same variables.
593   */
594  public static PptRelation newMergeChildRel(PptTopLevel parent, PptTopLevel child) {
595
596    assert (parent != null) && (child != null);
597
598    PptRelation rel = new PptRelation(parent, child, PptRelationType.MERGE_CHILD);
599
600    // assert that parent vars match child vars
601    if (parent.var_infos.length != child.var_infos.length) {
602      System.out.println("newMergeChildRel: in ppt " + parent.name() + " vars don't match");
603      System.out.println("parent vars= " + Arrays.toString(parent.var_infos));
604      System.out.println("child vars=  " + Arrays.toString(child.var_infos));
605      assert parent.var_infos.length == child.var_infos.length;
606    }
607
608    // Create the parent-child variable map.  This one is easy as the
609    // variables should match exactly
610    for (int i = 0; i < parent.var_infos.length; i++) {
611      VarInfo vc = child.var_infos[i];
612      VarInfo vp = parent.var_infos[i];
613      if (!vc.name().equals(vp.name())) {
614        System.out.println(
615            "newMergeChildRel: in ppt " + parent.name() + " var " + vc.name() + " doesn't match");
616        System.out.println("par vars  = " + Arrays.toString(parent.var_infos));
617        System.out.println("child vars= " + Arrays.toString(child.var_infos));
618        assert vc.name().equals(vp.name());
619      }
620      rel.child_to_parent_map.put(vc, vp);
621      rel.parent_to_child_map.put(vp, vc);
622    }
623    return rel;
624  }

626  /**
627   * Copies the relation from its current ppts to the specified ppts. The new ppts must have the
628   * same variables in the same order as do the original ones.
629   */
630  public PptRelation copy(PptTopLevel new_parent, PptTopLevel new_child) {
631
632    PptRelation rel = new PptRelation(new_parent, new_child, relationship);
633    for (VarInfo vc : child_to_parent_map.keySet()) {
634      VarInfo vp = child_to_parent_map.get(vc);
635      VarInfo new_vc = new_child.var_infos[vc.varinfo_index];
636      VarInfo new_vp = new_parent.var_infos[vp.varinfo_index];
637      assert new_vc.name().equals(vc.name());
638      assert new_vp.name().equals(vp.name());
639      rel.child_to_parent_map.put(new_vc, new_vp);
640      rel.parent_to_child_map.put(new_vp, new_vc);
641    }
642    return rel;
643  }

655  /**
656   * Initialize the hierarchical relationship between ppts. Specifically process each ppt, find its
657   * parent(s) in the partial order, and fill this point into the children field in the parent. Note
658   * that children contains only the immediate descendants of the ppt.
659   *
660   * <p>This version should be used with the old version of declaration records. Use
661   * init_hierarchy_new() with new declaration records.
662   */
663  public static void init_hierarchy(PptMap all_ppts) {
664
665    for (PptTopLevel ppt : all_ppts.pptIterable()) {
666      PptName pname = ppt.ppt_name;
667      PptRelation rel = null;
668      Daikon.debugProgress.fine("Processing ppt " + pname);
669      debug.fine("Processing ppt " + pname);
670
671      // If this is an object ppt, parent is the class point
672      if (pname.isObjectInstanceSynthetic()) {
673        PptTopLevel parent = all_ppts.get(pname.makeClassStatic());
674        if (parent != null) {
675          rel = newClassObjectRel(parent, ppt);
676        }
677
678        // Else if it's a method and not a constructor, parent is
679        // object or class static methods will relate to the class,
680        // while non-static methods will relate to the object.
681        // Whether or not a method is static is not in the decls file.
682        // We infer this by looking to see if the variables match with
683        // the object ppt or the class ppt.
684      } else if ((pname.isEnterPoint() && !pname.isConstructor()) || pname.isCombinedExitPoint()) {
685
686        PptTopLevel parent = all_ppts.get(pname.makeObject());
687
688        if (parent != null) {
689          if (ppt.find_var_by_name(parent.var_infos[0].name()) != null) {
690            rel = newObjectMethodRel(parent, ppt);
691          } else {
692            parent = all_ppts.get(parent.ppt_name.makeClassStatic());
693            if (parent != null) rel = newObjectMethodRel(parent, ppt);
694          }
695        }
696
697        // Else if an exitNN point, parent is combined exit point
698      } else if (pname.isExitPoint()) {
699        PptTopLevel parent = all_ppts.get(pname.makeExit());
700        // System.out.printf("Parent of %s is %s%n", pname.name(),
701        //                   parent.name());
702        if (parent != null) rel = newCombinedExitExitNNRel(parent, ppt);
703      }
704
705      // If a relation was created, connect it into its ppts
706      if (rel != null) {
707        debug.fine(
708            "-- ppt parent is "
709                + rel.parent.name()
710                + " with connections ["
711                + rel.parent_to_child_var_string()
712                + "]");
713      } else {
714        debug.fine(" -- no ppt parent");
715      }
716
717      // Connect combined exit points to enter points over orig variables
718      if (pname.isCombinedExitPoint()) {
719        PptTopLevel enter = all_ppts.get(pname.makeEnter());
720        if (enter != null) {
721          rel = PptRelation.newEnterExitRel(enter, ppt);
722          debug.fine(
723              " -- exit to enter "
724                  + enter.name
725                  + " with connections ["
726                  + rel.parent_to_child_var_string()
727                  + "]");
728        } else {
729          debug.fine("-- No matching enter for exit");
730        }
731      }
732
733      // For all points, look for vars of a declared type for which we have
734      // a corresponding OBJECT ppt.  Essentially these are all of the
735      // users of the object.  Don't match if the variable already has
736      // a parent (since the parent will provide the link back to the
737      // object) For each variable of this type that we find, setup a
738      // parent-child relationship with its corresponding OBJECT
739      // variables.
740      //
741      // For example, consider class A with fields x and y and method
742      // B.foo (A arg1, A arg2).  We will setup two relations to this
743      // ppt -- one from A to b.foo.arg1 and one from A to b.foo.arg2.
744      // in each we will equate A.x with arg.x and A.y with arg.y.
745      //
746      // We skip variables named exactly 'this' so that we don't setup a
747      // recursive relationship from the object to itself.
748
749      // DaikonSimple can not see these relations, so don't create them
750      // if we'll be comparing to DaikonSimple.
751      if (dkconfig_enable_object_user) {
752
753        debug.fine("-- Looking for variables with an OBJECT ppt");
754        for (VarInfo vc : ppt.var_infos) {
755          String dstr = "-- -- var '" + vc.name() + "' - ";
756          if (ppt.has_parent(vc)) {
757            debug.fine(dstr + " Skipping, already has a parent");
758            continue;
759          }
760          if (vc.isThis()) {
761            debug.fine(dstr + " skipping, name is 'this'");
762            continue;
763          }
764          PptTopLevel object_ppt = vc.find_object_ppt(all_ppts);
765          if (object_ppt != null) {
766            if (object_ppt == ppt) {
767              debug.fine(dstr + " skipping, OBJECT (" + object_ppt + ") is the same as this");
768              continue;
769            }
770            rel = PptRelation.newObjectUserRel(object_ppt, ppt, vc);
771            debug.fine(
772                dstr
773                    + " Connected to Object ppt "
774                    + object_ppt.name()
775                    + " with connections ["
776                    + rel.parent_to_child_var_string()
777                    + "]");
778          } else {
779            debug.fine(dstr + " No object ppt");
780          }
781        }
782      }
783      // Connect any conditional ppt variables.  Only connect to the
784      // first splitter, since each splitter should yield the same
785      // results at the parent (since each splitter sees the same
786      // points)  This should only happen at the leaves (numbered
787      // exit points) since all other points should be built from
788      // their other children.  But since we need the relation
789      // from the child's point of view when printing, we create
790      // under all cases and then remove it from non-leaves children
791      // list.  This doesn't seem like the best solution.
792      if (ppt.has_splitters()) {
793        assert ppt.splitters != null; // guaranteed by call to has_splitters
794        PptSplitter ppt_split = ppt.splitters.get(0);
795        for (int ii = 0; ii < ppt_split.ppts.length; ii++) {
796          rel = newPptPptConditional(ppt, ppt_split.ppts[ii]);
797          debug.fine(
798              " -- Connected down to ppt conditional "
799                  + ppt_split.ppts[ii].name()
800                  + " with connections ["
801                  + rel.parent_to_child_var_string()
802                  + "]");
803          if (!ppt.ppt_name.isNumberedExitPoint()) {
804            ppt.children.remove(rel);
805          }
806        }
807      }
808    }
809
810    // Create relations between conditional ppts and their children.
811    // The relationship between conditional ppts matches exactly
812    // the relationship between each their parents.  For example,
813    // presume ppt A has a child ppt B.  A has two conditional
814    // ppts (AC1, AC2) and B has two conditional ppts (BC1, BC2)
815    // Then AC1 is the parent of BC1 and AC2 is the parent of BC2
816
817    // Loop over each ppt and process each non-leaf with splitters
818    for (PptTopLevel ppt : all_ppts.pptIterable()) {
819      if (ppt.ppt_name.isNumberedExitPoint()) {
820        continue;
821      }
822      if (!ppt.has_splitters()) {
823        continue;
824      }
825
826      // System.out.printf("processing splitter '%s' [%s] %b%n", ppt.name(),
827      //                    ppt.ppt_name.getPoint(),
828      //                    ppt.ppt_name.isNumberedExitPoint());
829
830      // Loop over each splitter
831      splitter_loop:
832      for (Iterator<PptSplitter> ii = ppt.splitters.iterator(); ii.hasNext(); ) {
833        PptSplitter ppt_split = ii.next();
834
835        // list of children that match this splitter
836        List<SplitChild> split_children = new ArrayList<>();
837
838        // Create a list of children for this splitter
839        child_loop:
840        for (PptRelation rel : ppt.children) {
841          if (!rel.child.has_splitters()) {
842            break;
843          }
844          for (PptSplitter csplit : rel.child.splitters) {
845            if (ppt_split.splitter == csplit.splitter) {
846              split_children.add(new SplitChild(rel, csplit));
847              continue child_loop;
848            }
849          }
850          break;
851        }
852
853        // If we didn't find a matching splitter at each child, can't merge
854        // this point.  Just remove it from the list of splitters
855        if (split_children.size() != ppt.children.size()) {
856          ii.remove();
857          continue;
858        }
859
860        // Build the PptRelations for each child.  The PptRelation from
861        // the conditional point is of the same type as the original
862        // relation from parent to child
863        for (SplitChild sc : split_children) {
864          ppt_split.add_relation(sc.rel, sc.ppt_split);
865        }
866      }
867    }
868
869    // Debug print the hierarchy in a more readable manner
870    if (debug.isLoggable(Level.FINE)) {
871      debug.fine("PPT Hierarchy");
872      for (PptTopLevel ppt : all_ppts.pptIterable()) {
873        if (ppt.parents.size() == 0) ppt.debug_print_tree(debug, 0, null);
874      }
875    }
876
877    // Debug print the equality sets for each ppt
878    if (debug.isLoggable(Level.FINE)) {
879      for (PptTopLevel ppt : all_ppts.pptIterable()) {
880        debug.fine(ppt.name() + " equality sets: " + ppt.equality_sets_txt());
881      }
882    }
883  }

885  /**
886   * Initialize the hierarchical relationship between ppts. Specifically process each ppt, find its
887   * parent(s) in the partial order, and fill this point into the children field in the parent. Note
888   * that children contains only the immediate descendants of the ppt.
889   */
890  public static void init_hierarchy_new(PptMap all_ppts) {
891
892    for (PptTopLevel ppt : all_ppts.pptIterable()) {
893      PptName pname = ppt.ppt_name;
894      // rels is solely for debugging; each relation is stored in the
895      // parent and child ppts
896      List<PptRelation> rels = new ArrayList<>();
897      Daikon.debugProgress.fine("Processing ppt " + pname);
898      debug.fine("Processing ppt " + pname);
899
900      assert ppt.parent_relations != null : "missing parent_relations in ppt " + ppt.name();
901
902      // Process the front-end specified relations
903      for (ParentRelation pr : ppt.parent_relations) {
904        // Skip all relations in subexits.  These relations will be handled
905        // in the combined exit point.
906        if (ppt.is_subexit()) {
907          continue;
908        }
909
910        PptTopLevel parent = all_ppts.get(pr.parent_ppt_name);
911        if (parent == null) {
912          throw new RuntimeException(
913              "parent ppt " + pr.parent_ppt_name + " not found for ppt " + ppt.name());
914        }
915        if ((pr.rel_type == PptRelationType.USER) && !dkconfig_enable_object_user) {
916          continue;
917        }
918        // System.out.printf("processing hierarchy rel from '%s' to '%s'%n",
919        //                    ppt.name(), pr.parent_ppt_name);
920        rels.add(newParentRelation(pr, parent, ppt));
921      }
922
923      // if an exitNN point, parent is combined exit point
924      if (ppt.is_subexit()) {
925        PptTopLevel parent = all_ppts.get(pname.makeExit());
926        if (parent != null) {
927          rels.add(newCombinedExitExitNNRel(parent, ppt));
928        }
929
930        // Connect combined exit points to enter points over orig variables
931      } else if (ppt.is_combined_exit()) {
932        PptTopLevel enter = all_ppts.get(pname.makeEnter());
933        if (enter != null) {
934          rels.add(PptRelation.newEnterExitRel(enter, ppt));
935        }
936      }
937
938      // Connect any conditional ppt variables.  Only connect to the
939      // first splitter, since each splitter should yield the same
940      // results at the parent (since each splitter sees the same
941      // points)  This should only happen at the leaves (numbered
942      // exit points) since all other points should be built from
943      // their other children.  But since we need the relation
944      // from the child's point of view when printing, we create
945      // under all cases and then remove it from non-leaves children
946      // list.  This doesn't seem like the best solution.
947      if (ppt.has_splitters()) {
948        assert ppt.splitters != null; // guaranteed by call to has_splitters
949        PptSplitter ppt_split = ppt.splitters.get(0);
950        for (int ii = 0; ii < ppt_split.ppts.length; ii++) {
951          PptRelation rel = newPptPptConditional(ppt, ppt_split.ppts[ii]);
952          rels.add(rel);
953          if (!ppt.is_subexit()) {
954            ppt.children.remove(rel);
955          }
956        }
957      }
958      // Debug print the created relations
959      for (PptRelation rel : rels) {
960        debug.fine(
961            "-- ppt parent is "
962                + rel.parent.name()
963                + " with connections ["
964                + rel.parent_to_child_var_string()
965                + "]");
966      }
967    }
968
969    // Create relations between conditional ppts and their children.
970    // The relationship between conditional ppts matches exactly
971    // the relationship between each their parents.  For example,
972    // presume ppt A has a child ppt B.  A has two conditional
973    // ppts (AC1, AC2) and B has two conditional ppts (BC1, BC2)
974    // Then AC1 is the parent of BC1 and AC2 is the parent of BC2
975
976    // Loop over each ppt and process each non-leaf with splitters
977    for (PptTopLevel ppt : all_ppts.pptIterable()) {
978      if (ppt.is_subexit()) {
979        continue;
980      }
981      if (!ppt.has_splitters()) {
982        continue;
983      }
984
985      // System.out.printf("processing splitter %s%n", ppt.name());
986
987      // Loop over each splitter
988      splitter_loop:
989      for (Iterator<PptSplitter> ii = ppt.splitters.iterator(); ii.hasNext(); ) {
990        PptSplitter ppt_split = ii.next();
991
992        // list of children that match this splitter
993        List<SplitChild> split_children = new ArrayList<>();
994
995        // Create a list of children for this splitter
996        child_loop:
997        for (PptRelation rel : ppt.children) {
998          if (!rel.child.has_splitters()) {
999            break;
1000          }
1001          for (PptSplitter csplit : rel.child.splitters) {
1002            if (ppt_split.splitter == csplit.splitter) {
1003              split_children.add(new SplitChild(rel, csplit));
1004              continue child_loop;
1005            }
1006          }
1007          break;
1008        }
1009
1010        // If we didn't find a matching splitter at each child, can't merge
1011        // this point.  Just remove it from the list of splitters
1012        if (split_children.size() != ppt.children.size()) {
1013          ii.remove();
1014          continue;
1015        }
1016
1017        // Build the PptRelations for each child.  The PptRelation from
1018        // the conditional point is of the same type as the original
1019        // relation from parent to child
1020        for (SplitChild sc : split_children) {
1021          ppt_split.add_relation(sc.rel, sc.ppt_split);
1022        }
1023      }
1024    }
1025
1026    // Loop over each ppt and create an equality view and invariants for
1027    // any ppt without children that doesn't already have them.  This can
1028    // happen when there are ppts such as OBJECT or CLASS that don't end up
1029    // with any children (due to the program source or because of ppt filtering).
1030    for (PptTopLevel ppt : all_ppts.pptIterable()) {
1031      if ((ppt.children.size() == 0) && (ppt.equality_view == null)) {
1032        assert ppt.is_object() || ppt.is_class() || ppt.is_enter() : ppt;
1033        ppt.equality_view = new PptSliceEquality(ppt);
1034        ppt.equality_view.instantiate_invariants();
1035      }
1036    }
1037
1038    // Debug print the hierarchy in a more readable manner
1039    if (debug.isLoggable(Level.FINE)) {
1040      debug.fine("PPT Hierarchy");
1041      for (PptTopLevel ppt : all_ppts.pptIterable()) {
1042        if (ppt.parents.size() == 0) ppt.debug_print_tree(debug, 0, null);
1043      }
1044    }
1045
1046    // Debug print the equality sets for each ppt
1047    if (debug.isLoggable(Level.FINE)) {
1048      for (PptTopLevel ppt : all_ppts.pptIterable()) {
1049        debug.fine(ppt.name() + " equality sets: " + ppt.equality_sets_txt());
1050      }
1051    }
1052  }

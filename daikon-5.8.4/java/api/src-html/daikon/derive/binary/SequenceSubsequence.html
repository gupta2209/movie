001package daikon.derive.binary;
002
003import daikon.Quantify;
004import daikon.VarInfo;
005import org.checkerframework.checker.lock.qual.GuardSatisfied;
006import org.checkerframework.dataflow.qual.SideEffectFree;
007
008/** Derivations of the form A[0..i] or A[i..<em>end</em>], derived from A and i. */
009public abstract class SequenceSubsequence extends BinaryDerivation {
010  // We are Serializable, so we specify a version to allow changes to
011  // method signatures without breaking serialization.  If you add or
012  // remove fields, you should change this number to the current date.
013  static final long serialVersionUID = 20020801L;
014
015  // Variables starting with dkconfig_ should only be set via the
016  // daikon.config.Configuration interface.
017
018  // base1 is the sequence
019  // base2 is the scalar
020  public VarInfo seqvar(@GuardSatisfied SequenceSubsequence this) {
021    return base1;
022  }
023
024  public VarInfo sclvar(@GuardSatisfied SequenceSubsequence this) {
025    return base2;
026  }
027
028  // Indicates whether the subscript is an index of valid data or a limit
029  // (one element beyond the data of interest).  The first (or last)
030  // element of the derived variable is at index seqvar()+index_shift.
031  public final int index_shift;
032
033  // True for deriving from the start of the sequence to the scalar: B[0..I]
034  // False for deriving from the scalar to the end of the sequence: B[I..]
035  public final boolean from_start;
036
037  /**
038   * @param from_start true means the range goes 0..n; false means the range goes n..end. (n might
039   *     be fudged through off_by_one.)
040   * @param off_by_one true means we should exclude the scalar from the range; false means we should
041   *     include it
042   */
043  protected SequenceSubsequence(VarInfo vi1, VarInfo vi2, boolean from_start, boolean off_by_one) {
044    super(vi1, vi2);
045    this.from_start = from_start;
046    if (off_by_one) {
047      index_shift = from_start ? -1 : +1;
048    } else {
049      index_shift = 0;
050    }
051  }
052
053  @Override
054  protected VarInfo makeVarInfo() {
055    VarInfo seqvar = seqvar();
056    VarInfo sclvar = sclvar();
057
058    VarInfo vi;
059    if (from_start) {
060      vi = VarInfo.make_subsequence(seqvar, null, 0, sclvar, index_shift);
061    } else {
062      vi = VarInfo.make_subsequence(seqvar, sclvar, index_shift, null, 0);
063    }
064
065    return vi;
066  }
067
068  /** Returns the lower bound of the slice. */
069  @Override
070  public Quantify.Term get_lower_bound() {
071    if (from_start) {
072      return new Quantify.Constant(0);
073    } else {
074      return new Quantify.VarPlusOffset(sclvar(), index_shift);
075    }
076  }
077
078  /** Returns the upper bound of the slice. */
079  @Override
080  public Quantify.Term get_upper_bound() {
081    if (from_start) {
082      return new Quantify.VarPlusOffset(sclvar(), index_shift);
083    } else {
084      return new Quantify.Length(seqvar(), -1);
085    }
086  }
087
088  /** Returns the array variable for this slice. */
089  @Override
090  public VarInfo get_array_var() {
091    return seqvar();
092  }
093
094  @SideEffectFree
095  @Override
096  public String csharp_name(String index) {
097    // String lower = get_lower_bound().csharp_name();
098    // String upper = get_upper_bound().csharp_name();
099    // We do not need to check if seqvar().isPrestate() because it is redundant.
100    // return seqvar().csharp_name() + ".Slice(" + lower + ", " + upper + ")";
101    return "\"SequenceSubsequence.java.jpp unimplemented\" != null"; // "interned"
102  }
103
104  /** Returns the ESC name. */
105  @SideEffectFree
106  @Override
107  public String esc_name(String index) {
108    return String.format(
109        "%s[%s..%s]",
110        seqvar().esc_name(), get_lower_bound().esc_name(), get_upper_bound().esc_name());
111  }
112
113  /** returns the JML name for the slice */
114  @Override
115  @SuppressWarnings("nullness")
116  public String jml_name(String index) {
117
118    // The slice routine needs the actual length as opposed to the
119    // highest legal index.
120    Quantify.Term upper = get_upper_bound();
121    if (upper instanceof Quantify.Length) {
122      ((Quantify.Length) upper).set_offset(0);
123    }
124
125    if (seqvar().isPrestate()) {
126      return String.format(
127          "\\old(daikon.Quant.slice(%s, %s, %s))",
128          seqvar().enclosing_var.postState.jml_name(),
129          get_lower_bound().jml_name(true),
130          upper.jml_name(true));
131    } else {
132      return String.format(
133          "daikon.Quant.slice(%s, %s, %s)",
134          seqvar().enclosing_var.jml_name(), get_lower_bound().jml_name(), upper.jml_name());
135    }
136  }
137
138  /** Adds one to the default complexity if index_shift is not 0. */
139  @Override
140  public int complexity() {
141    return super.complexity() + ((index_shift != 0) ? 1 : 0);
142  }
143}

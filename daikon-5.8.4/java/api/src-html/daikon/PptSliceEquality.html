085  /**
086   * Encapsulates a VarInfo and its Comparability so that the two can be used to create sets of
087   * VarInfos that are initially equal. Two VarInfoAndComparability's are true iff they are
088   * VarComparability.comparable() to each other.
089   */
090  private static class VarInfoAndComparability {
091    public VarInfo vi;
092
093    @Pure
094    @Override
095    public int hashCode(@GuardSatisfied VarInfoAndComparability this) {
096      // This is very coarse but is about as good as we can do it.  Can't do hashcode of
097      // the comparability because two comparabilities may be
098      // comparable and yet be not the same.
099      // (e.g. VarComparabilityExplicit).
100      return vi.file_rep_type.hashCode();
101    }
102
103    @EnsuresNonNullIf(result = true, expression = "#1")
104    @Pure
105    @Override
106    public boolean equals(
107        @GuardSatisfied VarInfoAndComparability this, @GuardSatisfied @Nullable Object o) {
108      if (!(o instanceof VarInfoAndComparability)) {
109        return false;
110      }
111      return equalsVarInfoAndComparability((VarInfoAndComparability) o);
112    }
113
114    /**
115     * Whether two VarInfos can be set to be equal to each other is whether they are comparableNWay.
116     * Since we do not yet handle inheritance, we require that the comparability go both ways.
117     */
118    @EnsuresNonNullIf(result = true, expression = "#1")
119    @Pure
120    public boolean equalsVarInfoAndComparability(
121        @GuardSatisfied VarInfoAndComparability this, @GuardSatisfied VarInfoAndComparability o) {
122
123      return (vi.comparableNWay(o.vi)
124          && vi.comparability.equality_set_ok(o.vi.comparability)
125          && vi.aux.equals_for_instantiation(o.vi.aux));
126    }
127
128    public VarInfoAndComparability(VarInfo vi) {
129      this.vi = vi;
130    }
131  }

133  /** Actually instantiate the equality sets. */
134  @Override
135  void instantiate_invariants() {
136
137    // If each variable gets its own set, create those sets and return
138    if (dkconfig_set_per_var) {
139      // Debug.debugTrack.fine ("Vars for " + parent.name());
140      for (VarInfo vi : var_infos) {
141        List<VarInfo> vi_list = Collections.singletonList(vi);
142        Equality eq = new Equality(vi_list, this);
143        invs.add(eq);
144        // System.out.println ("  eq set = " + eq.shortString());
145      }
146      return;
147    }
148
149    // Start with everything comparable being equal.
150    if (debug.isLoggable(Level.FINE)) {
151      debug.fine("PptSliceEquality.instantiate_invariants: " + parent.name() + " vars:");
152    }
153    LinkedHashMap<VarInfoAndComparability, List<VarInfo>> multiMap = new LinkedHashMap<>();
154    for (VarInfo vi : var_infos) {
155      VarInfoAndComparability viac = new VarInfoAndComparability(vi);
156      addToBindingList(multiMap, viac, vi);
157      if (debug.isLoggable(Level.FINE)) {
158        debug.fine("  " + vi.name() + ": " + vi.comparability);
159      }
160    }
161    if (debug.isLoggable(Level.FINE)) {
162      debug.fine(
163          "PptSliceEquality.instantiate_invariants "
164              + parent.name()
165              + ": "
166              + Integer.toString(multiMap.keySet().size())
167              + " VarInfoAndComparability keys");
168    }
169    Equality[] newInvs = new Equality[multiMap.keySet().size()];
170    int varCount = 0;
171    int invCount = 0;
172    for (List<VarInfo> list : multiMap.values()) {
173      varCount += list.size();
174
175      Equality eq = new Equality(list, this);
176      newInvs[invCount] = eq;
177      if (debug.isLoggable(Level.FINE)) {
178        debug.fine(" Created: " + eq);
179        for (VarInfo vi : list) {
180          debug.fine("   vi: " + vi + " aux : " + vi.aux);
181        }
182      }
183      if (Debug.logOn()) Debug.log(getClass(), parent, Debug.vis(eq.leader()), "Created");
184      invCount++;
185    }
186    // Ensure determinism
187    Arrays.sort(newInvs, EqualityComparator.theInstance);
188    invs.addAll(Arrays.<Invariant>asList(newInvs));
189    assert varCount == var_infos.length; // Check that we get all vis
190  }

192  /**
193   * Instantiate the full equality sets from a set of variable pairs where each member of a pair is
194   * equal to the other.
195   */
196  public void instantiate_from_pairs(Set<VarInfo.Pair> eset) {
197
198    // Build a map from each variable to all those that are equal to it
199    Map<VarInfo, List<VarInfo>> varmap = new LinkedHashMap<>();
200    Map<VarInfo, Integer> sample_cnt_map = new LinkedHashMap<>();
201    for (VarInfo.Pair cp : eset) {
202      List<VarInfo> vlist = varmap.get(cp.v1);
203      if (vlist == null) {
204        vlist = new ArrayList<VarInfo>();
205        vlist.add(cp.v1);
206        varmap.put(cp.v1, vlist);
207        sample_cnt_map.put(cp.v1, cp.samples);
208      }
209      vlist.add(cp.v2);
210      vlist = varmap.get(cp.v2);
211      if (vlist == null) {
212        vlist = new ArrayList<VarInfo>();
213        vlist.add(cp.v2);
214        varmap.put(cp.v2, vlist);
215        sample_cnt_map.put(cp.v2, cp.samples);
216      }
217      vlist.add(cp.v1);
218    }
219
220    // Loop through each variable, building the appropriate equality set
221    // for each.  Note that variables that are distinct still have an
222    // equality set (albeit with only the one variable)
223    ArrayList<Invariant> newInvs = new ArrayList<>();
224    for (VarInfo v : var_infos) {
225      if (v.equalitySet != null) {
226        continue;
227      }
228      List<VarInfo> vlist = varmap.get(v);
229      if (vlist == null) {
230        vlist = new ArrayList<VarInfo>(1);
231        vlist.add(v);
232      }
233      Equality eq = new Equality(vlist, this);
234      Integer sample_cnt = sample_cnt_map.get(v);
235      if (sample_cnt != null) eq.setSamples(sample_cnt.intValue());
236      v.equalitySet = eq;
237      newInvs.add(eq);
238    }
239    invs.addAll(newInvs);
240  }

242  /**
243   * Returns a List of Invariants that have been weakened/destroyed. However, this handles the
244   * creation of new Equality invariants and the instantiation of other invariants.
245   *
246   * @return a List of invariants that have been weakened
247   */
248  // The basic approach is as follows:
249  //    - Loop through each equality set
250  //        - look for any variables that are no longer equal
251  //        - Create new equality sets (call createEqualityInvs)
252  //        - Get the new leaders
253  //        - Create new slices and invariants (call CopyInvsFromLeader)
254  //
255  @Override
256  public List<Invariant> add(ValueTuple vt, int count) {
257
258    ArrayList<Equality> allNewInvs = new ArrayList<>();
259    ArrayList<Invariant> weakenedInvs = new ArrayList<>();
260
261    // Loop through each existing equality invariant
262    for (Invariant invar : invs) {
263      Equality inv = (Equality) invar;
264
265      // Add this sample to the invariant and track any vars that fall
266      // out of the set.
267      List<VarInfo> nonEqualVis = inv.add(vt, count);
268
269      // If some vars fell out
270      if (nonEqualVis.size() > 0) {
271
272        // Create new equality sets for all of the non-equal vars
273        List<Equality> newInvs = createEqualityInvs(nonEqualVis, vt, inv, count);
274
275        // Get a list of all of the new non-missing leaders
276        List<VarInfo> newInvsLeaders = new ArrayList<>(newInvs.size());
277        for (Equality eq : newInvs) {
278          if ((parent.constants == null) || !parent.constants.is_missing(eq.leader())) {
279            newInvsLeaders.add(eq.leader());
280          }
281        }
282
283        // Debug print the new leaders
284        if (Debug.logOn()) {
285          for (VarInfo nileader : newInvsLeaders) {
286            Debug.log(
287                getClass(),
288                parent,
289                Debug.vis(nileader),
290                "Split off from previous leader "
291                    + inv.leader().name()
292                    + ": new set = "
293                    + nileader.equalitySet
294                    + ": old set = "
295                    + inv);
296          }
297        }
298
299        // Create new slices and invariants for each new leader
300        weakenedInvs.addAll(copyInvsFromLeader(inv.leader(), newInvsLeaders));
301
302        // Keep track of all of the new invariants created.
303        allNewInvs.addAll(newInvs);
304      }
305    }
306
307    // Add all of the new equality sets to our list
308    invs.addAll(allNewInvs);
309
310    return weakenedInvs;
311  }

318  /**
319   * Create a List of Equality invariants based on the values given by vt for the VarInfos in vis.
320   * Any variables that are out of bounds are forced into a separate equality set (since they no
321   * longer make sense and certainly shouldn't be equal to anything else)
322   *
323   * <p>Requires: vis.size() &gt; 0
324   *
325   * <p>Ensures: result.size() &gt; 0
326   *
327   * @param vis the VarInfos that were different from leader
328   * @param vt the ValueTuple associated with the VarInfos now
329   * @param leader the original leader of VarInfos
330   * @param count the number of samples seen (needed to set the number of samples for the new
331   *     Equality invariants)
332   * @return a List of Equality invariants bundling together same values from vis, and if needed,
333   *     another representing all the missing values
334   */
335  private List<Equality> createEqualityInvs(
336      List<VarInfo> vis, ValueTuple vt, Equality leader, int count) {
337    assert vis.size() > 0;
338    HashMap<Object, List<VarInfo>> multiMap = new HashMap<>(); /* key is a value */
339    List<VarInfo> out_of_bounds = new ArrayList<>();
340    for (VarInfo vi : vis) {
341      if (vi.missingOutOfBounds()) {
342        out_of_bounds.add(vi);
343      } else if (vt.isMissing(vi)) {
344        addToBindingList(multiMap, dummyMissing, vi);
345      } else {
346        if (vi.getValue(vt) == null) {
347          System.out.printf(
348              "null value for variable %s, mod=%d at ppt %s%n",
349              vi.name(), vt.getModified(vi), parent.name());
350          VarInfo rv = parent.find_var_by_name("return");
351          assert rv != null : "@AssumeAssertion(nullness)";
352          System.out.println("return value = " + Debug.toString(rv.getValue(vt)));
353          System.out.println("At line number " + FileIO.get_linenum());
354        }
355        addToBindingList(multiMap, vi.getValue(vt), vi);
356      }
357    }
358    // Why use an array?  Because we'll be sorting shortly
359    /*NNC:@MonotonicNonNull*/ Equality[] resultArray =
360        new Equality[multiMap.values().size() + out_of_bounds.size()];
361    int resultCount = 0;
362    for (Map.Entry<@KeyFor("multiMap") Object, List<VarInfo>> entry : multiMap.entrySet()) {
363      Object key = entry.getKey();
364      List<VarInfo> list = entry.getValue();
365      assert list.size() > 0;
366      Equality eq = new Equality(list, this);
367      @SuppressWarnings("interning") // special value
368      boolean isMissing = (key == dummyMissing);
369      if (isMissing) {
370        eq.setSamples(leader.numSamples() - count);
371      } else {
372        eq.setSamples(leader.numSamples());
373      }
374      if (debug.isLoggable(Level.FINE)) {
375        debug.fine("  created new inv: " + eq + " samples: " + eq.numSamples());
376      }
377      resultArray[resultCount] = eq;
378      resultCount++;
379    }
380    for (VarInfo oob : out_of_bounds) {
381      List<VarInfo> list = Collections.singletonList(oob);
382      resultArray[resultCount] = new Equality(list, this);
383      resultCount++;
384    }
385    resultArray = castNonNullDeep(resultArray); // https://tinyurl.com/cfissue/986
386
387    // Sort for determinism
388    Arrays.sort(resultArray, EqualityComparator.theInstance);
389    List<Equality> result = Arrays.<Equality>asList(resultArray);
390    assert result.size() > 0;
391    return result;
392  }

394  /**
395   * Create a List of Equality invariants based on the VarInfos in vis. Assumes that the VarInfos in
396   * vis are not missing. The method is used exclusively for reversing optimizations in Daikon.
397   *
398   * <p>Requires: vis.size() &gt; 0
399   *
400   * <p>Ensures: result.size() &gt; 0
401   *
402   * @param vis the VarInfos that were different from leader
403   * @param leader the original leader of VarInfos
404   * @return a List of Equality invariants bundling together same values from vis
405   */
406  public List<Equality> createEqualityInvs(List<VarInfo> vis, Equality leader) {
407    assert vis.size() > 0;
408
409    // Why use an array?  Because we'll be sorting shortly
410    /*NNC:@MonotonicNonNull*/ Equality[] resultArray = new Equality[vis.size()];
411    for (int i = 0; i < vis.size(); i++) {
412      VarInfo vi = vis.get(i);
413      List<VarInfo> list = new ArrayList<>();
414      list.add(vi);
415      Equality eq = new Equality(list, this);
416      eq.setSamples(leader.numSamples());
417      resultArray[i] = eq;
418    }
419    resultArray = castNonNullDeep(resultArray); // https://tinyurl.com/cfissue/986
420
421    // Sort for determinism
422    Arrays.sort(resultArray, PptSliceEquality.EqualityComparator.theInstance);
423    List<Equality> result = Arrays.<Equality>asList(resultArray);
424    assert result.size() > 0;
425    return result;
426  }

428  /**
429   * Map maps keys to non-empty lists of elements. This method adds var to the list mapped by key,
430   * creating a new list for key if one doesn't already exist.
431   *
432   * <p>Requires: Each value in map is a list of size 1 or greater
433   *
434   * <p>Ensures: Each value in map is a list of size 1 or greater
435   *
436   * @param map the map to add the bindings to
437   * @param key if there is already a List associated with key, then add value to key. Otherwise
438   *     create a new List associated with key and insert value.
439   * @param value the value to insert into the List mapped to key
440   */
441  private <T> void addToBindingList(Map<T, List<VarInfo>> map, T key, VarInfo value) {
442    if (key == null) {
443      throw new IllegalArgumentException();
444    }
445    List<VarInfo> elements = map.get(key);
446    if (elements == null) {
447      elements = new ArrayList<VarInfo>();
448      map.put(key, elements);
449    }
450    elements.add(value);
451  }

453  /**
454   * Instantiate invariants from each inv's leader. This is like instantiate_invariants at the start
455   * of reading the trace file, where we create new PptSliceNs. This is called when newVis have just
456   * split off from leader, and we want the leaders of newVis to have the same invariants as leader.
457   *
458   * <p>post: Adds the newly instantiated invariants and slices to this.parent.
459   *
460   * @param leader the old leader
461   * @param newVis a List of new VarInfos that used to be equal to leader. Actually, it's the list
462   *     of canonical that were equal to leader, representing their own newly-created equality sets.
463   */
464  public List<Invariant> copyInvsFromLeader(VarInfo leader, List<VarInfo> newVis) {
465
466    List<Invariant> falsified_invs = new ArrayList<>();
467    List<PptSlice> newSlices = new ArrayList<>();
468    if (debug.isLoggable(Level.FINE)) {
469      debug.fine(
470          "copyInvsFromLeader: "
471              + parent.name()
472              + ": leader "
473              + leader.name()
474              + ": new leaders = "
475              + newVis);
476      debug.fine("  orig slices count:" + parent.numViews());
477    }
478
479    // Copy all possible combinations from the current ppt (with repetition)
480    // of replacing leader with different members of newVis.
481
482    // Loop through each slice.
483    // Uses old-style for loop and Iterator because it will be side-effected.
484    for (Iterator<PptSlice> i = parent.views_iterator(); i.hasNext(); ) {
485      PptSlice slice = i.next();
486
487      if (debug.isLoggable(Level.FINE)) {
488        debug.fine("  Slice is: " + slice.toString());
489        debug.fine("  With invs: " + slice.invs);
490      }
491
492      // If this slice contains the old leader
493      if (slice.containsVar(leader)) {
494
495        // Substitute new leader for old leader and create new slices/invs
496        VarInfo[] toFill = new VarInfo[slice.var_infos.length];
497        copyInvsFromLeaderHelper(leader, newVis, slice, newSlices, 0, -1, toFill);
498
499        // Remove any statically obvious invariants in the old slice.
500        // This is called here because breaking up the equality set may
501        // cause some invariants to become statically obvious (because
502        // they will now be the only item in their set)
503        for (Invariant inv : slice.invs) {
504          if (!Daikon.dkconfig_undo_opts) {
505            if (inv.isObviousStatically_AllInEquality()) {
506              inv.falsify();
507              falsified_invs.add(inv);
508            }
509          }
510        }
511        if (slice.invs.size() == 0) i.remove();
512      }
513    }
514
515    // Add each new slice with invariants
516    for (PptSlice slice : newSlices) {
517      if (slice.invs.size() == 0) {
518        continue;
519      }
520      assert (parent.findSlice(slice.var_infos) == null) : parent.findSlice(slice.var_infos);
521      slice.repCheck();
522      parent.addSlice(slice);
523    }
524
525    parent.repCheck();
526
527    if (debug.isLoggable(Level.FINE)) {
528      debug.fine("  new slices count:" + parent.numViews());
529    }
530    return falsified_invs;
531  }

533  /**
534   * Clones slice (zero or more times) such that instances of leader are replaced by members of
535   * newVis; places new slices in newSlices. The replacement is such that we get all combinations,
536   * with repetition of newVis and leader in every slot in slice where there used to be leader. For
537   * example, if slice contained (A1, A1, B) and A1 is leader and newVis contains A2 and A3, then
538   * the slices we produce would be: (A1, A2, B), (A1, A3, B), (A2, A2, B) (A2, A3, B), (A3, A3, B).
539   * We do not produce (A1, A1, B) because it is already there. We do not produce (A2, A1, B)
540   * because it is the same as (A1, A2, B) wrt combinations. This method does the main work of
541   * copyInvsFromLeader so that each new equality set that spawned off leader has the correct
542   * slices. It works as a nested series of for loops, whose depth is equal to the length of
543   * slice.var_infos. The position and loop arguments along with the call stack keep track of the
544   * loop nesting. When position reaches the end of slice.var_infos, this method attempts to
545   * instantiate the slice that has been produced. The standard start for position is 0, and for
546   * loop is -1.
547   *
548   * @param leader the variable to replace in slice
549   * @param newVis of VarInfos that will replace leader in combination in slice
550   * @param slice the slice to clone
551   * @param newSlices where to put the cloned slices
552   * @param position the position currently being replaced in source. Starts at 0.
553   * @param loop the iteration of the loop for this position. If -1, means the previous replacement
554   *     is leader.
555   * @param soFar buffer to which assignments temporarily go before becoming instantiated. Has to
556   *     equal slice.var_infos in length.
557   */
558  private void copyInvsFromLeaderHelper(
559      VarInfo leader,
560      List<VarInfo> newVis,
561      PptSlice slice,
562      List<PptSlice> newSlices,
563      int position,
564      int loop,
565      VarInfo[] soFar) {
566
567    // Track debug if any variables are in newVis
568    Debug dlog = null;
569    if (Debug.logOn()) {
570      dlog = new Debug(getClass(), parent, newVis);
571    }
572
573    if (position >= slice.var_infos.length) {
574      // Done with assigning positions and recursion
575      if (parent.findSlice_unordered(soFar) == null) {
576        // If slice is already there, no need to clone.
577
578        if (parent.is_slice_ok(soFar, slice.arity())) {
579          PptSlice newSlice = slice.cloneAndPivot(soFar);
580          // Debug.debugTrack.fine ("LeaderHelper: Created Slice " + newSlice);
581          if (Debug.logOn()) {
582            assert dlog != null : "@AssumeAssertion(nullness): dependent: set if Debug.logOn()";
583            dlog.log(
584                "Created slice " + newSlice + " Leader equality set = " + soFar[0].equalitySet);
585            Debug.log(getClass(), newSlice, "Created this slice");
586          }
587          List<Invariant> invs = newSlice.invs;
588          for (Iterator<Invariant> iInvs = invs.iterator(); iInvs.hasNext(); ) {
589            Invariant inv = iInvs.next();
590            if (!Daikon.dkconfig_undo_opts) {
591              if (inv.isObviousStatically_AllInEquality()) {
592                iInvs.remove();
593              }
594            }
595          }
596          if (newSlice.invs.size() == 0) {
597            Debug.log(debug, getClass(), newSlice, soFar, "slice not added because 0 invs");
598          } else {
599            newSlices.add(newSlice);
600          }
601        }
602      } else {
603        if (Debug.logOn()) {
604          assert dlog != null : "@AssumeAssertion(nullness): dependent: set if Debug.logOn()";
605          dlog.log("Slice already existed " + parent.findSlice_unordered(soFar));
606        }
607      }
608      return;
609    } else {
610      // Not yet done with recursion, keep assigning to soFar
611      if (slice.var_infos[position] == leader) {
612        // If leader does need replacing
613        // newLoop starts at loop so that we don't have repeats
614        for (int newLoop = loop; newLoop < newVis.size(); newLoop++) {
615          VarInfo vi = newLoop == -1 ? leader : newVis.get(newLoop);
616          soFar[position] = vi;
617          // Advance position to next step, let next loop variable be
618          // this loop's counter.
619          copyInvsFromLeaderHelper(leader, newVis, slice, newSlices, position + 1, newLoop, soFar);
620        }
621      } else {
622        // Non leader position, just keep going after assigning soFar
623        soFar[position] = slice.var_infos[position];
624        copyInvsFromLeaderHelper(leader, newVis, slice, newSlices, position + 1, loop, soFar);
625      }
626    }
627  }

629  @Override
630  public void repCheck() {
631    for (Invariant inv : invs) {
632      inv.repCheck();
633      assert inv.ppt == this;
634    }

649  /** Order Equality invariants by the indices of leaders. */
650  public static final class EqualityComparator implements Comparator<Equality> {
651    public static final EqualityComparator theInstance = new EqualityComparator();
652
653    private EqualityComparator() {}
654
655    @Pure
656    @Override
657    public int compare(Equality eq1, Equality eq2) {
658      return VarInfo.IndexComparator.theInstance.compare(eq1.leader(), eq2.leader());
659    }

662  /** Returns an array of all of the leaders sorted by varinfo_index for this equality view. */
663  public VarInfo[] get_leaders_sorted() {
664    List<VarInfo> leaders = new ArrayList<>(invs.size());
665    for (Invariant inv : invs) {
666      VarInfo leader = ((Equality) inv).leader();
667      assert leader != null;
668      leaders.add(leader);
669    }
670    Collections.sort(leaders, VarInfo.IndexComparator.getInstance());
671    return (leaders.toArray(new VarInfo[leaders.size()]));
672  }

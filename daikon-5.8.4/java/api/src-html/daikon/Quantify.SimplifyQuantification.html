017  /** Flags describing how quantifications are to be built. */
018  public enum QuantFlags {
019    /** two indices where they refer to corresponding positions */
020    ELEMENT_WISE,
021    /** two indices where the second is one more than the first */
022    ADJACENT,
023    /** two indices are different */
024    DISTINCT,
025    /** Return the names of the index variables. */
026    INCLUDE_INDEX;
027
028    /** set with just ELEMENT_WISE turned on */
029    public static EnumSet<QuantFlags> element_wise() {
030      return EnumSet.of(QuantFlags.ELEMENT_WISE);
031    }
032
033    /** set with just ADJACENT turned on */
034    public static EnumSet<QuantFlags> adjacent() {
035      return EnumSet.of(QuantFlags.ADJACENT);
036    }
037
038    /** set with just DISTINCT turned on */
039    public static EnumSet<QuantFlags> distinct() {
040      return EnumSet.of(QuantFlags.DISTINCT);
041    }
042
043    /** set with just INCLUDE_INDEX turned on */
044    public static EnumSet<QuantFlags> include_index() {
045      return EnumSet.of(QuantFlags.INCLUDE_INDEX);
046    }
047  }
048
049  /** Returns a set with ELEMENT_WISE turned on if specified. */
050  public static EnumSet<QuantFlags> get_flags(boolean elementwise) {
051    if (elementwise) {
052      return EnumSet.of(QuantFlags.ELEMENT_WISE);
053    } else {
054      return EnumSet.noneOf(QuantFlags.class);
055    }
056  }

058  /**
059   * Class the represents terms that can be used in variable expressions. These include constants
060   * (such as 0 and 1), free variables used for quantification (i, j, etc), and normal Daikon
061   * variables.
062   */
063  public abstract static class Term {
064    @SideEffectFree
065    public abstract String name(@GuardSatisfied Term this);
066
067    @SideEffectFree
068    public String esc_name() {
069      return name();
070    }
071
072    @SideEffectFree
073    public String jml_name() {
074      return esc_name();
075    }
076
077    @SideEffectFree
078    public String jml_name(boolean in_prestate) {
079      return jml_name();
080    }
081
082    @SideEffectFree
083    public String simplify_name() {
084      return name();
085    }
086
087    @SideEffectFree
088    public String csharp_name() {
089      return name();
090    }
091
092    @SideEffectFree
093    protected static String name_with_offset(String name, int offset) {
094      if (offset == 0) {
095        return name;
096      } else {
097        return String.format("%s%+d", name, offset);
098      }
099    }

143  public static class Length extends Term {
144    VarInfo sequence;
145    int offset;
146
147    public Length(VarInfo sequence, int offset) {
148      this.sequence = sequence;
149      this.offset = offset;
150    }
151
152    @SideEffectFree
153    @Override
154    public String toString(@GuardSatisfied Length this) {
155      return name();
156    }
157
158    @SideEffectFree
159    @Override

162    }
163
164    @SideEffectFree
165    @Override
166    public String esc_name() {
167      VarInfo arr_var = get_check_array_var("ESC");
168      if (arr_var.isPrestate()) {
169        assert arr_var.postState != null; // because isPrestate() = true
170        return String.format(
171            "\\old(%s)", name_with_offset(arr_var.postState.esc_name() + ".length", offset));
172      } else { // array is not orig
173        return name_with_offset(arr_var.esc_name() + ".length", offset);
174      }
175    }
176
177    @SideEffectFree
178    @Override
179    public String jml_name() {
180      VarInfo arr_var = get_check_array_var("JML");
181      if (arr_var.isPrestate()) {
182        assert arr_var.postState != null; // because isPrestate() = true
183        String name = String.format("daikon.Quant.size(%s)", arr_var.postState.jml_name());
184        return name_with_offset(String.format("\\old(%s)", name), offset);
185        // return String.format ("\\old(%s)", name_with_offset (name, offset));
186      } else {
187        String name = String.format("daikon.Quant.size(%s)", arr_var.jml_name());
188        return name_with_offset(name, offset);
189      }
190    }
191
192    @SideEffectFree
193    @Override
194    public String jml_name(boolean in_prestate) {
195      if (!in_prestate) {
196        return jml_name();
197      }
198
199      VarInfo arr_var = get_check_array_var("JML");
200      if (arr_var.isPrestate()) {
201        assert arr_var.postState != null; // because isPrestate() = true
202        String name = String.format("daikon.Quant.size(%s)", arr_var.postState.jml_name());
203        return name_with_offset(name, offset);
204      } else {
205        String name = String.format("daikon.Quant.size(\\new(%s))", arr_var.jml_name());
206        return name_with_offset(name, offset);
207      }
208    }
209
210    @SideEffectFree
211    @Override
212    public String simplify_name() {
213      VarInfo arr_var = get_check_array_var("Simplify");
214      String length = String.format("(arrayLength %s)", arr_var.simplify_name());
215      if (offset < 0) {
216        return String.format("(- %s %d)", length, -offset);
217      } else if (offset > 0) {
218        return String.format("(+ %s %d)", length, offset);
219      } else {
220        return length;
221      }
222    }
223
224    @SideEffectFree
225    @Override
226    public String csharp_name() {
227      VarInfo arr_var = get_check_array_var("CHARPCONTRACT");
228      return name_with_offset(arr_var.csharp_name() + ".Count()", offset);
229    }

231    public void set_offset(int offset) {

233    }

235    /**
236     * Looks up the array variable which is the base of this array. Throws an exception if one does
237     * not exist.
238     */
239    @SuppressWarnings("all:sideeffectfree") // throws exception in case of error
240    @SideEffectFree
241    private VarInfo get_check_array_var(String output_format) {
242      VarInfo arr_var = sequence.get_base_array_hashcode();
243      if (arr_var != null) {
244        return arr_var;
245      }
246
247      throw new Daikon.UserError(
248          String.format(
249              "Error: Can't create %s expression for the size of an array: "
250                  + "No base array (hashcode) variable declared for array '%s'"
251                  + " in program point %s",
252              output_format, sequence.name(), sequence.ppt.name()));
253    }

330  /**
331   * Given a list of sequences, determines a free variable that can be used as a subscript for each
332   * sequence. If any of the vars are not sequences, no index is calculated for them.
333   */
334  public static QuantifyReturn[] quantify(VarInfo[] vars) {
335    assert vars != null;
336
337    // create empty result
338    QuantifyReturn[] result = new QuantifyReturn[vars.length];
339    for (int ii = 0; ii < vars.length; ii++) {
340      result[ii] = new QuantifyReturn(vars[ii]);
341    }
342
343    // Determine all of the simple identifiers used by the given variables (vars)
344    Set<String> simples = new HashSet<>();
345    for (VarInfo vi : vars) {
346      for (String name : vi.get_all_simple_names()) {
347        simples.add(name);
348      }
349    }
350    // System.out.printf("simple names = %s%n", simples);
351
352    // Loop through each of the variables, choosing an index for each
353    char tmp = 'i';
354    for (int ii = 0; ii < vars.length; ii++) {
355      VarInfo vi = vars[ii];
356
357      // If this variable is not an array, there is not much to do
358      if (!vi.file_rep_type.isArray()) {
359        continue;
360      }
361
362      // Get a unique free variable name
363      String idx_name;
364      do {
365        idx_name = String.valueOf(tmp++);
366      } while (simples.contains(idx_name));
367      assert tmp <= 'z' : "Ran out of letters in quantification";
368      result[ii].index = new FreeVar(idx_name);
369    }
370    return result;
371  }

373  /** Class that represents an ESC quantification over one or two variables. */
374  public static class ESCQuantification {
375
376    private EnumSet<QuantFlags> flags;
377    // private VarInfo[] vars;
378    private VarInfo[] arr_vars;
379    private String[] arr_vars_indexed;
380    private String[] quants;
381    private String quant;
382    private Term[] indices;
383
384    public ESCQuantification(EnumSet<QuantFlags> flags, VarInfo... vars) {
385      this.flags = flags.clone();
386
387      assert vars != null;
388      assert (vars.length == 1) || (vars.length == 2) : vars.length;
389      assert vars[0].file_rep_type.isArray();
390
391      // quantification for first var
392      Term index1 = new FreeVar("i");
393      String quant1 = bld_quant(vars[0], index1);
394      VarInfo arr_var1 = vars[0].get_array_var();
395      String arr_var1_index = arr_var1.esc_name(index1.esc_name());
396
397      // If there is a second array variable, get quant for it
398      if ((vars.length > 1) && vars[1].file_rep_type.isArray()) {
399        Term index2 = new FreeVar("j");
400        String quant2 = bld_quant(vars[1], index2);
401        indices = new Term[] {index1, index2};
402        quants = new String[] {quant1, quant2};
403        if (flags.contains(QuantFlags.ELEMENT_WISE)) {
404          quant =
405              String.format(
406                  "(\\forall int %s, %s; (%s && %s && %s == %s)",
407                  index1.esc_name(),
408                  index2.esc_name(),
409                  quant1,
410                  quant2,
411                  index1.esc_name(),
412                  index2.esc_name());
413        } else {
414          quant =
415              String.format(
416                  "(\\forall int %s, %s; (%s && %s)",
417                  index1.esc_name(), index2.esc_name(), quant1, quant2);
418        }
419
420        VarInfo arr_var2 = vars[1].get_array_var();
421        arr_vars = new VarInfo[] {arr_var1, arr_var2};
422        String arr_var2_index = arr_var2.esc_name(index2.esc_name());
423        arr_vars_indexed = new String[] {arr_var1_index, arr_var2_index};
424      } else { // only one array variable
425        indices = new Term[] {index1};
426        quants = new String[] {quant1};
427        quant = String.format("(\\forall int %s; (%s)", index1.esc_name(), quant1);
428        arr_vars = new VarInfo[] {arr_var1};
429        arr_vars_indexed = new String[] {arr_var1_index};
430      }
431    }
432
433    /**
434     * Returns a string quantification expression for the array variable var using index. The
435     * expression is of the form
436     *
437     * <pre>{@code lower_bound <= index && index <= upper_bound}</pre>
438     */
439    private static String bld_quant(VarInfo var, Term index) {
440      return String.format(
441          "%s <= %s && %s <= %s",
442          var.get_lower_bound().esc_name(),
443          index.esc_name(),
444          index.esc_name(),
445          var.get_upper_bound().esc_name());
446    }
447
448    /**
449     * Returns the quantification string. For example, if there is one array variable (a[]) that is
450     * not a slice, it will return
451     *
452     * <pre>{@code
453     * '(\forall int i; (0 <= i <= size(a[]) ==> '
454     * }</pre>
455     */
456    public String get_quantification() {
457      return quant + " ==> ";
458    }
459
460    /**
461     * Returns the specified array variable indexed by its index. For example, if the array variable
462     * is 'a.b[]' and the index is 'i', returns a.b[i].
463     */
464    public String get_arr_vars_indexed(int num) {
465      return arr_vars_indexed[num];
466    }
467  }

469  /** Class that represents an Simplify quantification over one or two variables. */
470  public static class SimplifyQuantification {
471
472    EnumSet<QuantFlags> flags;
473    String quantification;
474    String[] arr_vars_indexed;
475    @Nullable String[] indices;
476
477    public SimplifyQuantification(EnumSet<QuantFlags> flags, VarInfo... vars) {
478      this.flags = flags.clone();
479
480      assert vars != null;
481      assert (vars.length == 1) || (vars.length == 2) : vars.length;
482      assert vars[0].file_rep_type.isArray();
483
484      if (flags.contains(QuantFlags.ADJACENT) || flags.contains(QuantFlags.DISTINCT)) {
485        assert vars.length == 2;
486      }
487
488      QuantifyReturn[] qrets = quantify(vars);
489
490      // build the forall predicate
491      StringBuilder int_list, conditions;
492      {
493        // "i j ..."
494        int_list = new StringBuilder();
495        // "(AND (<= ai i) (<= i bi) (<= aj j) (<= j bj) ...)"
496        // if elementwise, also insert "(EQ (- i ai) (- j aj)) ..."
497        conditions = new StringBuilder();
498        for (int i = 0; i < qrets.length; i++) {
499          Term idx = qrets[i].index;
500          if (idx == null) {
501            continue;
502          }
503          VarInfo vi = qrets[i].var;
504          Term low = vi.get_lower_bound();
505          Term high = vi.get_upper_bound();
506          if (i != 0) {
507            int_list.append(" ");
508            conditions.append(" ");
509          }
510          int_list.append(idx.simplify_name());
511          conditions.append("(<= " + low.simplify_name() + " " + idx.simplify_name() + ")");
512          conditions.append(" (<= " + idx.simplify_name() + " " + high.simplify_name() + ")");
513          if (flags.contains(QuantFlags.ELEMENT_WISE) && (i >= 1)) {
514            // Term[] _boundv = qret.bound_vars.get(i-1);
515            // Term _idx = _boundv[0], _low = _boundv[1];
516            @SuppressWarnings(
517                "nullness") // if variable is a sequence (is it?), then index is non-null
518            @NonNull Term _idx = qrets[i - 1].index;
519            Term _low = qrets[i - 1].var.get_lower_bound();
520            if (_low.simplify_name().equals(low.simplify_name())) {
521              conditions.append(" (EQ " + _idx.simplify_name() + " " + idx.simplify_name() + ")");
522            } else {
523              conditions.append(
524                  " (EQ (- " + _idx.simplify_name() + " " + _low.simplify_name() + ")");
525              conditions.append(" (- " + idx.simplify_name() + " " + low.simplify_name() + "))");
526            }
527          }
528          if (i == 1
529              && (flags.contains(QuantFlags.ADJACENT) || flags.contains(QuantFlags.DISTINCT))) {
530            // Term[] _boundv = qret.bound_vars.get(i-1);
531            // Term prev_idx = _boundv[0];
532            @SuppressWarnings(
533                "nullness") // if variable is a sequence (is it?), then index is non-null
534            @NonNull Term prev_idx = qrets[i - 1].index;
535            if (flags.contains(QuantFlags.ADJACENT)) {
536              conditions.append(
537                  " (EQ (+ " + prev_idx.simplify_name() + " 1) " + idx.simplify_name() + ")");
538            }
539            if (flags.contains(QuantFlags.DISTINCT)) {
540              conditions.append(
541                  " (NEQ " + prev_idx.simplify_name() + " " + idx.simplify_name() + ")");
542            }
543          }
544        }
545      }
546      quantification = "(FORALL (" + int_list + ") (IMPLIES (AND " + conditions + ") ";
547
548      // stringify the terms
549      List<String> avi_list = new ArrayList<>(vars.length);
550      for (QuantifyReturn qret : qrets) {
551        String arr_var_indexed;
552        if (qret.index != null) {
553          Term index = qret.index;
554          VarInfo arr_var = qret.var.get_array_var();
555          arr_var_indexed = arr_var.simplify_name(index.simplify_name());
556          // System.out.printf("vi = %s, arr_var = %s%n", vi, arr_var);
557        } else {
558          arr_var_indexed = qret.var.simplify_name();
559        }
560        avi_list.add(arr_var_indexed);
561        // result[i + 1] = qret.root_primes[i].simplify_name();
562      }
563      arr_vars_indexed = avi_list.toArray(new String[avi_list.size()]);
564
565      // stringify the indices,
566      // note that the index should be relative to the slice, not relative
567      // to the original array (we used to get this wrong)
568      indices = new @Nullable String[vars.length];
569      for (int i = 0; i < qrets.length; i++) {
570        // Term[] boundv = qret.bound_vars.get(i);
571        // Term idx_var = boundv[0];
572        QuantifyReturn qret = qrets[i];
573        if (qret.index == null) {
574          continue;
575        }
576        String idx_var_name = qret.index.simplify_name();
577        String lower_bound = qret.var.get_lower_bound().simplify_name();
578        String idx_expr = "(- " + idx_var_name + " " + lower_bound + ")";
579        indices[i] = idx_expr;
580      }
581    }
582
583    /** Returns the quantification string that quantifies over each of the free variables. */
584    public String get_quantification() {
585      return quantification;
586    }
587
588    /**
589     * Returns the specified array variable indexed by its index. For example, if the array variable
590     * is 'a[]' and the index is 'i', returns 'select i a'.
591     */
592    public String get_arr_vars_indexed(int num) {
593      return arr_vars_indexed[num];
594    }
595
596    /** Returns the specified index. */
597    @SuppressWarnings("nullness:return.type.incompatible") // possible application invariant?
598    public String get_index(int num) {
599      assert indices[num] != null; // will this assertion fail?
600      return indices[num];
601    }
602
603    /** Returns the string to be appended to the end of the quantification. */
604    public String get_closer() {
605      return "))"; // close IMPLIES, FORALL
606    }

019/**
020 * AnnotateNullable reads a Daikon invariant file and determines which reference variables have seen
021 * any null values. It writes to standard out an <a
022 * href="https://checkerframework.org/annotation-file-utilities/annotation-file-format.html">annotation
023 * file</a> with those variables. It determines which variables have seen null values by looking at
024 * the NonZero invariant. If that invariant is NOT present, then the variable must have been null at
025 * least once.
026 *
027 * <p>Since only the NonZero invariant is used, Daikon processing time can be significantly reduced
028 * by turning off derived variables and all invariants other than daikon.inv.unary.scalar.NonZero.
029 * This is not necessary, however, for correct operation. File {@code annotate_nullable.config} in
030 * the distribution does this.
031 */
032public class AnnotateNullable {
033
034  // Why is this variable static?
035  static PptMap ppts = new PptMap(); // dummy value, to satisfy Nullness Checker
036
037  // static SimpleLog verbose = new SimpleLog(/*enabled=*/ false);
038  // static SimpleLog debug = new SimpleLog(/*enabled=*/ false);
039
040  /** Map from a class name to the list of static functions for that class. */
041  static Map<String, List<PptTopLevel>> class_map = new LinkedHashMap<>();
042
043  // The package for the previous class.  Used to reduce duplication in
044  // output file.
045  static @Interned String last_package = "";
046
047  /**
048   * Write an output file in the stub class format (see the Checker Framework Manual), instead of in
049   * annotation file format.
050   */
051  @Option("Use the stub class file format")
052  public static boolean stub_format = false;
053
054  /**
055   * Adds NonNull annotations as well as Nullable annotations. Unlike Nullable annotations, NonNull
056   * annotations are not necessarily correct.
057   */
058  @Option("-n Insert NonNull as well as Nullable annotations")
059  public static boolean nonnull_annotations = false;
060
061  public static void main(String[] args) throws IOException {
062
063    Options options =
064        new Options("plume.AnnotateNullable [options] <inv_file>", AnnotateNullable.class);
065    String[] inv_files = options.parse(true, args);
066    assert inv_files.length == 1;
067
068    // Read the serialized invariant file
069    File inv_file = new File(inv_files[0]);
070    ppts = FileIO.read_serialized_pptmap(inv_file, true);
071    Daikon.all_ppts = ppts;
072    // verbose.log("Finished reading %d program points", ppts.size());
073
074    // Setup the list of proto invariants and initialize NIS suppressions
075    Daikon.setup_proto_invs();
076    Daikon.setup_NISuppression();
077
078    // Write out the definitions of our annotations
079    if (stub_format) {
080      System.out.println("import org.checkerframework.checker.nullness.qual.Nullable;");
081      System.out.println("import org.checkerframework.checker.nullness.qual.NonNull;");
082      System.out.println();
083    } else {
084      System.out.println("package org.checkerframework.checker.nullness.qual:");
085      System.out.println("annotation @Nullable:");
086      System.out.println("annotation @NonNull:");
087      System.out.println();
088    }
089
090    // Find all exit ppts that do not have a parent and determine what
091    // class they are associated with.  These are static methods for classes
092    // without any static variables (no class ppt is created if there are no
093    // static variables)
094
095    // First find all of the classes
096    for (PptTopLevel ppt : ppts.pptIterable()) {
097      if (ppt.is_object()) {
098        String classname = ppt.name().replace(":::OBJECT", "");
099        assert !class_map.containsKey(classname) : classname;
100        List<PptTopLevel> static_methods = new ArrayList<>();
101        class_map.put(classname, static_methods);
102      }
103    }
104
105    // Then, add combined exit points for static methods to their class.  A
106    // static method can be identified because it will not have the OBJECT
107    // point as a parent.
108    for (PptTopLevel ppt : ppts.pptIterable()) {
109      if (!ppt.is_combined_exit() || !is_static_method(ppt)) {
110        continue;
111      }
112
113      String name = ppt.name().replaceFirst("[(].*$", "");
114      int lastdot = name.lastIndexOf('.');
115      @SuppressWarnings("keyfor") // application invariant:  KeyFor and substring
116      // @KeyFor because class_map has entry per class, and this method is in some class
117      @KeyFor("class_map") String classname = name.substring(0, lastdot);
118      // System.out.printf("classname for ppt %s is '%s'%n", name, classname);
119      @NonNull List<PptTopLevel> static_methods = class_map.get(classname);
120      assert static_methods != null : classname;
121      static_methods.add(ppt);
122    }
123
124    // Debug print all of the static methods
125    if (false) {
126      for (String classname : class_map.keySet()) {
127        System.out.printf("class %s static methods: %s%n", classname, class_map.get(classname));
128      }
129    }
130
131    // Make sure that the static methods found by inference, match those
132    // found for any class ppts
133    for (PptTopLevel ppt : ppts.pptIterable()) {
134      if (ppt.is_class()) {
135        @SuppressWarnings(
136            "nullness") // map: retrieve class name from class Ppt name, with string manipulation
137        @NonNull List<PptTopLevel> static_methods = class_map.get(ppt.name().replace(":::CLASS", ""));
138        int child_cnt = 0;
139        // TODO: Once Checker Framework issue 565 has been fixed (https://tinyurl.com/cfissue/565),
140        // change the following two lines back to
141        // for (PptRelation child_rel : ppt.children) {
142        for (int i = 0; i < ppt.children.size(); i++) {
143          PptRelation child_rel = ppt.children.get(i);
144          PptTopLevel child = child_rel.child;
145          // Skip enter ppts, all of the info is at the exit.
146          if ((child.type == PptType.ENTER) || (child.type == PptType.OBJECT)) {
147            continue;
148          }
149          child_cnt++;
150          assert static_methods.contains(child) : child;
151        }
152        assert child_cnt == static_methods.size() : static_methods;
153      }
154    }
155
156    // Process each class.
157    for (PptTopLevel ppt : ppts.pptIterable()) {
158
159      // Skip synthetic program points
160      if (ppt.name().startsWith("$")) {
161        continue;
162      }
163
164      // Skip program points that are not OBJECT ppts
165      if (ppt.is_object()) {
166        process_class(ppt);
167      }
168    }
169  }
170
171  // Returns null if no corresponding class ppt exists
172  private static @Nullable PptTopLevel class_for_object(PptTopLevel object_ppt) {
173    if (object_ppt.parents.size() == 0) {
174      return null;
175    }
176    assert object_ppt.parents.size() == 1 : object_ppt;
177    return object_ppt.parents.get(0).parent;
178  }

180  // Process a class, including all its methods.
181  // Takes the object program point as its argument.
182  public static void process_class(PptTopLevel object_ppt) {
183
184    // Get the class program point (if any)
185    PptTopLevel class_ppt = class_for_object(object_ppt);
186
187    String class_samples = "-";
188    if (class_ppt != null) {
189      class_samples = String.format("%d", class_ppt.num_samples());
190    }
191    String ppt_package = object_ppt.ppt_name.getPackageName();
192    if (ppt_package == null) {
193      ppt_package = "";
194    } else {
195      ppt_package = ppt_package.intern();
196    }
197    if (stub_format) {
198      if (ppt_package != last_package) {
199        // This will print the empty string if we switch from a package to the
200        // unnamed package.  That is intentional.
201        System.out.printf("package %s;%n%n", ppt_package);
202        last_package = ppt_package;
203      }
204      System.out.printf(
205          "class %s { // %d/%s obj/class samples%n",
206          object_ppt.ppt_name.getFullClassName(), object_ppt.num_samples(), class_samples);
207    } else {
208      System.out.printf("package %s:%n", ppt_package);
209      System.out.printf(
210          "class %s : // %d/%s obj/class samples%n",
211          object_ppt.ppt_name.getShortClassName(), object_ppt.num_samples(), class_samples);
212    }
213
214    // Process static fields
215    if (class_ppt != null) {
216      process_obj_fields(class_ppt);
217    }
218
219    // Process member (non-static) fields
220    process_obj_fields(object_ppt);
221
222    // Process static methods
223    if (class_ppt != null) {
224      for (PptRelation child_rel : class_ppt.children) {
225        PptTopLevel child = child_rel.child;
226        // Skip enter ppts, all of the info is at the exit.
227        if ((child.type == PptType.ENTER) || (child.type == PptType.OBJECT)) {
228          continue;
229        }
230        // debug.log("processing static method %s, type %s", child, child.type);
231        process_method(child);
232      }
233    } else {
234      String classname = object_ppt.ppt_name.getFullClassName();
235      assert classname != null;
236      @SuppressWarnings("nullness") // map: class_map has entry per classname
237      @NonNull List<PptTopLevel> static_methods = class_map.get(classname);
238      assert static_methods != null : classname;
239      for (PptTopLevel child : static_methods) {
240        process_method(child);
241      }
242    }
243
244    // Process member (non-static) methods
245    for (PptRelation child_rel : object_ppt.children) {
246      PptTopLevel child = child_rel.child;
247      // Skip enter ppts, all of the info is at the exit.
248      if (child.type == PptType.ENTER) {
249        continue;
250      }
251      // debug.log("processing method %s, type %s", child, child.type);
252      process_method(child);
253    }
254
255    if (stub_format) {
256      System.out.printf("}");
257    }
258    System.out.println();
259    System.out.println();
260  }

262  /**
263   * Get the annotation for the specified variable. Returns @Nullable if samples were found for this
264   * variable and at least one sample contained a null value. Returns an (interned) empty string if
265   * no annotation is applicable. Otherwise, the return value contains a trailing space.
266   */
267  public static String get_annotation(PptTopLevel ppt, VarInfo vi) {
268
269    if (vi.type.isPrimitive()) {
270      return "";
271    }
272
273    String annotation = (nonnull_annotations ? "NonNull" : "");
274    if ((ppt.num_samples(vi) > 0) && !ppt.is_nonzero(vi)) {
275      annotation = "Nullable";
276    }
277    if (annotation != "") { // interned
278      // if (! stub_format) {
279      //   annotation = "org.checkerframework.checker.nullness.qual." + annotation;
280      // }
281      annotation = "@" + annotation;
282    }
283    return annotation;
284  }

286  /** Print out the annotations for the specified method. */
287  public static void process_method(PptTopLevel ppt) {
288
289    assert ppt.type == PptType.EXIT : ppt;
290
291    // Get all of the parameters to the method and the return value
292    List<VarInfo> params = new ArrayList<>();
293    VarInfo retvar = null;
294    for (VarInfo vi : ppt.var_infos) {
295      if (vi.var_kind == VarInfo.VarKind.RETURN) {
296        retvar = vi;
297      } else {
298        if (vi.isParam()
299            && (vi.name() != "this") // interned
300            && !vi.isPrestate()) {
301          params.add(vi);
302        }
303      }
304    }
305
306    // The formatted annotation for the return value with a leading space, or empty string
307    String return_annotation = (retvar == null ? "" : " " + get_annotation(ppt, retvar));
308
309    // Look up the annotation for each parameter.
310    List<String> names = new ArrayList<>();
311    List<String> annos = new ArrayList<>();
312    for (VarInfo param : params) {
313      String annotation = "";
314      names.add(param.name());
315      if (param.file_rep_type.isHashcode()) {
316        annotation = get_annotation(ppt, param);
317      }
318      annos.add(annotation);
319    }
320
321    // Print out the method declaration
322    if (stub_format) {
323      System.out.printf(" %s %s(", return_annotation, ppt.ppt_name.getMethodName());
324      for (int i = 0; i < params.size(); i++) {
325        if (i != 0) System.out.printf(" ,");
326        System.out.printf("%s %s %s", annos.get(i), "type-goes-here", names.get(i));
327      }
328      System.out.printf("); // %d samples%n", ppt.num_samples());
329    } else {
330      System.out.printf("  method %s : // %d samples%n", jvm_signature(ppt), ppt.num_samples());
331      System.out.printf("    return:%s%n", return_annotation);
332      for (int i = 0; i < params.size(); i++) {
333        // Print the annotation for this parameter
334        System.out.printf("    parameter #%d : %s // %s%n", i, annos.get(i), names.get(i));
335      }
336    }
337  }

339  /** Print out the annotations for each field in the object or class. */
340  public static void process_obj_fields(PptTopLevel ppt) {
341
342    for (VarInfo vi : ppt.var_infos) {
343
344      assert (!vi.isPrestate()) : vi;
345
346      // Skip anyone with a parent in the hierarchy.  We are only
347      // interested in them at the top (e.g., we don't want to see
348      // object fields in each method).
349      if (!vi.parents.isEmpty()) {
350        continue;
351      }
352
353      // Skip variables that are always non-null.
354      if (vi.aux.isNonNull()) {
355        continue;
356      }
357
358      // Skip any variable that is enclosed by a variable other than 'this'.
359      // These are fields and can only be annotated where they are declared.
360      VarInfo evar = vi.get_enclosing_var();
361      if ((evar != null) && !evar.name().equals("this")) {
362        // System.out.printf("  enclosed %s %s%n", vi.type, vi.name());
363        continue;
364      }
365
366      // print out the entry for this field
367      String annotation = "";
368      if (vi.file_rep_type.isHashcode()) {
369        annotation = get_annotation(ppt, vi);
370      }
371      if (stub_format) {
372        System.out.printf("  field %s %s {} // %s%n", field_name(vi), annotation, vi.type);
373      } else {
374        System.out.printf("  field %s : %s // %s%n", field_name(vi), annotation, vi.type);
375      }
376    }
377  }

379  /** Returns a JVM signature for the method. */
380  public static String jvm_signature(PptTopLevel ppt) {
381
382    @SuppressWarnings("nullness") // Java method, so getMethodName() != null
383    @NonNull String method = ppt.ppt_name.getMethodName();
384    @SuppressWarnings("nullness") // Java method, so getSignature() != null
385    @NonNull String java_sig = ppt.ppt_name.getSignature();
386    String java_args = java_sig.replace(method, "");
387    // System.out.printf("m/s/a = %s %s %s%n", method, java_sig, java_args);
388    if (method.equals(ppt.ppt_name.getShortClassName())) {
389      method = "<init>";
390    }
391
392    // Problem:  I need the return type, but Chicory does not output it.
393    // So, I could try to retrieve it from the "return" variable in the
394    // program point (which is, fortunately, always an exit point), or
395    // change Chicory to output it.
396    VarInfo returnVar = ppt.find_var_by_name("return");
397    @SuppressWarnings(
398        "signature" // application invariant: returnVar.type.toString() is a binary name (if
399    // returnVar is non-null), because we are processing a Java program
400    )
401    String returnType =
402        returnVar == null ? "V" : Signatures.binaryNameToFieldDescriptor(returnVar.type.toString());
403
404    return method + Signatures.arglistToJvm(java_args) + returnType;
405  }

407  /**
408   * Returns the field name of the specified variable. This is the relative name for instance
409   * fields, but the relative name is not specified for static fields (because there is no enclosing
410   * variable with the full name). The field name is obtained in that case, by removing the
411   * package/class specifier.
412   */
413  public static String field_name(VarInfo vi) {
414
415    if (vi.relative_name != null) {
416      return vi.relative_name;
417    }
418
419    String field_name = vi.name();
420    int pt = field_name.lastIndexOf('.');
421    if (pt == -1) {
422      return field_name;
423    } else {
424      return field_name.substring(pt + 1);
425    }
426  }

428  /**
429   * Returns whether or not the method of the specified ppt is static or not. The ppt must be an
430   * exit ppt. Exit ppts that do not have an object as a parent are inferred to be static. This does
431   * not work for enter ppts, because constructors do not have the object as a parent on entry.
432   */
433  @Pure
434  public static boolean is_static_method(PptTopLevel ppt) {
435
436    assert ppt.is_exit() : ppt;
437    for (PptRelation rel : ppt.parents) {
438      if (rel.parent.is_object()) {
439        return false;
440      }
441    }
442
443    return true;
444  }

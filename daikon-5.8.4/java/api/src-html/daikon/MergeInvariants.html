004import daikon.suppress.NIS;
005import gnu.getopt.*;
006import java.io.File;
007import java.io.FileNotFoundException;
008import java.io.IOException;
009import java.io.OptionalDataException;
010import java.io.StreamCorruptedException;
011import java.util.ArrayList;
012import java.util.List;
013import java.util.Set;
014import java.util.TreeSet;
015import java.util.concurrent.TimeUnit;
016import java.util.logging.Level;
017import java.util.logging.Logger;
018import org.checkerframework.checker.nullness.qual.Nullable;
019import org.plumelib.util.UtilPlume;

021/**
022 * Merges invariants from multiple invariant files into a single invariant file. It does this by
023 * forming a hierarchy over the ppts from each invariant file and using the normal hierarchy merging
024 * code to merge the invariants.
025 *
026 * <p>The ppts from each invariant file are merged to create a single ppt map that contains the ppts
027 * from all of the files. At each leaf of the merged map, a hierarchy is formed to the ppts from
028 * each of the input files.
029 */
030public final class MergeInvariants {
031  private MergeInvariants() {
032    throw new Error("do not instantiate");
033  }
034
035  public static final Logger debug = Logger.getLogger("daikon.MergeInvariants");
036
037  public static final Logger debugProgress = Logger.getLogger("daikon.MergeInvariants.progress");
038
039  public static @Nullable File output_inv_file;
040
041  private static String usage =
042      UtilPlume.joinLines(
043          "Usage: java daikon.MergeInvariants [OPTION]... FILE",
044          "  -h, --" + Daikon.help_SWITCH,
045          "      Display this usage message",
046          "  --" + Daikon.config_option_SWITCH,
047          "      Specify a configuration option ",
048          "  --" + Daikon.debug_SWITCH,
049          "      Specify a logger to enable",
050          "  --" + Daikon.track_SWITCH,
051          "      Specify a class, varinfos, and ppt to debug track.  Format"
052              + "is class<var1,var2,var3>@ppt",
053          "   -o ",
054          "      Specify an output inv file.  If not specified, the results are printed");
055
056  public static void main(final String[] args)
057      throws FileNotFoundException, StreamCorruptedException, OptionalDataException, IOException,
058          ClassNotFoundException {
059    try {
060      mainHelper(args);
061    } catch (Daikon.DaikonTerminationException e) {
062      Daikon.handleDaikonTerminationException(e);
063    }
064  }
065
066  /**
067   * This does the work of {@link #main(String[])}, but it never calls System.exit, so it is
068   * appropriate to be called progrmmatically.
069   */
070  @SuppressWarnings("nullness:contracts.precondition.not.satisfied") // private field
071  public static void mainHelper(String[] args)
072      throws FileNotFoundException, StreamCorruptedException, OptionalDataException, IOException,
073          ClassNotFoundException {
074
075    daikon.LogHelper.setupLogs(daikon.LogHelper.INFO);
076
077    LongOpt[] longopts =
078        new LongOpt[] {
079          new LongOpt(Daikon.help_SWITCH, LongOpt.NO_ARGUMENT, null, 0),
080          new LongOpt(Daikon.config_option_SWITCH, LongOpt.REQUIRED_ARGUMENT, null, 0),
081          new LongOpt(Daikon.debugAll_SWITCH, LongOpt.NO_ARGUMENT, null, 0),
082          new LongOpt(Daikon.debug_SWITCH, LongOpt.REQUIRED_ARGUMENT, null, 0),
083          new LongOpt(Daikon.track_SWITCH, LongOpt.REQUIRED_ARGUMENT, null, 0),
084        };
085
086    Getopt g = new Getopt("daikon.MergeInvariants", args, "ho:", longopts);
087    int c;
088    while ((c = g.getopt()) != -1) {
089      switch (c) {
090
091          // long option
092        case 0:
093          String option_name = longopts[g.getLongind()].getName();
094          if (Daikon.help_SWITCH.equals(option_name)) {
095            System.out.println(usage);
096            throw new Daikon.NormalTermination();
097          } else if (Daikon.config_option_SWITCH.equals(option_name)) {
098            String item = Daikon.getOptarg(g);
099            daikon.config.Configuration.getInstance().apply(item);
100            break;
101
102          } else if (Daikon.debugAll_SWITCH.equals(option_name)) {
103            Global.debugAll = true;
104
105          } else if (Daikon.debug_SWITCH.equals(option_name)) {
106            LogHelper.setLevel(Daikon.getOptarg(g), LogHelper.FINE);
107          } else if (Daikon.track_SWITCH.equals(option_name)) {
108            LogHelper.setLevel("daikon.Debug", LogHelper.FINE);
109            String error = Debug.add_track(Daikon.getOptarg(g));
110            if (error != null) {
111              throw new Daikon.UserError(
112                  "Error parsing track argument '" + Daikon.getOptarg(g) + "' - " + error);
113            }
114          } else {
115            throw new Daikon.UserError("Unknown long option received: " + option_name);
116          }
117          break;
118
119        case 'h':
120          System.out.println(usage);
121          throw new Daikon.NormalTermination();
122
123        case 'o':
124          String output_inv_filename = Daikon.getOptarg(g);
125
126          if (output_inv_file != null) {
127            throw new Daikon.UserError(
128                "multiple serialization output files supplied on command line: "
129                    + output_inv_file
130                    + " "
131                    + output_inv_filename);
132          }
133
134          output_inv_file = new File(output_inv_filename);
135
136          if (!UtilPlume.canCreateAndWrite(output_inv_file)) {
137            throw new Daikon.UserError(
138                "Cannot write to serialization output file " + output_inv_file);
139          }
140          break;
141
142        case '?':
143          break; // getopt() already printed an error
144
145        default:
146          System.out.println("getopt() returned " + c);
147          break;
148      }
149    }
150
151    daikon.LogHelper.setupLogs(Global.debugAll ? LogHelper.FINE : LogHelper.INFO);
152
153    List<File> inv_files = new ArrayList<>();
154    File decl_file = null;
155    Set<File> splitter_files = new TreeSet<>();
156
157    // Get each file specified
158    for (int i = g.getOptind(); i < args.length; i++) {
159      File file = new File(args[i]);
160      if (!file.exists()) {
161        throw new Daikon.UserError("File " + file + " not found.");
162      }
163      if (file.toString().indexOf(".inv") != -1) {
164        inv_files.add(file);
165      } else if (file.toString().indexOf(".decls") != -1) {
166        if (decl_file != null) {
167          throw new Daikon.UserError("Only one decl file may be specified");
168        }
169        decl_file = file;
170      } else if (file.toString().indexOf(".spinfo") != -1) {
171        splitter_files.add(file);
172      } else {
173        throw new Daikon.UserError("Unrecognized file type: " + file);
174      }
175    }
176
177    // Make sure at least two files were specified
178    if (inv_files.size() < 2) {
179      throw new Daikon.UserError(
180          "Must specify at least two inv files; only specified "
181              + UtilPlume.nplural(inv_files.size(), "file"));
182    }
183
184    // Setup the default for guarding
185    PrintInvariants.validateGuardNulls();
186
187    // Initialize the prototype invariants and NI suppressions
188    Daikon.setup_proto_invs();
189    NIS.init_ni_suppression();
190
191    // Read in each of the specified maps
192    List<PptMap> pptmaps = new ArrayList<>();
193    for (File file : inv_files) {
194      debugProgress.fine("Processing " + file);
195      PptMap ppts = FileIO.read_serialized_pptmap(file, true);
196      ppts.repCheck();
197      pptmaps.add(ppts);
198      Debug.check(ppts, "After initial reading of " + file);
199    }
200
201    // Merged ppt map (result of merging each specified inv file)
202    PptMap merge_ppts = null;
203
204    // if no decls file was specified
205    if (decl_file == null) {
206      if (splitter_files.size() > 0) {
207        throw new Daikon.UserError(".spinfo files may only be specified along with a .decls file");
208      }
209
210      // Read in each of the maps again to build a template that contains all
211      // of the program points from each map.
212      for (File file : inv_files) {
213        debugProgress.fine("Reading " + file + " as merge template");
214        if (merge_ppts == null) {
215          merge_ppts = FileIO.read_serialized_pptmap(file, true);
216        } else {
217          PptMap pmap = FileIO.read_serialized_pptmap(file, true);
218          for (PptTopLevel ppt : pmap.pptIterable()) {
219            if (merge_ppts.containsName(ppt.name())) {
220              // System.out.printf("Not adding ppt %s from %s%n", ppt, file);
221              continue;
222            }
223            merge_ppts.add(ppt);
224            // System.out.printf("Adding ppt %s from %s%n", ppt, file);
225
226            // Make sure that the parents of this ppt are already in
227            // the map.  This will be true if all possible children of
228            // any ppt are always included in the same invariant file.
229            // For example, all possible enter/exit points should be
230            // included with each object point.  This is true for Chicory
231            // as long as ppt filtering didn't remove some ppts.
232            for (PptRelation rel : ppt.parents) {
233              assert merge_ppts.get(rel.parent.name()) == rel.parent : ppt + " - " + rel;
234            }
235          }
236        }
237      }
238      assert merge_ppts != null
239          : "@AssumeAssertion(nullness): inv_files is non-empty, so for-loop body executed";
240
241      // Remove all of the slices, equality sets, to start
242      debugProgress.fine("Cleaning ppt map in preparation for merge");
243      for (PptTopLevel ppt : merge_ppts.ppt_all_iterable()) {
244        ppt.clean_for_merge();
245      }
246
247    } else {
248
249      // Build the result pptmap from the specific decls file
250      debugProgress.fine("Building result ppt map from decls file");
251      Daikon.create_splitters(splitter_files);
252      List<File> decl_files = new ArrayList<>();
253      decl_files.add(decl_file);
254      merge_ppts = FileIO.read_declaration_files(decl_files);
255      merge_ppts.trimToSize();
256      PptRelation.init_hierarchy(merge_ppts);
257    }
258
259    // Create a hierarchy between the merge exitNN points and the
260    // corresponding points in each of the specified maps.  This
261    // should only be created at the exitNN points (i.e., the leaves)
262    // so that the normal processing will create the invariants at
263    // upper points.
264    debugProgress.fine("Building hierarchy between leaves of the maps");
265    for (PptTopLevel ppt : merge_ppts.pptIterable()) {
266
267      // Skip everything that is not a final exit point
268      if (!ppt.ppt_name.isExitPoint()) {
269        assert ppt.children.size() > 0 : ppt;
270        continue;
271      }
272      if (ppt.ppt_name.isCombinedExitPoint()) {
273        assert ppt.children.size() > 0 : ppt;
274        continue;
275      }
276
277      // System.out.printf("Including ppt %s, %d children%n", ppt,
278      //                   ppt.children.size());
279
280      // Splitters should not have any children to begin with
281      if (ppt.has_splitters()) {
282        assert ppt.splitters != null; // because ppt.has_splitters() = true
283        for (PptSplitter ppt_split : ppt.splitters) {
284          for (PptTopLevel p : ppt_split.ppts) {
285            assert p.children.size() == 0 : p;
286          }
287        }
288      }
289
290      // Loop over each of the input ppt maps, looking for the same ppt
291      for (int j = 0; j < pptmaps.size(); j++) {
292        PptMap pmap = pptmaps.get(j);
293        PptTopLevel child = pmap.get(ppt.name());
294        // System.out.printf("found child %s from pmap %d%n", child, j);
295        if (child == null) {
296          continue;
297        }
298        if (child.equality_view == null) {
299          System.out.println(
300              "equality_view == null in child ppt: "
301                  + child.name()
302                  + " ("
303                  + inv_files.get(j)
304                  + ")");
305        } else if (child.equality_view.invs == null) {
306          System.out.println(
307              "equality_view.invs == null in child ppt: "
308                  + child.name()
309                  + " ("
310                  + inv_files.get(j)
311                  + ")"
312                  + " samples = "
313                  + child.num_samples());
314        }
315
316        // Remove the equality invariants added during equality post
317        // processing.  These are not over leaders and will cause problems
318        // in the merge
319        child.remove_equality_invariants();
320        child.in_merge = false;
321
322        // Remove implications, they don't merge correctly
323        child.remove_implications();
324
325        // If the ppt has splitters, attach the child's splitters to the
326        // splitters.  Don't attach the ppt itself, as its invariants can
327        // be built from the invariants in the splitters.
328        if (ppt.has_splitters()) {
329          assert ppt.splitters != null; // because ppt.has_splitters() = true
330          setup_conditional_merge(ppt, child);
331        } else {
332          PptRelation.newMergeChildRel(ppt, child);
333        }
334      }
335
336      // Make sure at least one child was found
337      assert ppt.children.size() > 0 : ppt;
338      if (ppt.has_splitters()) {
339        assert ppt.splitters != null; // because ppt.has_splitters() = true
340        for (PptSplitter ppt_split : ppt.splitters) {
341          for (PptTopLevel p : ppt_split.ppts) {
342            assert p.children.size() > 0 : p;
343          }
344        }
345      }
346    }
347
348    // Check the resulting PptMap for consistency
349    merge_ppts.repCheck();
350
351    // Debug print the hierarchy is a more readable manner
352    if (debug.isLoggable(Level.FINE)) {
353      debug.fine("PPT Hierarchy");
354      for (PptTopLevel ppt : merge_ppts.pptIterable()) {
355        if (ppt.parents.size() == 0) ppt.debug_print_tree(debug, 0, null);
356      }
357    }
358
359    // Merge the invariants
360    debugProgress.fine("Merging invariants");
361    Daikon.createUpperPpts(merge_ppts);
362
363    // Equality post processing
364    debugProgress.fine("Equality Post Processing");
365    for (PptTopLevel ppt : merge_ppts.ppt_all_iterable()) {
366      ppt.postProcessEquality();
367    }
368
369    // Implications
370    long startTime = System.nanoTime();
371    // System.out.println("Creating implications ");
372    debugProgress.fine("Adding Implications ... ");
373    for (PptTopLevel ppt : merge_ppts.pptIterable()) {
374      if (ppt.num_samples() > 0) ppt.addImplications();
375    }
376    long duration = System.nanoTime() - startTime;
377    debugProgress.fine("Time spent in implications: " + TimeUnit.NANOSECONDS.toSeconds(duration));
378
379    // Remove the PptRelation links so that when the file is written
380    // out it only includes the new information
381    for (PptTopLevel ppt : merge_ppts.pptIterable()) {
382      if (!ppt.ppt_name.isExitPoint()) {
383        continue;
384      }
385      if (ppt.ppt_name.isCombinedExitPoint()) {
386        continue;
387      }
388      ppt.children.clear();
389      for (PptConditional cond : ppt.cond_iterable()) {
390        cond.children.clear();
391      }
392    }
393
394    // Write serialized output
395    debugProgress.fine("Writing Output");
396    if (output_inv_file != null) {
397      try {
398        FileIO.write_serialized_pptmap(merge_ppts, output_inv_file);
399      } catch (IOException e) {
400        throw new RuntimeException(
401            "Error while writing .inv file '" + output_inv_file + "': " + e.toString());
402      }
403    } else {
404      // Print the invariants
405      PrintInvariants.print_invariants(merge_ppts);
406    }
407  }
408
409  /**
410   * Ses up the specified relation beteween each of the conditionals in ppt and the matching
411   * conditionals in child. Each must have the same number of splitters setup in the same order. The
412   * splitter match can't be checked because splitters can't be read back in.
413   */
414  private static void setup_conditional_merge(PptTopLevel ppt, PptTopLevel child) {
415
416    // Both ppt and child should have splitters
417    if (ppt.has_splitters() != child.has_splitters()) {
418      throw new Error(
419          "Merge ppt "
420              + ppt.name
421              + (ppt.has_splitters() ? " has " : "doesn't have ")
422              + "splitters, but child ppt "
423              + child.name
424              + (child.has_splitters() ? " does" : " doesn't"));
425    }
426
427    // Nothing to do if there are no splitters here
428    if (!ppt.has_splitters()) {
429      return;
430    }
431
432    assert child.splitters != null
433        : "@AssumeAssertion(nullness): correlated: ppt.has_splitters() == child.has_splitters(), and ppt.has_splitters() == true";
434
435    // Both ppt and child should have the same number of splitters
436    if (ppt.splitters.size() != child.splitters.size()) {
437      throw new Error(
438          "Merge ppt "
439              + ppt.name
440              + " has "
441              + ((ppt.splitters.size() > child.splitters.size()) ? "more" : "fewer")
442              + " splitters ("
443              + ppt.splitters.size()
444              + ") than child ppt "
445              + child.name
446              + " ("
447              + child.splitters.size()
448              + ")");
449    }
450
451    // Create a relation from each conditional ppt to its corresponding
452    // conditional ppt in the child.
453    for (int ii = 0; ii < ppt.splitters.size(); ii++) {
454      PptSplitter ppt_split = ppt.splitters.get(ii);
455      PptSplitter child_split = child.splitters.get(ii);
456      for (int jj = 0; jj < ppt_split.ppts.length; jj++) {
457        child_split.ppts[jj].remove_equality_invariants();
458        child_split.ppts[jj].in_merge = false;
459        PptRelation.newMergeChildRel(ppt_split.ppts[jj], child_split.ppts[jj]);
460      }
461    }
462  }

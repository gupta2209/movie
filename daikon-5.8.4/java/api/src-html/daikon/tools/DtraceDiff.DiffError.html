087  /**
088   * This does the work of {@link #main(String[])}, but it never calls System.exit, so it is
089   * appropriate to be called progrmmatically.
090   *
091   * @param args command-line arguments, like those of {@link #main}
092   */
093  public static void mainHelper(final String[] args) {
094    Set<File> declsfile1 = new HashSet<>();
095    String dtracefile1 = null;
096    Set<File> declsfile2 = new HashSet<>();
097    String dtracefile2 = null;
098
099    LongOpt[] longopts =
100        new LongOpt[] {
101          // Process only part of the trace file
102          new LongOpt(Daikon.ppt_regexp_SWITCH, LongOpt.REQUIRED_ARGUMENT, null, 0),
103          new LongOpt(Daikon.ppt_omit_regexp_SWITCH, LongOpt.REQUIRED_ARGUMENT, null, 0),
104          new LongOpt(Daikon.var_regexp_SWITCH, LongOpt.REQUIRED_ARGUMENT, null, 0),
105          new LongOpt(Daikon.var_omit_regexp_SWITCH, LongOpt.REQUIRED_ARGUMENT, null, 0),
106          // Configuration options
107          new LongOpt(Daikon.config_SWITCH, LongOpt.REQUIRED_ARGUMENT, null, 0),
108          new LongOpt(Daikon.config_option_SWITCH, LongOpt.REQUIRED_ARGUMENT, null, 0),
109        };
110
111    Getopt g = new Getopt("daikon.tools.DtraceDiff", args, "h:", longopts);
112    int c;
113    while ((c = g.getopt()) != -1) {
114      switch (c) {
115
116          // long option
117        case 0:
118          String option_name = longopts[g.getLongind()].getName();
119          if (Daikon.help_SWITCH.equals(option_name)) {
120            System.out.println(usage);
121            throw new Daikon.NormalTermination();
122          } else if (Daikon.ppt_regexp_SWITCH.equals(option_name)) {
123            if (Daikon.ppt_regexp != null) {
124              throw new Error(
125                  "multiple --"
126                      + Daikon.ppt_regexp_SWITCH
127                      + " regular expressions supplied on command line");
128            }
129            String regexp_string = Daikon.getOptarg(g);
130            if (!RegexUtil.isRegex(regexp_string)) {
131              throw new Daikon.UserError(
132                  "Bad regexp "
133                      + regexp_string
134                      + " for "
135                      + Daikon.ppt_regexp_SWITCH
136                      + ": "
137                      + RegexUtil.regexError(regexp_string));
138            }
139            Daikon.ppt_regexp = Pattern.compile(regexp_string);
140            break;
141          } else if (Daikon.ppt_omit_regexp_SWITCH.equals(option_name)) {
142            if (Daikon.ppt_omit_regexp != null) {
143              throw new Error(
144                  "multiple --"
145                      + Daikon.ppt_omit_regexp_SWITCH
146                      + " regular expressions supplied on command line");
147            }
148            String regexp_string = Daikon.getOptarg(g);
149            if (!RegexUtil.isRegex(regexp_string)) {
150              throw new Daikon.UserError(
151                  "Bad regexp "
152                      + regexp_string
153                      + " for "
154                      + Daikon.ppt_omit_regexp_SWITCH
155                      + ": "
156                      + RegexUtil.regexError(regexp_string));
157            }
158            Daikon.ppt_omit_regexp = Pattern.compile(regexp_string);
159            break;
160          } else if (Daikon.var_regexp_SWITCH.equals(option_name)) {
161            if (Daikon.var_regexp != null) {
162              throw new Error(
163                  "multiple --"
164                      + Daikon.var_regexp_SWITCH
165                      + " regular expressions supplied on command line");
166            }
167            String regexp_string = Daikon.getOptarg(g);
168            if (!RegexUtil.isRegex(regexp_string)) {
169              throw new Daikon.UserError(
170                  "Bad regexp "
171                      + regexp_string
172                      + " for "
173                      + Daikon.var_regexp_SWITCH
174                      + ": "
175                      + RegexUtil.regexError(regexp_string));
176            }
177            Daikon.var_regexp = Pattern.compile(regexp_string);
178            break;
179          } else if (Daikon.var_omit_regexp_SWITCH.equals(option_name)) {
180            if (Daikon.var_omit_regexp != null) {
181              throw new Error(
182                  "multiple --"
183                      + Daikon.var_omit_regexp_SWITCH
184                      + " regular expressions supplied on command line");
185            }
186            String regexp_string = Daikon.getOptarg(g);
187            if (!RegexUtil.isRegex(regexp_string)) {
188              throw new Daikon.UserError(
189                  "Bad regexp "
190                      + regexp_string
191                      + " for "
192                      + Daikon.var_omit_regexp_SWITCH
193                      + ": "
194                      + RegexUtil.regexError(regexp_string));
195            }
196            Daikon.var_omit_regexp = Pattern.compile(regexp_string);
197            break;
198          } else if (Daikon.config_SWITCH.equals(option_name)) {
199            String config_file = Daikon.getOptarg(g);
200            try {
201              InputStream stream = new FileInputStream(config_file);
202              Configuration.getInstance().apply(stream);
203            } catch (IOException e) {
204              throw new RuntimeException("Could not open config file " + config_file);
205            }
206            break;
207          } else if (Daikon.config_option_SWITCH.equals(option_name)) {
208            String item = Daikon.getOptarg(g);
209            Configuration.getInstance().apply(item);
210            break;
211          } else {
212            throw new RuntimeException("Unknown long option received: " + option_name);
213          }
214
215          // short options
216        case 'h':
217          System.out.println(usage);
218          throw new Daikon.NormalTermination();
219
220        case '?':
221          break; // getopt() already printed an error
222
223        default:
224          System.out.println("getopt() returned " + c);
225          break;
226      }
227    }
228
229    for (int i = g.getOptind(); i < args.length; i++) {
230      if (args[i].indexOf(".decls") != -1) {
231        if (dtracefile1 == null) declsfile1.add(new File(args[i]));
232        else if (dtracefile2 == null) declsfile2.add(new File(args[i]));
233        else {
234          throw new daikon.Daikon.UserError(usage);
235        }
236      } else { // presume any other file is a dtrace file
237        if (dtracefile1 == null) dtracefile1 = args[i];
238        else if (dtracefile2 == null) dtracefile2 = args[i];
239        else {
240          throw new daikon.Daikon.UserError(usage);
241        }
242      }
243    }
244    if ((dtracefile1 == null) || (dtracefile2 == null)) {
245      throw new daikon.Daikon.UserError(usage);
246    }
247    dtraceDiff(declsfile1, dtracefile1, declsfile2, dtracefile2);
248  }

250  public static void dtraceDiff(
251      Set<File> declsfile1, String dtracefile1, Set<File> declsfile2, String dtracefile2) {
252
253    // System.out.printf("dtrace files = %s, %s%n", dtracefile1, dtracefile2);
254    FileIO.resetNewDeclFormat();
255
256    try {
257      Map<PptTopLevel, PptTopLevel> pptmap = new HashMap<>(); // map ppts1 -> ppts2
258      PptMap ppts1 = FileIO.read_declaration_files(declsfile1);
259      PptMap ppts2 = FileIO.read_declaration_files(declsfile2);
260
261      FileIO.ParseState state1 = new FileIO.ParseState(dtracefile1, false, true, ppts1);
262      FileIO.ParseState state2 = new FileIO.ParseState(dtracefile2, false, true, ppts2);
263
264      while (true) {
265        // *** should do some kind of progress bar here?
266        // read from dtracefile1 until we get a data trace record or EOF
267        while (true) {
268          FileIO.read_data_trace_record_setstate(state1);
269          if (state1.rtype == FileIO.RecordType.SAMPLE) {
270            break;
271          } else if ((state1.rtype == FileIO.RecordType.EOF)
272              || (state1.rtype == FileIO.RecordType.TRUNCATED)) {
273            break;
274          }
275        }
276        // read from dtracefile2 until we get a data trace record or EOF
277        while (true) {
278          FileIO.read_data_trace_record_setstate(state2);
279          if (state2.rtype == FileIO.RecordType.SAMPLE) {
280            break;
281          } else if ((state2.rtype == FileIO.RecordType.EOF)
282              || (state2.rtype == FileIO.RecordType.TRUNCATED)) {
283            break;
284          }
285        }
286
287        // things had better be the same
288        if (state1.rtype == state2.rtype) {
289          if (state1.rtype == FileIO.RecordType.SAMPLE) {
290            @SuppressWarnings("nullness") // dependent:  state1 is SAMPLE
291            @NonNull PptTopLevel ppt1 = state1.ppt;
292            @SuppressWarnings("nullness") // dependent:  state1 is SAMPLE
293            @NonNull PptTopLevel ppt2 = state2.ppt;
294            @SuppressWarnings("nullness") // dependent:  state1 is SAMPLE
295            @NonNull ValueTuple vt1 = state1.vt;
296            @SuppressWarnings("nullness") // dependent:  state2 is SAMPLE
297            @NonNull ValueTuple vt2 = state2.vt;
298            VarInfo[] vis1 = ppt1.var_infos;
299            VarInfo[] vis2 = ppt2.var_infos;
300
301            // Check to see that Ppts match the first time we encounter them
302            PptTopLevel foundppt = pptmap.get(ppt1);
303            if (foundppt == null) {
304              if (!ppt1.name.equals(ppt2.name)) {
305                ppt_mismatch_error(state1, dtracefile1, state2, dtracefile2);
306              }
307              for (int i = 0; (i < ppt1.num_tracevars) && (i < ppt2.num_tracevars); i++) {
308                // *** what about comparability and aux info?
309                if (!vis1[i].name().equals(vis2[i].name())
310                    || (vis1[i].is_static_constant != vis2[i].is_static_constant)
311                    || (vis1[i].isStaticConstant()
312                        && vis2[i].isStaticConstant()
313                        && !values_are_equal(
314                            vis1[i], vis1[i].constantValue(), vis2[i].constantValue()))
315                    || ((vis1[i].type != vis2[i].type)
316                        || (vis1[i].file_rep_type != vis2[i].file_rep_type)))
317                  ppt_var_decl_error(vis1[i], state1, dtracefile1, vis2[i], state2, dtracefile2);
318              }
319              if (ppt1.num_tracevars != ppt2.num_tracevars) {
320                ppt_decl_error(state1, dtracefile1, state2, dtracefile2);
321              }
322              pptmap.put(ppt1, ppt2);
323            } else if (foundppt != ppt2) {
324              ppt_mismatch_error(state1, dtracefile1, state2, dtracefile2);
325            }
326
327            // check to see that variables on this pair of samples match
328            for (int i = 0; i < ppt1.num_tracevars; i++) {
329              if (vis1[i].is_static_constant) {
330                continue;
331              }
332              boolean missing1 = vt1.isMissingNonsensical(vis1[i]);
333              boolean missing2 = vt2.isMissingNonsensical(vis2[i]);
334              Object val1 = vt1.getValueOrNull(vis1[i]);
335              Object val2 = vt2.getValueOrNull(vis2[i]);
336              // Require that missing1 == missing2.  Also require that if
337              // the values are present, they are the same.
338              if (!((missing1 == missing2)
339                  && (missing1
340                      // At this point, missing1 == false, missing2 == false,
341                      // val1 != null, val2 != null.
342                      || values_are_equal(
343                          vis1[i], castNonNull(val1), castNonNull(val2))))) // application invariant
344              ppt_var_value_error(
345                    vis1[i], val1, state1, dtracefile1, vis2[i], val2, state2, dtracefile2);
346            }
347          } else {
348            return; // EOF on both files ==> normal return
349          }
350        } else if ((state1.rtype == FileIO.RecordType.TRUNCATED)
351            || (state2.rtype == FileIO.RecordType.TRUNCATED))
352          return; // either file reached truncation limit, return quietly
353        else if (state1.rtype == FileIO.RecordType.EOF) {
354          assert state2.ppt != null
355              : "@AssumeAssertion(nullness): application invariant: status is not EOF or TRUNCATED";
356          throw new DiffError(
357              String.format(
358                  "ppt %s is at line %d in %s but is missing at end of %s",
359                  state2.ppt.name(), state2.get_linenum(), dtracefile2, dtracefile1));
360        } else {
361          assert state1.ppt != null
362              : "@AssumeAssertion(nullness): application invariant: status is not EOF or TRUNCATED";
363          throw new DiffError(
364              String.format(
365                  "ppt %s is at line %d in %s but is missing at end of %s",
366                  state1.ppt.name(), state1.get_linenum(), dtracefile1, dtracefile2));
367        }
368      }
369    } catch (IOException e) {
370      System.out.println();
371      e.printStackTrace();
372      throw new Error(e);
373    }
374  }

376  private static boolean values_are_equal(VarInfo vi, Object val1, Object val2) {
377    ProglangType type = vi.file_rep_type;
378    // System.out.printf("values_are_equal type = %s%n", type);
379    if (type.isArray()) {
380      // array case
381      if (type.isPointerFileRep()) {
382        long[] v1 = (long[]) val1;
383        long[] v2 = (long[]) val2;
384        if (v1.length != v2.length) {
385          return false;
386        }
387        for (int i = 0; i < v1.length; i++) {
388          if (((v1[i] == 0) || (v2[i] == 0)) && (v1[i] != v2[i])) {
389            return false;
390          }
391        }
392        return true;
393      } else if (type.baseIsScalar()) {
394        long[] v1 = (long[]) val1;
395        long[] v2 = (long[]) val2;
396        if (v1.length != v2.length) {
397          return false;
398        }
399        for (int i = 0; i < v1.length; i++) {
400          if (v1[i] != v2[i]) {
401            return false;
402          }
403        }
404        return true;
405      } else if (type.baseIsFloat()) {
406        double[] v1 = (double[]) val1;
407        double[] v2 = (double[]) val2;
408        if (v1.length != v2.length) {
409          return false;
410        }
411        for (int i = 0; i < v1.length; i++) {
412          if (!((Double.isNaN(v1[i]) && Double.isNaN(v2[i])) || Global.fuzzy.eq(v1[i], v2[i]))) {
413            return false;
414          }
415        }
416        return true;
417      } else if (type.baseIsString()) {
418        String[] v1 = (String[]) val1;
419        String[] v2 = (String[]) val2;
420        if (v1.length != v2.length) {
421          return false;
422        }
423        for (int i = 0; i < v1.length; i++) {
424          // System.out.printf("string array[%d] %s %s%n", i, v1[i], v2[i]);
425          if ((v1[i] == null) && (v2[i] == null)) {
426            // nothing to do
427          } else if ((v1[i] == null) || (v2[i] == null)) {
428            return false;
429          } else if (!v1[i].equals(v2[i])) {
430            return false;
431          }
432        }
433        return true;
434      }
435    } else {
436      // scalar case
437      if (type.isPointerFileRep()) {
438        long v1 = ((Long) val1).longValue();
439        long v2 = ((Long) val2).longValue();
440        return !(((v1 == 0) || (v2 == 0)) && (v1 != v2));
441      } else if (type.isScalar()) {
442        return (((Long) val1).longValue() == ((Long) val2).longValue());
443      } else if (type.isFloat()) {
444        double d1 = ((Double) val1).doubleValue();
445        double d2 = ((Double) val2).doubleValue();
446        return ((Double.isNaN(d1) && Double.isNaN(d2)) || Global.fuzzy.eq(d1, d2));
447      } else if (type.isString()) {
448        return (((String) val1).equals((String) val2));
449      }
450    }
451    throw new Error("Unexpected value type found"); // should never happen
452  }

454  @SuppressWarnings("nullness") // dependent: ParseState for error reporting
455  private static void ppt_mismatch_error(
456      FileIO.ParseState state1, String dtracefile1, FileIO.ParseState state2, String dtracefile2) {
457    throw new DiffError(
458        String.format(
459            "Mismatched program point:%n  ppt %s at %s:%d%n  ppt %s at %s:%d",
460            state1.ppt.name,
461            dtracefile1,
462            state1.get_linenum(),
463            state2.ppt.name,
464            dtracefile2,
465            state2.get_linenum()));
466  }

468  @SuppressWarnings("nullness") // dependent: ParseState for error reporting
469  private static void ppt_decl_error(
470      FileIO.ParseState state1, String dtracefile1, FileIO.ParseState state2, String dtracefile2) {
471    throw new DiffError(
472        String.format(
473            "Mismatched program point declaration:%n  ppt %s at %s:%d%n  ppt %s at %s:%d",
474            state1.ppt.name,
475            dtracefile1,
476            state1.get_linenum(),
477            state2.ppt.name,
478            dtracefile2,
479            state2.get_linenum()));
480  }
481
482  @SuppressWarnings("nullness") // dependent: ParseState for error reporting
483  private static void ppt_var_decl_error(
484      VarInfo vi1,
485      FileIO.ParseState state1,
486      String dtracefile1,
487      VarInfo vi2,
488      FileIO.ParseState state2,
489      String dtracefile2) {
490    assert state1.ppt.name.equals(state2.ppt.name);
491    throw new DiffError(
492        String.format(
493            "Mismatched variable declaration in program point %s:%n"
494                + "  variable %s at %s:%d%n"
495                + "  variable %s at %s:%d",
496            state1.ppt.name,
497            vi1.name(),
498            dtracefile1,
499            state1.get_linenum(),
500            vi2.name(),
501            dtracefile2,
502            state2.get_linenum()));
503  }
504
505  @SuppressWarnings("nullness") // nullable parameters suppress warnings at call sites
506  private static void ppt_var_value_error(
507      VarInfo vi1,
508      @Nullable Object val1,
509      FileIO.ParseState state1,
510      String dtracefile1,
511      VarInfo vi2,
512      @Nullable Object val2,
513      FileIO.ParseState state2,
514      String dtracefile2) {
515    assert vi1.name().equals(vi2.name());
516    assert state1.ppt.name.equals(state2.ppt.name);
517    throw new DiffError(
518        String.format(
519            "Mismatched values for variable %s in program point %s:%n"
520                + "  value %s at %s:%d%n"
521                + "  value %s at %s:%d",
522            vi1.name(),
523            state1.ppt.name,
524            val1,
525            dtracefile1,
526            state1.get_linenum(),
527            val2,
528            dtracefile2,
529            state2.get_linenum()));
530  }

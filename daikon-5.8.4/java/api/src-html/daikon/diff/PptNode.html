001package daikon.diff;
002
003import daikon.PptTopLevel;
004import org.checkerframework.checker.nullness.qual.Nullable;
005import org.checkerframework.dataflow.qual.Pure;
006import org.plumelib.util.Pair;
007
008/**
009 * Contains a pair of Ppts. Resides in the second level of the tree. All its children are InvNodes.
010 */
011public class PptNode extends Node<@Nullable PptTopLevel, InvNode> {
012
013  /** Either ppt1 or ppt2 may be null, but not both. */
014  public PptNode(@Nullable PptTopLevel ppt1, @Nullable PptTopLevel ppt2) {
015    super(Pair.of(ppt1, ppt2));
016    assert !(ppt1 == null && ppt2 == null) : "Both program points may not be null";
017  }
018
019  @Pure
020  public @Nullable PptTopLevel getPpt1() {
021    return getUserLeft();
022  }
023
024  @Pure
025  public @Nullable PptTopLevel getPpt2() {
026    return getUserRight();
027  }
028
029  @Override
030  public void accept(Visitor v) {
031    v.visit(this);
032  }
033}

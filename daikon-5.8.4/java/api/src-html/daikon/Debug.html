015/**
016 * Debug class used with the logger to create standardized output. It can be setup to track
017 * combinations of classes, program points, and variables. The most common class to track is an
018 * invariant, but any class can be used.
019 *

021 * printed without getting lost in a mass of other information (which is a particular problem in
022 * Daikon due to the volume of data considered).
023 *
024 * <p>Note that each of the three items (class, ppt, variable) must match in order for printing to
025 * occur.
026 */

186  /**
187   * Ordinarily, a client would have to supply a Class, Ppt, and {@code List<Varinfo>} with each
188   * call to a log method. This constructor sets as defaults c, ppt, and whatever variable (if any)
189   * from vis that is on the debugTrackVar list. Essentially this creates a debug object that will
190   * print if any of the variables in vis are being tracked (and c and ppt match).

192  public Debug(Class<?> c, Ppt ppt, List<VarInfo> vis) {
193
194    VarInfo v = visTracked(vis);
195    if (v != null) {
196      set(c, ppt, new VarInfo[] {v});
197    } else if (vis.size() > 0) {
198      set(c, ppt, new VarInfo[] {vis.get(0)});
199    } else {
200      set(c, ppt, null);
201    }
202  }

204  /**
205   * Looks for each of the variables in vis in the DebugTrackVar list. If any match, returns that
206   * variable. Returns null if there are no matches.
207   */
208  public @Nullable VarInfo visTracked(@UnknownInitialization Debug this, List<VarInfo> vis) {
209
210    for (VarInfo v : vis) {
211      Set<VarInfo> evars = null;
212      if (v.equalitySet != null) {
213        evars = v.equalitySet.getVars();
214      }
215      if (evars != null) {
216        for (VarInfo ev : evars) {
217          for (int k = 0; k < debugTrackVars.length; k++) {
218            if (ev.name().equals(debugTrackVars[k][0])) {
219              return v;
220            }
221          }
222        }
223      }
224    }
225
226    return null;
227  }

345  /**
346   * Logs a description of the class, ppt, variables and the specified msg via the logger. The
347   * class, ppt, and variables are checked against those described in {@link #debugTrackClass},
348   * {@link #debugTrackPpt}, and {@link #debugTrackVars}. Only those that match are printed.
349   * Variables will match if they are in the same equality set. The information is written as:
350   *
351   * <p>{@code class: ppt : var1 : var2 : var3 : msg }
352   *
353   * <p>Note that if {@link #debugTrack} is not enabled then nothing is printed. It is somewhat
354   * faster to check {@link #logOn()} directly rather than relying on the check here.
355   *
356   * <p>Other versions of this method (noted below) work without the Logger parameter and take
357   * class, ppt, and vis from the cached values, which were set by the constructor or by the set()
358   * method.
359   *
360   * @param debug a second Logger to query if debug tracking is turned off or does not match. If
361   *     this logger is enabled, the same information will be written to it. Note that the
362   *     information is never written to both loggers.
363   * @param inv_class the class. Can be obtained in a static context by ClassName.class
364   * @param ppt program point
365   * @param vis variables at the program point. These are sometimes different from the ones in the
366   *     ppt itself.
367   * @param msg string message to log
368   * @see #logOn()
369   * @see #logDetail()
370   * @see #log(Class, Ppt, VarInfo[], String)
371   * @see #log(Class, Ppt, String)
372   * @see #log(Logger, String)
373   * @see #log(String)
374   */
375  public static void log(
376      Logger debug,
377      @Nullable Class<?> inv_class,
378      @Nullable Ppt ppt,
379      VarInfo @Nullable [] vis,
380      String msg) {
381
382    // Try to log via the logger first
383    if (log(inv_class, ppt, vis, msg)) {
384      return;
385    }
386
387    // If debug isn't turned on, there is nothing to do
388    if (!debug.isLoggable(Level.FINE)) {
389      return;
390    }
391
392    // Get the non-qualified class name
393    String class_str;
394    if (inv_class == null) {
395      // when is inv_class null?
396      class_str = "null";
397    } else {
398      @SuppressWarnings("nullness") // getPackage(): invariant class always has a package
399      @NonNull String packageName = inv_class.getPackage().getName() + ".";
400      class_str = UtilPlume.replaceString(inv_class.getName(), packageName, "");
401    }
402
403    String vars = "";
404    if (vis == null) {
405      System.out.println("no var infos");
406    } else {
407
408      // Get a string with all of the variable names.  Each is separated by ': '
409      // 3 variable slots are always setup for consistency
410      for (VarInfo v : vis) {
411        vars += v.name() + ": ";
412      }
413      for (int i = vis.length; i < 3; i++) {
414        vars += ": ";
415      }
416    }
417
418    // Figure out the sample count if possible
419    String samp_str = "";
420    if (ppt instanceof PptSlice) {
421      PptSlice pslice = (PptSlice) ppt;
422      samp_str = " s" + pslice.num_samples();
423    }
424
425    String line = " line=" + FileIO.get_linenum();
426
427    String name = "ppt is null";
428    if (ppt != null) {
429      name = ppt.name();
430    }
431
432    debug.fine(class_str + ": " + name + samp_str + line + ": " + vars + msg);
433    if (dkconfig_showTraceback) {
434      Throwable stack = new Throwable("debug traceback");
435      stack.fillInStackTrace();
436      stack.printStackTrace();
437    }
438  }

466  /**
467   * Logs a description of the class, ppt, variables and the specified msg via the logger as
468   * described in {@link #log(Logger, Class, Ppt, String)}. Accepts vis because sometimes the
469   * variables are different from those in the ppt.
470   *
471   * @return whether or not it logged anything
472   */
473  // 4-argument form
474  public static boolean log(
475      @Nullable Class<?> inv_class,
476      @Nullable @UnknownInitialization(PptTopLevel.class) Ppt ppt,
477      VarInfo @Nullable [] vis,
478      String msg) {
479
480    if (!debugTrack.isLoggable(Level.FINE)) {
481      return false;
482    }
483
484    // Make sure the class matches
485    if (!class_match(inv_class)) {
486      return false;
487    }
488
489    // Make sure the Ppt matches
490    if (!ppt_match(ppt)) {
491      return false;
492    }
493
494    // Make sure the variables match
495    if (!var_match(vis)) {
496      return false;
497    }
498
499    // Get the non-qualified class name
500    String class_str = "null";
501    if (inv_class != null) {
502      @SuppressWarnings("nullness") // getPackage(): invariant class always has a package
503      @NonNull String packageName = inv_class.getPackage().getName() + ".";
504      class_str = UtilPlume.replaceString(inv_class.getName(), packageName, "");
505    }
506
507    // Get a string with all of the variable names.  Each is separated by ': '.
508    // 3 variable slots are always setup for consistency.
509    String vars = "";
510    if (vis != null) {
511      int numvars = vis.length;
512      for (int i = 0; i < numvars; i++) {
513        VarInfo v = vis[i];
514        vars += v.name();
515        if (ourvars[i] != null) {
516          vars += " {" + ourvars[i] + "}";
517        }
518        vars += ": ";
519      }
520      for (int i = numvars; i < 3; i++) {
521        vars += ": ";
522      }
523    }
524
525    // Figure out the sample count if possible
526    String samp_str = "";
527    if (ppt instanceof PptSlice) {
528      PptSlice pslice = (PptSlice) ppt;
529      samp_str = " s" + pslice.num_samples();
530    }
531
532    String line = " line=" + FileIO.get_linenum();
533
534    debugTrack.fine(
535        class_str
536            + ": "
537            + ((ppt == null) ? "null" : ppt.name())
538            + samp_str
539            + line
540            + ": "
541            + vars
542            + msg);
543    if (dkconfig_showTraceback) {
544      Throwable stack = new Throwable("debug traceback");
545      stack.fillInStackTrace();
546      stack.printStackTrace();
547    }
548
549    return true;
550  }

571  /**
572   * Returns whether or not the specified vars match the ones being tracked. Also, sets
573   * Debug.ourvars with the names of the variables matched if they are not the leader of their
574   * equality sets.
575   */
576  public static boolean var_match(VarInfo @Nullable [] vis) {
577
578    if (debugTrackVars.length == 0) {
579      return true;
580    }
581    if (vis == null) {
582      return false;
583    }
584
585    boolean match = false;
586
587    // Loop through each set of specified debug variables.
588    outer:
589    for (String[] cv : debugTrackVars) {
590      if (cv.length != vis.length) {
591        continue;
592      }
593      for (int j = 0; j < ourvars.length; j++) {
594        ourvars[j] = null;
595      }
596
597      // Flags to insure that we don't match a variable more than once
598      boolean[] used = {false, false, false};
599
600      // Loop through each variable in this set of debug variables
601      for (int j = 0; j < cv.length; j++) {
602        boolean this_match = false;
603
604        // Loop through each variable at this point
605        eachvis:
606        for (int k = 0; k < vis.length; k++) {
607
608          // Get the matching equality set
609          Set<VarInfo> evars = null;
610          if (vis[k].equalitySet != null) {
611            evars = vis[k].equalitySet.getVars();
612          }
613
614          // If there is an equality set
615          if ((evars != null) && vis[k].isCanonical()) {
616
617            // Loop through each variable in the equality set
618            for (VarInfo v : evars) {
619              if (!used[k] && (cv[j].equals("*") || cv[j].equals(v.name()))) {
620                used[k] = true;
621                this_match = true;
622                if (!cv[j].equals(vis[j].name())) {
623                  ourvars[j] = v.name();
624                  if (j != k) ourvars[j] += " (" + j + "/" + k + ")";
625                  if (v.isCanonical()) ourvars[j] += " (Leader)";
626                }
627                break eachvis;
628              }
629            }
630          } else { // sometimes, no equality set
631            if (cv[j].equals("*") || cv[j].equals(vis[k].name())) this_match = true;
632          }
633        }
634        if (!this_match) continue outer;
635      }
636      match = true;
637      break outer;
638    }
639
640    return match;
641  }

643  /** Looks for an element in arr that is a substring of str. */
644  private static boolean strContainsElem(String str, String[] arr) {
645
646    for (String elt : arr) {
647      if (str.indexOf(elt) >= 0) {
648        return true;
649      }
650    }
651    return false;
652  }

654  /**
655   * Looks through entire ppt tree and checks for any items we are interested in. If found, prints
656   * them out.
657   */
658  public static void check(PptMap all_ppts, String msg) {
659
660    boolean found = false;
661
662    for (PptTopLevel ppt : all_ppts.ppt_all_iterable()) {
663      if (ppt_match(ppt)) debugTrack.fine("Matched ppt '" + ppt.name() + "' at " + msg);
664      for (PptSlice slice : ppt.views_iterable()) {
665        for (int k = 0; k < slice.invs.size(); k++) {
666          Invariant inv = slice.invs.get(k);
667          if (inv.log("%s: found #%s=%s in slice %s", msg, k, inv.format(), slice)) found = true;
668        }
669      }
670    }
671    if (!found) debugTrack.fine("Found no points at '" + msg + "'");
672  }

674  /** Returns a string containing the integer variables and their values. */
675  public static String int_vars(PptTopLevel ppt, ValueTuple vt) {
676
677    String out = "";
678
679    for (VarInfo v : ppt.var_infos) {
680      if (!v.isCanonical()) {
681        continue;
682      }
683      if (v.file_rep_type != ProglangType.INT) {
684        continue;
685      }
686      out += v.name() + "=" + toString(v.getValueOrNull(vt)) + " [" + vt.getModified(v) + "]: ";
687    }
688    return out;
689  }

692   * Returns a string containing the variable values for any variables that are currently being
693   * tracked in ppt. The string is of the form 'v1 = val1: v2 = val2, etc.
694   */
695  public static String related_vars(PptTopLevel ppt, ValueTuple vt) {
696
697    String out = "";
698
699    for (VarInfo v : ppt.var_infos) {
700      for (String[] cv : debugTrackVars) {
701        for (String cv_elt : cv) {
702          if (cv_elt.equals(v.name())) {
703            Object val = v.getValue(vt);
704            int mod = vt.getModified(v);
705            out += v.name() + "=";
706            out += toString(val);
707            if ((mod == ValueTuple.MISSING_FLOW) || (mod == ValueTuple.MISSING_NONSENSICAL)) {
708              out += " (missing)";
709            }
710            if (v.missingOutOfBounds()) out += " (out of bounds)";
711            if (v.equalitySet != null) {
712              if (!v.isCanonical()) out += " (leader=" + v.canonicalRep().name() + ")";
713            }
714            // out += " mod=" + mod;
715            out += ": ";
716          }
717        }
718      }
719    }
720
721    return out;
722  }

724  /** Like Object.toString(), but handles null, and has special handling for arrays. */
725  public static String toString(@Nullable Object val) {
726    if (val == null) {
727      return "none";
728    }
729    if (val instanceof String) {
730      return "\"" + val + "\"";
731    }
732    if (val instanceof VarInfo[]) {
733      return Arrays.toString((VarInfo[]) val);
734    }
735    if (val instanceof String[]) {
736      return Arrays.toString((String[]) val);
737    }
738    if (val instanceof boolean[]) {
739      return Arrays.toString((boolean[]) val);
740    }
741    if (val instanceof byte[]) {
742      return Arrays.toString((byte[]) val);
743    }
744    if (val instanceof char[]) {
745      return Arrays.toString((char[]) val);
746    }
747    if (val instanceof double[]) {
748      return Arrays.toString((double[]) val);
749    }
750    if (val instanceof float[]) {
751      return Arrays.toString((float[]) val);
752    }
753    if (val instanceof int[]) {
754      return Arrays.toString((int[]) val);
755    }
756    if (val instanceof long[]) {
757      return Arrays.toString((long[]) val);
758    }
759    if (val instanceof short[]) {
760      return Arrays.toString((short[]) val);
761    }
762    return val.toString();
763  }

791  /**
792   * Parses the specified argument to {@code --track} and sets up the track arrays accordingly. The
793   * syntax of the argument is
794   *
795   * <pre>{@code class|class|...<var,var,var>@ppt}</pre>
796   *
797   * As shown, multiple class arguments can be specified separated by pipe symbols (|). The
798   * variables are specified in angle brackets (&lt;&gt;) and the program point is preceeded by an
799   * at sign (@). Each is optional and can be left out. The add_track routine can be called multiple
800   * times. An invariant that matches any of the specifications will be tracked.
801   */
802  public static @Nullable String add_track(String def) {
803
804    String classes = null;
805    String vars = null;
806    String ppt = null;
807
808    // Get the classes, vars, and ppt
809    int var_start = def.indexOf('<');
810    int ppt_start = def.indexOf("@");
811    if ((var_start == -1) && (ppt_start == -1)) {
812      classes = def;
813    } else if (var_start != -1) {
814      if (var_start > 0) classes = def.substring(0, var_start);
815      if (ppt_start == -1) {
816        vars = def.substring(var_start + 1, def.length() - 1);
817      } else {
818        vars = def.substring(var_start + 1, ppt_start - 1);
819        ppt = def.substring(ppt_start + 1, def.length());
820      }
821    } else {
822      if (ppt_start > 0) classes = def.substring(0, ppt_start);
823      ppt = def.substring(ppt_start + 1, def.length());
824    }
825
826    // If classes were specified, get each class
827    if (classes != null) {
828      String[] class_arr = classes.split("\\|");
829      debugTrackClass = ArraysPlume.concat(debugTrackClass, class_arr);
830    }
831
832    // If vars were specified, get each var
833    if (vars != null) {
834      String[] var_arr = vars.split(", *");
835      String[][] new_var = new String[debugTrackVars.length + 1][];
836      for (int ii = 0; ii < debugTrackVars.length; ii++) {
837        new_var[ii] = debugTrackVars[ii];
838      }
839      new_var[debugTrackVars.length] = var_arr;
840      debugTrackVars = new_var;
841    }
842
843    // if a ppt was specified, add it to the array of tracked ppts
844    if (ppt != null) {
845      String[] newPpt = new String[] {ppt};
846      debugTrackPpt = ArraysPlume.concat(debugTrackPpt, newPpt);
847    }
848
849    System.out.println();
850    debugTrack.fine("After --track: " + def);
851    debugTrack.fine("Track Classes: " + ArraysPlume.toString(debugTrackClass, false));
852    String vars_out = "";
853    for (int ii = 0; ii < debugTrackVars.length; ii++) {
854      vars_out += Arrays.toString(debugTrackVars[ii]) + " ";
855    }
856    debugTrack.fine("Track Vars: " + vars_out);
857    debugTrack.fine("Track Ppts: " + ArraysPlume.toString(debugTrackPpt, false));
858
859    return null;
860  }

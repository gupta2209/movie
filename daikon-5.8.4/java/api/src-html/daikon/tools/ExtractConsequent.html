037/**
038 * Extract the consequents of all Implication invariants that are predicated by membership in a
039 * cluster, from a {@code .inv} file. An example of such an implication would be "(cluster ==
040 * <em>NUM</em>) &rArr; consequent". The consequent is only true in certain clusters, but is not
041 * generally true for all executions of the program point to which the Implication belongs. These
042 * resulting implications are written to standard output in the format of a splitter info file.
043 */
044public class ExtractConsequent {
045
046  public static final Logger debug = Logger.getLogger("daikon.ExtractConsequent");
047  private static final String lineSep = Global.lineSep;
048
049  private static class HashedConsequent {
050    Invariant inv;
051
052    // This field, `fakeFor`, is a simplified/preferred version of the invariant.
053    // We prefer "x < y", "x > y", and "x == y" to the conditions
054    // "x >= y", "x <= y", and "x != y" that (respectively) give the
055    // same split.  When we see a dispreferred form, we index it by
056    // the preferred form, and if there's already an entry (from the
057    // real preferred one) we throw the new one out. Otherwise, we
058    // insert both the dispreferred form and an entry for the
059    // preferred form, with a pointer pack to the dispreferred
060    // form. If we later see the preferred form, we replace the
061    // placeholder and remove the dispreferred form.
062    final @Nullable String fakeFor;
063
064    HashedConsequent(Invariant inv, @Nullable String fakeFor) {
065      this.inv = inv;
066      this.fakeFor = fakeFor;
067    }
068  }

070  /* A HashMap whose keys are PPT names (Strings) and whose values are
071  HashMaps whose keys are predicate names (Strings) and whose values are
072   HashMaps whose keys are Strings (normalized java-format invariants)
073     and whose values are HashedConsequent objects. */
074  private static Map<String, Map<String, Map<String, HashedConsequent>>> pptname_to_conditions =
075      new HashMap<>();

098  /**
099   * This does the work of {@link #main(String[])}, but it never calls System.exit, so it is
100   * appropriate to be called progrmmatically.
101   *
102   * @param args command-line arguments, like those of {@link #main}
103   */
104  public static void mainHelper(final String[] args)
105      throws FileNotFoundException, IOException, ClassNotFoundException {
106    daikon.LogHelper.setupLogs(daikon.LogHelper.INFO);
107    LongOpt[] longopts =
108        new LongOpt[] {
109          new LongOpt(Daikon.suppress_redundant_SWITCH, LongOpt.NO_ARGUMENT, null, 0),
110          new LongOpt(Daikon.config_option_SWITCH, LongOpt.REQUIRED_ARGUMENT, null, 0),
111          new LongOpt(Daikon.debugAll_SWITCH, LongOpt.NO_ARGUMENT, null, 0),
112          new LongOpt(Daikon.debug_SWITCH, LongOpt.REQUIRED_ARGUMENT, null, 0),
113        };
114    Getopt g = new Getopt("daikon.ExtractConsequent", args, "h", longopts);
115    int c;
116    while ((c = g.getopt()) != -1) {
117      switch (c) {
118        case 0:
119          // got a long option
120          String option_name = longopts[g.getLongind()].getName();
121          if (Daikon.help_SWITCH.equals(option_name)) {
122            System.out.println(usage);
123            throw new Daikon.NormalTermination();
124          } else if (Daikon.suppress_redundant_SWITCH.equals(option_name)) {
125            Daikon.suppress_redundant_invariants_with_simplify = true;
126          } else if (Daikon.config_option_SWITCH.equals(option_name)) {
127            String item = Daikon.getOptarg(g);
128            daikon.config.Configuration.getInstance().apply(item);
129            break;
130          } else if (Daikon.debugAll_SWITCH.equals(option_name)) {
131            Global.debugAll = true;
132          } else if (Daikon.debug_SWITCH.equals(option_name)) {
133            LogHelper.setLevel(Daikon.getOptarg(g), LogHelper.FINE);
134          } else {
135            throw new RuntimeException("Unknown long option received: " + option_name);
136          }
137          break;
138        case 'h':
139          System.out.println(usage);
140          throw new Daikon.NormalTermination();
141        case '?':
142          break; // getopt() already printed an error
143        default:
144          System.out.println("getopt() returned " + c);
145          break;
146      }
147    }
148    // The index of the first non-option argument -- the name of the file
149    int fileIndex = g.getOptind();
150    if (args.length - fileIndex != 1) {
151      throw new Daikon.UserError("Wrong number of arguments." + Daikon.lineSep + usage);
152    }
153    String filename = args[fileIndex];
154    PptMap ppts =
155        FileIO.read_serialized_pptmap(
156            new File(filename), true // use saved config
157            );
158    extract_consequent(ppts);
159  }

161  public static void extract_consequent(PptMap ppts) {
162    // Retrieve Ppt objects in sorted order.
163    // Use a custom comparator for a specific ordering
164    Comparator<PptTopLevel> comparator = new Ppt.NameComparator();
165    TreeSet<PptTopLevel> ppts_sorted = new TreeSet<>(comparator);
166    ppts_sorted.addAll(ppts.asCollection());
167
168    for (PptTopLevel ppt : ppts_sorted) {
169      extract_consequent_maybe(ppt, ppts);
170    }
171
172    PrintWriter pw =
173        new PrintWriter(new BufferedWriter(new OutputStreamWriter(System.out, UTF_8)), true);
174
175    // All conditions at a program point.  A TreeSet to enable
176    // deterministic output.
177    TreeSet<String> allConds = new TreeSet<>();
178    for (String pptname : pptname_to_conditions.keySet()) {
179      Map<String, Map<String, HashedConsequent>> cluster_to_conditions =
180          pptname_to_conditions.get(pptname);
181      for (Map.Entry<@KeyFor("cluster_to_conditions") String, Map<String, HashedConsequent>> entry :
182          cluster_to_conditions.entrySet()) {
183        Map<String, HashedConsequent> conditions = entry.getValue();
184        StringBuilder conjunctionJava = new StringBuilder();
185        StringBuilder conjunctionDaikon = new StringBuilder();
186        StringBuilder conjunctionESC = new StringBuilder();
187        StringBuilder conjunctionSimplify = new StringBuilder("(AND ");
188        int count = 0;
189        for (Map.Entry<@KeyFor("conditions") String, HashedConsequent> entry2 :
190            conditions.entrySet()) {
191          count++;
192          String condIndex = entry2.getKey();
193          HashedConsequent cond = entry2.getValue();
194          if (cond.fakeFor != null) {
195            count--;
196            continue;
197          }
198          String javaStr = cond.inv.format_using(OutputFormat.JAVA);
199          String daikonStr = cond.inv.format_using(OutputFormat.DAIKON);
200          String escStr = cond.inv.format_using(OutputFormat.ESCJAVA);
201          String simplifyStr = cond.inv.format_using(OutputFormat.SIMPLIFY);
202          allConds.add(combineDummy(condIndex, "<dummy> " + daikonStr, escStr, simplifyStr));
203          //           allConds.add(condIndex);
204          if (count > 0) {
205            conjunctionJava.append(" && ");
206            conjunctionDaikon.append(" and ");
207            conjunctionESC.append(" && ");
208            conjunctionSimplify.append(" ");
209          }
210          conjunctionJava.append(javaStr);
211          conjunctionDaikon.append(daikonStr);
212          conjunctionESC.append(escStr);
213          conjunctionSimplify.append(simplifyStr);
214        }
215        conjunctionSimplify.append(")");
216        String conj = conjunctionJava.toString();
217        // Avoid inserting self-contradictory conditions such as "x == 1 &&
218        // x == 2", or conjunctions of only a single condition.
219        if (count < 2
220            || contradict_inv_pattern.matcher(conj).find()
221            || useless_inv_pattern_1.matcher(conj).find()
222            || useless_inv_pattern_2.matcher(conj).find()) {
223          // System.out.println("Suppressing: " + conj);
224        } else {
225          allConds.add(
226              combineDummy(
227                  conjunctionJava.toString(),
228                  conjunctionDaikon.toString(),
229                  conjunctionESC.toString(),
230                  conjunctionSimplify.toString()));
231        }
232      }
233
234      if (allConds.size() > 0) {
235        pw.println();
236        pw.println("PPT_NAME " + pptname);
237        for (String s : allConds) {
238          pw.println(s);
239        }
240      }
241      allConds.clear();
242    }
243
244    pw.flush();
245  }
246
247  static String combineDummy(String inv, String daikonStr, String esc, String simplify) {
248    StringBuilder combined = new StringBuilder(inv);
249    combined.append(lineSep + "\tDAIKON_FORMAT ");
250    combined.append(daikonStr);
251    combined.append(lineSep + "\tESC_FORMAT ");
252    combined.append(esc);
253    combined.append(lineSep + "\tSIMPLIFY_FORMAT ");
254    combined.append(simplify);
255    return combined.toString();
256  }

258  /**
259   * Extract consequents from a implications at a single program point. It only searches for
260   * top-level program points because Implications are produced only at those points.
261   */
262  public static void extract_consequent_maybe(PptTopLevel ppt, PptMap all_ppts) {
263    ppt.simplify_variable_names();
264
265    List<Invariant> invs = new ArrayList<>();
266    if (invs.size() > 0) {
267      String pptname = cleanup_pptname(ppt.name());
268      for (Invariant maybe_as_inv : invs) {
269        Implication maybe = (Implication) maybe_as_inv;
270
271        // don't print redundant invariants.
272        if (Daikon.suppress_redundant_invariants_with_simplify
273            && maybe.ppt.parent.redundant_invs.contains(maybe)) {
274          continue;
275        }
276
277        // don't print out invariants with min(), max(), or sum() variables
278        boolean mms = false;
279        VarInfo[] varbls = maybe.ppt.var_infos;
280        for (int v = 0; !mms && v < varbls.length; v++) {
281          mms |= varbls[v].isDerivedSequenceMinMaxSum();
282        }
283        if (mms) {
284          continue;
285        }
286
287        if (maybe.ppt.parent.ppt_name.isExitPoint()) {
288          for (int i = 0; i < maybe.ppt.var_infos.length; i++) {
289            VarInfo vi = maybe.ppt.var_infos[i];
290            if (vi.isDerivedParam()) {
291              continue;
292            }
293          }
294        }
295
296        Invariant consequent = maybe.consequent();
297        Invariant predicate = maybe.predicate();
298        Invariant inv, cluster_inv;
299        boolean cons_uses_cluster = false, pred_uses_cluster = false;
300        // extract the consequent (predicate) if the predicate
301        // (consequent) uses the variable "cluster".  Ignore if they
302        // both depend on "cluster"
303        if (consequent.usesVarDerived("cluster")) cons_uses_cluster = true;
304        if (predicate.usesVarDerived("cluster")) {
305          pred_uses_cluster = true;
306        }
307
308        if (!(pred_uses_cluster ^ cons_uses_cluster)) {
309          continue;
310        } else if (pred_uses_cluster) {
311          inv = consequent;
312          cluster_inv = predicate;
313        } else {
314          inv = predicate;
315          cluster_inv = consequent;
316        }
317
318        if (!inv.isWorthPrinting()) {
319          continue;
320        }
321
322        if (contains_constant_non_012(inv)) {
323          continue;
324        }
325
326        // filter out unwanted invariants
327
328        // 1) Invariants involving sequences
329        if (inv instanceof daikon.inv.binary.twoSequence.TwoSequence
330            || inv instanceof daikon.inv.binary.sequenceScalar.SequenceScalar
331            || inv instanceof daikon.inv.binary.sequenceString.SequenceString
332            || inv instanceof daikon.inv.unary.sequence.SingleSequence
333            || inv instanceof daikon.inv.unary.stringsequence.SingleStringSequence) {
334          continue;
335        }
336
337        if (inv instanceof daikon.inv.ternary.threeScalar.LinearTernary
338            || inv instanceof daikon.inv.binary.twoScalar.LinearBinary) {
339          continue;
340        }
341
342        String inv_string = inv.format_using(OutputFormat.JAVA);
343        if (orig_pattern.matcher(inv_string).find()
344            || dot_class_pattern.matcher(inv_string).find()) {
345          continue;
346        }
347        String fake_inv_string = simplify_inequalities(inv_string);
348        HashedConsequent real = new HashedConsequent(inv, null);
349        if (!fake_inv_string.equals(inv_string)) {
350          // For instance, inv_string is "x != y", fake_inv_string is "x == y"
351          HashedConsequent fake = new HashedConsequent(inv, inv_string);
352          boolean added =
353              store_invariant(
354                  cluster_inv.format_using(OutputFormat.JAVA), fake_inv_string, fake, pptname);
355          if (!added) {
356            // We couldn't add "x == y", (when we're "x != y") because
357            // it already exists; so don't add "x == y" either.
358            continue;
359          }
360        }
361        store_invariant(cluster_inv.format_using(OutputFormat.JAVA), inv_string, real, pptname);
362      }
363    }
364  }

366  // Store the invariant for later printing. Ignore duplicate
367  // invariants at the same program point.
368  private static boolean store_invariant(
369      String predicate, String index, HashedConsequent consequent, String pptname) {
370    if (!pptname_to_conditions.containsKey(pptname)) {
371      pptname_to_conditions.put(pptname, new HashMap<>());
372    }
373
374    Map<String, Map<String, HashedConsequent>> cluster_to_conditions =
375        pptname_to_conditions.get(pptname);
376    if (!cluster_to_conditions.containsKey(predicate)) {
377      cluster_to_conditions.put(predicate, new HashMap<>());
378    }
379
380    Map<String, HashedConsequent> conditions = cluster_to_conditions.get(predicate);
381    if (conditions.containsKey(index)) {
382      HashedConsequent old = conditions.get(index);
383      if (old.fakeFor != null && consequent.fakeFor == null) {
384        // We already saw (say) "x != y", but we're "x == y", so replace it.
385        conditions.remove(index);
386        conditions.remove(old.fakeFor);
387        conditions.put(index, consequent);
388        return true;
389      }
390      return false;
391    } else {
392      conditions.put(index, consequent);
393      return true;
394    }
395  }

397  private static boolean contains_constant_non_012(Invariant inv) {
398    if (inv instanceof daikon.inv.unary.scalar.OneOfScalar) {
399      daikon.inv.unary.scalar.OneOfScalar oneof = (daikon.inv.unary.scalar.OneOfScalar) inv;
400      // TODO: isInteresting has been removed.  Do we need to deal with it specially here?
401      // OneOf invariants that indicate a small set ( > 1 element) of
402      // possible values are not interesting, and have already been
403      // eliminated by the isInteresting check
404      long num = ((Long) oneof.elt()).longValue();
405      if (num > 2 || num < -1) {
406        return true;
407      }
408    }
409
410    return false;
411  }

413  // remove non-word characters and everything after ":::" from the
414  // program point name, leaving PackageName.ClassName.MethodName
415  private static String cleanup_pptname(String pptname) {
416    int index;
417    if ((index = pptname.indexOf("(")) > 0) {
418      pptname = pptname.substring(0, index);
419    }
420
421    if (pptname.endsWith(".")) {
422      pptname = pptname.substring(0, pptname.length() - 2);
423    }
424
425    Matcher m = non_word_pattern.matcher(pptname);
426    return m.replaceAll(".");
427  }

429  /**
430   * Prevents the occurence of "equivalent" inequalities, or inequalities which produce the same
431   * pair of splits at a program point, for example "x <= y" and "x &gt; y". Replaces "&ge;" with
432   * "<", "&le;" with ">", and "!=" with "==" so that the occurence of equivalent inequalities can
433   * be detected. However it tries not to be smart ... If there is more than one inequality in the
434   * expression, it doesn't perform a substitution.
435   */
436  private static String simplify_inequalities(String condition) {
437    if (contains_exactly_one(condition, inequality_pattern)) {
438      if (gteq_pattern.matcher(condition).find()) {
439        condition = gteq_pattern.matcher(condition).replaceFirst("<");
440      } else if (lteq_pattern.matcher(condition).find()) {
441        condition = lteq_pattern.matcher(condition).replaceFirst(">");
442      } else if (neq_pattern.matcher(condition).find()) {
443        condition = neq_pattern.matcher(condition).replaceFirst("==");
444      } else {
445        throw new Error("this can't happen");
446      }
447    }
448    return condition;
449  }
450
451  private static boolean contains_exactly_one(String string, Pattern pattern) {
452    Matcher m = pattern.matcher(string);
453    // return true if first call returns true and second returns false
454    return (m.find() && !m.find());
455  }

457  static Pattern orig_pattern, dot_class_pattern, non_word_pattern;
458  static Pattern gteq_pattern, lteq_pattern, neq_pattern, inequality_pattern;
459  static Pattern contradict_inv_pattern, useless_inv_pattern_1, useless_inv_pattern_2;
460
461  static {
462    try {
463      non_word_pattern = Pattern.compile("\\W+");
464      orig_pattern = Pattern.compile("orig\\s*\\(");
465      dot_class_pattern = Pattern.compile("\\.class");
466      inequality_pattern = Pattern.compile("[\\!<>]=");
467      gteq_pattern = Pattern.compile(">=");
468      lteq_pattern = Pattern.compile("<=");
469      neq_pattern = Pattern.compile("\\!=");
470      contradict_inv_pattern =
471          Pattern.compile("(^| && )(.*) == -?[0-9]+ &.*& \\2 == -?[0-9]+($| && )");
472      useless_inv_pattern_1 =
473          Pattern.compile("(^| && )(.*) > -?[0-9]+ &.*& \\2 > -?[0-9]+($| && )");
474      useless_inv_pattern_2 =
475          Pattern.compile("(^| && )(.*) < -?[0-9]+ &.*& \\2 < -?[0-9]+($| && )");
476    } catch (PatternSyntaxException me) {
477      throw new Error("ExtractConsequent: Error while compiling pattern " + me.getMessage());
478    }
479  }

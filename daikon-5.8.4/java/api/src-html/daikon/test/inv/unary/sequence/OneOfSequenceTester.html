001package daikon.test.inv.unary.sequence;
002
003import static org.junit.Assert.assertFalse;
004import static org.junit.Assert.assertTrue;
005
006import daikon.FileIO;
007import daikon.LogHelper;
008import daikon.PptSlice;
009import daikon.PptSlice1;
010import daikon.PptTopLevel;
011import daikon.ProglangType;
012import daikon.VarComparabilityNone;
013import daikon.VarInfo;
014import daikon.VarInfoAux;
015import daikon.inv.unary.sequence.OneOfSequence;
016import daikon.test.Common;
017import junit.framework.*;
018import org.checkerframework.checker.nullness.qual.NonNull;
019import org.junit.BeforeClass;
020import org.junit.Test;
021import org.plumelib.util.Intern;
022
023@SuppressWarnings("nullness") // testing code
024public class OneOfSequenceTester {
025
026  private VarInfo[] vars = {Common.makeHashcodeArrayVarInfo("x"), Common.makeIntArrayVarInfo("y")};
027  private PptTopLevel ppt = Common.makePptTopLevel("Foo.Baa(int):::ENTER", vars);
028  private PptSlice slicex = new PptSlice1(ppt, new VarInfo[] {vars[0]});
029  // private PptSlice slicey = new PptSlice1(ppt, new VarInfo[] {vars[1]});
030
031  private static final int DOESNT_MATTER = 0;
032
033  @BeforeClass
034  public static void setUpClass() {
035    daikon.LogHelper.setupLogs(LogHelper.INFO);
036    FileIO.new_decl_format = true;
037  }
038
039  @SuppressWarnings("interning")
040  public static VarInfo newIntArrayVarInfo(String name) {
041    VarInfo result =
042        new VarInfo(
043            name,
044            ProglangType.INT_ARRAY,
045            ProglangType.INT_ARRAY,
046            VarComparabilityNone.it,
047            VarInfoAux.getDefault());
048    return result;
049  }
050
051  @SuppressWarnings("interning")
052  public static VarInfo newHashcodeArrayVarInfo(String name) {
053    VarInfo result =
054        new VarInfo(
055            name,
056            ProglangType.HASHCODE_ARRAY,
057            ProglangType.HASHCODE_ARRAY,
058            VarComparabilityNone.it,
059            VarInfoAux.getDefault());
060    return result;
061  }
062
063  @Test
064  public void testNonNonNonNull() {
065    @NonNull OneOfSequence inv1 = (OneOfSequence) OneOfSequence.get_proto().instantiate(slicex);
066    @NonNull OneOfSequence inv2 = (OneOfSequence) OneOfSequence.get_proto().instantiate(slicex);
067
068    inv1.add_modified(Intern.intern(new long[] {19, 23}), DOESNT_MATTER);
069    inv2.add_modified(Intern.intern(new long[] {91, 0}), DOESNT_MATTER);
070
071    assertFalse(inv1.isSameFormula(inv2));
072  }
073
074  @Test
075  public void testNonNonNonNon() {
076    @NonNull OneOfSequence inv1 = (OneOfSequence) OneOfSequence.get_proto().instantiate(slicex);
077    @NonNull OneOfSequence inv2 = (OneOfSequence) OneOfSequence.get_proto().instantiate(slicex);
078
079    inv1.add_modified(Intern.intern(new long[] {19, 23}), DOESNT_MATTER);
080    inv2.add_modified(Intern.intern(new long[] {91, 32}), DOESNT_MATTER);
081
082    assertTrue(inv1.isSameFormula(inv2));
083  }
084
085  @Test
086  public void testNonNullNonNull() {
087    @NonNull OneOfSequence inv1 = (OneOfSequence) OneOfSequence.get_proto().instantiate(slicex);
088    @NonNull OneOfSequence inv2 = (OneOfSequence) OneOfSequence.get_proto().instantiate(slicex);
089
090    inv1.add_modified(Intern.intern(new long[] {19, 0}), DOESNT_MATTER);
091    inv2.add_modified(Intern.intern(new long[] {91, 0}), DOESNT_MATTER);
092
093    assertTrue(inv1.isSameFormula(inv2));
094  }
095
096  @Test
097  public void testNullNullNullNull() {
098    @NonNull OneOfSequence inv1 = (OneOfSequence) OneOfSequence.get_proto().instantiate(slicex);
099    @NonNull OneOfSequence inv2 = (OneOfSequence) OneOfSequence.get_proto().instantiate(slicex);
100
101    inv1.add_modified(Intern.intern(new long[] {0, 0}), DOESNT_MATTER);
102    inv2.add_modified(Intern.intern(new long[] {0, 0}), DOESNT_MATTER);
103
104    assertTrue(inv1.isSameFormula(inv2));
105  }
106
107  @Test
108  public void testDifferentLengths() {
109    @NonNull OneOfSequence inv1 = (OneOfSequence) OneOfSequence.get_proto().instantiate(slicex);
110    @NonNull OneOfSequence inv2 = (OneOfSequence) OneOfSequence.get_proto().instantiate(slicex);
111
112    inv1.add_modified(Intern.intern(new long[] {0, 0, 0}), DOESNT_MATTER);
113    inv2.add_modified(Intern.intern(new long[] {0, 0}), DOESNT_MATTER);
114
115    assertFalse(inv1.isSameFormula(inv2));
116  }
117}

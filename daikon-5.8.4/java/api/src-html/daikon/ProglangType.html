021/**
022 * Represents the type of a variable, for its declared type, dtrace file representation, and
023 * internal representations. ProgLangTypes are interned, so they can be == compared.
024 */
025
026// I could also consider using Class; however:
027//  * that ties this to a Java front end, as Class can't represent types of
028//    (say) C variables.  (not a compelling problem)
029//  * that loads the class, which requies that all classes available at
030//    runtime be available at inference time.  (not a compelling problem)
031//  * Class does not represent inheritance (but I can do that myself);
032//    and see isAssignableFrom, which might do all I need.
033//  * Class has no "dimensions" field.  isArray() exists, however, as does
034//    getComponentType().  I can always determine the number of dimensions
035//    from the number of leading "[" in the name, and dimensions is only
036//    really needed for parse_value.  parse_value can do literal checks and
037//    elementType() can use the name, I suppose.  Or maybe try to
038//    instantiate something, though that seems dicier.
039
040// Problem:  this doesn't currently represent inheritance, coercability, or
041// other relationships over the base types.
042

044// known_types = integral_types + ("pointer", "address")

109  @SuppressWarnings("super.invocation.invalid") // never called twice with the same arguments

112    this.base = basetype;

116  /**
117   * This can't be a constructor because it returns a canonical representation (that can be compared
118   * with ==), not necessarily a new object.
119   *
120   * @param rep the name of the type, optionally suffixed by (possibly multiple) "[]"
121   */
122  public static ProglangType parse(String rep) {
123    String new_base = rep;
124    int dims = 0;
125    while (new_base.endsWith("[]")) {
126      dims++;
127      new_base = new_base.substring(0, new_base.length() - 2);
128    }
129    return intern(new_base.intern(), dims);
130  }

132  /**
133   * Like parse, but normalizes representation types (such as converting "float" to "double"), in
134   * order to return real file representation types even if the file contains something slightly
135   * different than the prescribed format.
136   */
137  public static ProglangType rep_parse(String rep) {
138    ProglangType candidate = parse(rep);
139    if ((candidate.base == "address") || (candidate.base == "pointer")) { // interned
140      return intern("hashcode", candidate.dimensions);
141    } else if (candidate.base == "float") { // interned
142      return intern("double", candidate.dimensions);
143    } else if (candidate.base == "string") { // interned
144      return intern("java.lang.String", candidate.dimensions);
145    } else {
146      return candidate;
147    }
148  }

150  /** Convert a file representation type to an internal representation type. */
151  public ProglangType fileTypeToRepType() {
152    if ((base == BASE_HASHCODE)
153        || (base == BASE_BOOLEAN)
154        || (base == BASE_LONG)
155        || (base == BASE_LONG_LONG)
156        || (base == BASE_SHORT)) {
157      return intern(BASE_INT, dimensions);
158    }
159    return this;
160  }

178  /** @param t_base must be interned */
179  private static @Nullable ProglangType find(@Interned String t_base, int t_dims) {
180    // Disabled for performance reasons! this assertion is sound though:
181    //    assert t_base == t_base.intern();
182
183    // the string maps us to a vec of all ProglangTypes with that base
184    List<ProglangType> v = all_known_types.get(t_base);
185    if (v == null) {
186      return null;
187    }
188
189    // now search for the right dimension
190    for (ProglangType candidate : v) {
191      if (candidate.dimensions() == t_dims) {
192        return candidate;
193      }
194    }
195
196    return null;
197  }

213  private static ProglangType intern(@Interned String t_base, int t_dims) {
214    // Disabled for performance reasons! this assertion is sound though:
215    //    assert t_base == t_base.intern();
216    ProglangType existing = find(t_base, t_dims);
217    if (existing != null) {
218      return existing;
219    }
220    @SuppressWarnings("interning") // test above did not find one, so the new one is interned
221    @Interned ProglangType result = new ProglangType(t_base, t_dims);
222
223    List<ProglangType> v = all_known_types.get(t_base);
224    if (v == null) {
225      v = new ArrayList<ProglangType>();
226      all_known_types.put(t_base, v);
227    }
228
229    v.add(result);
230
231    return result;
232  }

234  /**
235   * Returns the type of elements of this. They may themselves be arrays if this is
236   * multidimensional.
237   */
238  public ProglangType elementType(@GuardSatisfied ProglangType this) {
239    // Presume that if there are no dimensions, this must be a list of
240    // objects.  Callers should really find this out from other information
241    // in the variable, but this will old code that relied on the pseudo
242    // dimensions of lists to work
243    if (dimensions == 0) {
244      return OBJECT;
245    }
246    assert base == base.intern() : "Uninterned base " + base;
247    return ProglangType.intern(base, dimensions - 1);
248  }

316  /**
317   * Like Long.parseLong(), but transform large unsigned longs (as from C's unsigned long long) into
318   * the corresponding negative Java longs. Also handles hex values that begin with 0x.
319   */
320  @SuppressWarnings("ConstantOverflow")
321  private static long myParseLong(String value) {
322    if ((value.length() == 20 && value.charAt(0) == '1')
323        || (value.length() == 19
324            && value.charAt(0) == '9'
325            && value.compareTo("9223372036854775808") >= 0)) {
326      // Oops, we got a large unsigned long, which Java, having
327      // only signed longs, will refuse to parse. We'll have to
328      // turn it into the corresponding negative long value.
329      String rest; // A substring that will be a valid positive long
330      long subtracted; // The amount we effectively subtracted to make it
331      if (value.length() == 20) {
332        rest = value.substring(1);
333        subtracted = 100000L * 100000 * 100000 * 10000; // 10^19; overflows
334      } else {
335        rest = value.substring(1);
336        subtracted = 9L * 100000 * 100000 * 100000 * 1000; // 9*10^18
337      }
338      return Long.parseLong(rest) + subtracted;
339    } else {
340      long val;
341      if ((value.length() > 2) && (value.charAt(0) == '0') && (value.charAt(1) == 'x')) {
342        val = Long.parseLong(value.substring(2), 16);
343      } else {
344        val = Long.parseLong(value);
345      }
346      // presume that 32 bit values are signed
347      if (dkconfig_convert_to_signed
348          && (((val & 0x80000000L) == 0x80000000L) && ((val & 0xFFFFFFFF00000000L) == 0))) {
349        // long orig = val;
350        val |= 0xFFFFFFFF00000000L;
351        // System.out.printf("Warning: converted %d to %d%n", orig, val);
352      }
353      return val;
354    }
355  }

357  /**
358   * Given a string representation of a value (of the type represented by this ProglangType), return
359   * the (canonicalized) interpretation of that value.
360   *
361   * <p>This ProglangType (the method receiver) is assumed to be a representation type, not an
362   * arbitrary type in the underlying programming language.
363   *
364   * <p>If the type is an array and there are any nonsensical elements in the array, the entire
365   * array is considered to be nonsensical (indicated by returning null). This is not really
366   * correct, but it is a reasonable path to take for now. (jhp, Feb 12, 2005)
367   */
368  public final @Nullable @Interned Object parse_value(
369      String value, LineNumberReader reader, String filename) {
370    // System.out.println(format() + ".parse(\"" + value + "\")");
371
372    switch (dimensions) {
373      case 0:
374        return parse_value_scalar(value, reader, filename);
375      case 1:
376        return parse_value_array_1d(value, reader, filename);
377      case 2:
378        return parse_value_array_2d(value, reader, filename);
379      default:
380        throw new Error("Can't parse a value of type " + format());
381    }
382  }

384  public final @Nullable @Interned Object parse_value_scalar(
385      String value, LineNumberReader reader, String filename) {
386    // System.out.println(format() + ".parse(\"" + value + "\")");
387
388    assert dimensions == 0;
389
390    if (base == BASE_STRING) {
391      if (value.equals("null")) {
392        return null;
393      }
394      // assert value.startsWith("\"") && value.endsWith("\"");
395      if (value.startsWith("\"") && value.endsWith("\"")) {
396        value = value.substring(1, value.length() - 1);
397      } else {
398        // Unfortunately, there is not a convenient way to communicate what
399        // the variable name is, which would make the error message even
400        // more specific.
401        if (!value.startsWith("\"")) {
402          System.out.printf(
403              "Warning: unquoted string value at %s line %d: %s%n",
404              filename, reader.getLineNumber(), value);
405        } else {
406          assert !value.endsWith("\"");
407          System.out.printf(
408              "Warning: unterminated string value at %s line %d: %s%n",
409              filename, reader.getLineNumber(), value);
410        }
411        System.out.printf(
412            "Proceeding anyway.  Please report a bug in the tool that made the data trace file.");
413      }
414      value = UtilPlume.unescapeJava(value);
415      return value.intern();
416    } else if (base == BASE_CHAR) {
417      // This will fail if the character is output as an integer
418      // (as I believe the C front end does).
419      char c;
420      if (value.length() == 1) {
421        c = value.charAt(0);
422      } else if ((value.length() == 2) && (value.charAt(0) == '\\')) {
423        c = UtilPlume.unescapeJava(value).charAt(0);
424      } else if ((value.length() == 4) && (value.charAt(0) == '\\')) {
425        Byte b = Byte.decode("0" + value.substring(1));
426        return Intern.internedLong(b.longValue());
427      } else {
428        throw new IllegalArgumentException("Bad character: " + value);
429      }
430      return Intern.internedLong(Character.getNumericValue(c));
431    }
432    // When parse_value is called from FileIO.read_ppt_decl, we have
433    // not set file_rep_type. Hence, rep_type is still file_rep_type
434    // and BASE_BOOLEAN is legal.
435    else if ((base == BASE_INT) || (base == BASE_BOOLEAN)) {
436      // File rep type might be int, boolean, or hashcode.
437      // If we had the declared type, we could do error-checking here.
438      // (Example:  no hashcode should be negative, nor any boolean > 1.)
439      if (value.equals("nonsensical")) {
440        return null;
441      }
442      if (value.equals("false") || value.equals("0")) {
443        return LongZero;
444      }
445      if (value.equals("true") || value.equals("1")) {
446        return LongOne;
447      }
448      if (value.equals("null")) {
449        return LongZero;
450      }
451      return Intern.internedLong(myParseLong(value));
452    } else if (base == BASE_DOUBLE) {
453      // Must ignore case, because dfej outputs "NaN", while dfec
454      // outputs "nan".  dfec outputs "nan", because this string
455      // comes from the C++ library.
456      if (value.equalsIgnoreCase("NaN")) {
457        return DoubleNaN;
458      }
459      if (value.equalsIgnoreCase("Infinity") || value.equals("inf")) {
460        return DoublePositiveInfinity;
461      }
462      if (value.equalsIgnoreCase("-Infinity") || value.equals("-inf")) {
463        return DoubleNegativeInfinity;
464      }
465      return Intern.internedDouble(value);
466    } else if ((base == BASE_HASHCODE)
467        || (base == BASE_LONG)
468        || (base == BASE_LONG_LONG)
469        || (base == BASE_SHORT)) {
470      throw new Error("not a rep type: illegal base type " + base);
471    } else {
472      throw new Error("unrecognized base type " + base);
473    }
474  }

476  public final @Nullable @Interned Object parse_value_array_1d(
477      String value, LineNumberReader reader, String filename) {
478    // System.out.println(format() + ".parse(\"" + value + "\")");
479
480    String value_orig = value; // we will side-effect the parameter
481
482    // variable is an array
483    assert dimensions == 1;
484
485    value = value.trim();
486
487    // Deal with [] surrounding array output (permits
488    // distinguishing between null and an array containing just null).
489    if (value.startsWith("[") && value.endsWith("]")) {
490      value = value.substring(1, value.length() - 1).trim();
491    } else {
492      throw new IllegalArgumentException("Array values must be enlosed in square brackets");
493    }
494
495    // Elements of value_strings can be null only if base == BASE_STRING.
496    // [I think they can never be null.  -MDE]
497    String[] value_strings;
498    if (value.length() == 0) {
499      value_strings = new String[0];
500    } else if (base == BASE_STRING) {
501      // This properly handles strings containing embedded spaces.
502      List<@Nullable String> v = new ArrayList<>();
503      StreamTokenizer parser = new StreamTokenizer(new StringReader(value));
504      parser.quoteChar('\"');
505      try {
506        while (parser.nextToken() != StreamTokenizer.TT_EOF) {
507          if (parser.ttype == '\"') {
508            v.add(parser.sval);
509          } else if (parser.ttype == StreamTokenizer.TT_WORD) {
510            assert parser.sval != null
511                : "@AssumeAssertion(nullness): dependent: representation invariant of StreamTokenizer";
512            if (parser.sval.equals("nonsensical")) {
513              return null;
514            }
515            assert parser.sval.equals("null");
516            v.add(null);
517          } else if (parser.ttype == StreamTokenizer.TT_NUMBER) {
518            v.add(Integer.toString((int) parser.nval));
519          } else {
520            System.out.printf(
521                "Warning: at %s line %d%n  bad ttype %c [int=%d] while parsing %s%n  Proceeding with value 'null'%n",
522                filename, reader.getLineNumber(), (char) parser.ttype, parser.ttype, value_orig);
523            v.add(null);
524          }
525        }
526      } catch (Exception e) {
527        throw new Error(e);
528      }
529      // Avoid nullness warnings about elements of value_strings
530      @SuppressWarnings("nullness")
531      String[] value_strings_result = v.toArray(new @Nullable String[0]);
532      value_strings = value_strings_result;
533    } else {
534      value_strings = Global.ws_regexp.split(value);
535    }
536    int len = value_strings.length;
537
538    // This big if ... else should deal with all the primitive types --
539    // or at least all the ones that can be rep_types.
540    // ("long" and "short" cannot be rep_types; for simplicity, variables
541    // declared as long or short have the "int" rep_type.)
542    if (base == BASE_INT) {
543      long[] result = new long[len];
544      for (int i = 0; i < len; i++) {
545        if (value_strings[i].equals("nonsensical")) {
546          return null;
547        } else if (value_strings[i].equals("null")) result[i] = 0;
548        else if (value_strings[i].equals("false")) result[i] = 0;
549        else if (value_strings[i].equals("true")) result[i] = 1;
550        else {
551          result[i] = myParseLong(value_strings[i]);
552        }
553      }
554      return Intern.intern(result);
555    } else if (base == BASE_DOUBLE) {
556      double[] result = new double[len];
557      for (int i = 0; i < len; i++) {
558        if (value_strings[i].equals("nonsensical")) {
559          return null;
560        } else if (value_strings[i].equals("null")) result[i] = 0;
561        else if (value_strings[i].equalsIgnoreCase("NaN")) result[i] = Double.NaN;
562        else if (value_strings[i].equalsIgnoreCase("Infinity") || value_strings[i].equals("inf")) {
563          result[i] = Double.POSITIVE_INFINITY;
564        } else if (value_strings[i].equalsIgnoreCase("-Infinity")
565            || value_strings[i].equals("-inf")) {
566          result[i] = Double.NEGATIVE_INFINITY;
567        } else {
568          result[i] = Double.parseDouble(value_strings[i]);
569        }
570      }
571      return Intern.intern(result);
572    } else if (base == BASE_STRING) {
573      // First, intern each String in the array ...
574      @Interned String[] value_strings_elts_interned = Intern.internStrings(value_strings);
575      // ... then, intern the entire array, and return it
576      return Intern.intern(value_strings_elts_interned);
577    } else {
578      throw new Error("Can't yet parse array of base type " + base);
579    }
580
581    // This is a more general technique; but when will we need

727  /**
728   * Return true if these two types can be sensibly compared to one another, or if one can be cast
729   * to the other. For instance, int is castable to long, but boolean is not castable to float, and
730   * int is not castable to int[]. This is a reflexive relationship, but not a transitive one
731   * because it might not be true for two children of a superclass, even though it's true for the
732   * superclass.
733   */
734  public boolean comparableOrSuperclassEitherWay(ProglangType other) {
735    if (this == other) { // ProglangType objects are interned
736      return true;
737    }
738    if (this.dimensions != other.dimensions) {
739      return false;
740    }
741    boolean thisIntegral = this.baseIsIntegral();
742    boolean otherIntegral = other.baseIsIntegral();
743    if (thisIntegral && otherIntegral) {
744      return true;
745    }
746    // Make Object castable to everything, except booleans
747    if (((this.base == BASE_OBJECT) && other.baseIsObject()) // interned strings
748        || ((other.base == BASE_OBJECT) && baseIsObject())) { // interned strings
749      return true;
750    }
751
752    return false;
753  }
754
755  /**
756   * Return true if these two types can be sensibly compared to one another, and if non-integral,
757   * whether this could be a superclass of other. A List is comparableOrSuperclassOf to a ArrayList,
758   * but not the other way around. This is a transitive method, but not reflexive.
759   */
760  public boolean comparableOrSuperclassOf(ProglangType other) {
761    if (this == other) // ProglangType objects are interned
762    return true;
763    if (this.dimensions != other.dimensions) {
764      return false;
765    }
766    boolean thisIntegral = this.baseIsIntegral();
767    boolean otherIntegral = other.baseIsIntegral();
768    if (thisIntegral && otherIntegral) {
769      return true;
770    }
771    // Make Object castable to everything, except booleans
772    if ((this.base == BASE_OBJECT) && other.baseIsObject()) // interned strings
773    return true;
774
775    return false;
776  }

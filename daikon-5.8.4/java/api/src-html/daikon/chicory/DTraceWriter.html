013/**
014 * DTraceWriter writes {@code .dtrace} program points to an output stream. It uses the trees created
015 * by the {@link DeclWriter}.
016 */
017@SuppressWarnings("nullness")
018public class DTraceWriter extends DaikonWriter {
019  // Notes:
020  //
021  //  - methodExit(): handles exits from a method
022  //    - printReturnValue: prints return value and related vars
023  //      - checkForRuntimeClass: prints return.class and value
024  //    - traceMethod(): prints out this and arguments of method
025  //      - traceMethodVars: prints out this and arguments of method
026  //        - traceLocalVars: prints arguments
027  //          - traceLocalVar: prints one argument
028  //          - checkForVarRecursion: recursive check on on argument
029  //        - traceClassVars: prints fields in a class
030
031  /** instance of a nonsensical value */
032  private static NonsensicalObject nonsenseValue = NonsensicalObject.getInstance();
033  /** instance of a nonsensical list */
034  private static List<Object> nonsenseList = NonsensicalList.getInstance();
035
036  // certain class names
037  protected static final String classClassName = "java.lang.Class";
038  protected static final String stringClassName = "java.lang.String";

040  /** Where to print output. */
041  private PrintStream outFile;

043  /** debug information about daikon variables */
044  private boolean debug_vars = false;

056  /** Prints the method entry program point in the dtrace file. */
057  public void methodEntry(
058      @GuardSatisfied DTraceWriter this,
059      MethodInfo mi,
060      int nonceVal,
061      @Nullable Object obj,
062      Object[] args) {
063    // don't print
064    if (Runtime.dtrace_closed) {
065      return;
066    }
067
068    Member member = mi.member;
069
070    // get the root of the method's traversal pattern
071    RootInfo root = mi.traversalEnter;
072    if (root == null) {
073      throw new RuntimeException("Traversal pattern not initialized at method " + mi.method_name);
074    }
075
076    if (debug_vars) {
077      System.out.printf("Entering %s%n%s%n", DaikonWriter.methodEntryName(member), root);
078      Throwable stack = new Throwable("enter traceback");
079      stack.fillInStackTrace();
080      stack.printStackTrace(System.out);
081    }
082    outFile.println(DaikonWriter.methodEntryName(member));
083    printNonce(nonceVal);
084    traverse(mi, root, args, obj, nonsenseValue);
085
086    outFile.println();
087
088    Runtime.incrementRecords();
089  }

103  /** Prints the method exit program point in the dtrace file. */
104  public void methodExit(
105      @GuardSatisfied DTraceWriter this,
106      MethodInfo mi,
107      int nonceVal,
108      @Nullable Object obj,
109      Object[] args,
110      Object ret_val,
111      int lineNum) {
112    if (Runtime.dtrace_closed) {
113      return;
114    }
115
116    Member member = mi.member;
117
118    // gets the traversal pattern root for this method exit
119    RootInfo root = mi.traversalExit;
120    if (root == null) {
121      throw new RuntimeException(
122          "Traversal pattern not initialized for method " + mi.method_name + " at line " + lineNum);
123    }
124
125    // make sure the line number is valid
126    // i.e., it is one of the exit locations in the MethodInfo for this method
127    if (mi.exit_locations == null || !mi.exit_locations.contains(lineNum)) {
128      throw new RuntimeException(
129          "The line number "
130              + lineNum
131              + " is not found in the MethodInfo for method "
132              + mi.method_name
133              + DaikonWriter.lineSep
134              + "No exit locations found in exit_locations set!");
135    }
136
137    outFile.println(DaikonWriter.methodExitName(member, lineNum));
138    printNonce(nonceVal);
139    traverse(mi, root, args, obj, ret_val);
140
141    outFile.println();
142
143    Runtime.incrementRecords();
144  }

158  // prints an invocation nonce entry in the dtrace
159  private void printNonce(@GuardSatisfied DTraceWriter this, int val) {
160    outFile.println("this_invocation_nonce");
161    outFile.println(val);
162  }

164  /**
165   * Prints the method's return value and all relevant variables. Uses the tree of
166   * DaikonVariableInfo objects.
167   *
168   * @param mi the method whose program point we are printing
169   * @param root the root of the program point's tree
170   * @param args the arguments to the method corrsponding to mi. Must be in the same order as the
171   *     .decls info is in (which is the declared order in the source code).
172   * @param thisObj the value of the "this" object at this point in the execution
173   * @param ret_val the value returned from this method, only used for exit program points
174   */
175  private void traverse(
176      @GuardSatisfied DTraceWriter this,
177      MethodInfo mi,
178      RootInfo root,
179      Object[] args,
180      Object thisObj,
181      Object ret_val) {
182    // go through all of the node's children
183    for (DaikonVariableInfo child : root) {
184
185      Object val;
186
187      if (child instanceof ReturnInfo) {
188        val = ret_val;
189      } else if (child instanceof ThisObjInfo) {
190        val = thisObj;
191      } else if (child instanceof ParameterInfo) {
192        val = args[((ParameterInfo) child).getArgNum()];
193      } else if (child instanceof FieldInfo) {
194        // can only occur for static fields
195        // non-static fields will appear as children of "this"
196
197        val = child.getMyValFromParentVal(null);
198      } else if (child instanceof StaticObjInfo) {
199        val = null;
200      } else {
201        throw new Error(
202            "Unknown DaikonVariableInfo subtype "
203                + child.getClass()
204                + " in traversePattern in DTraceWriter for info named "
205                + child.getName()
206                + " in class "
207                + "for method "
208                + mi);
209      }
210
211      traverseValue(mi, child, val);
212    }
213  }
214
215  // traverse from the traversal pattern data structure and recurse
216  private void traverseValue(
217      @GuardSatisfied DTraceWriter this, MethodInfo mi, DaikonVariableInfo curInfo, Object val) {
218
219    if (curInfo.dTraceShouldPrint()) {
220      if (!(curInfo instanceof StaticObjInfo)) {
221        outFile.println(curInfo.getName());
222        outFile.println(curInfo.getDTraceValueString(val));
223      }
224
225      if (debug_vars) {
226        String out = curInfo.getDTraceValueString(val);
227        if (out.length() > 20) out = out.substring(0, 20);
228        System.out.printf(
229            "  --variable %s [%d]= %s%n", curInfo.getName(), curInfo.children.size(), out);
230      }
231    }
232
233    // go through all of the current node's children
234    // and recurse on their values
235    if (curInfo.dTraceShouldPrintChildren()) {
236      for (DaikonVariableInfo child : curInfo) {
237        Object childVal = child.getMyValFromParentVal(val);
238        traverseValue(mi, child, childVal);
239      }
240    }
241  }

243  /**
244   * Returns a list of values of the field for each Object in theObjects.
245   *
246   * @param theObjects list of Objects, each must have the Field field
247   * @param field which field of theObjects we are probing
248   */
249  public static List<Object> getFieldValues(Field field, List<Object> theObjects) {
250    if (theObjects == null || theObjects instanceof NonsensicalList) {
251      return nonsenseList;
252    }
253
254    List<Object> fieldVals = new ArrayList<>();
255
256    for (Object theObj : theObjects) {
257      if (theObj == null) {
258        fieldVals.add(nonsenseValue);
259      } else {
260        fieldVals.add(getValue(field, theObj));
261      }
262    }
263
264    return fieldVals;
265  }

267  /**
268   * Get the value of a certain field in theObj.
269   *
270   * @param classField which field we are interested in
271   * @param theObj the object whose field we are examining. TheoObj must be null, Nonsensical, or of
272   *     a type which contains the field classField.
273   * @return the value of the classField field in theObj
274   */
275  @SuppressWarnings("deprecation") // in Java 9+, use canAccess instead of isAccessible
276  public static Object getValue(Field classField, Object theObj) {
277    // if we don't have a real object, return NonsensicalValue
278    if ((theObj == null) || (theObj instanceof NonsensicalObject)) {
279      return nonsenseValue;
280    }
281
282    Class<?> fieldType = classField.getType();
283
284    if (!classField.isAccessible()) {
285      classField.setAccessible(true);
286    }
287
288    try {
289      if (fieldType.equals(int.class)) {
290        return new Runtime.IntWrap(classField.getInt(theObj));
291      } else if (fieldType.equals(long.class)) {
292        return new Runtime.LongWrap(classField.getLong(theObj));
293      } else if (fieldType.equals(boolean.class)) {
294        return new Runtime.BooleanWrap(classField.getBoolean(theObj));
295      } else if (fieldType.equals(float.class)) {
296        return new Runtime.FloatWrap(classField.getFloat(theObj));
297      } else if (fieldType.equals(byte.class)) {
298        return new Runtime.ByteWrap(classField.getByte(theObj));
299      } else if (fieldType.equals(char.class)) {
300        return new Runtime.CharWrap(classField.getChar(theObj));
301      } else if (fieldType.equals(short.class)) {
302        return new Runtime.ShortWrap(classField.getShort(theObj));
303      } else if (fieldType.equals(double.class)) {
304        return new Runtime.DoubleWrap(classField.getDouble(theObj));
305      } else {
306        return classField.get(theObj);
307      }
308    } catch (IllegalArgumentException e) {
309      throw new Error(e);
310    } catch (IllegalAccessException e) {
311      throw new Error(e);
312    }
313  }

315  /**
316   * Similar to {@link DTraceWriter#getValue}, but used for static fields.
317   *
318   * @param classField the field whose static value to return
319   * @return the static value of the field
320   */
321  @SuppressWarnings("deprecation") // in Java 9+, use canAccess instead of isAccessible
322  public static Object getStaticValue(Field classField) {
323    if (!classField.isAccessible()) {
324      classField.setAccessible(true);
325    }
326
327    Class<?> fieldType = classField.getType();
328
329    if (Chicory.checkStaticInit) {
330      // don't force initialization!
331      if (!Runtime.isInitialized(classField.getDeclaringClass().getName())) {
332        return nonsenseValue;
333      }
334    }
335
336    try {
337      if (fieldType.equals(int.class)) {
338        return new Runtime.IntWrap(classField.getInt(null));
339      } else if (fieldType.equals(long.class)) {
340        return new Runtime.LongWrap(classField.getLong(null));
341      } else if (fieldType.equals(boolean.class)) {
342        return new Runtime.BooleanWrap(classField.getBoolean(null));
343      } else if (fieldType.equals(float.class)) {
344        return new Runtime.FloatWrap(classField.getFloat(null));
345      } else if (fieldType.equals(byte.class)) {
346        return new Runtime.ByteWrap(classField.getByte(null));
347      } else if (fieldType.equals(char.class)) {
348        return new Runtime.CharWrap(classField.getChar(null));
349      } else if (fieldType.equals(short.class)) {
350        return new Runtime.ShortWrap(classField.getShort(null));
351      } else if (fieldType.equals(double.class)) {
352        return new Runtime.DoubleWrap(classField.getDouble(null));
353      } else {
354        return classField.get(null);
355      }
356    } catch (IllegalArgumentException e) {
357      throw new Error(e);
358    } catch (IllegalAccessException e) {
359      throw new Error(e);
360    }
361  }

363  /**
364   * Return a List derived from an aray.
365   *
366   * @param arrayVal must be an array type
367   * @return a List (with correct primitive wrappers) corresponding to the array
368   */
369  public static List<Object> getListFromArray(Object arrayVal) {
370    if (arrayVal instanceof NonsensicalObject) {
371      return nonsenseList;
372    }
373
374    if (!arrayVal.getClass().isArray()) {
375      throw new RuntimeException(
376          String.format(
377              "The object \"%s\" of type %s is not an array", arrayVal, arrayVal.getClass()));
378    }
379
380    int len = Array.getLength(arrayVal);
381    List<Object> arrList = new ArrayList<>(len);
382
383    Class<?> arrType = arrayVal.getClass().getComponentType();
384
385    // have to wrap primitives in our wrappers
386    // otherwise, couldn't distinguish from a wrapped object in the
387    // target app
388    if (arrType.equals(int.class)) {
389      for (int i = 0; i < len; i++) {
390        arrList.add(new Runtime.IntWrap(Array.getInt(arrayVal, i)));
391      }
392    } else if (arrType.equals(long.class)) {
393      for (int i = 0; i < len; i++) {
394        arrList.add(new Runtime.LongWrap(Array.getLong(arrayVal, i)));
395      }
396    } else if (arrType.equals(boolean.class)) {
397      for (int i = 0; i < len; i++) {
398        arrList.add(new Runtime.BooleanWrap(Array.getBoolean(arrayVal, i)));
399      }
400    } else if (arrType.equals(float.class)) {
401      for (int i = 0; i < len; i++) {
402        arrList.add(new Runtime.FloatWrap(Array.getFloat(arrayVal, i)));
403      }
404    } else if (arrType.equals(byte.class)) {
405      for (int i = 0; i < len; i++) {
406        arrList.add(new Runtime.ByteWrap(Array.getByte(arrayVal, i)));
407      }
408    } else if (arrType.equals(char.class)) {
409      for (int i = 0; i < len; i++) {
410        arrList.add(new Runtime.CharWrap(Array.getChar(arrayVal, i)));
411      }
412    } else if (arrType.equals(short.class)) {
413      for (int i = 0; i < len; i++) {
414        arrList.add(new Runtime.ShortWrap(Array.getShort(arrayVal, i)));
415      }
416    } else if (arrType.equals(double.class)) {
417      for (int i = 0; i < len; i++) {
418        arrList.add(new Runtime.DoubleWrap(Array.getDouble(arrayVal, i)));
419      }
420    } else {
421      for (int i = 0; i < len; i++) {
422        // non-primitives
423        arrList.add(Array.get(arrayVal, i));
424      }
425    }
426
427    return arrList;
428  }

430  /**
431   * Returns a list of Strings which are the names of the runtime types in the theVals param.
432   *
433   * @param theVals list of ObjectReferences
434   * @return a list of Strings which are the names of the runtime types in the theVals param
435   */
436  public static @Nullable List<String> getTypeNameList(List<Object> theVals) {
437    // Return null rather than NonsensicalList as NonsensicalList is
438    // an array of Object and not String.
439    if (theVals == null || theVals instanceof NonsensicalList) {
440      return null;
441    }
442
443    List<String> typeNames = new ArrayList<>(theVals.size());
444
445    for (Object ref : theVals) {
446      if (ref == null) {
447        typeNames.add(null);
448      } else {
449        Class<?> type = ref.getClass();
450        type = removeWrappers(ref, type, true);
451        typeNames.add(type.getCanonicalName());
452      }
453    }
454
455    return typeNames;
456  }

458  /**
459   * Get the type of val, removing any PrimitiveWrapper if it exists For example, if we execute
460   * removeWRappers(val, boolean.class, true) where (val instanceof Runtime.PrimitiveWrapper), then
461   * the method returns boolean.class
462   *
463   * @param val the object whose type we are examining
464   * @param declared the declared type of the variable corresponding to val
465   * @param runtime should we use the runtime type or declared type?
466   * @return the variable's type, with primitive wrappers removed, or null if the value is non-null
467   */
468  public static @Nullable Class<?> removeWrappers(Object val, Class<?> declared, boolean runtime) {
469    if (!runtime) {
470      return declared;
471    }
472
473    if (val instanceof Runtime.PrimitiveWrapper) {
474      return ((Runtime.PrimitiveWrapper) val).primitiveClass();
475    }
476
477    if (val == null) {
478      return null;
479    }
480
481    return val.getClass();
482  }

500  private InvariantStatus runValue(long[] a, int count, boolean mutate) {
501    InvariantStatus finalStatus = InvariantStatus.NO_CHANGE;
502    for (int ai = 0; ai <a.length; ai++) {
503      InvariantStatus status;
504      if (mutate) {
505        status = add_mod_elem(a[ai], count);
506      } else {
507        status = check_mod_elem(a[ai], count);
508      }
509      if (status == InvariantStatus.FALSIFIED) {
510        return InvariantStatus.FALSIFIED;
511      } else if (status == InvariantStatus.WEAKENED) {
512        finalStatus = InvariantStatus.WEAKENED;
513      }
514    }
515    return finalStatus;
516  }

523  public InvariantStatus add_mod_elem(long v, int count) {

532  /**
533   * Checks a single sample to the invariant. Returns
534   * the appropriate InvariantStatus from the result
535   * of adding the sample to this.
536   */
537  public InvariantStatus check_mod_elem(long v, int count) {
538
539    // Look for v in our list of previously seen values.  If it's
540    // found, we're all set.
541    for (int i = 0; i < num_elts; i++) {
542      // if (logDetail())
543      //  log ("add_modified (" + v + ")");
544      if (((elts[i]) == ( v))) {
545        return InvariantStatus.NO_CHANGE;
546      }
547    }
548
549    if (num_elts == dkconfig_size) {
550      return InvariantStatus.FALSIFIED;
551    }
552
553    if ((is_boolean() && (num_elts == 1)) || (is_hashcode() && (num_elts == 2))) {
554      return InvariantStatus.FALSIFIED;
555    }
556
557    if (is_hashcode() && (num_elts == 1)) {
558      // Permit two object values only if one of them is null
559      if ((elts[0] != 0) && (v != 0)) {
560        return InvariantStatus.FALSIFIED;
561      }
562    }
563
564    return InvariantStatus.WEAKENED;
565  }

620        setValuesInvariant.add(e);
621      }
622      if (setValidValues.equals(setValuesInvariant)) {
623        return new DiscardInfo(this, DiscardCode.obvious,
624          "The value list matches the allowed value list");
625      }
626    }
627
628    // Look for the same property over a supersequence of this one.
629    PptTopLevel pptt = ppt.parent;
630    for (Iterator<Invariant> inv_itor = pptt.invariants_iterator(); inv_itor.hasNext(); ) {
631      Invariant inv = inv_itor.next();
632      if (inv == this) {
633        continue;
634      }
635      if (inv instanceof EltOneOf) {
636        EltOneOf other = (EltOneOf) inv;
637        if (isSameFormula(other)
638          && SubSequence.isObviousSubSequenceDynamically(this, v, other.var())) {
639          debug.fine("isObviousDyn: Returning true because isObviousSubSequenceDynamically");
640          return new DiscardInfo(this, DiscardCode.obvious, "The same property holds over a supersequence " + other.var().name());
641        }
642      }
643    }
644
645    return null;
646  }

657  @Pure
658  @Override
659  public boolean isSameFormula(Invariant o) {
660    EltOneOf other = (EltOneOf) o;
661    if (num_elts != other.num_elts) {
662      return false;
663    }
664    if (num_elts == 0 && other.num_elts == 0) {
665      return true;
666    }
667
668    sort_rep();
669    other.sort_rep();
670
671    // All nonzero hashcode values should be considered equal to each other
672    //
673    // Examples:
674    // inv1  inv2  result
675    // ----  ----  ------
676    // 19    0     false
677    // 19    22    true
678    // 0     0     true
679
680    if (is_hashcode() && other.is_hashcode()) {
681      if (num_elts == 1 && other.num_elts == 1) {
682        return ((elts[0] == 0 && other.elts[0] == 0)
683                || (elts[0] != 0 && other.elts[0] != 0));
684      } else if (num_elts == 2 && other.num_elts == 2) {
685        // add_modified allows two elements iff one is null
686        assert elts[0] == 0;
687        assert other.elts[0] == 0;
688        assert elts[1] != 0;
689        assert other.elts[1] != 0;
690
691        // Since we know the first elements of each invariant are
692        // zero, and the second elements are nonzero, we can immediately
693        // return true
694        return true;
695      } else {
696        return false;
697      }
698    }
699
700    for (int i = 0; i < num_elts; i++) {
701      if (!((elts[i]) == (other.elts[i]))) {
702        return false;
703      }
704    }
705
706    return true;
707  }

709  @Pure
710  @Override
711  public boolean isExclusiveFormula(Invariant o) {
712    if (o instanceof EltOneOf) {
713      EltOneOf other = (EltOneOf) o;
714
715      if (num_elts == 0 || other.num_elts == 0) {
716        return false;
717      }
718      for (int i = 0; i < num_elts; i++) {
719        for (int j = 0; j < other.num_elts; j++) {
720          if (((elts[i]) == (other.elts[j]))) // elements are interned
721            return false;
722        }
723      }
724
725      // Don't consider two instances of "non-null" as exclusive.
726      if (is_hashcode() && num_elts == 1
727          && other.is_hashcode() && other.num_elts == 1
728          && elts[0] != 0 && other.elts[0] != 0) {
729        return false;
730      }
731
732      // Be even more aggressive about rejecting these for use in
733      // implications in this case, since, we'd be printing them as
734      // "true"
735      /*
736      if (dkconfig_omit_hashcode_values_Simplify
737          && (is_hashcode() || other.is_hashcode())) {
738        return false;
739      }
740      */
741
742      return true;
743    }
744
745    // Many more checks can be added here:  against nonzero, modulus, etc.
746    if ((o instanceof EltNonZero) && (num_elts == 1) && (elts[0] == 0)) {
747      return true;
748    }
749    long elts_min = Long.MAX_VALUE;
750    long elts_max = Long.MIN_VALUE;
751    for (int i = 0; i < num_elts; i++) {
752      elts_min = Math.min(elts_min, elts[i]);
753      elts_max = Math.max(elts_max, elts[i]);
754    }
755    if ((o instanceof LowerBound) && (elts_max < ((LowerBound)o).min())) {
756      return true;
757    }
758    if ((o instanceof UpperBound) && (elts_min > ((UpperBound)o).max())) {
759      return true;
760    }
761
762    return false;
763  }

805  /**
806   * Merge the invariants in invs to form a new invariant. Each must be a EltOneOf invariant.
807   * This code finds all of the oneof values from each of the invariants and returns the merged
808   * invariant (if any).
809   *
810   * @param invs list of invariants to merge. The invariants must all be of the same type and should
811   *     come from the children of parent_ppt. They should also all be permuted to match the
812   *     variable order in parent_ppt.
813   * @param parent_ppt slice that will contain the new invariant
814   */
815  @Override
816  public @Nullable Invariant merge(List<Invariant> invs, PptSlice parent_ppt) {
817
818    // Create the initial parent invariant from the first child
819    EltOneOf  first = (EltOneOf) invs.get(0);
820    EltOneOf result = first.clone();
821    result.ppt = parent_ppt;
822
823    // Loop through the rest of the child invariants
824    for (int i = 1; i < invs.size(); i++ ) {
825
826      // Get this invariant
827      EltOneOf inv = (EltOneOf) invs.get(i);
828
829      // Loop through each distinct value found in this child and add
830      // it to the parent.  If the invariant is falsified, there is no parent
831      // invariant
832      for (int j = 0; j < inv.num_elts; j++) {
833        long val = inv.elts[j];
834
835        InvariantStatus status = result.add_mod_elem(val, 1);
836        if (status == InvariantStatus.FALSIFIED) {
837
838          result.log("%s", "child value '" + val + "' destroyed oneof");
839
840          return null;
841        }
842      }
843    }
844
845    result.log("Merged '%s' from %s child invariants", result.format(), invs.size());
846    return result;
847  }

861  /**
862   * Returns true if every element in this invariant is contained in the specified state. For
863   * example if x = 1 and the state contains 1 and 2, true will be returned.
864   */
865  @Override
866  public boolean state_match(Object state) {
867
868    if (num_elts == 0) {
869      return false;
870    }
871
872    if (!(state instanceof long[])) {
873      System.out.println("state is of class '" + state.getClass().getName()
874                          + "'");
875    }
876    long[] e = (long[]) state;
877    for (int i = 0; i < num_elts; i++) {
878      boolean match = false;
879      for (int j = 0; j < e.length; j++) {
880        if (elts[i] == e[j]) {
881          match = true;
882          break;
883        }
884      }
885      if (!match) {
886        return false;
887      }
888    }
889    return true;
890  }

892  /**
893   * Map that holds dummy hashcode values for hashcodes. Each hashcode seen is assigned a small
894   * integer in the order they are seen. These values will be consistent as long as new hashcodes do
895   * not appear in the output. Not a perfect fix for regressions consistency, but workable.
896   */
897  private static Map<Long,Long> dummy_hashcode_vals = new LinkedHashMap<>();
898  private static long next_dummy_hashcode = 1001;
899
900  private long get_hashcode_val(@GuardSatisfied EltOneOf this, long hashcode) {
901    if (!dkconfig_omit_hashcode_values_Simplify) {
902      return hashcode;
903    }
904
905    Long val = dummy_hashcode_vals.get(hashcode);
906    if (val != null) {
907      return val;
908    }
909    dummy_hashcode_vals.put(hashcode, next_dummy_hashcode);
910    return (next_dummy_hashcode++);
911  }

101  /**
102   * Create a binary PptSplitter with the specied splitter for the specified PptTopLevel parent. The
103   * parent should be a leaf (i.e., a numbered exit point).
104   */
105  public PptSplitter(PptTopLevel parent, Splitter splitter) {
106
107    this.parent = parent;
108    this.splitter = splitter;
109    ppts =
110        new PptTopLevel[] {
111          new PptConditional(parent, splitter, false), new PptConditional(parent, splitter, true)

116          "PptSplitter("
117              + parent.name()

119              + splitter.condition()
120              + "): "
121              + parent.var_infos.length
122              + " VarInfos");
123      for (int ii = 0; ii < parent.var_infos.length; ii++) {
124        debug.fine(
125            "  VarInfo #"
126                + ii
127                + ": at parent="
128                + parent.var_infos[ii].name()
129                + " at ppts[0]="
130                + ppts[0].var_infos[ii].name()
131                + " at ppts[1]="
132                + ppts[1].var_infos[ii].name());
133      }
134    }
135  }

188  /**
189   * Chooses the correct conditional point based on the values in this sample. Returns null if none
190   * is applicable.
191   */
192  public @Nullable PptConditional choose_conditional(ValueTuple vt) {
193
194    // Currently only binary implications are supported
195    assert ppts.length == 2;
196
197    boolean splitter_test;
198    try {
199      splitter_test = ((PptConditional) ppts[0]).splitter.test(vt);
200    } catch (Throwable e) {
201      // If an exception is thrown, don't put the data on either side
202      // of the split.
203      if (false) { // need to add a debugging switch
204        System.out.println(
205            "Exception thrown in PptSplitter.choose_conditional() for " + ppts[0].name());
206        System.out.println("Vars = " + Debug.related_vars(ppts[0], vt));
207      }
208      return null;
209    }
210
211    // Choose the appropriate conditional point based on the condition result
212    return ((PptConditional) ppts[splitter_test ? 0 : 1]);
213  }

215  /** Adds implication invariants based on the invariants found on each side of the split. */
216  @SuppressWarnings("flowexpr.parse.error") // private field
217  @RequiresNonNull({"parent.equality_view", "NIS.all_suppressions", "NIS.suppressor_map"})
218  public void add_implications() {
219
220    // Currently only binary implications are supported
221    assert ppts.length == 2;
222
223    // Create any NIS suppressed invariants in each conditional
224    @SuppressWarnings({"unchecked", "rawtypes"})
225    List<Invariant> suppressed_invs[] =
226        (ArrayList<Invariant>[]) new @Nullable ArrayList[ppts.length];
227    for (int i = 0; i < ppts.length; i++) {
228      suppressed_invs[i] = NIS.create_suppressed_invs(ppts[i]);
229    }
230
231    add_implications_pair();
232
233    // Remove all of the NIS suppressed invariants that we just created.
234    for (int i = 0; i < ppts.length; i++) {
235      ppts[i].remove_invs(suppressed_invs[i]);
236    }
237  }

239  /**
240   * Given a pair of conditional program points, form implications from the invariants true at each
241   * one.
242   *
243   * <p>The algorithm first divides the invariants into two groups:
244   *
245   * <ol>
246   *   <li>the "same" invariants: those that are true at both program points.
247   *   <li>the "different" invariants: all other invariants.
248   * </ol>
249   *
250   * The "exclusive" invariants (a subset of the "different" inviariants) are true at one program
251   * point, and their negation is true at the other program point.
252   *
253   * <p>At the first program point, for each exclusive invariant and each different invariant,
254   * create a conditional of the form "exclusive &hArr; different". Do the same at the second
255   * program point.
256   *
257   * <p>This method is correct only if the two conditional program points fully partition the input
258   * space (their conditions are negations of one another). For instance, suppose there is a
259   * three-way split with the following invariants detected at each:
260   *
261   * <pre>
262   *   {A,B}  {!A,!B}  {A,!B}
263   * </pre>
264   *
265   * Examining just the first two would suggest that "A &hArr; B" is valid, but in fact that is a
266   * false inference. Note that this situation can occur if the splitting condition uses variables
267   * that can ever be missing. (Or, presumably, if the condition ever cannot be computed.)
268   */
269  @RequiresNonNull("parent.equality_view")
270  private void add_implications_pair() {
271
272    for (PptTopLevel pchild : ppts) {
273      // System.out.printf("splitter child = %s%n", pchild.name());
274      if (pchild.equality_view == null) {
275        System.out.printf("this: %s%n", this);
276        System.out.printf("pchild: %s[%s]%n", pchild, System.identityHashCode(pchild));
277        System.out.printf("pchild.children: %s%n", pchild.children);
278        for (PptRelation rel : pchild.children) {
279          System.out.printf("  relation = %s%n", rel);
280        }
281        System.out.printf("parent: %s%n", parent);
282        throw new Error();
283      }
284    }
285
286    debug.fine("add_implications_pair: parent = " + parent.name);
287
288    // Maps permuted invariants to their original invariants
289    Map<Invariant, Invariant> orig_invs = new LinkedHashMap<>();
290
291    List<@KeyFor("orig_invs") Invariant[]> exclusive_invs_vec =
292        new ArrayList<@KeyFor("orig_invs") Invariant[]>();
293
294    // Does not contain anything that is in exclusive_invs_vec.
295    // (Those may be added temporarily, but are removed later.)
296    List<@Nullable @KeyFor("orig_invs") Invariant[]> different_invs_vec =
297        new ArrayList<@Nullable @KeyFor("orig_invs") Invariant[]>();
298
299    /// ??? MDE
300    // Loop through each possible parent slice
301    List<VarInfo[]> slices = possible_slices();
302
303    int num_children = ppts.length;
304
305    for (VarInfo[] vis : slices) {
306
307      // Each element of invs[i] is an invariant from the i-th child, permuted to
308      // the parent (and with a parent slice as its ppt slot).
309      @SuppressWarnings({"unchecked", "rawtypes"})
310      /*NNC:@MonotonicNonNull*/ List<Invariant> invs[] =
311          (ArrayList<Invariant>[]) new @Nullable ArrayList[num_children];
312
313      // find the parent slice
314      PptSlice pslice = parent.get_or_instantiate_slice(vis);
315
316      // Daikon.debugProgress.fine ("    slice: " + pslice.name());
317
318      // Loop through each child ppt
319      for (int childno = 0; childno < num_children; childno++) {
320        PptTopLevel child_ppt = ppts[childno];
321
322        assert child_ppt.equality_view != null : child_ppt.name();
323        assert parent.equality_view != null : parent.name();
324
325        invs[childno] = new ArrayList<Invariant>(); // permuted to parent
326
327        // vis is in parent order.  Find corresponding child vis, in child order.
328        // Each of these arrays contains child vis.
329        /*NNC:@MonotonicNonNull*/ VarInfo[] cvis_non_canonical = new VarInfo[vis.length];
330        /*NNC:@MonotonicNonNull*/ VarInfo[] cvis = new VarInfo[vis.length];
331        /*NNC:@MonotonicNonNull*/ VarInfo[] cvis_sorted = new VarInfo[vis.length];
332        for (int kk = 0; kk < vis.length; kk++) {
333          cvis_non_canonical[kk] = matching_var(child_ppt, vis[kk]);
334          cvis[kk] = cvis_non_canonical[kk].canonicalRep();
335          cvis_sorted[kk] = cvis[kk];
336        }
337        Arrays.sort(cvis_sorted, VarInfo.IndexComparator.getInstance());
338
339        cvis_non_canonical = castNonNullDeep(cvis_non_canonical); // https://tinyurl.com/cfissue/986
340        cvis = castNonNullDeep(cvis); // https://tinyurl.com/cfissue/986
341        cvis_sorted = castNonNullDeep(cvis_sorted); // https://tinyurl.com/cfissue/986
342
343        // Look for an equality invariant in the non-canonical slice (if any).
344        // Note that only an equality invariant can exist in a non-canonical
345        // slice.  If it does exist, we want it rather than the canonical
346        // form (which for equality invariants will always be of the form
347        // 'a == a').
348        Invariant eq_inv = null;
349        if (!Arrays.equals(cvis_non_canonical, cvis)) {
350          PptSlice nc_slice = child_ppt.findSlice(cvis_non_canonical);
351          if (nc_slice != null) {
352            if (nc_slice.invs.size() != 1) {
353              // Impossible: multiple invariants found
354              System.out.println("Found " + nc_slice.invs.size() + " invs at " + nc_slice);
355              for (Invariant inv2 : nc_slice.invs) {
356                System.out.println(" -- inv = " + inv2);
357              }
358              for (VarInfo cvi : cvis_non_canonical) {
359                System.out.println(" -- equality set = " + cvi.equalitySet.shortString());
360              }
361              throw new Error("nc_slice.invs.size() == " + nc_slice.invs.size());
362            }
363            eq_inv = nc_slice.invs.get(0);
364            debug.fine("Found eq inv " + eq_inv);
365          }
366        }
367
368        // Find the corresponding slice
369        PptSlice cslice = child_ppt.findSlice(cvis_sorted);
370        if (cslice == null) {
371          if (eq_inv != null) {
372            // There is trouble.  Print a lot of debugging information.
373            System.out.println("cvis_non_canonical:");
374            for (VarInfo cvi : cvis_non_canonical) {
375              System.out.println("  " + cvi);
376            }
377            System.out.println("cvis:");
378            for (VarInfo cvi : cvis) {
379              System.out.println("  " + cvi);
380            }
381            System.out.println("cvis_sorted:");
382            for (VarInfo cvi : cvis_sorted) {
383              System.out.println("  " + cvi);
384            }
385            if (DynamicConstants.dkconfig_use_dynamic_constant_optimization) {
386              assert child_ppt.constants != null
387                  : "@AssumeAssertion(nullness):  dependent:  config var";
388              System.out.println("constant values for cvis_sorted:");
389              for (VarInfo cvi : cvis_sorted) {
390                System.out.println("  " + child_ppt.constants.getConstant(cvi));
391              }
392            }
393            String eq_inv_ppt = eq_inv.ppt.toString();
394            assert eq_inv.ppt.equals(child_ppt.findSlice(cvis_non_canonical));
395
396            System.out.println("All child_ppt slices: ");
397            for (PptSlice slice : child_ppt.views_iterable()) {
398              System.out.println("  " + slice);
399            }
400
401            // found slice on non-canonical, but didn't find it here
402            throw new RuntimeException(
403                String.join(
404                    System.lineSeparator(),
405                    "found eq_inv",
406                    "  " + eq_inv,
407                    "  @" + eq_inv_ppt,
408                    "  but can't find slice for " + Arrays.toString(cvis_sorted)));
409          }
410          // If no slice, just give up?
411          continue;
412        }
413
414        // Copy each invariant permuted to the parent.
415        // This permits them to be directly compared to one another.
416        int[] permute = PptTopLevel.build_permute(cvis_sorted, cvis);
417        for (Invariant orig_inv : cslice.invs) {
418          Invariant inv = orig_inv.clone_and_permute(permute);
419          inv.ppt = pslice;
420          if ((eq_inv != null) && orig_inv.getClass().equals(eq_inv.getClass())) orig_inv = eq_inv;
421          assert !orig_invs.containsKey(inv);
422          orig_invs.put(inv, orig_inv);
423          invs[childno].add(inv);
424        }
425      } // children loop
426
427      invs = castNonNullDeep(invs); // https://tinyurl.com/cfissue/986
428
429      // If neither child slice has invariants there is nothing to do
430      if ((invs[0].size() == 0) && (invs[1].size() == 0)) {
431        if (pslice.invs.size() == 0) parent.removeSlice(pslice);
432        continue;
433      }
434
435      if (pslice.invs.size() == 0) {
436        debug.fine("PptSplitter: created new slice " + Arrays.toString(vis) + " @" + parent.name);
437      }
438
439      // Add any exclusive conditions for this slice to the list
440      @SuppressWarnings("keyfor") // need qualifier parameter to Invariants
441      List<@KeyFor("orig_invs") Invariant[]> ec = exclusive_conditions(invs[0], invs[1]);
442      exclusive_invs_vec.addAll(ec);
443
444      // Add any invariants that are different to the list
445      @SuppressWarnings("keyfor") // need qualifier parameter to Invariants
446      List<@Nullable @KeyFor("orig_invs") Invariant[]> di = different_invariants(invs[0], invs[1]);
447      different_invs_vec.addAll(di);
448    } // slices.iterator() loop
449
450    if (debug.isLoggable(Level.FINE)) {
451      debug.fine("Found " + exclusive_invs_vec.size() + " exclusive conditions ");
452      for (Invariant[] invs : exclusive_invs_vec) {
453        invs[0].log("exclusive condition with %s", invs[1].format());
454        invs[1].log("exclusive condition with %s", invs[0].format());
455        debug.fine("-- " + invs[0] + " -- " + invs[1]);
456      }
457      debug.fine("Found " + different_invs_vec.size() + " different invariants ");
458      for (@Nullable Invariant[] invs : different_invs_vec) {
459        if (invs[0] != null) invs[0].log("%s differs from %s", invs[0], invs[1]);
460        if (invs[1] != null) invs[1].log("%s differs from %s", invs[0], invs[1]);
461        debug.fine("-- " + invs[0] + " -- " + invs[1]);
462      }
463    }
464
465    PptTopLevel ppt1 = ppts[0];
466    PptTopLevel ppt2 = ppts[1];
467
468    // Add the splitting condition as an exclusive condition if requested
469    if ((splitter != null) && dkconfig_dummy_invariant_level > 0) {
470      if (exclusive_invs_vec.size() == 0 || dkconfig_dummy_invariant_level >= 2) {
471        // As a last resort, try using the user's supplied DummyInvariant
472        debug.fine("addImplications: resorting to dummy");
473        PptConditional cond1 = (PptConditional) ppt1;
474        PptConditional cond2 = (PptConditional) ppt2;
475        debug.fine("addImplications: cond1 " + cond1 + " cond2 " + cond2);
476        cond1.splitter.instantiateDummy(ppt1);
477        cond2.splitter.instantiateDummy(ppt2);
478        DummyInvariant dummy1 = cond1.dummyInvariant();
479        DummyInvariant dummy2 = cond2.dummyInvariant();
480        debug.fine("addImplications: dummy1 " + dummy1 + " dummy2 " + dummy2);
481        if (dummy1 != null && dummy1.valid && dummy2 != null && dummy2.valid) {
482          assert !cond1.splitter_inverse;
483          assert cond2.splitter_inverse;
484          dummy2.negate();
485          orig_invs.put(dummy1, dummy1);
486          orig_invs.put(dummy2, dummy2);
487          @KeyFor("orig_invs") Invariant[] dummy_pair = new @KeyFor("orig_invs") Invariant[] {dummy1, dummy2};
488          exclusive_invs_vec.add(dummy_pair);
489          // Don't add the dummy_pair, as it would just be removed afterward.
490          // different_invs_vec.add(dummy_pair);
491        } else {
492          // nothing to do
493        }
494      }
495    }
496
497    // If there are no exclusive conditions, we can do nothing here
498    if (exclusive_invs_vec.size() == 0) {
499      debug.fine("addImplications: no exclusive conditions");
500      return;
501    }
502
503    // Remove exclusive invariants from the different invariants list.
504    // It would be better not to have added them in the first place,
505    // but this is easier for now.
506    for (Iterator<@Nullable @KeyFor("orig_invs") Invariant[]> ii = different_invs_vec.iterator();
507        ii.hasNext(); ) {
508      @Nullable Invariant[] diff_invs = ii.next();
509      if (diff_invs[0] != null) {
510        assert diff_invs[1] == null;
511        // debug.fine ("Considering inv0 " + diff_invs[0]);
512        for (Invariant[] ex_invs : exclusive_invs_vec) {
513          if (ex_invs[0] == diff_invs[0]) {
514            debug.fine("removed exclusive invariant " + ex_invs[0]);
515            ii.remove();
516            break;
517          }
518        }
519      } else {
520        assert diff_invs[1] != null;
521        // debug.fine ("Considering inv1 " + diff_invs[1]);
522        for (Invariant[] ex_invs : exclusive_invs_vec) {
523          if (ex_invs[1] == diff_invs[1]) {
524            debug.fine("removed exclusive invariant " + ex_invs[1]);
525            ii.remove();
526            break;
527          }
528        }
529      }
530    }
531
532    // Get the canonical predicate invariants from the exclusive list.
533    // We pick the first one that is neither obvious nor suppressed.
534    // If all are either obvious or suppressed, we just pick the first
535    // one in the list.
536    // TODO: Why do we want canonical predicate invariants?  How will they be used?  It seems that
537    // different elements of this list have different semantics.
538    // TODO: After this loop, might the two canonical invariants not be exclusive with one another?
539    // TODO: con_invs should probably be renamed to canon_invs.
540    /*NNC:@MonotonicNonNull*/ Invariant[] con_invs = new Invariant[2];
541    for (Invariant[] invs : exclusive_invs_vec) {
542      assert invs.length == 2;
543      for (int jj = 0; jj < con_invs.length; jj++) {
544        if (con_invs[jj] == null) {
545          @SuppressWarnings("nullness") // map
546          @NonNull Invariant orig = orig_invs.get(invs[jj]);
547          assert orig != null : "Not in orig_invs: " + invs[jj] + " " + invs[jj].getClass();
548          if ((orig.isObvious() == null) && !orig.is_ni_suppressed()) con_invs[jj] = invs[jj];
549        }
550      }
551    }
552    Invariant[] first = exclusive_invs_vec.get(0);
553    for (int jj = 0; jj < con_invs.length; jj++) {
554      if (con_invs[jj] == null) {
555        System.out.println(
556            "Warning: No non-obvious non-suppressed exclusive"
557                + " invariants found in "
558                + parent.name);
559        // throw new Error();
560        con_invs[jj] = first[jj];
561      }
562    }
563    con_invs = castNonNullDeep(con_invs); // https://tinyurl.com/cfissue/986
564
565    // Create double-implications for each exclusive invariant
566    for (Invariant[] invs : exclusive_invs_vec) {
567      for (int jj = 0; jj < con_invs.length; jj++) {
568        if (con_invs[jj] != invs[jj]) {
569          add_implication(parent, con_invs[jj], invs[jj], true, orig_invs);
570        }
571      }
572    }
573
574    // Create single implication for each different invariant
575    for (@Nullable Invariant[] invs : different_invs_vec) {
576      for (int jj = 0; jj < con_invs.length; jj++) {
577        if (invs[jj] != null) add_implication(parent, con_invs[jj], invs[jj], false, orig_invs);
578      }
579    }
580  } // add_implications_pair

582  /**
583   * Returns a list of all possible slices that may appear at the parent. The parent must have
584   * already been created by merging the invariants from its child conditionals.
585   *
586   * <p>This is a subset of the slices that actually exist at the parent because the parent may have
587   * implications created from invariants in child slices that only exist in one child (and thus
588   * don't exist in the parent).
589   */
590  @RequiresNonNull("parent.equality_view")
591  private List<VarInfo[]> possible_slices() {
592
593    List<VarInfo[]> result = new ArrayList<>();
594
595    // Get an array of leaders at the parent to build slices over
596    VarInfo[] leaders = parent.equality_view.get_leaders_sorted();
597
598    // Create unary views
599    for (int i = 0; i < leaders.length; i++) {
600      if (parent.is_slice_ok(leaders[i])) {
601        result.add(new VarInfo[] {leaders[i]});
602      }
603    }
604
605    // Create binary views
606    for (int i = 0; i < leaders.length; i++) {
607      for (int j = i; j < leaders.length; j++) {
608        if (parent.is_slice_ok(leaders[i], leaders[j])) {
609          result.add(new VarInfo[] {leaders[i], leaders[j]});
610        }
611      }
612    }
613
614    /// Expensive!
615    /// ??? MDE
616    // Create ternary views
617    for (int i = 0; i < leaders.length; i++) {
618      for (int j = i; j < leaders.length; j++) {
619        for (int k = j; k < leaders.length; k++) {
620          if (parent.is_slice_ok(leaders[i], leaders[j], leaders[k])) {
621            result.add(new VarInfo[] {leaders[i], leaders[j], leaders[k]});
622          }
623        }
624      }
625    }
626
627    return result;
628  }

632  private boolean at_same_ppt(List<Invariant> invs1, List<Invariant> invs2) {

635        new CollectionsPlume.MergedIterator2<Invariant>(invs1.iterator(), invs2.iterator());

659  List<Invariant[]> exclusive_conditions(List<Invariant> invs1, List<Invariant> invs2) {

661    List<Invariant[]> result = new ArrayList<>();
662    for (Invariant inv1 : invs1) {
663      for (Invariant inv2 : invs2) {
664        // // This is a debugging tool, to make sure that various versions
665        // // of isExclusiveFormula remain coordinated.  (That's also one
666        // // reason we don't break out of the loop early:  also, there will
667        // // be few invariants in a slice, so breaking out is of minimal
668        // // benefit.)
669        // assert inv1.isExclusiveFormula(inv2)
670        //                  == inv2.isExclusiveFormula(inv1)
671        //              : "Bad exclusivity: " + inv1.isExclusiveFormula(inv2)
672        //               + " " + inv2.isExclusiveFormula(inv1)
673        //               + "    " + inv1.format() + "    " + inv2.format();
674        if (inv1.isExclusiveFormula(inv2)) {
675          result.add(new Invariant[] {inv1, inv2});
676        }
677      }
678    }
679    return result;

745  /**
746   * If the implication specified by predicate and consequent is a valid implication, adds it to the
747   * joiner view of parent.
748   *
749   * @param orig_invs maps permuted invariants to their original invariants
750   */
751  public void add_implication(
752      PptTopLevel ppt,
753      Invariant predicate,
754      Invariant consequent,
755      boolean iff,
756      Map<Invariant, Invariant> orig_invs) {
757    debug.fine("add_implication " + ppt + " " + predicate + " " + consequent + " " + iff);
758
759    assert predicate != null;
760    assert consequent != null;
761
762    @SuppressWarnings("nullness") // map: method precondition
763    @NonNull Invariant orig_pred = orig_invs.get(predicate);
764    Invariant orig_cons = orig_invs.get(consequent);
765    if (orig_cons == null) {
766      assert (consequent instanceof DummyInvariant);
767      orig_cons = consequent;
768    }
769    assert orig_pred != null : "predicate is not in orig_invs: " + predicate;
770    assert orig_cons != null;
771
772    // Don't add consequents that are obvious or suppressed.
773    // JHP: Jan 2005: It might be better to create them anyway and
774    // only suppress them in printing.  Also, this could possibly be
775    // better implemented by changing the way that we create the list
776    // of invariants that is in one conditional and not in the other
777    // to not include an invariant if it is suppressed on the other
778    // side.  This would have the pleasant side effect of not forcing
779    // all of the suppressed invariants to be created before
780    // determining implications.
781    if (orig_cons.isObvious() != null) {
782      debug.fine("add_implication obvious: " + orig_cons.isObvious().format());
783      return;
784    }
785    if (orig_cons.is_ni_suppressed()) {
786      debug.fine("add_implication suppressed: " + orig_cons.is_ni_suppressed());
787      return;
788    }
789
790    // System.out.println("add_implication:");
791    // System.out.println("  predicate = " + predicate.format());
792    // System.out.println("  consequent= " + consequent.format());
793    // System.out.println("  orig_pred = " + orig_pred.format());
794    // System.out.println("  orig_cons = " + orig_cons.format());
795
796    if (dkconfig_split_bi_implications && iff) {
797      Implication imp =
798          Implication.makeImplication(ppt, predicate, consequent, false, orig_pred, orig_cons);
799      if (imp != null) ppt.joiner_view.addInvariant(imp);
800      imp = Implication.makeImplication(ppt, consequent, predicate, false, orig_cons, orig_pred);
801      if (imp != null) {
802        ppt.joiner_view.addInvariant(imp);
803      }
804
805      return;
806    }
807
808    Implication imp =
809        Implication.makeImplication(ppt, predicate, consequent, iff, orig_pred, orig_cons);
810    if (imp == null) {
811      // The predicate is the same as the consequent, or the implication
812      // already exists.
813      debug.fine("add_implication imp == null");
814      return;
815    }
816
817    ppt.joiner_view.addInvariant(imp);
818  }

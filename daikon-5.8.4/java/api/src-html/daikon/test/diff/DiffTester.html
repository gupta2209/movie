535  private static String printTree(RootNode root) {
536    ByteArrayOutputStream baos = new ByteArrayOutputStream();
537    PrintStream ps = new PrintStream(baos);
538    PrintAllVisitor v = new PrintAllVisitor(ps, false, true);
539    root.accept(v);
540    return baos.toString();

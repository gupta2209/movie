043  // Reads an AST from the input stream, applies the visitor to the AST,
044  // reformats only to insert comments, and writes the resulting AST to the
045  // output stream.
046  public static void applyVisitorInsertComments(
047      String javafilename, Node root, Writer output, AnnotateVisitor visitor) {
048    root.accept(visitor);
049    root.accept(new InsertCommentFormatter(visitor.addedComments));
050    PrintWriter writer = new PrintWriter(output, true);
051    for (int i = 0; i < visitor.javaFileLines.size(); i++) {
052      writer.println(visitor.javaFileLines.get(i));
053    }
054    writer.close();
055
056    // root.accept(new TreeDumper(output));
057  }

059  // Reads an AST from the input stream, applies the visitor to the AST,
060  // completely reformats the Ast (losing previous formating), and writes
061  // the resulting AST to the output stream.
062  public static void applyVisitorReformat(Reader input, Writer output, Visitor visitor) {
063    JavaParser parser = new JavaParser(input);
064    Node root;
065    try {
066      root = parser.CompilationUnit();
067    } catch (ParseException e) {
068      e.printStackTrace();
069      throw new Daikon.UserError("ParseException in applyVisitorReformat");
070    }
071    root.accept(visitor);
072    // This is unfortunately necessary because TreeDumper dies if line or
073    // column numbers are out of sync.  Also see InsertCommentFormatter and
074    // applyVisitorInsertComments.
075    root.accept(new TreeFormatter(2, 80));
076    root.accept(new TreeDumper(output));
077  }

083  // Formats an AST as a String.
084  // This version does not reformat the tree (which blows away formatting
085  // information).  The call to "removeWhitespace" may do the wrong thing
086  // for embedded strings, however.  In any event, the result is not
087  // intended for direct human consumption.
088  public static String format(Node n) {
089    StringWriter w = new StringWriter();
090    n.accept(new TreeDumper(w));
091    // This is incorrect. A "//" comment ending in a period, for example, will
092    // cause the line after it to become part of the comment as well.
093    // If not intended for human consumption, why remove whitespace?
094    // return removeWhitespace(w.toString());
095    return removeWhitespace(quickFixForInternalComment(w.toString()));
096  }

098  // Formats an AST as a String.
099  // This is a debugging version that outputs all tree nodes:
100  // terminals and non-terminals alike.
101  public static String formatEntireTree(Node n) {
102    StringWriter w = new StringWriter();
103    n.accept(new TreeDumper(w, true));
104    // delete empty lines, but otherwise leave text alone
105    return w.toString().replaceAll("(?m)^[ \t]*\r?\n", "");

108  // This translates a line that looks like this:
109  //  a statement; // a comment
110  // into
111  //  a statement; // a comment //
112  public static String quickFixForInternalComment(String s) {
113    StringBuilder b = new StringBuilder();
114    String[] split = UtilPlume.splitLines(s);
115    for (int i = 0; i < split.length; i++) {
116      String line = split[i];
117      b.append(line);
118      if (line.indexOf("//") != -1) {
119        b.append("//");
120        b.append(lineSep);
121        b.append("/* */");
122      }
123      b.append(lineSep);
124    }
125    return b.toString();
126  }

128  // Formats the line enclosing a node
129  public static String formatCurrentLine(Node n) {
130    Node current = n;
131    while (current.getParent() != null && print(current.getParent()).indexOf(lineSep) < 0) {
132      current = current.getParent();
133    }
134    return print(current);
135  }

154  // Creates an AST from a String
155  public static Node create(String type, Class<?>[] argTypes, Object[] args, String stringRep) {
156    JavaParser parser = new JavaParser(new StringReader(stringRep));
157    Node n;
158    try {
159      Method m = JavaParser.class.getMethod(type, argTypes);
160      n = (Node) m.invoke(parser, args);
161    } catch (Exception e) {
162      System.err.println("create(" + type + ", \"" + stringRep + "\")");
163      e.printStackTrace();
164      throw new Daikon.UserError("Error in Ast.create");
165    }
166    return n;
167  }

179    String name = print(p.f4);

183    } else {

188  // f2 -> Type()
189  // f4 -> VariableDeclaratorId()
190  public static String getType(FormalParameter fp) {
191
192    StringWriter w = new StringWriter();
193    fp.accept(new TreeDumper(w));
194
195    FormalParameter p = (FormalParameter) create("FormalParameter", w.toString());
196
197    p.accept(new TreeFormatter());
198
199    String type = print(p.f2);
200    String name = print(p.f4);
201
202    // print() removes whitespace around brackets, so this test is safe.
203    while (name.endsWith("[]")) {
204      type += "[]";
205      name = name.substring(0, name.length() - 2);
206    }
207
208    return type;
209  }

254  /**
255   * Returns the classname if the given type declaration declares a ClassOrInterfaceDeclaration.
256   * Otherwise returns null.
257   */
258  public static @Nullable @BinaryName String getClassNameForType(TypeDeclaration d) {
259
260    // Grammar production for TypeDeclaration:
261    // f0 -> ";"
262    //       | Modifiers() ( ClassOrInterfaceDeclaration(modifiers) | EnumDeclaration(modifiers) |
263    // AnnotationTypeDeclaration(modifiers) )
264
265    NodeChoice c = d.f0;
266    if (c.which == 0) {
267      return null;
268    } else {
269      NodeSequence seq = (NodeSequence) c.choice;
270      NodeChoice c2 = (NodeChoice) seq.elementAt(1);
271      if (c2.choice instanceof ClassOrInterfaceDeclaration) {
272        return getClassName(c2.choice);
273      } else {
274        return null;
275      }
276    }
277  }

279  /** Return the fully qualified name of the class containing the node. */
280  public static @BinaryName String getClassName(Node d) {
281
282    ClassOrInterfaceDeclaration n =
283        (d instanceof ClassOrInterfaceDeclaration)
284            ? (ClassOrInterfaceDeclaration) d
285            : (ClassOrInterfaceDeclaration) Ast.getParent(ClassOrInterfaceDeclaration.class, d);
286
287    String packageName;
288    CompilationUnit unit = (CompilationUnit) getParent(CompilationUnit.class, n);
289    String getPackage = getPackage(unit);
290    if (getPackage != null) {
291      packageName = getPackage + ".";
292    } else {
293      packageName = "";
294    }
295
296    String className = n.f1.tokenImage + "."; // f1 -> <IDENTIFIER>
297
298    Node currentNode = n;
299    while (true) {
300      ClassOrInterfaceBody b =
301          (ClassOrInterfaceBody) getParent(ClassOrInterfaceBody.class, currentNode);
302      if (b == null) {
303        break;
304      }
305      Node n1 = b.getParent();
306      assert n1 instanceof ClassOrInterfaceDeclaration;
307      if (isInner((ClassOrInterfaceDeclaration) n1)) {
308        // TODO: This works for anonymous classes (maybe), but is wrong for
309        // non-anonymous inner classes.
310        className = "$inner." + className;
311        currentNode = b;
312      } else {
313        String s = ((ClassOrInterfaceDeclaration) n1).f1.tokenImage;
314        className = s + "." + className;
315        currentNode = n1;
316      }
317    }
318
319    String result = packageName + className;
320    if (result.endsWith(".")) {
321      result = result.substring(0, result.length() - 1);
322    }
323
324    @SuppressWarnings("signature") // string concatenation, etc.
325    @BinaryName String result_bnfna = result;
326    return result_bnfna;
327  }

334  // f1 -> [ AssignmentOperator() Expression() ]
335  // Return the primary expression on the left-hand side of an assignment
336  public static PrimaryExpression assignment2primaryexpression(Expression n) {
337    // All this could perhaps be replaced with an ad-hoc visitor, as is
338    // done in nodeTokenAfter().  But it is written now, so leave it as is.
339
340    assert n.f1.present();
341    ConditionalExpression ce = n.f0;
342    assert !ce.f1.present();
343    ConditionalOrExpression coe = ce.f0;
344    assert !coe.f1.present();
345    ConditionalAndExpression cae = coe.f0;
346    assert !cae.f1.present();
347    InclusiveOrExpression ioe = cae.f0;
348    assert !ioe.f1.present();
349    ExclusiveOrExpression eoe = ioe.f0;
350    assert !eoe.f1.present();
351    AndExpression ande = eoe.f0;
352    assert !ande.f1.present();
353    EqualityExpression ee = ande.f0;
354    assert !ee.f1.present();
355    InstanceOfExpression iofe = ee.f0;
356    assert !iofe.f1.present();
357    RelationalExpression re = iofe.f0;
358    assert !re.f1.present();
359    ShiftExpression se = re.f0;
360    assert !se.f1.present();
361    AdditiveExpression adde = se.f0;
362    assert !adde.f1.present();
363    MultiplicativeExpression me = adde.f0;
364    assert !me.f1.present();
365    UnaryExpression ue = me.f0;
366    UnaryExpressionNotPlusMinus uenpm = (UnaryExpressionNotPlusMinus) ue.f0.choice;
367    PostfixExpression pfe = (PostfixExpression) uenpm.f0.choice;
368    assert !pfe.f1.present();
369    PrimaryExpression pe = pfe.f0;
370    return pe;
371  }

373  public static String fieldName(PrimaryExpression pe) {
374
375    // System.out.println("fieldName(" + pe + ")");
376
377    // PrimaryExpression:
378    // f0 -> PrimaryPrefix()
379    // f1 -> ( PrimarySuffix() )*
380
381    // First, try to get a name from the PrimarySuffix.
382
383    NodeListOptional pslist = pe.f1;
384    if (pslist.size() > 0) {
385      PrimarySuffix ps = (PrimarySuffix) pslist.elementAt(pslist.size() - 1);
386      NodeChoice psnc = ps.f0;
387      // PrimarySuffix:
388      // f0 -> "." "super"
389      //       | "." "this"
390      //       | "." AllocationExpression()
391      //       | MemberSelector()
392      //       | "[" Expression() "]"
393      //       | "." <IDENTIFIER>
394      //       | Arguments()
395      switch (psnc.which) {
396        case 5:
397          NodeSequence sn = (NodeSequence) psnc.choice;
398          assert sn.size() == 2;
399          return ((NodeToken) sn.elementAt(1)).tokenImage;
400      }
401    }
402
403    // If it was impossible to get a name from the PrimarySuffix,
404    // try the PrimaryPrefix.
405
406    // PrimaryPrefix:
407    // f0 -> Literal()
408    //       | ( <IDENTIFIER> "." )* "this"
409    //       | "super" "." <IDENTIFIER>
410    //       | ClassOrInterfaceType() "." "super" "." <IDENTIFIER>
411    //       | "(" Expression() ")"
412    //       | AllocationExpression()
413    //       | ResultType() "." "class"
414    //       | Name()
415    PrimaryPrefix pp = pe.f0;
416    NodeChoice ppnc = pp.f0;
417    switch (ppnc.which) {
418      case 2:
419        NodeSequence sn = (NodeSequence) ppnc.choice;
420        assert sn.size() == 3;
421        return ((NodeToken) sn.elementAt(2)).tokenImage;
422      case 7:
423        return fieldName((Name) ppnc.choice);
424    }
425
426    return null;
427  }

526  /**
527   * Add the comment to the first regular token in the tree. If first is true, then it is inserted
528   * before all other special tokens; otherwise, it is inserted after them.
529   *
530   * <p>Exception: If first is true, may heuristically place the comment it in the middle of the
531   * list of special tokens (comments), in order to place it at the same column as the real node.
532   *
533   * <p>Postcondition (make sure you preserve it if you modify this method): comment.beginLine and
534   * comment.beginColumn have been assigned the line and column number where this comment will go.
535   */
536  public static void findLineAndCol(Node n, NodeToken comment, boolean first) {
537    class AddCommentVisitor extends DepthFirstVisitor {
538      private boolean seenToken = false;
539      private final NodeToken comment;
540      private final boolean first;
541
542      public AddCommentVisitor(NodeToken comment, boolean first) {
543        this.comment = comment;
544        this.first = first;
545      }
546
547      @Override
548      public void visit(NodeToken node) {
549        if (!seenToken) {
550          seenToken = true;
551          if (first
552              && (node.numSpecials() > 1)
553              && (node.getSpecialAt(0).beginColumn
554                  != node.getSpecialAt(node.numSpecials() - 1).beginColumn)) {
555            // There are multiple comments, and the first and last ones
556            // start at different columns.  Search from the end, finding
557            // the first one with a different column.
558            int i = node.numSpecials() - 1;
559            while (node.getSpecialAt(i - 1).beginColumn == node.getSpecialAt(i).beginColumn) {
560              i--;
561            }
562            comment.beginLine = node.getSpecialAt(i).beginLine;
563            comment.beginColumn = node.getSpecialAt(i).beginColumn;
564            // addNthSpecial(node, comment, i);
565          } else if (first && (node.numSpecials() > 0)) {
566            comment.beginLine = node.getSpecialAt(0).beginLine;
567            comment.beginColumn = node.getSpecialAt(0).beginColumn;
568            // System.out.println("findLineAndCol: set from special <<<" + node.getSpecialAt(0) +
569            // ">>>");
570            // addFirstSpecial(node, comment);
571          } else {
572            comment.beginLine = node.beginLine;
573            comment.beginColumn = node.beginColumn;
574            // node.addSpecial(comment);
575          }
576          // System.out.printf("comment placed at (%d,%d) = <<<%s>>> for node (%d,%d)= <<<%s>>>%n",
577          //                   comment.beginLine, comment.beginColumn, comment.tokenImage.trim(),
578          //                   node.beginLine, node.beginColumn, node.tokenImage.trim());
579        }
580      }
581    }
582    n.accept(new AddCommentVisitor(comment, first));
583  }

600  // Return the first NodeToken after (all of) the specified Node.
601  public static NodeToken nodeTokenAfter(Node n) {
602    // After the traversal, the "lastNodeToken" slot contains the
603    // last NodeToken visited.
604    class LastNodeTokenVisitor extends DepthFirstVisitor {
605      public NodeToken lastNodeToken = null;
606
607      @Override
608      public void visit(NodeToken node) {
609        lastNodeToken = node;
610      }
611    }
612    // After the traversal, the "nextNodeToken" slot contains the token
613    // visited immediately after "predecessor".  ("predecessor" should be a
614    // descendant of the token from whcih traversal starts.)
615    class NextNodeTokenVisitor extends DepthFirstVisitor {
616      private boolean seenPredecessor = false;
617      public NodeToken nextNodeToken;
618      private final NodeToken predecessor;
619
620      public NextNodeTokenVisitor(NodeToken predecessor) {
621        this.predecessor = predecessor;
622      }
623
624      @Override
625      @SuppressWarnings("interning")
626      public void visit(NodeToken node) {
627        if (!seenPredecessor) {
628          if (node == predecessor) {
629            seenPredecessor = true;
630          }
631        } else if (nextNodeToken == null) {
632          nextNodeToken = node;
633        }
634      }
635    }
636
637    // set predecessor
638    LastNodeTokenVisitor lntv = new LastNodeTokenVisitor();
639    n.accept(lntv);
640    NodeToken predecessor = lntv.lastNodeToken;
641    if (predecessor == null) {
642      throw new Error("No lastNodeToken for " + n);
643    }
644
645    // We don't know how high in the tree we need to go in order to find a
646    // successor, so iteratively go higher until success.  This has bad
647    // worst-case performance, but should be acceptable in practice.
648    NodeToken result = null;
649    Node parent = n.getParent();
650    while ((result == null) && (parent != null)) {
651      NextNodeTokenVisitor nntv = new NextNodeTokenVisitor(predecessor);
652      parent.accept(nntv);
653      result = nntv.nextNodeToken;
654      parent = parent.getParent();
655    }
656    if (result == null) {
657      throw new Error("No nextNodeToken for " + n);
658    }
659    return result;
660  }

710  public static Class<?> getClass(@ClassGetName String s) {
711    try {
712      Class<?> c = Class.forName(s);
713      assert c != null;
714      return c;
715    } catch (ClassNotFoundException e) {
716      String orig_s = s;
717      // System.out.println("Lookup failed: " + s);
718      // We should have found it.  Maybe there is a name mangling problem.
719      // Systematically change each "." to "$" in an attempt to fix it.
720      while (true) {
721        int dot_pos = s.lastIndexOf('.');
722        if (dot_pos == -1) {
723          throw new Error("Didn't find class " + orig_s);
724        }
725        @SuppressWarnings("signature") // string concatenation
726        @ClassGetName String new_s = s.substring(0, dot_pos) + "$" + s.substring(dot_pos + 1);
727        s = new_s;
728        // System.out.println("Lookup trying: " + s);
729        try {
730          Class<?> c = Class.forName(s);
731          assert c != null;
732          return c;
733        } catch (ClassNotFoundException ex) {
734          throw new Error("This can't happen");
735        }
736      }
737      // throw new Error("Control cannot reach here");
738    }
739  }

746  public static Method getMethod(Class<?> c, MethodDeclaration methoddecl) {
747    String ast_methodname = getName(methoddecl);
748    List<FormalParameter> ast_params = getParameters(methoddecl);
749
750    List<Method> publicMethods = Arrays.<Method>asList(c.getMethods());
751    List<Method> declaredMethods = Arrays.<Method>asList(c.getDeclaredMethods());
752    List<Method> allMethods = new ArrayList<>();
753    allMethods.addAll(publicMethods);
754    allMethods.addAll(declaredMethods);
755
756    Method[] meths = allMethods.toArray(new Method[0]);
757
758    for (int i = 0; i < meths.length; i++) {
759
760      Method meth = meths[i];
761      // System.out.println("getMethod(" + c.getName() + ", " + getName(methoddecl) + ") checking "
762      // + meth.getName());
763      if (!typeMatch(meth.getName(), ast_methodname)) {
764        continue;
765      }
766
767      Class<?>[] params = meth.getParameterTypes();
768      if (paramsMatch(params, ast_params)) {
769        // System.out.println("getMatch succeeded: " + ppt.name());
770        return meth;
771      }
772    }
773    return null;
774  }

903  // the first node of the specified type or a subtype.  Returns null
904  // if no parent of that type.
905  public static @Nullable Node getParent(Class<?> type, Node child) {
906    Node currentNode = child.getParent();
907    while (true) {
908      if (type.isInstance(currentNode)) {
909        return currentNode;
910      } else {
911        if (currentNode == null) {
912          return null;
913        }
914        currentNode = currentNode.getParent();

1283    // Grammar production:
1284    // f0 -> ( ( "public" | "static" | "protected" | "private" | "final" | "abstract" |

1323    return isInterface((ClassOrInterfaceDeclaration) n.getParent());
1324  }

1326  public static boolean isInterface(ClassOrInterfaceDeclaration n) {
1327    // Grammar production for ClassOrInterfaceDeclaration:
1328    // f0 -> ( "class" | "interface" )
1329    // f1 -> <IDENTIFIER>
1330    // f2 -> [ TypeParameters() ]
1331    // f3 -> [ ExtendsList(isInterface) ]
1332    // f4 -> [ ImplementsList(isInterface) ]
1333    // f5 -> ClassOrInterfaceBody(isInterface)
1334    NodeToken t = (NodeToken) n.f0.choice;
1335    String token = t.tokenImage;
1336    if (token.equals("interface")) {
1337      return true;
1338    } else {
1339      return false;
1340    }
1341  }

1357  public static boolean isArray(jtb.syntaxtree.Type n) {
1358    // Grammar production:
1359    // f0 -> PrimitiveType() ( "[" "]" )+
1360    //       | ( ClassOrInterfaceType() ) ( "[" "]" )*
1361    if (isPrimitive(n)) {
1362      return false;
1363    }
1364    assert isReference(n);
1365    ReferenceType refType = (ReferenceType) n.f0.choice;
1366    NodeSequence seq = (NodeSequence) refType.f0.choice;
1367    if (seq.elementAt(0) instanceof PrimitiveType) {
1368      return true; // see grammar--must be array in this case
1369    }
1370    NodeListOptional opt = (NodeListOptional) seq.elementAt(1);
1371    return opt.present();
1372  }

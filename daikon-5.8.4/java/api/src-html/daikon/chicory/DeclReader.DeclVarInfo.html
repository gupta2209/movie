021/**
022 * Reads declaration files and provides methods to access the information within them. A declaration
023 * file consists of a number of program points and the variables for each program point.
024 */
025public class DeclReader {
026
027  /** Prints only the average set size for each specified file. */
028  @Option("Print average set size (only) for each specified file")
029  public static boolean avg_size = false;

031  /**

036  @Option("Output a decl file with primitive declaration comparability")

064  /** Information about variables within a program point. */
065  public static class DeclVarInfo {
066    public String name;
067    public String type;
068    public String rep_type;
069    public String comparability;
070    public int index;
071
072    public DeclVarInfo(String name, String type, String rep_type, String comparability, int index) {
073      this.name = name;
074      this.type = type;
075      this.rep_type = rep_type;
076      this.comparability = comparability;
077      this.index = index;
078    }
079
080    /** Returns the variable's name. */
081    public String get_name() {
082      return name;
083    }
084
085    /** Returns the variable's declared type as specified in the decl file b. */
086    public String get_type() {
087      return type;
088    }
089
090    /**
091     * Returns the type name. get_type() returns the entire entry including auxiliary information.
092     */
093    public String get_type_name() {
094      return type.replaceFirst(" .*", "");
095    }
096
097    /**
098     * Returns the representation type of the variable as specified in the decl file. The static
099     * value (if any) is discarded.
100     */
101    public String get_rep_type() {
102      return rep_type.replaceFirst(" .*", "");
103    }
104
105    /**
106     * Return true if this variable has floating-point type (float or double).
107     *
108     * @return ture if this variable has floating-point type
109     */
110    @Pure
111    public boolean is_double() {
112      return (rep_type.equals("double") || rep_type.equals("float"));
113    }
114
115    @Pure
116    public boolean is_string() {
117      return (rep_type.equals("string"));
118    }
119
120    @Pure
121    public boolean is_string_array() {
122      return (rep_type.equals("string[]"));
123    }
124
125    @Pure
126    public boolean is_int() {
127      return (rep_type.equals("int"));
128    }
129
130    /** Returns the comparability string from the decl file. */
131    public String get_comparability() {
132      return comparability;
133    }
134
135    public String get_basic_comparability() {
136      return comparability.replaceFirst("\\[.*\\]", "");
137    }
138
139    public void set_comparability(String comparability) {
140      this.comparability = comparability;
141    }
142
143    @SideEffectFree
144    @Override
145    public String toString(@GuardSatisfied DeclVarInfo this) {
146      return String.format("%s [%s] %s", type, rep_type, name);
147    }
148
149    /**
150     * Reads a single value for this variable and returns it. The result is null exactly if the
151     * value is nonsensical. The return value is interned.
152     */
153    public @Nullable @Interned Object read_data(EntryReader reader) throws IOException {
154      String var_name = reader.readLine();
155      if (var_name == null) {
156        throw new Error(
157            "file " + reader.getFileName() + " terminated prematurely, expeced var " + this.name);
158      }
159      if (!var_name.equals(this.name)) {
160        throw new Error(var_name + " found where " + this.name + " expected ");
161      }
162      String value = reader.readLine();
163      String mod_bit = reader.readLine();
164      if ((value == null) || (mod_bit == null)) {
165        throw new Error("file " + reader.getFileName() + " terminated prematurely");
166      }
167      if (value.equals("nonsensical")) {
168        return null;
169      } else if (is_int()) {
170        try {
171          return (Intern.intern(value)); // interning bugfix
172        } catch (NumberFormatException t) {
173          throw new Error("Unexpected value '" + value + "' for integer variable " + this.name);
174        }
175      } else if (is_double()) {
176        try {
177          return (Intern.intern(value)); // interning bugfix
178        } catch (NumberFormatException t) {
179          throw new Error("Unexpected string '" + value + "' for double variable " + this.name);
180        }
181      } else if (is_string()) {
182        if (value.startsWith("\"") && value.endsWith("\"")) {
183          value = value.substring(1, value.length() - 1);
184        }
185        return value.intern();
186      } else if (is_string_array()) {
187        return (null); // treating arrays as nonsensical for now
188      } else {
189        throw new Error("unexpected rep type " + rep_type);
190      }
191    }
192  }

194  /**
195   * Information about the program point that is contained in the decl file. This consists of the
196   * ppt name and a list of the declared variables.
197   */
198  public static class DeclPpt {
199    public String name;
200    public HashMap<String, DeclVarInfo> vars = new LinkedHashMap<>();
201
202    /**
203     * List of values for the program point. There is one entry in the list for each time the
204     * program point is executed. That entry is a list of the values for each variable in the same
205     * order as the variables were defined.
206     */
207    List<List<@Interned Object>> data_values = new ArrayList<>();
208
209    public DeclPpt(String name) {
210      this.name = name;
211    }
212
213    /**
214     * Read a single variable declaration from decl_file. The file must be positioned immediately
215     * before the variable name.
216     */
217    public DeclVarInfo read_var(EntryReader decl_file) throws java.io.IOException {
218
219      String name = decl_file.readLine();
220      String type = decl_file.readLine();
221      String rep_type = decl_file.readLine();
222      String comparability = decl_file.readLine();
223      if ((name == null) || (type == null) || (rep_type == null) || (comparability == null)) {
224        throw new Error("File " + decl_file.getFileName() + " ends prematurely");
225      }
226
227      // I don't see the point of this interning.  No code seems to take
228      // advantage of it.  Is it just for space?  -MDE
229      DeclVarInfo var =
230          new DeclVarInfo(
231              name.intern(), type.intern(), rep_type.intern(), comparability.intern(), vars.size());
232      vars.put(name, var);
233      return var;
234    }
235
236    /**
237     * Adds a record of data for this ppt. The data must have one element for each variable in the
238     * ppt and be ordered in the same way.
239     */
240    public void add_var_data(List<@Interned Object> var_data_list) {
241      assert var_data_list.size() == vars.size();
242      data_values.add(var_data_list);
243    }
244
245    public List<List<@Interned Object>> get_var_data() {
246      return data_values;
247    }
248
249    /** Returns the DeclVarInfo named var_name or null if it doesn't exist. */
250    public @Nullable DeclVarInfo find_var(String var_name) {
251      return vars.get(var_name);
252    }
253
254    /** Returns the ppt name. */
255    public String get_name() {
256      return name;
257    }
258
259    /** Returns the name without the :::EXIT, :::ENTER, etc. */
260    public String get_short_name() {
261      return name.replaceFirst(":::.*", "");
262    }
263
264    @SideEffectFree
265    @Override
266    public String toString(@GuardSatisfied DeclPpt this) {
267      return name;
268    }
269
270    /** Returns the list of variables in their standard order. */
271    public List<DeclVarInfo> get_all_vars() {
272      return new ArrayList<DeclVarInfo>(vars.values());
273    }
274  }

278  /** Read declarations from the specified pathname. */
279  public void read(File pathname) throws IOException {
280
281    // have caller deal with FileNotFound
282    EntryReader decl_file = new EntryReader(pathname, "^(//|#).*", null);
283
284    try {
285      for (String line = decl_file.readLine(); line != null; line = decl_file.readLine()) {
286        if (!line.equals("DECLARE")) {
287          continue;
288        }
289
290        // Read the declaration
291        read_decl(decl_file);
292      }
293    } catch (Exception e) {
294      throw new Error("Error reading comparability decl file", e);
295    }
296  }

298  /**
299   * Reads a single declaration from decl_file. The opening "DECLARE" line should have already been
300   * read. Returns the ppt.
301   */
302  protected DeclPpt read_decl(EntryReader decl_file) throws IOException {
303
304    // Read the name of the program point
305    String pptname = decl_file.readLine();
306    if (pptname == null) throw new Error("file " + decl_file.getFileName() + " ends prematurely");
307    assert pptname.contains(":::");
308    DeclPpt ppt = new DeclPpt(pptname);
309    ppts.put(pptname, ppt);
310
311    // Read each of the variables in this program point.  The variables
312    // are terminated by a blank line.
313    String line = decl_file.readLine();
314    while ((line != null) && (line.length() != 0)) {
315      decl_file.putback(line);
316      ppt.read_var(decl_file);
317      line = decl_file.readLine();
318    }
319
320    return ppt;
321  }

323  public void dump_decl() {
324
325    for (String ppt_name : ppts.keySet()) {
326      System.out.printf("Comp Ppt: %s%n", ppt_name);
327      DeclPpt ppt = ppts.get(ppt_name);
328      for (String var_name : ppt.vars.keySet()) {
329        System.out.printf("  var %s%n", var_name);
330      }
331    }
332  }

334  // This can return null.  Example:  when DynComp is run to compute
335  // comparability information, it produces no information (not even a
336  // declaration) for program points that are never executed.  But, Chicory
337  // outputs a declaration for every program point, and this lookup can
338  // fail when using the --comparability-file=... command-line argument
339  // with a file produced by DynComp.
340  public @Nullable DeclPpt find_ppt(String ppt_name) {
341    DeclPpt result = ppts.get(ppt_name);
342    return result;
343  }

345  /** Reads a decl file and dumps statistics. */
346  public static void main(String[] args) throws IOException {
347
348    Options options = new Options("DeclReader [options] decl-files...", DeclReader.class);
349    String[] files = options.parse(true, args);
350    boolean print_each_set = !avg_size;
351
352    if (files.length < 1) {
353      System.err.println("No decl-files provided.");
354      System.exit(1);
355    }
356
357    // If reading/dumping dtrace file, just read one file and dump it
358    if (dump_dtrace) {
359      if (files.length != 1) {
360        System.err.println("One decl-file expected, received " + files.length + ".");
361        System.exit(1);
362      }
363      DTraceReader trace = new DTraceReader();
364      trace.read(new File(files[0]));
365      trace.dump_data();
366      return;
367    }
368
369    // If determining primitive declaration type comparability, setup
370    // the comparability based on the declared type of primitives and
371    // write out the result.
372    if (primitive_declaration_type_comparability) {
373      if (files.length != 2) {
374        System.err.println("Two decl-files expected, received " + files.length + ".");
375        System.exit(1);
376      }
377      DeclReader dr = new DeclReader();
378      dr.read(new File(files[0]));
379      dr.primitive_declaration_types();
380      dr.write_decl(files[1], comparability);
381      return;
382    }
383
384    // If determining declaration type comparability, setup the comparability
385    // based on the declared types and write out the result.
386    if (declaration_type_comparability) {
387      if (files.length != 2) {
388        System.err.println("Two decl-files expected, received " + files.length + ".");
389        System.exit(1);
390      }
391      DeclReader dr = new DeclReader();
392      dr.read(new File(files[0]));
393      dr.declaration_types();
394      dr.write_decl(files[1], comparability);
395      return;
396    }
397
398    // If determining representation type comparability, setup comparability
399    // on the rep type of each variable and write out the result.
400    if (rep_type_comparability) {
401      if (files.length != 2) {
402        System.err.println("Two decl-files expected, received " + files.length + ".");
403        System.exit(1);
404      }
405      DeclReader dr = new DeclReader();
406      dr.read(new File(files[0]));
407      dr.rep_types();
408      dr.write_decl(files[1], comparability);
409      return;
410    }
411
412    for (String filename : files) {
413      DeclReader dr = new DeclReader();
414      dr.read(new File(filename));
415
416      int num_sets = 0;
417      int total_set_size = 0;
418      int num_type_integer_vars = 0;
419      int num_rep_type_integer_vars = 0;
420      int num_vars = 0;
421
422      // Map from rep type to the count of each declared type for that rep
423      Map<String, Map<String, Integer>> rep_map = new LinkedHashMap<>();
424
425      // Loop through each ppt
426      for (DeclPpt ppt : dr.ppts.values()) {
427        if (print_each_set) {
428          System.out.printf("ppt %s%n", ppt.name);
429        }
430
431        int ppt_num_sets = 0;
432        int ppt_total_set_size = 0;
433
434        // Build a map from comparabilty to all of the variables with that comp
435        Map<String, List<DeclVarInfo>> comp_map = new LinkedHashMap<>();
436        for (DeclVarInfo vi : ppt.vars.values()) {
437
438          // Update the map that tracks the declared types for each rep type
439          Map<String, Integer> dec_map = rep_map.get(vi.get_rep_type());
440          if (dec_map == null) {
441            dec_map = new LinkedHashMap<>();
442            rep_map.put(vi.get_rep_type(), dec_map);
443          }
444          Integer cnt = dec_map.get(vi.get_type_name());
445          if (cnt == null) cnt = 0;
446          dec_map.put(vi.get_type_name(), ++cnt);
447
448          num_vars++;
449          if (vi.get_type_name().equals("int")) num_type_integer_vars++;
450          if (vi.get_rep_type().equals("int")) num_rep_type_integer_vars++;
451          if (!vi.get_type_name().equals("int") && vi.get_rep_type().equals("int")) {
452            System.out.printf("huh '%s' - '%s'%n", vi.get_type_name(), vi.rep_type);
453          }
454          String comp = vi.get_basic_comparability();
455          if (!comp_map.containsKey(comp)) {
456            comp_map.put(comp, new ArrayList<DeclVarInfo>());
457          }
458          assert comp_map.containsKey(comp); // XXX flow could compute this
459          List<DeclVarInfo> vi_list = comp_map.get(comp);
460          vi_list.add(vi);
461        }
462
463        // Print out the variables with each comparability
464        for (List<DeclVarInfo> vi_list : comp_map.values()) {
465          num_sets++;
466          total_set_size += vi_list.size();
467          if (print_each_set) {
468            ppt_num_sets++;
469            ppt_total_set_size += vi_list.size();
470            System.out.printf(
471                "  %-5s : [%d] %s%n",
472                vi_list.get(0).get_basic_comparability(), vi_list.size(), vi_list);
473          }
474        }
475        if (print_each_set && (ppt_num_sets > 0)) {
476          System.out.printf(
477              "  %d sets of average size %f%n",
478              ppt_num_sets, ((double) ppt_total_set_size) / ppt_num_sets);
479        }
480      }
481
482      if (avg_size) {
483        System.out.printf(
484            "%-35s %,6d sets of average size %f found%n",
485            filename, num_sets, ((double) total_set_size) / num_sets);
486      }
487      for (String rep_type : rep_map.keySet()) {
488        Map<String, Integer> dec_map = rep_map.get(rep_type);
489        System.out.printf("Rep Type %s (%d declared types):%n", rep_type, dec_map.size());
490        for (String dec_type : dec_map.keySet()) {
491          System.out.printf("  %5d - %s%n", dec_map.get(dec_type), dec_type);
492        }
493      }
494      System.out.printf("Total variables of declared type int = %d%n", num_type_integer_vars);
495      System.out.printf("Total variables of rep type int = %d%n", num_rep_type_integer_vars);
496      System.out.printf("Total number of variables = %d%n", num_vars);
497    }
498  }

500  /**
501   * Sets the comparability to match declaration types. The comparability for each variable is set
502   * so that each declaration type is in a separate set. Each hashcode with a different type name
503   * will be in a different set. Primitives with different type names will be in different sets.
504   */
505  public void declaration_types() {
506
507    Map<String, Integer> type_comp = new LinkedHashMap<>();
508    int next_comparability = 1;
509
510    // Loop through each program point
511    for (DeclPpt ppt : ppts.values()) {
512
513      // Loop through each variable
514      for (DeclVarInfo vi : ppt.vars.values()) {
515
516        // Determine the comparabilty for this declared type.  Hashcodes
517        // are not included because hashcodes of different declared types
518        // can still be sensibly compared (eg, subclasses/superclasses)
519        String declared_type = vi.get_type_name();
520        Integer comparability = type_comp.get(declared_type);
521        if (comparability == null) {
522          comparability = Integer.valueOf(next_comparability);
523          next_comparability++;
524          type_comp.put(declared_type, comparability);
525          System.out.printf(
526              "declared type %s has comparability %d%n", declared_type, comparability);
527        }
528        vi.set_comparability(comparability.toString());
529      }
530    }
531  }
532  /**
533   * Sets the comparability to match primitive declaration types. The comparability for each
534   * non-hashcode is set so that each declaration type is in a separate set. Hashcodes are all set
535   * to a single comparability regardless of their declared type.
536   */
537  public void primitive_declaration_types() {
538
539    Map<String, Integer> type_comp = new LinkedHashMap<>();
540    int next_comparability = 1;
541
542    // Loop through each program point
543    for (DeclPpt ppt : ppts.values()) {
544
545      // Loop through each variable
546      for (DeclVarInfo vi : ppt.vars.values()) {
547
548        // Determine the comparabilty for this declared type.  Hashcodes
549        // are not included because hashcodes of different declared types
550        // can still be sensibly compared (eg, subclasses/superclasses)
551        String declared_type;
552        if (vi.get_rep_type().startsWith("hashcode")) {
553          declared_type = "hashcode";
554        } else {
555          declared_type = vi.get_type_name();
556        }
557        Integer comparability = type_comp.get(declared_type);
558        if (comparability == null) {
559          comparability = Integer.valueOf(next_comparability);
560          next_comparability++;
561          type_comp.put(declared_type, comparability);
562          System.out.printf(
563              "declared type %s has comparability %d%n", declared_type, comparability);
564        }
565        vi.set_comparability(comparability.toString());
566      }
567    }
568  }

597  /** Writes the declaration to the specified file. If the filename is "-", writes to stdout. */
598  public void write_decl(String filename, String comparability) throws IOException {
599
600    // Get the output stream
601    PrintStream decl_file;
602    if (filename.equals("-")) {
603      decl_file = System.out;
604    } else {
605      decl_file = new PrintStream(filename);
606    }
607
608    decl_file.printf("// Declaration file written by DeclReader%n%n");
609    decl_file.printf("VarComparability%n%s%n%n", comparability);
610
611    // Loop through each program point
612    for (DeclPpt ppt : ppts.values()) {
613
614      decl_file.printf("%nDECLARE%n%s%n", ppt.name);
615
616      // Loop through each variable
617      for (DeclVarInfo vi : ppt.vars.values()) {
618        decl_file.printf(
619            "%s%n%s%n%s%n%s%n",
620            vi.get_name(), vi.get_type(), vi.get_rep_type(), vi.get_comparability());
621      }
622    }
623
624    decl_file.close();
625  }

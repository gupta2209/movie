038/**
039 * BuildJDK uses {@link DCInstrument} to add comparability instrumentation to Java class files, then
040 * stores the modified files into a directory identified by a (required) command line argument.
041 *
042 * <p>DCInstrument duplicates each method of a class file. The new methods are distinguished by the
043 * addition of a final parameter of type DCompMarker and are instrumented to track comparability.
044 * Based on its invocation arguments, DynComp will decide whether to call the instrumented or
045 * uninstrumented version of a method.
046 */
047public class BuildJDK {
048
049  /**
050   * The "java.home" system property. Note that there is also a JAVA_HOME variable that contains
051   * {@code System.getenv("JAVA_HOME")}.
052   */
053  public static final String java_home = System.getProperty("java.home");
054
055  /** Whether to print information about the classes being instrumented. */
056  private static boolean verbose = false;
057
058  /** Number of class files processed; used for progress display. */
059  private int _numFilesProcessed = 0;

061  /** Name of file in output jar containing the static-fields map. */
062  private static String static_field_id_filename = "dcomp_jdk_static_field_id";

080  /**
081   * Instruments each class file in the Java runtime and puts the result in the first non-option
082   * command-line argument.
083   *
084   * <p>By default, BuildJDK will locate the appropriate Java runtime library and instrument each of
085   * its member class files. However, if there are additional arguments on the command line after
086   * the destination directory, then we assume these are class files to be instrumented and the Java
087   * runtime library is not used. This usage is primarily for testing purposes.
088   *
089   * @param args arguments being passed to BuildJDK
090   * @throws IOException if unable to read or write file {@code dcomp_jdk_static_field_id} or if
091   *     unable to write {@code jdk_classes.txt}
092   */
093  public static void main(String[] args) throws IOException {
094
095    System.out.println("BuildJDK starting at " + LocalDateTime.now(ZoneId.systemDefault()));
096
097    BuildJDK build = new BuildJDK();
098
099    Options options =
100        new Options(
101            "daikon.BuildJDK [options] dest_dir [classfiles...]",
102            DynComp.class,
103            DCInstrument.class);
104    String[] cl_args = options.parse(true, args);
105    if (cl_args.length < 1) {
106      System.out.println("must specify destination dir");
107      options.printUsage();
108      System.exit(1);
109    }
110    verbose = DynComp.verbose;
111
112    File dest_dir = new File(cl_args[0]);
113
114    /**
115     * Key is a class file name, value is a stream that opens that file name.
116     *
117     * <p>We want to share code to read and instrument the Java class file members of a jar file
118     * (JDK 8) or a module file (JDK 9+). However, jar files and module files are located in two
119     * completely different file systems. So we open an InputStream for each class file we wish to
120     * instrument and save it in the class_stream_map with the file name as the key. From that point
121     * the code to instrument a class file can be shared.
122     */
123    Map<String, InputStream> class_stream_map;
124
125    if (cl_args.length > 1) {
126
127      // Arguments are <destdir> [<classfiles>...]
128      @SuppressWarnings("nullness:assignment.type.incompatible") // https://tinyurl.com/cfissue/3224
129      @NonNull String[] class_files = Arrays.copyOfRange(cl_args, 1, cl_args.length);
130
131      // Instrumenting a specific list of class files is usually used for testing.
132      // But if we're using it to fix a broken classfile, then we need
133      // to restore the static-fields map from when our runtime jar was originally
134      // built.  We assume it is in the destination directory.
135      DCInstrument.restore_static_field_id(new File(dest_dir, static_field_id_filename));
136      System.out.printf(
137          "Restored %d entries in static map.%n", DCInstrument.static_field_id.size());
138
139      class_stream_map = new HashMap<>();
140      for (String classFileName : class_files) {
141        try {
142          class_stream_map.put(classFileName, new FileInputStream(classFileName));
143        } catch (FileNotFoundException e) {
144          throw new Error("File not found: " + classFileName, e);
145        }
146      }
147
148      // Instrument the classes identified in class_stream_map.
149      build.instrument_classes(dest_dir, class_stream_map);
150
151    } else {
152
153      check_java_home();
154
155      if (BcelUtil.javaVersion > 8) {
156        class_stream_map = build.gather_runtime_from_modules();
157      } else {
158        class_stream_map = build.gather_runtime_from_jar();
159      }
160
161      // Instrument the Java runtime classes identified in class_stream_map.
162      build.instrument_classes(dest_dir, class_stream_map);
163
164      // We've finished instrumenting all the class files. Now we create some
165      // abstract interface classes for use by the DynComp runtime.
166      build.addInterfaceClasses(dest_dir.getName());
167
168      // Write out the file containing the static-fields map.
169      System.out.printf("Found %d static fields.%n", DCInstrument.static_field_id.size());
170      DCInstrument.save_static_field_id(new File(dest_dir, static_field_id_filename));
171
172      // Write out the list of all classes in the jar file
173      File jdk_classes_file = new File(dest_dir, "java/lang/jdk_classes.txt");
174      System.out.printf("Writing a list of class names to %s%n", jdk_classes_file);
175      // Class names are written in internal form.
176      try (PrintWriter pw = new PrintWriter(jdk_classes_file, UTF_8.name())) {
177        pw.println("no_primitives: " + DynComp.no_primitives);
178        for (String classFileName : class_stream_map.keySet()) {
179          pw.println(classFileName.replace(".class", ""));
180        }
181      }
182    }
183
184    // Print out any methods that could not be instrumented
185    print_skipped_methods();
186
187    System.out.println("BuildJDK done at " + LocalDateTime.now(ZoneId.systemDefault()));
188  }

190  /** Verify that java.home and JAVA_HOME match. Exits the JVM if there is an error. */
191  public static void check_java_home() {
192
193    // We are going to instrument the default Java runtime library.
194    // We need to verify where we should look for it.
195
196    String JAVA_HOME = System.getenv("JAVA_HOME");
197    if (JAVA_HOME == null) {
198      if (verbose) {
199        System.out.println("JAVA_HOME not defined; using java.home: " + java_home);
200      }
201      JAVA_HOME = java_home;
202    }
203
204    File jrt = new File(JAVA_HOME);
205    if (!jrt.exists()) {
206      System.out.printf("Java home directory %s does not exist.%n", jrt);
207      System.exit(1);
208    }
209
210    try {
211      jrt = jrt.getCanonicalFile();
212    } catch (Exception e) {
213      System.out.printf("Error geting canonical file for %s: %s", jrt, e.getMessage());
214      System.exit(1);
215    }
216
217    JAVA_HOME = jrt.getAbsolutePath();
218    if (!java_home.startsWith(JAVA_HOME)) {
219      System.out.printf(
220          "JAVA_HOME (%s) does not agree with java.home (%s).%n", JAVA_HOME, java_home);
221      System.out.printf("Please correct your Java environment.%n");
222      System.exit(1);
223    }
224  }

226  /**
227   * For Java 8 the Java runtime is located in rt.jar. This method creates an InputStream for each
228   * class in rt.jar and returns this information in a map from class file name to InputStream.
229   *
230   * @return a map from class file name to the associated InputStream
231   */
232  @SuppressWarnings("JdkObsolete") // JarEntry.entries() returns Enumeration
233  Map<String, InputStream> gather_runtime_from_jar() {
234
235    Map<String, InputStream> class_stream_map = new HashMap<>();
236    String jar_name = java_home + "/lib/rt.jar";
237    System.out.printf("using jar file %s%n", jar_name);
238    try {
239      JarFile jfile = new JarFile(jar_name);
240      // Get each class to be instrumented and store it away
241      Enumeration<JarEntry> entries = jfile.entries();
242      while (entries.hasMoreElements()) {
243        JarEntry entry = entries.nextElement();
244        // System.out.printf("processing entry %s%n", entry);
245        final String entryName = entry.getName();
246        if (entryName.endsWith("/") || entryName.endsWith("~")) {
247          continue;
248        }
249
250        // Get the InputStream for this file
251        InputStream is = jfile.getInputStream(entry);
252        class_stream_map.put(entryName, is);
253      }
254    } catch (Exception e) {
255      throw new Error("Problem while reading " + jar_name, e);
256    }
257    return class_stream_map;
258  }

260  /**
261   * For Java 9+ the Java runtime is located in a series of modules. At this time, we are only
262   * pre-instrumenting the java.base module. This method initializes the DirectoryStream used to
263   * explore java.base. It calls gather_runtime_from_modules_directory to process the directory
264   * structure.
265   *
266   * @return a map from class file name to the associated InputStream
267   */
268  Map<String, InputStream> gather_runtime_from_modules() {
269
270    Map<String, InputStream> class_stream_map = new HashMap<>();
271    FileSystem fs = FileSystems.getFileSystem(URI.create("jrt:/"));
272    Path modules = fs.getPath("/modules");
273    // The path java_home+/lib/modules is the file in the host file system that
274    // corresponds to the modules file in the jrt: file system.
275    System.out.printf("using modules directory %s%n", java_home + "/lib/modules");
276    try (DirectoryStream<Path> directoryStream = Files.newDirectoryStream(modules, "java.base*")) {
277      for (Path moduleDir : directoryStream) {
278        gather_runtime_from_modules_directory(
279            moduleDir, moduleDir.toString().length(), class_stream_map);
280      }
281    } catch (IOException e) {
282      throw new Error(e);
283    }
284    return class_stream_map;
285  }

287  /**
288   * This is a helper method for {@link #gather_runtime_from_modules}. It recurses down the module
289   * directory tree, selects the classes we want to instrument, creates an InputStream for each of
290   * these classes, and adds this information to the {@code class_stream_map} argument.
291   *
292   * @param path module file, which might be subdirectory
293   * @param modulePrefixLength length of "/module/..." path prefix before start of actual member
294   *     path
295   * @param class_stream_map a map from class file name to InputStream that collects the results
296   */
297  void gather_runtime_from_modules_directory(
298      Path path, int modulePrefixLength, Map<String, InputStream> class_stream_map) {
299
300    if (Files.isDirectory(path)) {
301      try (DirectoryStream<Path> directoryStream = Files.newDirectoryStream(path)) {
302        for (Path subpath : directoryStream) {
303          gather_runtime_from_modules_directory(subpath, modulePrefixLength, class_stream_map);
304        }
305      } catch (IOException e) {
306        throw new Error(e);
307      }
308    } else {
309      String entryName = path.toString().substring(modulePrefixLength + 1);
310      try {
311        // Get the InputStream for this file
312        InputStream is = Files.newInputStream(path);
313        class_stream_map.put(entryName, is);
314      } catch (Exception e) {
315        throw new Error(e);
316      }
317    }
318  }

320  /**
321   * Instrument each of the classes indentified by the class_stream_map argument.
322   *
323   * @param dest_dir where to store the instrumented classes
324   * @param class_stream_map maps from class file name to an input stream on that file
325   */
326  void instrument_classes(File dest_dir, Map<String, InputStream> class_stream_map) {
327
328    try {
329      // Create the destination directory
330      dest_dir.mkdirs();
331
332      // Process each file.
333      for (String classFileName : class_stream_map.keySet()) {
334        if (verbose) {
335          System.out.println("instrument_classes: " + classFileName);
336        }
337
338        if (classFileName.equals("module-info.class")) {
339          System.out.printf("Skipping file %s%n", classFileName);
340          continue;
341        }
342
343        // Handle non-.class files and Object.class.  In JDK 8, copy them unchanged.
344        // For JDK 9+ we do not copy as these items will be loaded from the original module file.
345        if (!classFileName.endsWith(".class") || classFileName.equals("java/lang/Object.class")) {
346          if (BcelUtil.javaVersion > 8) {
347            if (verbose) System.out.printf("Skipping file %s%n", classFileName);
348            continue;
349          }
350          // This File constructor ignores dest_dir if classFileName is absolute.
351          File classFile = new File(dest_dir, classFileName);
352          if (classFile.getParentFile() == null) {
353            throw new Error("This can't happen: " + classFile);
354          }
355          classFile.getParentFile().mkdirs();
356          if (verbose) System.out.println("Copying Object.class or non-classfile: " + classFile);
357          try (InputStream in = class_stream_map.get(classFileName)) {
358            Files.copy(in, classFile.toPath());
359          }
360          continue;
361        }
362
363        // Get the binary for this class
364        InputStream is = class_stream_map.get(classFileName);
365        JavaClass jc;
366        try {
367          ClassParser parser = new ClassParser(is, classFileName);
368          jc = parser.parse();
369        } catch (Throwable e) {
370          throw new Error("Failed to parse classfile " + classFileName, e);
371        }
372
373        // Instrument the class file.
374        try {
375          instrumentClassFile(jc, dest_dir, classFileName, class_stream_map.size());
376        } catch (Throwable e) {
377          throw new Error("Couldn't instrument " + classFileName, e);
378        }
379      }
380    } catch (Exception e) {
381      throw new Error(e);
382    }
383  }

385  /**
386   * Add abstract interface classes needed by DynComp runtime.
387   *
388   * @param destDir where to store the interface classes
389   */
390  private void addInterfaceClasses(String destDir) {
391    // Create the DcompMarker class which is used to identify instrumented calls.
392    createDCompClass(destDir, "DCompMarker", false);
393
394    // The remainer of the generated classes are needed for JDK 9+ only.
395    if (BcelUtil.javaVersion > 8) {
396      createDCompClass(destDir, "DCompInstrumented", true);
397      createDCompClass(destDir, "DCompClone", false);
398      createDCompClass(destDir, "DCompToString", false);
399    }
400  }

402  /**
403   * Create an abstract interface class for use by the DynComp runtime.
404   *
405   * @param destDir where to store the new class
406   * @param className name of class
407   * @param dcompInstrumented c$if true, add equals_dcomp_instrumented method to class
408   */
409  private void createDCompClass(
410      String destDir, @BinaryName String className, boolean dcompInstrumented) {
411    try {
412      ClassGen dcomp_class =
413          new ClassGen(
414              Signatures.addPackage("java.lang", className),
415              "java.lang.Object",
416              "daikon.dcomp.BuildJDK tool",
417              Const.ACC_INTERFACE | Const.ACC_PUBLIC | Const.ACC_ABSTRACT,
418              new String[0]);
419      dcomp_class.setMinor(0);
420      // Convert from JDK version number to ClassFile major_version.
421      // A bit of a hack, but seems OK.
422      dcomp_class.setMajor(BcelUtil.javaVersion + 44);
423
424      if (dcompInstrumented) {
425        @SuppressWarnings(
426            "nullness:argument.type.incompatible") // null instruction list is ok for abstract
427        MethodGen mg =
428            new MethodGen(
429                Const.ACC_PUBLIC | Const.ACC_ABSTRACT,
430                Type.BOOLEAN,
431                new Type[] {Type.OBJECT},
432                new String[] {"o"},
433                "equals_dcomp_instrumented",
434                dcomp_class.getClassName(),
435                null,
436                dcomp_class.getConstantPool());
437        dcomp_class.addMethod(mg.getMethod());
438      }
439
440      dcomp_class
441          .getJavaClass()
442          .dump(
443              new File(
444                  destDir,
445                  "java" + File.separator + "lang" + File.separator + className + ".class"));
446    } catch (Exception e) {
447      throw new Error(e);
448    }
449  }

451  /**
452   * Instruments the JavaClass {@code jc} (whose name is {@code classFileName}). Writes the
453   * resulting class to its corresponding location in the directory outputDir.
454   *
455   * @param jc JavaClass to be instrumented
456   * @param outputDir output directory for instrumented class
457   * @param classFileName name of class to be instrumented
458   * @param classTotal total number of classes to be processed; used for progress display
459   * @throws IOException if unable to write out instrumented class
460   */
461  private void instrumentClassFile(
462      JavaClass jc, File outputDir, String classFileName, int classTotal)
463      throws java.io.IOException {
464    if (verbose) System.out.printf("processing target %s%n", classFileName);
465    DCInstrument dci = new DCInstrument(jc, true, null);
466    JavaClass inst_jc;
467    if (DynComp.no_primitives) {
468      inst_jc = dci.instrument_jdk_refs_only();
469    } else {
470      inst_jc = dci.instrument_jdk();
471    }
472    skipped_methods.addAll(dci.get_skipped_methods());
473    File classfile = new File(classFileName);
474    File dir;
475    if (classfile.getParent() == null) {
476      dir = outputDir;
477    } else {
478      dir = new File(outputDir, classfile.getParent());
479    }
480    dir.mkdirs();
481    File classpath = new File(dir, classfile.getName());
482    if (verbose) System.out.printf("writing to file %s%n", classpath);
483    inst_jc.dump(classpath);
484    _numFilesProcessed++;
485    if (((_numFilesProcessed % 100) == 0) && (System.console() != null)) {
486      System.out.printf(
487          "Processed %d/%d classes at %s%n",
488          _numFilesProcessed, classTotal, LocalDateTime.now(ZoneId.systemDefault()));
489    }
490  }

492  /**
493   * Print information about methods that were not instrumented. This happens when a method fails
494   * BCEL's verifier (which is more strict than Java's).
495   */
496  private static void print_skipped_methods() {
497
498    if (skipped_methods.isEmpty()) {
499      // System.out.printf("No methods were skipped.%n");
500      return;
501    }
502
503    System.out.println(
504        "Warning: The following JDK methods could not be instrumented. DynComp will");
505    System.out.println("still work as long as these methods are not called by your application.");
506    System.out.println("If your application calls one, it will throw a NoSuchMethodException.");
507
508    List<String> unknown = new ArrayList<>(skipped_methods);
509    unknown.removeAll(known_uninstrumentable_methods);
510    List<String> known = new ArrayList<>(skipped_methods);
511    known.retainAll(known_uninstrumentable_methods);
512
513    if (!unknown.isEmpty()) {
514      System.out.println("Please report the following problems to the Daikon maintainers.");
515      System.out.println(
516          "Please give sufficient details; see \"Reporting problems\" in the Daikon manual.");
517      for (String method : unknown) {
518        System.out.printf("  %s%n", method);
519      }
520    }
521    if (!known.isEmpty()) {
522      System.out.printf("The following are known problems; you do not need to report them.");
523      for (String method : known) {
524        System.out.printf("  %s%n", method);
525      }
526    }
527  }

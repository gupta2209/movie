001package daikon.test;
002
003import static org.junit.Assert.assertEquals;
004
005import daikon.tools.jtb.*;
006import junit.framework.*;
007import org.junit.Test;
008
009/** Tests that Annotate respects tabs. */
010public final class TestAnnotate {
011
012  @Test
013  public void testGetTabbedIndex() {
014
015    String[] tabbed =
016        new String[] {
017          "private int[]\telems;",
018          "\tprivate\tint\tnumberOfElements\t;",
019          "\t\tprivate\t\tint\t\tmax;",
020          "   \tpublic   \tuniqueBoundedStack()   \t{",
021          "   \t   \tnumberOfElements\t= 0;",
022          "\tmax = 2;\t\t",
023          "\telems\t=\tnew int[max]\t;",
024          "\tpublic void push(int k) {",
025          "\t       int index;",
026          "\t       for (index = 0; index<numberOfElements; index++) {",
027          "                        if (k==elems[index]) {"
028        };
029
030    String[] untabbed =
031        new String[] {
032          "private int[]   elems;",
033          "        private int     numberOfElements        ;",
034          "                private         int             max;",
035          "        public          uniqueBoundedStack()    {",
036          "                numberOfElements        = 0;",
037          "        max = 2;",
038          "        elems   =       new int[max]    ;",
039          "        public void push(int k) {",
040          "               int index;",
041          "               for (index = 0; index<numberOfElements; index++) {",
042          "                        if (k==elems[index]) {"
043        };
044
045    for (int i = 0; i < tabbed.length; i++) {
046
047      String tabbedString = tabbed[i];
048      String untabbedString = untabbed[i];
049      for (int j = 0; j < untabbedString.length(); j++) {
050        char untabbedChar = untabbedString.charAt(j);
051        if (untabbedChar != ' ') {
052          char tabbedChar = tabbedString.charAt(AnnotateVisitor.getTabbedIndex(j, tabbedString));
053          assertEquals(untabbedChar, tabbedChar);
054        }
055      }
056    }
057  }
058}

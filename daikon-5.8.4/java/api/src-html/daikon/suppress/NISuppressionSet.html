024/**
025 * Class that defines a set of non-instantiating suppressions for a single invariant (suppressee).
026 *
027 * <p>Not immutable: see recurse_definitions().
028 */
029public class NISuppressionSet implements Iterable<NISuppression> {
030
031  public static final Logger debug = Logger.getLogger("daikon.suppress.NISuppressionSet");
032
033  NISuppression[] suppression_set;
034
035  public NISuppressionSet(NISuppression[] suppressions) {
036    assert suppressions != null;
037    assert suppressions.length != 0;
038    suppression_set = suppressions;
039  }
040
041  @Override
042  public Iterator<NISuppression> iterator() {
043    List<NISuppression> asList = Arrays.<NISuppression>asList(suppression_set);
044    return asList.iterator();
045  }

047  /**
048   * Adds this set to the suppressor map. The map is from the class of the suppressor to this. If
049   * the same suppressor class appears more than once, the suppression is only added once.
050   */
051  public void add_to_suppressor_map(
052      Map<Class<? extends Invariant>, List<NISuppressionSet>> suppressor_map) {
053
054    Set<Class<? extends Invariant>> all_suppressors =
055        new LinkedHashSet<Class<? extends Invariant>>();
056
057    // Loop through each suppression in the suppression set
058    for (int i = 0; i < suppression_set.length; i++) {
059      NISuppression suppression = suppression_set[i];
060
061      // Loop through each suppressor in the suppression
062      for (Iterator<NISuppressor> j = suppression.suppressor_iterator(); j.hasNext(); ) {
063        NISuppressor suppressor = j.next();
064
065        // If we have seen this suppressor already, skip it
066        if (all_suppressors.contains(suppressor.get_inv_class())) {
067          continue;
068        }
069
070        // Note that we have now seen this suppressor invariant class
071        all_suppressors.add(suppressor.get_inv_class());
072
073        // Get the list of suppression sets for this suppressor.  Create it
074        // if this is the first one.  Add this set to the list
075        List<NISuppressionSet> suppression_set_list =
076            suppressor_map.get(suppressor.get_inv_class());
077        if (suppression_set_list == null) {
078          suppression_set_list = new ArrayList<NISuppressionSet>();
079          suppressor_map.put(suppressor.get_inv_class(), suppression_set_list);
080        }
081        suppression_set_list.add(this);
082      }
083    }
084  }

086  /**
087   * NIS process a falsified invariant. This method should be called for each falsified invariant in
088   * turn. Any invariants for which inv is the last valid suppressor are added to new_invs.
089   *
090   * <p>Note, this is no longer the preferred approach, but is kept for informational purposes. Use
091   * NIS.process_falsified_invs() instead.
092   */
093  public void falsified(Invariant inv, List<Invariant> new_invs) {
094
095    // Get the ppt we are working in
096    PptTopLevel ppt = inv.ppt.parent;
097    assert ppt.equality_view != null
098        : "@AssumeAssertion(nullness): haven't reasoned through the reason";
099
100    // For now all suppressors are unary/binary and
101    // all suppressees are unary, binary or ternary
102    assert inv.ppt.var_infos.length < 3;
103
104    // check unary, binary and ternary suppressees separately
105
106    // unary suppressee
107    if (suppression_set[0].suppressee.var_count == 1) {
108      // Create all of the valid unary slices that use the vars from inv
109      // and check to see if the invariant should be created for each slice
110      if (inv.ppt.var_infos.length == 1) {
111        VarInfo v1 = inv.ppt.var_infos[0];
112        VarInfo[] vis = new VarInfo[] {v1};
113
114        // Make sure the slice is interesting and has valid types over the
115        // suppressee invariant
116        if (!v1.missingOutOfBounds() && ppt.is_slice_ok(v1)) {
117          if (suppression_set[0].suppressee.sample_inv.valid_types(vis)) {
118            check_falsified(ppt, vis, inv, new_invs);
119          }
120        }
121      }
122      return;
123    }
124
125    // binary suppressee
126    if (suppression_set[0].suppressee.var_count == 2) {
127      // Create all of the valid binary slices that use the vars from inv
128      // and check to see if the invariant should be created for each slice
129      if (inv.ppt.var_infos.length == 2) {
130        VarInfo v1 = inv.ppt.var_infos[0];
131        VarInfo v2 = inv.ppt.var_infos[1];
132        VarInfo[] vis = new VarInfo[] {v1, v2};
133
134        // Make sure the slice is interesting and has valid types over the
135        // suppressee invariant
136        if (!v1.missingOutOfBounds() && !v2.missingOutOfBounds() && ppt.is_slice_ok(v1, v2)) {
137          if (suppression_set[0].suppressee.sample_inv.valid_types(vis)) {
138            check_falsified(ppt, vis, inv, new_invs);
139          }
140        }
141
142      } else /* must be unary */ {
143        VarInfo v1 = inv.ppt.var_infos[0];
144        VarInfo[] leaders = ppt.equality_view.get_leaders_sorted();
145        for (int i = 0; i < leaders.length; i++) {
146          VarInfo l1 = leaders[i];
147
148          // hashcode types are not involved in suppressions
149          if (NIS.dkconfig_skip_hashcode_type) {
150            if (l1.file_rep_type.isHashcode()) {
151              continue;
152            }
153          }
154
155          // Make sure the slice is interesting
156          if (v1.missingOutOfBounds() || l1.missingOutOfBounds()) {
157            continue;
158          }
159          if (!ppt.is_slice_ok(v1, l1)) {
160            continue;
161          }
162
163          VarInfo[] vis;
164
165          // Sort the variables
166          if (v1.varinfo_index <= l1.varinfo_index) {
167            vis = new VarInfo[] {v1, l1};
168          } else {
169            vis = new VarInfo[] {l1, v1};
170          }
171
172          if (!suppression_set[0].suppressee.sample_inv.valid_types(vis)) {
173            continue;
174          }
175
176          if (NIS.debug.isLoggable(Level.FINE)) {
177            NIS.debug.fine(
178                "processing slice "
179                    + Arrays.toString(vis)
180                    + " in ppt "
181                    + ppt.name()
182                    + " with "
183                    + ppt.numViews());
184          }
185
186          check_falsified(ppt, vis, inv, new_invs);
187        }
188      }
189      return;
190    }
191
192    // ternary suppressee
193    if (suppression_set[0].suppressee.var_count == 3) {
194      // Create all of the valid ternary slices that use the vars from inv
195      // and check to see if the invariant should be created for each slice
196      if (inv.ppt.var_infos.length == 2) {
197        VarInfo v1 = inv.ppt.var_infos[0];
198        VarInfo v2 = inv.ppt.var_infos[1];
199        VarInfo[] leaders = ppt.equality_view.get_leaders_sorted();
200        for (int i = 0; i < leaders.length; i++) {
201          VarInfo l = leaders[i];
202
203          if (NIS.dkconfig_skip_hashcode_type) {
204            if (l.file_rep_type.isHashcode()) {
205              continue;
206            }
207          }
208
209          if (!ppt.is_slice_ok(l, v1, v2)) {
210            continue;
211          }
212          if (l.missingOutOfBounds() || v1.missingOutOfBounds() || v2.missingOutOfBounds()) {
213            continue;
214          }
215
216          VarInfo[] vis;
217
218          // Order the variables,
219          if (l.varinfo_index <= v1.varinfo_index) {
220            vis = new VarInfo[] {l, v1, v2};
221          } else if (l.varinfo_index <= v2.varinfo_index) {
222            vis = new VarInfo[] {v1, l, v2};
223          } else {
224            vis = new VarInfo[] {v1, v2, l};
225          }
226
227          if (!suppression_set[0].suppressee.sample_inv.valid_types(vis)) {
228            continue;
229          }
230
231          if (NIS.debug.isLoggable(Level.FINE)) {
232            NIS.debug.fine(
233                "processing slice "
234                    + Arrays.toString(vis)
235                    + " in ppt "
236                    + ppt.name()
237                    + " with "
238                    + ppt.numViews());
239          }
240
241          check_falsified(ppt, vis, inv, new_invs);
242        }
243      } else /* must be unary */ {
244        VarInfo v1 = inv.ppt.var_infos[0];
245        VarInfo[] leaders = ppt.equality_view.get_leaders_sorted();
246        for (int i = 0; i < leaders.length; i++) {
247          VarInfo l1 = leaders[i];
248
249          if (NIS.dkconfig_skip_hashcode_type) {
250            if (l1.file_rep_type.isHashcode()) {
251              continue;
252            }
253          }
254
255          for (int j = i; j < leaders.length; j++) {
256            VarInfo l2 = leaders[j];
257
258            if (NIS.dkconfig_skip_hashcode_type) {
259              if (l2.file_rep_type.isHashcode()) {
260                continue;
261              }
262            }
263
264            // Make sure the slice is interesting
265            if (v1.missingOutOfBounds() || l1.missingOutOfBounds() || l2.missingOutOfBounds()) {
266              continue;
267            }
268            if (!ppt.is_slice_ok(v1, l1, l2)) {
269              continue;
270            }
271
272            VarInfo[] vis;
273
274            // Sort the variables
275            if (v1.varinfo_index <= l1.varinfo_index) {
276              vis = new VarInfo[] {v1, l1, l2};
277            } else if (v1.varinfo_index <= l2.varinfo_index) {
278              vis = new VarInfo[] {l1, v1, l2};
279            } else {
280              vis = new VarInfo[] {l1, l2, v1};
281            }
282
283            if (!suppression_set[0].suppressee.sample_inv.valid_types(vis)) {
284              continue;
285            }
286
287            if (NIS.debug.isLoggable(Level.FINE)) {
288              NIS.debug.fine(
289                  "processing slice "
290                      + Arrays.toString(vis)
291                      + " in ppt "
292                      + ppt.name()
293                      + " with "
294                      + ppt.numViews());
295            }
296
297            check_falsified(ppt, vis, inv, new_invs);
298          }
299        }
300      }
301      return;
302    }
303  }

305  /**
306   * Checks the falsified invariant against the slice specified by vis. If the falsification of inv
307   * removed the last valid suppression, then instantiates the suppressee.
308   */
309  private void check_falsified(
310      PptTopLevel ppt, VarInfo[] vis, Invariant inv, List<Invariant> new_invs) {
311
312    // process each suppression in the set, marking each suppressor as
313    // to whether it is true, false, or matches the falsified inv
314    // If any particular suppression is still valid, just return as there
315    // is nothing to be done (the suppressee is still suppressed)
316
317    for (int i = 0; i < suppression_set.length; i++) {
318
319      NIS.SuppressState status = suppression_set[i].check(ppt, vis, inv);
320      if (status == NIS.SuppressState.VALID) {
321        if (NIS.debug.isLoggable(Level.FINE)) {
322          NIS.debug.fine("suppression " + suppression_set[i] + " is valid");
323        }
324        return;
325      }
326      assert status != NIS.SuppressState.NONSENSICAL;
327    }
328
329    if (NIS.debug.isLoggable(Level.FINE)) {
330      NIS.debug.fine("After check, suppression set: " + this);
331    }
332
333    // There are no remaining valid (true) suppressions.  If inv is the
334    // first suppressor to be removed from any suppressions, then this
335    // falsification removed the last valid suppression.  In that case we
336    // need to instantiate the suppressee.
337    for (int i = 0; i < suppression_set.length; i++) {
338      if (suppression_set[i].invalidated()) {
339
340        Invariant v = suppression_set[i].suppressee.instantiate(vis, ppt);
341        if (v != null) new_invs.add(v);
342        return;
343      }
344    }
345  }

347  /**
348   * Determines whether or not the suppression set is valid in the specified slice. The suppression
349   * set is valid if any of its suppressions are valid. A suppression is valid if all of its
350   * suppressors are true.
351   *
352   * <p>Also updates the debug information in each suppressor.
353   *
354   * @see #is_instantiate_ok(PptSlice) for a check that considers missing
355   */
356  public boolean suppressed(PptSlice slice) {
357
358    return (suppressed(slice.parent, slice.var_infos));
359  }

361  /**
362   * Determines whether or not the suppression set is valid in the specified ppt and var_infos. The
363   * suppression set is valid if any of its suppressions are valid. A suppression is valid if all of
364   * its suppressors are true.
365   *
366   * <p>Also updates the debug information in each suppressor.
367   *
368   * @see #is_instantiate_ok(PptTopLevel,VarInfo[]) for a check that considers missing
369   */
370  public boolean suppressed(PptTopLevel ppt, VarInfo[] var_infos) {
371
372    // Check each suppression to see if it is valid
373    for (int i = 0; i < suppression_set.length; i++) {
374      NIS.SuppressState status = suppression_set[i].check(ppt, var_infos, null);
375      if (status == NIS.SuppressState.VALID) {
376        if (Debug.logOn() || NIS.debug.isLoggable(Level.FINE)) {
377          Debug.log(
378              NIS.debug,
379              getClass(),
380              ppt,
381              var_infos,
382              "suppression "
383                  + suppression_set[i]
384                  + " is "
385                  + status
386                  + " in ppt "
387                  + ppt
388                  + " with var infos "
389                  + Arrays.toString(var_infos));
390        }
391        return true;
392      }
393    }
394
395    if (Debug.logOn() || NIS.debug.isLoggable(Level.FINE)) {
396      Debug.log(
397          NIS.debug,
398          getClass(),
399          ppt,
400          var_infos,
401          "suppression "
402              + this
403              + " is not valid in ppt "
404              + ppt
405              + " with var infos "
406              + Arrays.toString(var_infos));
407    }
408    return false;
409  }

422  /**
423   * Determines whether or not the suppressee of the suppression set should be instantiated.
424   * Instantiation is ok only if each suppression is invalid. A suppression is valid if all of its
425   * non-missing suppressors are true.
426   */
427  @Pure
428  public boolean is_instantiate_ok(PptTopLevel ppt, VarInfo[] var_infos) {
429
430    // Check each suppression to see if it is valid
431    for (int i = 0; i < suppression_set.length; i++) {
432      NIS.SuppressState status = suppression_set[i].check(ppt, var_infos, null);
433      if ((status == NIS.SuppressState.VALID) || (status == NIS.SuppressState.NONSENSICAL)) {
434        if (Debug.logOn() || NIS.debug.isLoggable(Level.FINE)) {
435          Debug.log(
436              NIS.debug,
437              getClass(),
438              ppt,
439              var_infos,
440              "suppression "
441                  + suppression_set[i]
442                  + " is "
443                  + status
444                  + " in ppt "
445                  + ppt
446                  + " with var infos "
447                  + Arrays.toString(var_infos));
448        }
449        return false;
450      }
451    }
452
453    if (Debug.logOn() || NIS.debug.isLoggable(Level.FINE)) {
454      Debug.log(
455          NIS.debug,
456          getClass(),
457          ppt,
458          var_infos,
459          "suppression "
460              + this
461              + " is not valid in ppt "
462              + ppt
463              + " with var infos "
464              + Arrays.toString(var_infos));
465    }
466    return true;
467  }

469  /**
470   * Side-effects this NISuppressionSet. Each suppression where a suppressor matches the suppressee
471   * in ss is augmented by additional suppression(s) where the suppressor is replaced by each of its
472   * suppressions. This allows recursive suppressions.
473   *
474   * <p>For example, consider the suppressions:
475   *
476   * <pre>
477   *    (r == arg1) &and; (arg2 &le; arg1) &rArr; r = max(arg1,arg2)
478   *    (arg2 == arg1) &rArr; arg2 &le; arg1
479   * </pre>
480   *
481   * The suppressor (arg2 &le; arg1) in the first suppression matches the suppressee in the second
482   * suppression. In order for the first suppression to work even when (arg2 &le; arg1) is
483   * suppressed, the second suppression is added to the first:
484   *
485   * <pre>
486   *    (r == arg1) &and; (arg2 &le; arg1) &rArr; r = max(arg1,arg2)
487   *    (r == arg1) &and; (arg2 == arg1) &rArr; r = max(arg1,arg2)
488   * </pre>
489   *
490   * When (arg2 &le; arg1) is suppressed, the second suppression for max will still suppress max. If
491   * (arg2 == arg1) is falsified, the (arg2 &le; arg1) invariant will be created and can continue to
492   * suppress max (as long as it is not falsified itself).
493   */
494  public void recurse_definitions(NISuppressionSet ss) {
495
496    // Get all of the new suppressions
497    List<NISuppression> new_suppressions = new ArrayList<>();
498    for (int i = 0; i < suppression_set.length; i++) {
499      new_suppressions.addAll(suppression_set[i].recurse_definition(ss));
500    }
501    // This isn't necessarily true if the suppressee is of the same
502    // class but doesn't match due to variable swapping.
503    // assert new_suppressions.size() > 0;
504
505    // Create a new suppression set with all of the suppressions.
506    /*NNC:@MonotonicNonNull*/ NISuppression[] new_array =
507        new NISuppression[suppression_set.length + new_suppressions.size()];
508    for (int i = 0; i < suppression_set.length; i++) {
509      new_array[i] = suppression_set[i];
510    }
511    for (int i = 0; i < new_suppressions.size(); i++) {
512      new_array[suppression_set.length + i] = new_suppressions.get(i);
513    }
514    new_array = castNonNullDeep(new_array); // https://tinyurl.com/cfissue/986
515    suppression_set = new_array;
516  }

518  /**
519   * Swaps each suppressor and suppressee to the opposite variable order. Valid only on unary and
520   * binary suppressors and suppressees.
521   */
522  public NISuppressionSet swap() {
523
524    NISuppression[] swap_sups = new NISuppression[suppression_set.length];
525    for (int i = 0; i < swap_sups.length; i++) {
526      NISuppression std_sup = suppression_set[i];
527      /*NNC:@MonotonicNonNull*/ NISuppressor[] sors = new NISuppressor[std_sup.suppressors.length];
528      for (int j = 0; j < sors.length; j++) {
529        sors[j] = std_sup.suppressors[j].swap();
530      }
531      sors = castNonNullDeep(sors); // https://tinyurl.com/cfissue/986
532      swap_sups[i] = new NISuppression(sors, std_sup.suppressee.swap());
533    }
534    NISuppressionSet new_ss = new NISuppressionSet(swap_sups);
535    return new_ss;
536  }

538  /** Returns the suppressee. */
539  public NISuppressee get_suppressee() {
540    return suppression_set[0].suppressee;
541  }

543  /** Clears the suppressor state in each suppression. */
544  public void clear_state() {
545    for (int i = 0; i < suppression_set.length; i++) {
546      suppression_set[i].clear_state();
547    }
548  }

550  /** Returns a string containing each suppression separated by commas. */
551  @SideEffectFree
552  @Override
553  public String toString(@GuardSatisfied NISuppressionSet this) {
554    return "{ " + UtilPlume.join(", ", suppression_set) + " }";

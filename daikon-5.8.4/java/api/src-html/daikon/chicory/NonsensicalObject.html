008/**
009 * A NonsensicalObject is used during data trace output for variables whose value is "nonsensical"
010 * to print. For instance, say class A has a field name. If variable x is of type A is null, then we
011 * print "null" for x's value. However, we print "nonsensical" for x.name's value.
012 */
013public class NonsensicalObject {
014  private static NonsensicalObject instance = new NonsensicalObject();
015
016  private NonsensicalObject() {}
017
018  public static NonsensicalObject getInstance() {
019    return instance;
020  }
021}

360   * Given ((x0,y0),(x1,y1)), finds a,b and c such that ax + by + c = 0. Places a, b and c (if they
361   * exist) into result at indexes 0, 1 and 2 respectively.
362   *
363   * @return true if such an a and b exist
364   */
365  private static boolean find_bi_linear(
366      long x0, long x1, long y0, long y1, double[] result) {
367    if (x1 - x0 == 0) {         // not "x0 == x1", due to roundoff
368      // x being constant would have been discovered elsewhere (and this
369      // invariant would not have been instantiated).
370      return false;
371    }
372    if (y1 - y0 == 0) {         // not "y0 == y1", due to roundoff
373      // y being constant would have been discovered elsewhere (and this
374      // invariant would not have been instantiated).
375      return false;
376    }
377
378    long i_var = y1 - y0;
379    long j_var = x0 - x1;
380    long Stand_const = y0 * x1 - x0 * y1;
381
382    // Put the coefficients in lowest terms by dividing by the gcd.
383    // If the gcd is 1, no harm done -- divide by 1.
384    // Type is double to ensure floating-point division.
385    double myGCD = MathPlume.gcd(MathPlume.gcd(i_var, j_var), Stand_const);
386    double standard_i_var = i_var / myGCD;
387    double standard_j_var = j_var / myGCD;
388    double standard_Stand_const = Stand_const / myGCD;
389
390    // Ensure postive x coefficient by negating if "x" term is negative.
391    if (i_var < 0) {
392      standard_i_var = -standard_i_var;
393      standard_j_var = -standard_j_var;
394      standard_Stand_const = -standard_Stand_const;
395    }
396
397    result[0] = standard_i_var;
398    result[1] = standard_j_var;
399    result[2] = standard_Stand_const;
400
401    return true;
402  }

565  /**
566   * Merge the linear binary cores in cores to form a new core. Any core in the list that has seen
567   * enough points to define a line, must define the same line. Any cores that have not yet seen
568   * enough points, will have each of their points applied to that invariant. The merged core is
569   * returned. Null is returned if the cores don't describe the same line
570   *
571   * @param cores list of LinearBinary cores to merge. They should all be permuted to match the
572   *     variable order in ppt.
573   */
574  public @Nullable LinearBinaryCore merge(List<LinearBinaryCore> cores, Invariant wrapper) {
575
576    // Look for any active lines.  All must define the same line
577    LinearBinaryCore result = null;
578    for (LinearBinaryCore c : cores) {
579      if (!c.isActive()) {
580        continue;
581      }
582      if (result == null) {
583        result = c.clone();
584      } else {
585        if (!Global.fuzzy.eq(result.a, c.a)
586            || !Global.fuzzy.eq(result.b, c.b)
587            || !Global.fuzzy.eq(result.c, c.c)) {
588          return null;
589        }
590      }
591    }
592
593    // If no active lines were found, created an empty core
594    if (result == null) {
595      result = new LinearBinaryCore(wrapper);
596    } else {
597      result.wrapper = wrapper;
598    }
599
600    // Merge in any points from non-active cores
601    for (LinearBinaryCore c : cores) {
602      if (c.isActive()) {
603        continue;
604      }
605      for (int j = 0; j < c.values_seen; j++) {
606        if (result.add_modified(c.x_cache[j], c.y_cache[j], 1) == InvariantStatus.FALSIFIED) {
607          //        if (wrapper.falsified)
608          return null;
609        }
610      }
611    }
612
613    return result;
614  }

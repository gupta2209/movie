123  /**
124   * One of the last phases for DynComp is to write out the comparability values after the user
125   * program completes execution. One of the steps is to assign values to the arguments of methods
126   * that have not been executed. We use reflection to get type information about these arguments,
127   * which causes the method to be loaded; which causes the main part of DynComp to try to
128   * instrument the method. As the user program has completed execution, doing instrumentation at
129   * this point can lead to problems. The correct fix for this problem is to use BCEL to get the
130   * type information instead of reflection, thus avoiding loading the method into the JVM. This
131   * will be a large change, so a temporary fix is to indicate if the program is in shutdown mode
132   * and not instrument any methods when this flag is true. TODO: Couldn't we just call
133   * removeTransformer at the start of shutdown?
134   */
135  protected static boolean in_shutdown = false;

146  /**
147   * This method is the entry point of the java agent. Its main purpose is to set up the transformer
148   * so that when classes from the target app are loaded, they are first transformed in order to add
149   * comparability instrumentation.
150   *
151   * <p>If this code is running on Java 9+ and jdk_instrumented is true it also retransforms any JDK
152   * methods that were loaded prior to premain getting control.
153   *
154   * @param agentArgs string containing the arguments passed to this agent
155   * @param inst instrumentation instance to be used to transform classes
156   * @throws IOException if jdk_classes.txt cannot be read or if the correct version of BCEL cannot
157   *     be found or loaded
158   */
159  public static void premain(String agentArgs, Instrumentation inst) throws IOException {
160    // For debugging
161    // this.instr = inst;
162
163    // Because DynComp started Premain in a separate process, we must rescan
164    // the options to set up the DynComp static variables.
165    Options options = new Options(DynComp.synopsis, DynComp.class, Premain.class);
166    String[] args = options.parse(true, agentArgs.split("  *"));
167    if (args.length > 0) {
168      System.err.printf("Unexpected arguments %s%n", Arrays.toString(args));
169      options.printUsage();
170      System.exit(-1);
171    }
172    if (DynComp.rt_file != null && DynComp.rt_file.getName().equalsIgnoreCase("NONE")) {
173      DynComp.no_jdk = true;
174      DynComp.rt_file = null;
175    }
176
177    // Note that the following references to static fields have an important
178    // side effect: They cause the corresponding class to be loaded. This helps
179    // Instrument.transform() avoid ClassCircularityErrors during initialization.
180    DaikonVariableInfo.std_visibility = DynComp.std_visibility;
181    DCRuntime.depth = DynComp.nesting_depth;
182    DCInstrument.jdk_instrumented = !DynComp.no_jdk;
183    @SuppressWarnings("UnusedVariable") // loads the BcelUtil class; otherwise, Premain gives errors
184    int junk = BcelUtil.javaVersion;
185
186    // Another 'trick' to force needed classes to be loaded prior to retransformation.
187    String buffer =
188        String.format("In dcomp premain, agentargs ='%s', Instrumentation = '%s'", agentArgs, inst);
189    if (DynComp.verbose) {
190      System.out.println(buffer);
191      System.out.printf("Options settings: %n%s%n", options.settings());
192    }
193
194    // Read the list of pre-instrumented classes.
195    if (DCInstrument.jdk_instrumented) {
196      // location is: daikon/java/dcomp-rt/java/lang/jdk_classes.txt .
197      InputStream strm = Object.class.getResourceAsStream("jdk_classes.txt");
198      if (strm == null) {
199        System.err.println(
200            "Can't find jdk_classes.txt; see Daikon manual, section \"Instrumenting the JDK with DynComp\"");
201        System.exit(1);
202      }
203      BufferedReader reader = new BufferedReader(new InputStreamReader(strm, UTF_8));
204
205      // Verify that the current no-primitives flag setting matches the setting used to
206      // generate dcomp_rt.jar (via the BuildJDK tool).
207      String noPrimitivesLine = reader.readLine();
208      if (noPrimitivesLine == null) {
209        System.err.println("jdk_classes.txt is an empty file.");
210        System.exit(1);
211      }
212      noPrimitivesLine = noPrimitivesLine.trim();
213      if (!noPrimitivesLine.startsWith("no_primitives: ")) {
214        System.err.println("First line of jdk_classes.txt does not contain no_primitives flag.");
215        System.exit(1);
216      }
217      if (DynComp.no_primitives != noPrimitivesLine.equalsIgnoreCase("no_primitives: true")) {
218        System.err.println(
219            "no-primitives flag does not match setting used to generate dcomp_rt.jar.");
220        System.exit(1);
221      }
222
223      // Read in the list of pre-instrumented classes
224      while (true) {
225        String line = reader.readLine();
226        if (line == null) {
227          break;
228        }
229        // System.out.printf("adding '%s'%n", line);
230        pre_instrumented.add(line);
231      }
232    }
233
234    // Setup the shutdown hook
235    Thread shutdown_thread = new ShutdownThread();
236    java.lang.Runtime.getRuntime().addShutdownHook(shutdown_thread);
237
238    // Setup the transformer
239    ClassFileTransformer transformer;
240    // use a special classloader to ensure correct version of BCEL is used
241    ClassLoader loader = new daikon.chicory.ChicoryPremain.ChicoryLoader();
242    try {
243      transformer =
244          (ClassFileTransformer)
245              loader.loadClass("daikon.dcomp.Instrument").getDeclaredConstructor().newInstance();
246    } catch (Exception e) {
247      throw new RuntimeException("Unexpected error loading Instrument", e);
248    }
249    if (DynComp.verbose) {
250      System.out.printf(
251          "Classloader of tranformer = %s%n", transformer.getClass().getClassLoader());
252    }
253
254    // Check that we got a newer version of BCEL that includes JDK 11 support. At present,
255    // this is only the PLSE 6.4.1.1 release version.  We can verify this version by the
256    // presence of the method FieldGenOrMethodGen.removeAnnotationEntries().
257    try {
258      Class<?> c = loader.loadClass("org.apache.bcel.generic.FieldGenOrMethodGen");
259      c.getMethod("removeAnnotationEntries", (Class<?>[]) null);
260    } catch (Exception e) {
261      System.err.printf("%nBCEL jar found is not the version included with the Daikon release.%n");
262      System.exit(1);
263    }
264
265    // now turn on instrumentation
266    if (DynComp.verbose) {
267      System.out.println("call addTransformer");
268    }
269    inst.addTransformer(transformer, true);
270
271    // For Java 9+ we only partially instrument the JDK as part of building DynComp.
272    // We complete the instrumentation when a JDK class is loaded during program execution.
273    // However, there is Catch-22: "All future class definitions will be seen by the transformer,
274    // except definitions of classes upon which any registered transformer is dependent"
275    // (from the documentation for Instrumentation.addTransformer()).
276    // Thus a JDK class used by DynComp will not be seen by our transformer.  To get around this
277    // we get the list of all the classes already loaded and call retransformClasses on them.
278    // Unfortunately, this process may need to be repeated multiple times as each time we call
279    // retransform on a class it may require the use of a previously unexecuted part of DynComp
280    // which may, in turn, cause more JDK classes to be loaded without our knowledge.
281    //
282    // Possibility:
283    // There may be a way to use the JDeps tool to get a list of all the dependencies as part of
284    // the DynComp build process and then pass this list to Premain for processing.
285    //
286    // Future work:
287    // We should revisit the problems associated with pre-instrumenting the JDK via BuildJDK.
288    // The current system has a noticeable performance hit at startup in addition to the
289    // retransformation complications noted above.
290    //
291    if (BcelUtil.javaVersion > 8 && DCInstrument.jdk_instrumented) {
292
293      retransform_preloads = true;
294      Class<?>[] classes_to_retransform = get_retransform_list(inst);
295
296      while (classes_to_retransform.length > 0) {
297        if (DynComp.verbose) {
298          System.out.println("call retransformClasses");
299        }
300        try {
301          inst.retransformClasses(classes_to_retransform);
302        } catch (Exception e) {
303          System.err.println("Unable to retransformClasses.");
304          System.err.println(e);
305        }
306        classes_to_retransform = get_retransform_list(inst);
307      }
308    }
309    retransform_preloads = false;
310
311    // Iterator<Class<?>> value = previously_processed_classes.iterator();
312    // while (value.hasNext()) {
313    //   System.out.println(value.next());
314    // }
315
316    // Initialize the static tag array
317    if (DynComp.verbose) {
318      System.out.println("call DCRuntime.init");
319    }
320    DCRuntime.init();
321
322    if (DynComp.verbose) {
323      System.out.println("exit premain");
324    }
325  }

327  /**
328   * Get an array of already loaded classes that need to be retransformed.
329   *
330   * @param inst instrumentation instance to be used to transform classes
331   * @return an array containing the classes to be retransformed
332   */
333  private static Class<?>[] get_retransform_list(Instrumentation inst) {
334    if (DynComp.verbose) {
335      System.out.println("get retransformation list");
336    }
337
338    ArrayList<Class<?>> class_list = new ArrayList<>();
339
340    // Get the set of already loaded classes.
341    Class<?>[] loaded_classes = inst.getAllLoadedClasses();
342
343    for (Class<?> loaded_class : loaded_classes) {
344      // System.out.println(loaded_class + ": " + loaded_class.getClassLoader());
345      // Skip previously processed classes.
346      if (previously_processed_classes.contains(loaded_class)) continue;
347      // Skip Daikon classes.
348      if (loaded_class.getName().startsWith("daikon.")) continue;
349      // Skip BCEL classes.
350      if (loaded_class.getName().startsWith("org.apache.bcel.")) continue;
351      // Object cannot be instrumented due to its fundimental nature.
352      if (loaded_class.getName().equals("java.lang.Object")) continue;
353      if (inst.isModifiableClass(loaded_class)) {
354        // System.out.println(loaded_class);
355        class_list.add(loaded_class);
356      }
357    }
358    previously_processed_classes = new HashSet<>(Arrays.asList(loaded_classes));
359    return class_list.toArray(new Class<?>[class_list.size()]);
360  }

362  /** Shutdown thread that writes out the comparability results. */
363  public static class ShutdownThread extends Thread {
364
365    @Override
366    public void run() {
367
368      if (DynComp.verbose) {
369        System.out.println("in shutdown");
370      }
371      in_shutdown = true;
372
373      // for debugging
374      // Class<?>[] loaded_classes = instr.getAllLoadedClasses();
375      // for (Class<?> loaded_class : loaded_classes) {
376      //   System.out.println(loaded_class + ": " + loaded_class.getClassLoader());
377      // }
378
379      // If requested, write the comparability data to a file
380      if (DynComp.comparability_file != null) {
381        if (DynComp.verbose) {
382          System.out.println("Writing comparability sets to " + DynComp.comparability_file);
383        }
384        PrintWriter compare_out = open(DynComp.comparability_file);
385        long startTime = System.nanoTime();
386        if (DynComp.no_primitives) {
387          DCRuntime.print_all_comparable_refs_only(compare_out);
388        } else {
389          DCRuntime.print_all_comparable(compare_out);
390        }
391        compare_out.close();
392        if (DynComp.verbose) {
393          long duration = System.nanoTime() - startTime;
394          System.out.printf(
395              "Comparability sets written in %ds%n", TimeUnit.NANOSECONDS.toSeconds(duration));
396        }
397      }
398
399      if (DynComp.trace_file != null) {
400        if (DynComp.verbose) {
401          System.out.println("Writing traced comparability sets to " + DynComp.trace_file);
402        }
403        PrintWriter trace_out = open(DynComp.trace_file);
404        long startTime = System.nanoTime();
405        DCRuntime.trace_all_comparable(trace_out);
406        trace_out.close();
407        if (DynComp.verbose) {
408          long duration = System.nanoTime() - startTime;
409          System.out.printf(
410              "Traced comparability sets written in %ds%n",
411              TimeUnit.NANOSECONDS.toSeconds(duration));
412        }
413      } else {
414        // Writing comparability sets to standard output?
415      }
416
417      if (DynComp.verbose) {
418        DCRuntime.decl_stats();
419      }
420
421      // Write the decl file out
422      @SuppressWarnings(
423          "nullness:argument.type.incompatible") // DynComp guarantees decl_file is non null
424      File decl_file = new File(DynComp.output_dir, DynComp.decl_file);
425      if (DynComp.verbose) System.out.println("Writing decl file to " + decl_file);
426      PrintWriter decl_fp = open(decl_file);
427      long startTime = System.nanoTime();
428      DCRuntime.print_decl_file(decl_fp);
429      decl_fp.close();
430      if (DynComp.verbose) {
431        long duration = System.nanoTime() - startTime;
432        System.out.printf("Decl file written in %ds%n", TimeUnit.NANOSECONDS.toSeconds(duration));
433        System.out.printf("comp_list = %,d%n", DCRuntime.comp_list_ms);
434        System.out.printf("ppt name  = %,d%n", DCRuntime.ppt_name_ms);
435        System.out.printf("decl vars = %,d%n", DCRuntime.decl_vars_ms);
436        System.out.printf("total     = %,d%n", DCRuntime.total_ms);
437      }
438      if (DynComp.verbose) System.out.println("DynComp complete");
439    }
440  }

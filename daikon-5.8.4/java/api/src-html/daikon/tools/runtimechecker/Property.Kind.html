164  /**
165   * Parse a String and return the property that it represents. An example of the String
166   * representation of an property is:
167   *
168   * <pre>{@code
169   * <INVINFO>
170   * <INV> this.topOfStack <= this.theArray.length-1 </INV>
171   * <ENTER>
172   * <DAIKON>  this.topOfStack <= size(this.theArray[])-1  </DAIKON>
173   * <DAIKONCLASS>class daikon.inv.binary.twoScalar.IntLessEqual</DAIKONCLASS>
174   * <METHOD>  isEmpty()  </METHOD>
175   * </INVINFO>
176   * }</pre>
177   *
178   * <p>The above string should actually span only one line.
179   *
180   * <p>To be well-formed, a property should be enclosed in {@code <INVINFO>} tags, contain {@code
181   * <DAIKON>} and {@code <METHOD>} tags, and exactly one of {@code <ENTER>}, {@code <EXIT>}, {@code
182   * <OBJECT>}, or {@code <CLASS>}.
183   */
184  // [[ Using an XML parser seems like too strong a hammer here.
185  //    But should do some profiling to see if all the string
186  //    matching slows things sufficiently for us to want to optimize. ]]
187  public static Property get(String annoString) throws MalformedPropertyException {
188
189    // check well-formedness
190    if (!(annoString.matches(".*<INVINFO>.*</INVINFO>.*")
191        && annoString.matches(".*<DAIKON>(.*)</DAIKON>.*")
192        && annoString.matches(".*<METHOD>(.*)</METHOD>.*"))) {
193      throw new MalformedPropertyException(annoString);
194    }
195
196    // figure out what kind of property it is
197    Kind k;
198    if (annoString.matches(".*<ENTER>.*")) {
199      k = Kind.enter;
200    } else if (annoString.matches(".*<EXIT>.*")) {
201      k = Kind.exit;
202    } else if (annoString.matches(".*<OBJECT>.*")) {
203      k = Kind.objectInvariant;
204    } else if (annoString.matches(".*<CLASS>.*")) {
205      k = Kind.classInvariant;
206    } else {
207      throw new MalformedPropertyException(annoString);
208    }
209
210    String theDaikonRep = annoString.replaceFirst(".*<DAIKON>(.*)</DAIKON>.*", "$1").trim();
211    String theMethod = annoString.replaceFirst(".*<METHOD>(.*)</METHOD>.*", "$1").trim();
212
213    String theInvRep = annoString.replaceFirst(".*<INV>(.*)</INV>.*", "$1").trim();
214    String theDaikonClass =
215        annoString.replaceFirst(".*<DAIKONCLASS>(.*)</DAIKONCLASS>.*", "$1").trim();
216    double theConfidence = -1;
217    if (annoString.matches(".*<CONFIDENCE>(.*)</CONFIDENCE>.*")) {
218      theConfidence =
219          Double.parseDouble(
220              annoString.replaceFirst(".*<CONFIDENCE>(.*)</CONFIDENCE>.*", "$1").trim());
221    }
222    return Property.get(k, theDaikonRep, theMethod, theInvRep, theDaikonClass, theConfidence);
223  }

283  /**
284   * Find, parse and return all distinct properties found in a list of strings. Each string in
285   * {@code annoStrings} may contain none, one, or several properties. Malformed properties are
286   * ignored.
287   */
288  public static Property[] findProperties(List<String> annoStrings) {
289
290    if (annoStrings == null) {
291      return new Property[] {};
292    }
293    // Pattern p = Pattern.compile("(<INVINFO>.*</INVINFO>)");
294    Set<Property> annos = new HashSet<>();
295    for (String location : annoStrings) {
296      if (location == null || location.equals("")) {
297        continue;
298      }
299      String[] cutUp = location.split("<INVINFO>");
300      // Matcher m = p.matcher(location);
301      for (int splits = 0; splits < cutUp.length; splits++) {
302        // while (m.find()) {
303        try {
304          // String s = m.group(1);
305          String s = cutUp[splits];
306          Property anno = Property.get("<INVINFO>" + s);
307          // [[[ explain this! ]]]
308          annos.add(anno);
309        } catch (Exception e) {
310          // malformed property; just go to next iteration
311        }
312      }
313    }
314    return annos.toArray(new Property[] {});
315  }

317  /** Get the property with the given attributes. */
318  private static Property get(
319      Kind kind,
320      String daikonRep,
321      String method,
322      String jmlRep,
323      String daikonClass,
324      double confidence)
325      throws MalformedPropertyException {
326
327    Property anno = new Property(kind, daikonRep, method, jmlRep, daikonClass, confidence);
328    Integer key = anno.hashCode();
329    if (propertiesMap.containsKey(key)) {
330      return propertiesMap.get(key);
331    } else {
332      if (confidence == -1) {
333        anno.confidence = anno.calculateConfidence();
334      }
335      propertiesMap.put(key, anno);
336      return anno;
337    }
338  }

513  /** The maximum confidence value calculated among all Properties given. */
514  public static double maxConf(Property[] annos) {
515    if (annos.length == 0) {
516      return -1;
517    }
518    double max = 0.0;
519    for (int i = 0; i < annos.length; i++) {
520      if (annos[i].confidence > max) {
521        max = annos[i].confidence;
522      }
523    }
524    return max;
525  }

527  /** The confidence of a set of properties. Currently it's just the maximum confidence. */
528  public static double confidence(Property[] annos) {
529    return maxConf(annos);
530  }

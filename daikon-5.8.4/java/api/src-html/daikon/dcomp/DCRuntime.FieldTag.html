221  /**
222   * Handles calls to instrumented equals() methods. Makes the arguments comparable, and returns
223   * true if they are equals() to one another.
224   *
225   * @param o1 the first argument to equals()
226   * @param o2 the second argument to equals()
227   * @return whether the two values are equal
228   */
229  public static boolean dcomp_equals(Object o1, Object o2) {
230    // Make obj1 and obj2 comparable
231    if ((o1 != null) && (o2 != null)) {
232      TagEntry.union(o1, o2);
233    }
234
235    if (debug) {
236      System.out.printf("In dcomp_equals%n");
237    }
238
239    Method m;
240    try {
241      m =
242          o1.getClass()
243              .getMethod("equals_dcomp_instrumented", new Class<?>[] {java_lang_Object_class});
244    } catch (NoSuchMethodException e) {
245      m = null;
246    } catch (Exception e) {
247      throw new RuntimeException("unexpected error locating equal_dcomp_instrumented", e);
248    }
249
250    if (m != null) {
251      try {
252        return (Boolean) m.invoke(o1, o2);
253      } catch (Exception e) {
254        throw new RuntimeException("unexpected error invoking equal_dcomp_instrumented", e);
255      }
256    }
257
258    // Push tag for return value, and call the uninstrumented version
259    ThreadData td = thread_to_data.get(Thread.currentThread());
260    td.tag_stack.push(new Constant());
261    if (debug_tag_frame) System.out.printf("tag stack size: %d%n", td.tag_stack.size());
262    return o1.equals(o2);
263  }

267   *
268   * <p>Each time we make a call on a particular Object, we keep track of which superclass's equals
269   * method we called last. If that Object makes another call to super.equals before the original
270   * call is done, then invoke the equals method of the next-higher class in the class hierarchy
271   * (i.e. the next superclass).

274   * Object's original super.equals call. Once the equals call terminates, whether by returning a
275   * value or by throwing an exception, the corresponding key is removed from this map.
276   */

286  /**
287   * Handles {@code super.equals(Object)} calls. Makes the arguments comparable, and returns true if
288   * super.equals() returns true for them
289   *
290   * @param o1 the first argument to super.equals()
291   * @param o2 the second argument to super.equals()
292   * @return whether the two values are equal, according to super.equals()
293   * @see #active_equals_calls
294   */
295  public static boolean dcomp_super_equals(Object o1, Object o2) {
296    // Make obj1 and obj2 comparable
297    if ((o1 != null) && (o2 != null)) {
298      TagEntry.union(o1, o2);
299    }
300
301    if (debug) {
302      System.out.printf("In dcomp_super_equals%n");
303    }
304
305    Class<?> o1c = o1.getClass();
306    Class<?> o1super;
307
308    // Check to see if we're already in the middle of a super.equals
309    // call for this Object
310    if (null == active_equals_calls.get(o1)) {
311      // No, we are not
312      o1super = o1c.getSuperclass();
313    } else {
314      // Yes, we are -- continue up the class hierarchy
315      o1super = active_equals_calls.get(o1).getSuperclass();
316    }
317
318    // Update the active_equals_calls map
319    active_equals_calls.put(o1, o1super);
320
321    Class<?>[] o1superifaces = o1super.getInterfaces();
322
323    boolean instrumented = false;
324    for (Class<?> c : o1superifaces) {
325      if (c.getName().equals(DCInstrument.instrumentation_interface)) {
326        instrumented = true;
327        break;
328      }
329    }
330
331    boolean return_val;
332    try {
333      // if the superclass whose method we are calling is instrumented...
334      if (instrumented) {
335        // call the instrumented version
336        Method m =
337            o1super.getMethod(
338                "equals", new Class<?>[] {java_lang_Object_class, dcomp_marker_class});
339        return_val = ((Boolean) m.invoke(o1, o2, null));
340      } else {
341        // Push tag for return value, and call the uninstrumented version
342        ThreadData td = thread_to_data.get(Thread.currentThread());
343        td.tag_stack.push(new Constant());
344        Method m = o1super.getMethod("equals", new Class<?>[] {java_lang_Object_class});
345        return_val = ((Boolean) m.invoke(o1, o2));
346        if (debug_tag_frame) System.out.printf("tag stack size: %d%n", td.tag_stack.size());
347      }
348    } catch (NoSuchMethodException e) {
349      System.err.printf("dcomp_super_equals(%s, %s)%n", obj_str(o1), obj_str(o2));
350      System.err.printf("o1super %s%n", o1super);
351      for (Class<?> c : o1superifaces) {
352        System.err.printf("  o1super interface %s%n", c);
353      }
354      for (Method m : o1super.getDeclaredMethods()) {
355        System.err.printf("  o1super method %s%n", m);
356      }
357      throw new RuntimeException(e);
358    } catch (IllegalAccessException e) {
359      throw new RuntimeException(e);
360    } catch (InvocationTargetException e) {
361      e.printStackTrace();
362      throw new RuntimeException(e);
363    } catch (Exception e) {
364      throw new RuntimeException("unexpected error locating equals method", e);
365    }
366
367    // We are now done with the call, so remove the entry for this
368    // call from the active_equals_calls map
369    active_equals_calls.remove(o1);
370    return return_val;
371  }

373  /**
374   * Handles {@code clone()} calls.
375   *
376   * <p>This method throws Throwable because it may throw any checked exception that is thrown by
377   * {@code o.clone()}.
378   */
379  // XXX TODO consolidate this and dcomp_super_clone, since there is a
380  // lot of duplicated code
381  public static Object dcomp_clone(Object o) throws Throwable {
382    Class<?> target_class = o.getClass();
383
384    if (debug) {
385      System.out.printf("In dcomp_clone%n");
386    }
387
388    Object return_val;
389    Method m;
390
391    try {
392      if (target_class.getName().equals("[Ljava.lang.Class;")) {
393        Class<?>[] ca = (Class<?>[]) o;
394        return_val = ca.clone();
395      } else if (target_class.getName().equals("java.lang.Object")) {
396        // call the uninstrumented Object.clone()
397        // Use getDeclaredMethod instead of getMethod because clone is
398        // protected
399        m = target_class.getDeclaredMethod("clone", new Class<?>[] {});
400        return_val = m.invoke(o);
401      } else {
402        // every other class has an instrumented version
403        m = target_class.getDeclaredMethod("clone", new Class<?>[] {dcomp_marker_class});
404
405        // Use length-1 array containing null to distinguish from just
406        // null, which indicates 0 arguments
407        return_val = m.invoke(o, new Object[] {null});
408      }
409    } catch (IllegalAccessException e) {
410      throw new RuntimeException(e);
411    } catch (NoSuchMethodException e) {
412      throw new RuntimeException("error finding clone in " + target_class.getName(), e);
413    } catch (InvocationTargetException e) {
414      throw e.getCause();
415    }
416
417    // Make o and its clone comparable
418    if ((o != null) && (return_val != null)) {
419      TagEntry.union(o, return_val);
420    }
421
422    return return_val;
423  }

425  /**
426   * Handles {@code super.clone()} calls.
427   *
428   * @see #active_clone_calls
429   */
430  public static Object dcomp_super_clone(Object o) throws Throwable {
431    Class<?> oc = o.getClass(); // "Don't call it that."
432
433    if (debug) {
434      System.out.printf("In dcomp_super_clone%n");
435    }
436
437    Class<?> target_class; // The class whose method we will invoke
438    if (null == active_clone_calls.get(o)) target_class = oc;
439    else {
440      target_class = active_clone_calls.get(o).getSuperclass();
441    }
442    active_clone_calls.put(o, target_class);
443
444    Object return_val;
445    Method m;
446
447    try {
448      if (target_class.getName().startsWith("[L")) {
449        // Arrays can't use reflection
450        Object[] oa = (Object[]) o;
451        return_val = oa.clone();
452      } else if (target_class.getName().equals("java.lang.Object")) {
453        // call the uninstrumented Object.clone()
454        // Use getDeclaredMethod instead of getMethod because clone is
455        // protected
456        m = oc.getDeclaredMethod("clone", new Class<?>[] {});
457        return_val = m.invoke(o);
458      } else {
459        // every other class has an instrumented version
460        m = target_class.getDeclaredMethod("clone", new Class<?>[] {dcomp_marker_class});
461
462        // Use length-1 array containing null to distinguish from just
463        // null, which indicates 0 arguments
464        return_val = m.invoke(o, new Object[] {null});
465      }
466    } catch (IllegalAccessException e) {
467      // This shouldn't happen
468      active_clone_calls.remove(o);
469      throw new RuntimeException(e);
470    } catch (NoSuchMethodException e) {
471      // This shouldn't happen
472      active_clone_calls.remove(o);
473      throw new RuntimeException(e);
474    } catch (InvocationTargetException e) {
475      // This might happen - if an exception is thrown from clone(),
476      // propagate it by rethrowing it
477      active_clone_calls.remove(o);
478      throw e.getCause();
479    }
480
481    // Make o and its clone comparable
482    if ((o != null) && (return_val != null)) {
483      TagEntry.union(o, return_val);
484    }
485    active_clone_calls.remove(o);
486    return return_val;
487  }

497  public static boolean has_instrumented(Class<?> c, String method_name) {
498
499    if (debug) {
500      System.out.printf("In has_instrumented%n");
501    }
502
503    // System.out.printf("has_instrumented: %s %s %s%n", c, method_name, dcomp_marker_class);
504
505    Class<?>[] args = new Class<?>[] {dcomp_marker_class};
506    while (!c.getName().equals("java.lang.Object")) {
507      java.lang.reflect.Method m;
508      try {
509        m = c.getDeclaredMethod(method_name, args);
510      } catch (Exception e) {
511        m = null;
512      }
513      // System.out.printf("Class %s instrumented %s = %s%n", c, method_name, m);
514      if (m != null) {
515        return true;
516      }
517      c = c.getSuperclass();
518    }
519    return false;
520  }

586  /**
587   * Create the tag frame for this method. Pop the tags for any primitive parameters off of the tag
588   * stack and store them in the tag frame.
589   *
590   * @param params encodes the position of the primitive parameters into a string. The first
591   *     character is size of the tag frame. The remaining characters indicate where each parameter
592   *     on the tag stack should be stored into the frame. For example "20" allocates a tag frame
593   *     with two elements and stores the top of the tag stack into element 0. A string is used for
594   *     simplicity in code generation since strings can easily be placed into the constant portion
595   *     of the class file. Note that characters are determined by adding the integer value to '0'.
596   *     Values greater than 9 will have unintuitive (but printable) values.
597   * @return the allocated and initialized tag frame
598   */
599  public static Object[] create_tag_frame(String params) {
600    if (debug) {
601      System.out.printf("%nEnter: %s%n", caller_name());
602    }
603
604    if (DynComp.verbose) {
605      String method_name = caller_name();
606      Integer count = methodCountMap.get(method_name);
607      if (count == null) {
608        count = 1;
609      } else {
610        count = count.intValue() + 1;
611      }
612      methodCountMap.put(method_name, count);
613    }
614
615    // create_tag_frame is the first DCRuntime method called for an
616    // instrumented user method.  Since it might be on a new thread
617    // we need to check/set the per-thread data map.
618    Thread t = Thread.currentThread();
619    ThreadData td = thread_to_data.get(t);
620    if (td == null) {
621      td = new ThreadData();
622      thread_to_data.put(t, td);
623    }
624
625    int frame_size = ((int) params.charAt(0)) - '0';
626    // Character.digit (params.charAt(0), Character.MAX_RADIX);
627    Object[] tag_frame = new Object[frame_size];
628    if (debug_tag_frame) {
629      System.out.printf(
630          "Creating tag frame of size %d [%s] for %s%n", frame_size, params, caller_name());
631    }
632    for (int ii = 1; ii < params.length(); ii++) {
633      int offset = params.charAt(ii) - '0';
634      // Character.digit (params.charAt(ii), Character.MAX_RADIX);
635      assert td.tag_stack.peek() != method_marker;
636      tag_frame[offset] = td.tag_stack.pop();
637      if (debug_tag_frame) {
638        System.out.printf("popped %s into tag_frame[%d]%n", tag_frame[offset], offset);
639      }
640    }
641
642    // Push the method marker on the tag stack (now that we have removed
643    // the parameters
644    td.tag_stack.push(method_marker);
645    // save the tag stack call_depth for debugging
646    tag_frame[frame_size - 1] = ++td.tag_stack_call_depth;
647    if (debug_tag_frame) {
648      System.out.printf("push method marker tag: %s%n", method_marker);
649      System.out.printf("tag stack call_depth: %d%n", td.tag_stack_call_depth);
650      System.out.printf("tag stack size: %d%n", td.tag_stack.size());

659   * Make sure the tag stack for this method is empty before exit.
660   *
661   * @param tag_frame the tag frame
662   */
663  public static void normal_exit(Object[] tag_frame) {
664    if (debug) {
665      System.out.printf("Begin normal exit from %s%n", caller_name());
666    }
667
668    ThreadData td = thread_to_data.get(Thread.currentThread());
669    if (td.tag_stack.peek() != method_marker) {
670      // Something has gone wrong.  It's probably an exception that
671      // was handled by an exception handler other than one added by
672      // DCInstrument. From an <init> method or native code, for example.
673      if (debug_tag_frame) {
674        System.out.printf("tag stack value mismatch! top not method marker%n");
675        Thread.dumpStack();
676      }
677      // discard tags until we get to a method marker
678      while (td.tag_stack.peek() != method_marker) td.tag_stack.pop();
679    }
680
681    int orig_tag_stack_call_depth = ((Integer) tag_frame[tag_frame.length - 1]).intValue();
682    if (td.tag_stack_call_depth != orig_tag_stack_call_depth) {
683      // Something has gone wrong.  It's almost certainly an exception that
684      // was handled by an exception handler other than one added by
685      // DCInstrument. From an <init> method or native code, for example.
686      if (debug_tag_frame) {

688            "tag stack call_depth mismatch! at %d expected %d%n",
689            td.tag_stack_call_depth, orig_tag_stack_call_depth);
690        Thread.dumpStack();
691      }
692    }
693    while (td.tag_stack_call_depth > orig_tag_stack_call_depth) {
694      td.tag_stack.pop(); // discard marker
695      td.tag_stack_call_depth--;
696      // discard tags until we get to a method marker
697      while (td.tag_stack.peek() != method_marker) td.tag_stack.pop();
698    }

700    // now do normal exit process
701    td.tag_stack.pop(); // discard marker
702    td.tag_stack_call_depth--;
703    if (debug_tag_frame) {
704      System.out.printf("tag stack call_depth: %d%n", td.tag_stack_call_depth);
705      System.out.printf("tag stack size: %d%n", td.tag_stack.size());

1981  private static void add_dv_stats(RootInfo root) {
1982
1983    if (root == null) {
1984      return;
1985    }
1986    List<DaikonVariableInfo> dv_list = root.tree_as_list();
1987    for (DaikonVariableInfo dv : dv_list) {

1995      //                    dv.getTypeName());

1998      } else if (dv.isHashcodeArray()) {

2013        else if (Modifier.isStatic(modifiers)) static_cnt++;
2014        else if (dv.getName().startsWith("this")) this_instance_cnt++;
2015        else {

2017        }
2018      } else if (dv instanceof ParameterInfo) {
2019        parameter_cnt++;
2020      } else {
2021        other_cnt++;
2022      }
2023    }

2126   * Print the variables in sets to ps in DECL file format. Each variable in the same set is given
2127   * the same comparability. Constructed classname variables are made comparable to other classname
2128   * variables only.
2129   *
2130   * @param ps where to print the variables
2131   * @param sets the comparability sets
2132   * @param dv_tree the tree of variables
2133   * @param pptName used only for debugging output
2134   */
2135  private static void print_decl_vars(
2136      PrintWriter ps, List<DVSet> sets, RootInfo dv_tree, String pptName) {
2137
2138    time_decl.indent();
2139    time_decl.log("print_decl_vars start");
2140
2141    debug_decl_print.log("print_decl_vars(%s)%n", pptName);
2142
2143    // Map from array name to comparability for its indices (if any)
2144    Map<String, Integer> arr_index_map = new LinkedHashMap<>();
2145
2146    // Map from daikon variable to its comparability
2147    Map<DaikonVariableInfo, Integer> dv_comp_map =
2148        new IdentityHashMap<DaikonVariableInfo, Integer>(256);
2149
2150    // Initial comparability values
2151    int class_comp = 1;
2152    int base_comp = 2;
2153
2154    // Loop through each set of comparable variables
2155    for (DVSet set : sets) {
2156
2157      if ((set.size() == 1) && (set.get(0) instanceof StaticObjInfo)) {
2158        continue;
2159      }
2160
2161      // Determine if the set has both hashcode variables and integer
2162      // variables.  If it does, it is indicating index comparability
2163      boolean hashcode_vars = false;
2164      boolean non_hashcode_vars = false;
2165      // System.out.printf("Checking dv set %s%n", set);
2166      for (DaikonVariableInfo dv : set) {
2167        if (dv.isHashcode() || dv.isHashcodeArray()) hashcode_vars = true;
2168        else {
2169          non_hashcode_vars = true;
2170        }
2171        // System.out.printf("dv = %s, hashcode_var = %b%n",
2172        //                   dv, dv.isHashcode() || dv.isHashcodeArray());
2173      }

2176          set.size(), hashcode_vars, non_hashcode_vars);
2177
2178      // Loop through each variable and assign its comparability
2179      // Since hashcodes and their indices are in the same set, assign
2180      // hashcodes one higher comparability number.  Note that there is
2181      // not necessarily an array child for a hashcode that is comparable
2182      // to an integer.  This can happen when an array and a non-array object
2183      // become comparable to one another.  An integer that is comparable
2184      // to the array will also be comparable to the non-array object, but
2185      // that comparability isn't interesting (and it can't be expressed)

2189          dv_comp_map.put(dv, class_comp);
2190          assert set.size() == 1 : "odd set " + set;

2192        } else if (dv.isHashcode() && non_hashcode_vars) {
2193          assert !dv_comp_map.containsKey(dv) : dv + " " + base_comp;
2194          dv_comp_map.put(dv, base_comp + 1);
2195          DaikonVariableInfo array_child = dv.array_child();

2197            // System.out.printf("array_index_map put: %s, %d%n", array_child.getName(),
2198            // base_comp);

2200          }
2201        } else {
2202          assert !dv_comp_map.containsKey(dv) : dv + " " + base_comp;
2203          dv_comp_map.put(dv, base_comp);
2204        }
2205      }
2206
2207      // Increment the comparability number to the next valid number
2208      base_comp++;
2209      if (hashcode_vars && non_hashcode_vars) {
2210        base_comp++;
2211      }
2212    }

2215

2217    // Use the dv_tree rather than sets so that we print out in the
2218    // same order each time.  Note that arrays get two comparablities, one
2219    // for the elements of the array and one for indices.  Normally, these
2220    // comparabilities will be different, but if an index is placed in the
2221    // array the comparabilities can be the same.
2222    List<DaikonVariableInfo> dv_list = dv_tree.tree_as_list();
2223    time_decl.log("built tree as list with %d elements", dv_list.size());
2224    for (DaikonVariableInfo dv : dv_list) {
2225      if ((dv instanceof RootInfo) || (dv instanceof StaticObjInfo) || !dv.declShouldPrint()) {
2226        continue;
2227      }
2228      // System.out.printf("Output dv: %s ", dv);
2229      ps.println(dv.getName());
2230      ps.println(dv.getTypeName());
2231      ps.println(dv.getRepTypeName());
2232      int comp = dv_comp_map.get(dv);
2233      if (dv.isArray()) {
2234        String name = dv.getName();
2235        // If we an array of CLASSNAME or TO_STRING get the index
2236        // comparability from the base array.
2237        if (name.endsWith(DaikonVariableInfo.class_suffix)) {
2238          name = name.substring(0, name.length() - DaikonVariableInfo.class_suffix.length());
2239        } else if (name.endsWith(".toString")) {
2240          name = name.substring(0, name.length() - ".toString".length());
2241        }
2242        Integer index_comp = arr_index_map.get(name);
2243        // System.out.printf("compare: %d [ %s ] ", comp, index_comp);
2244        if (index_comp != null) {
2245          // System.out.println(comp + "[" + index_comp + "]");
2246          ps.println(comp + "[" + index_comp + "]");
2247        } else {
2248          // There is no index comparability, so just set it to a unique value.
2249          // System.out.println(comp + "[" + base_comp + "]");
2250          ps.println(comp + "[" + base_comp++ + "]");
2251        }
2252      } else {
2253        // System.out.println(comp);
2254        ps.println(comp);
2255      }
2256    }
2257

2571  /**
2572   * Merges comparability so that the same variable has the same comparability at all points in the
2573   * program point hierarchy. The comparability at the class/object points is calculated by merging
2574   * the comparability at each exit point (i.e., if two variables are in the same set at any exit
2575   * point, they are in the same set at the class point). That comparability is then applied back to
2576   * the exit points so that if two class variables are comparable at any exit point they are
2577   * comparable at each exit point. Finally exit point comparability is merged to the enter point so
2578   * that their comparabilities are the same.
2579   *
2580   * <p>This is not the only valid definition of comparability but it is the one that Daikon expects
2581   * because of how equality sets are handled.
2582   */
2583  static void merge_class_comparability(ClassInfo ci) {
2584
2585    // Get the variables at the object and class point
2586    assert ci.traversalObject != null : ci;
2587    assert ci.traversalClass != null : ci;
2588    // ci.init_traversal (depth);
2589
2590    // If any methods have not been executed, create their information
2591    // now (which will note all of their variables as not comparable)
2592    for (MethodInfo mi : ci.method_infos) {
2593      if (mi.is_class_init()) {
2594        continue;
2595      }
2596      if (mi.traversalEnter == null) {
2597        // mi.initViaReflection();
2598        mi.init_traversal(depth);
2599        // System.out.printf("Warning: Method %s never executed%n", mi);
2600      }
2601    }
2602
2603    // Merge the comparability from each exit point into the object point
2604    for (MethodInfo mi : ci.method_infos) {
2605      if (mi.is_class_init()) {
2606        continue;
2607      }
2608      merge_dv_comparability(mi.traversalExit, mi.traversalEnter, "Merging exit to enter: " + mi);
2609      merge_dv_comparability(mi.traversalExit, ci.traversalObject, "Merging exit to object: " + mi);
2610      merge_dv_comparability(mi.traversalEnter, ci.traversalObject, "Merging enter to object" + mi);
2611    }
2612
2613    // Merge the comparability from the object point back to each exit point
2614    for (MethodInfo mi : ci.method_infos) {
2615      if (mi.is_class_init()) {
2616        continue;
2617      }
2618      merge_dv_comparability(ci.traversalObject, mi.traversalExit, "Merging object to exit: " + mi);
2619    }
2620
2621    // Merge the comparability for each exit point back to the enter
2622    for (MethodInfo mi : ci.method_infos) {
2623      if (mi.is_class_init()) {
2624        continue;
2625      }
2626      merge_dv_comparability(mi.traversalExit, mi.traversalEnter, "Merging exit to enter: " + mi);
2627    }
2628
2629    // Merge the object comparability to the class
2630    merge_dv_comparability(ci.traversalObject, ci.traversalClass, "Merging object to class: " + ci);
2631  }

001package daikon;

003import java.util.HashSet;
004import java.util.Set;
005import java.util.logging.ConsoleHandler;
006import java.util.logging.Formatter;
007import java.util.logging.Handler;
008import java.util.logging.Level;
009import java.util.logging.LogRecord;
010import java.util.logging.Logger;
011import java.util.logging.SimpleFormatter;

013/**
014 * Standard methods for setting up logging. Allows creation of Console writers using one method.
015 * Logger methods should only be called in a shell class at setup, after which Logger calls should
016 * be used for logging.
017 */
018public final class LogHelper {
019  private LogHelper() {
020    throw new Error("do not instantiate");
021  }
022
023  // Class variables so user doesn't have to use "Level." prefix.
024  public static final Level FINE = Level.FINE;
025  public static final Level INFO = Level.INFO;
026  public static final Level WARNING = Level.WARNING;
027  public static final Level SEVERE = Level.SEVERE;
028
029  /**
030   * Sets up global logs with a given priority and logging output pattern. Creates one
031   * ConsoleHandler at root to receive default messages, setting priority to INFO. Removes previous
032   * appenders at root.
033   */
034  public static void setupLogs(Level l, Formatter formatter) {
035    // Send debug and other info messages to System.err
036    Handler app = new ConsoleHandler();
037    app.setLevel(Level.ALL);
038    app.setFormatter(formatter);
039
040    // Logger.global.removeAllAppenders();
041    {
042      // Java 5 version
043      @SuppressWarnings("deprecation")
044      Logger global = Logger.global;
045      // Java 6 version (doesn't work in Java 5)
046      // Logger global = Logger.getLogger(Logger.GLOBAL_LOGGER_NAME);
047      Handler[] handlers = global.getHandlers();
048      for (Handler handler : handlers) {
049        global.removeHandler(handler);
050      }
051    }
052
053    Logger root = Logger.getLogger("");
054    Handler[] handlers = root.getHandlers();
055    for (Handler handler : handlers) {
056      root.removeHandler(handler);
057    }
058    root.addHandler(app);
059    root.setLevel(l);
060    allLoggers.add(root);
061
062    // Logger.global.addHandler(app);
063    // Logger.global.setLevel(l);
064    // Logger.global.fine ("Installed logger at level " + l);
065  }
066
067  public static class DaikonLogFormatter extends SimpleFormatter {
068    @Override
069    public synchronized String format(LogRecord record) {
070      // // By default, take up 20 spaces min, and 20 spaces max for logger.
071      // // %c = Logger. %m = message, %n = newline
072      // // Example output: "@ daikon.Daikon: This is a message"
073      // setupLogs (l, "@ %20.20c: %m%n");
074
075      String loggerName = record.getLoggerName() + ":";
076
077      // If we aren't generating tracebacks, find the src class/method/line
078      // where the log was called from
079      String src = "";
080      if (!Debug.dkconfig_showTraceback) {
081        Throwable stack = new Throwable("debug traceback");
082        stack.fillInStackTrace();
083        StackTraceElement[] ste_arr = stack.getStackTrace();
084        for (int ii = ste_arr.length - 1; ii >= 0; ii--) {
085          StackTraceElement ste = ste_arr[ii];
086          if (ste.getClassName().startsWith("java")
087              || ste.getClassName().contains("daikon.LogHelper")
088              || ste.getMethodName().equals("log")
089              || ste.getClassName().contains("daikon.Debug")) {
090            continue;
091          }
092          src = ste.getFileName() + " " + ste.getLineNumber() + ": ";
093        }
094      }
095      // src = record.getSourceClassName().replaceAll ("\\w*\\.", "")
096      //        + "." + record.getSourceMethodName() + ": ";
097
098      return "@ " + loggerName + " " + src + record.getMessage() + Global.lineSep;
099    }
100  }
101
102  /** Default method for setting up global logs. */
103  public static void setupLogs() {
104    setupLogs(INFO);
105  }
106
107  /**
108   * Sets up global logs with a given priority. Creates one ConsoleHandler. Removes previous
109   * appenders at root.
110   */
111  public static void setupLogs(Level l) {
112    setupLogs(l, new DaikonLogFormatter());
113  }
114
115  private static final Set<Logger> allLoggers = new HashSet<>();
116
117  /**
118   * Changes the logging priority of a sub category. Also caches the logger to avoid
119   * garbage-collection and recreation with the old level.
120   */
121  public static void setLevel(Logger lg, Level l) {
122    lg.setLevel(l);
123    allLoggers.add(lg); // to prevent garbage-collection and re-initialization
124  }
125
126  /**
127   * Changes the logging priority of a sub category. Also caches the logger to avoid
128   * garbage-collection and recreation with the old level.
129   */
130  public static void setLevel(String s, Level l) {
131    setLevel(Logger.getLogger(s), l);
132  }
133}

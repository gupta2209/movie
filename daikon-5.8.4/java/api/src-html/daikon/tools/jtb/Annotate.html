019/**
020 * Merge Daikon-generated invariants into Java source code as ESC/JML/DBC annotations. All original
021 * {@code .java} files are left unmodified; copies are created.
022 *
023 * <p>The first argument is a Daikon {@code .inv} file -- a serialized file of Invariant objects.
024 * All subsequent arguments are {@code Foo.java} files that are rewritten into {@code
025 * Foo.java-jmlannotated} versions; alternately, use the {@code -r} flag to process every {@code
026 * .java} file under the current directory.
027 */
028public class Annotate {
029
030  // ESC format: Invariants are inserted as follows:
031  //  * invariants at method entry become "requires"
032  //  * invariants at method exit become "ensures".
033  //    ensures/exsures annotations should not mention orig(primitive arguments).
034  //  * invariants at method exceptional exit become "exsures"
035  //  * object invariants become object invariants, inserted at the beginning
036  //    of the java class
037  //  * "modifies" clauses are inserted for changed variables.
038  //    Use "a[*]", not "a[]" or "a[i]" or "a[i..]" or "a[...].field", for arrays;
039  //    consider adding a \forall to specify which indices are *not* changed.
040  //    Modifies clauses should not contain "size()", ".getClass()", or "~".
041  //    Modifies clauses should not contain final fields.
042  //  * use "also_requires", "also_ensures", "also_modifies" if this method
043  //    overrides another method/interface.
044
045  // spec_public:
046  // for each non-public field, add "/*@ spec_public */" to its declaration
047
048  // owner annotations:
049  // for each private field that is set via a call to new in the constructor
050  // (in the short term, just do this for all array fields):
051  //  * add object invariant "invariant field.owner == this"
052  //  * whenever the field is set, "set field.owner = this"
053
054  // JML format: The formatting is similar to ESC (heavyweight) format,
055  // but "modifies" clauses are omitted.
056
057  // DBC format: same as ESC format, but "requires" becomes "@pre",
058  // "ensures" becomes "@post", and we omit everything but invariants,
059  // pre and postconditions.  See Appendix A below for Jtest's (not
060  // very formal or detailed) specification of their DBC
061  // language.  Note that we avoided using most DBC constructs -- we
062  // found the DBC aspect of Jtest buggy, so we try to depend on their
063  // constructs as little as possible.
064
065  // Explanatory comments:
066  // Handle "The invariant on the following line means:".
067
068  // Optional behavior:
069  //  * With -i flag, invariants not supported by ESC are inserted with "!"
070  //    instead of "@"; by default these "inexpressible" invariants are
071  //    simply omitted.
072  //  * With -s flag, use // comments; by default, use /* comments.
073
074  public static final Logger debug = Logger.getLogger("daikon.tools.jtb.Annotate");
075
076  public static final String wrapXML_SWITCH = "wrap_xml";
077  public static final String max_invariants_pp_SWITCH = "max_invariants_pp";
078  public static final String no_reflection_SWITCH = "no_reflection";
079
080  private static String usage =
081      UtilPlume.joinLines(
082          "Usage:  java daikon.tools.Annotate FILE.inv FILE.java ...",
083          "  -h   Display this usage message",
084          "  -i   Insert invariants not supported by ESC with \"!\" instead of \"@\";",
085          "       by default these \"inexpressible\" invariants are simply omitted",
086          "  -r   Use all .java files under the current directory as arguments",
087          "  -s   Use // comments rather than /* comments",
088          "  --format name  Insert specifications in the given format: DBC, ESC, JML, Java",
089          "  --wrap_xml     Wrap each annotation and auxiliary information in XML tags",
090          "  --max_invariants_pp N",
091          "                 Annotate the sources with at most N invariants per program point",
092          "                 (the annotated invariants will be an arbitrary subset of the",
093          "                 total number of invariants for the program point).",
094          "  --no_reflection",
095          "                  (This is an experimental option.) Annotate uses reflection",
096          "                  to determine whether a method overrides/implements another",
097          "                  method. If this flag is given, Annotate will not use reflection",
098          "                  to access information about an instrumented class. This means",
099          "                  that in the JML and ESC formats, no \"also\" annotations",
100          "                  will be inserted.",
101          "  --dbg CATEGORY",
102          "  --debug",
103          "                  Enable one or all loggers, analogously to the Daikon option");

105  public static void main(String[] args) throws Exception {
106    try {
107      mainHelper(args);
108    } catch (Daikon.DaikonTerminationException e) {
109      Daikon.handleDaikonTerminationException(e);
110    }
111  }

113  /**
114   * This does the work of {@link #main(String[])}, but it never calls System.exit, so it is
115   * appropriate to be called progrmmatically.
116   */
117  public static void mainHelper(final String[] args) throws Exception {
118    boolean slashslash = false;
119    boolean insert_inexpressible = false;
120    boolean setLightweight = true;
121    boolean useReflection = true;
122    int maxInvariantsPP = -1;
123
124    Daikon.output_format = OutputFormat.ESCJAVA;
125    LongOpt[] longopts =
126        new LongOpt[] {
127          new LongOpt(Daikon.help_SWITCH, LongOpt.NO_ARGUMENT, null, 0),
128          new LongOpt(Daikon.debugAll_SWITCH, LongOpt.NO_ARGUMENT, null, 0),
129          new LongOpt(Daikon.debug_SWITCH, LongOpt.REQUIRED_ARGUMENT, null, 0),
130          new LongOpt(Daikon.format_SWITCH, LongOpt.REQUIRED_ARGUMENT, null, 0),
131          new LongOpt(wrapXML_SWITCH, LongOpt.NO_ARGUMENT, null, 0),
132          new LongOpt(max_invariants_pp_SWITCH, LongOpt.REQUIRED_ARGUMENT, null, 0),
133          new LongOpt(no_reflection_SWITCH, LongOpt.NO_ARGUMENT, null, 0)
134        };
135    Getopt g = new Getopt("daikon.tools.jtb.Annotate", args, "hs", longopts);
136    int c;
137    while ((c = g.getopt()) != -1) {
138      switch (c) {
139        case 0:
140          // got a long option
141          String option_name = longopts[g.getLongind()].getName();
142
143          if (Daikon.help_SWITCH.equals(option_name)) {
144            System.out.println(usage);
145            throw new Daikon.NormalTermination();
146          } else if (no_reflection_SWITCH.equals(option_name)) {
147            useReflection = false;
148          } else if (max_invariants_pp_SWITCH.equals(option_name)) {
149            try {
150              maxInvariantsPP = Integer.parseInt(Daikon.getOptarg(g));
151            } catch (NumberFormatException e) {
152              System.err.println(
153                  "Annotate: found the --max_invariants_pp option "
154                      + "followed by an invalid numeric argument. Annotate "
155                      + "will run without the option.");
156              maxInvariantsPP = -1;
157            }
158          } else if (wrapXML_SWITCH.equals(option_name)) {
159            PrintInvariants.wrap_xml = true;
160          } else if (Daikon.debugAll_SWITCH.equals(option_name)) {
161            Global.debugAll = true;
162          } else if (Daikon.debug_SWITCH.equals(option_name)) {
163            LogHelper.setLevel(Daikon.getOptarg(g), LogHelper.FINE);
164          } else if (Daikon.format_SWITCH.equals(option_name)) {
165            String format_name = Daikon.getOptarg(g);
166            Daikon.output_format = OutputFormat.get(format_name);
167            if (Daikon.output_format == null) {
168              throw new Daikon.UserError("Bad argument:  --format " + format_name);
169            }
170            if (Daikon.output_format == OutputFormat.JML) {
171              setLightweight = false;
172            } else {
173              setLightweight = true;
174            }
175          } else {
176            throw new Daikon.UserError("Unknown long option received: " + option_name);
177          }
178          break;
179        case 'h':
180          System.out.println(usage);
181          throw new Daikon.NormalTermination();
182        case 'i':
183          insert_inexpressible = true;
184          break;
185          // case 'r':
186          //   // Should do this witout calling out to the system.  (There must be
187          //   // an easy way to do this in Java.)
188          //   Process p = System.exec("find . -type f -name '*.java' -print");
189          //   p.waitFor();
190          //   StringBuilderInputStream sbis
191          //   break;
192        case 's':
193          slashslash = true;
194          break;
195        case '?':
196          break; // getopt() already printed an error
197        default:
198          System.out.println("getopt() returned " + c);
199          break;
200      }
201    }
202
203    // Set up debug traces; note this comes after reading command line options.
204    LogHelper.setupLogs(Global.debugAll ? LogHelper.FINE : LogHelper.INFO);
205
206    // The index of the first non-option argument -- the name of the .inv file
207    int argindex = g.getOptind();
208
209    if (argindex >= args.length) {
210      throw new Daikon.UserError(
211          "Error: No .inv file or .java file arguments supplied." + Global.lineSep + usage);
212    }
213    String invfile = args[argindex];
214    argindex++;
215    if (argindex >= args.length) {
216      throw new Daikon.UserError(
217          "Error: No .java file arguments supplied." + Global.lineSep + usage);
218    }
219    PptMap ppts = FileIO.read_serialized_pptmap(new File(invfile), /*use saved config=*/ true);
220
221    Daikon.suppress_redundant_invariants_with_simplify = true;
222
223    for (; argindex < args.length; argindex++) {
224      String javafilename = args[argindex];
225      if (!(javafilename.endsWith(".java") || javafilename.endsWith(".java-random-tabs"))) {
226        throw new Daikon.UserError("File does not end in .java: " + javafilename);
227      }
228      File outputFile;
229      if (Daikon.output_format == OutputFormat.ESCJAVA) {
230        outputFile = new File(javafilename + "-escannotated");
231      } else if (Daikon.output_format == OutputFormat.JML) {
232        outputFile = new File(javafilename + "-jmlannotated");
233      } else if (Daikon.output_format == OutputFormat.JAVA) {
234        outputFile = new File(javafilename + "-javaannotated");
235      } else if (Daikon.output_format == OutputFormat.DBCJAVA) {
236        outputFile = new File(javafilename + "-dbcannotated");
237      } else {
238        throw new Error("unsupported output file format " + Daikon.output_format);
239      }
240      // outputFile.getParentFile().mkdirs();
241      Writer output = Files.newBufferedWriter(outputFile.toPath(), UTF_8);
242
243      debug.fine("Parsing file " + javafilename);
244
245      // Annotate the file
246      Reader input;
247      try {
248        input = Files.newBufferedReader(Paths.get(javafilename), UTF_8);
249      } catch (FileNotFoundException e) {
250        throw new Error(e);
251      }
252
253      JavaParser parser = new JavaParser(input);
254      Node root;
255      try {
256        root = parser.CompilationUnit();
257      } catch (ParseException e) {
258        // e.printStackTrace();
259        System.err.println(javafilename + ": " + e);
260        throw new Daikon.UserError("ParseException in applyVisitorInsertComments");
261      }
262
263      debug.fine("Processing file " + javafilename);
264
265      try {
266        Ast.applyVisitorInsertComments(
267            javafilename,
268            root,
269            output,
270            new AnnotateVisitor(
271                javafilename,
272                root,
273                ppts,
274                slashslash,
275                insert_inexpressible,
276                setLightweight,
277                useReflection,
278                maxInvariantsPP));
279      } catch (Error e) {
280        String message = e.getMessage();
281        if (message != null && message.startsWith("Didn't find class ")) {
282          throw new Daikon.UserError(
283              String.join(
284                  System.lineSeparator(),
285                  message + ".",
286                  "Be sure to put .class files on the classpath when calling Annotate.",
287                  "The classpath is: " + System.getProperty("java.class.path")));
288        }
289        throw e;
290      }
291    }
292  }

295// Appendix A: Jtest DBC format (taken from documentation provided by
296// Jtest with their tool, version 1.x)
297
298//                     Design by Contract for Java
299//
300//
301//    This document describes the syntax and semantics for the "Design by
302// Contract" (DbC) specification supported by Jcontract 1.X.
303//
304//    The "Design by Contract" contracts are expressed with Java code
305// embedded in Javadoc comments in the .java source file.
306//
307//
308// 1. Javadoc Tags for Design by Contract.
309// --------------------------------------
310//
311//    The reserved javadoc tags for DbC are:
312//
313// - @invariant: specifies class invariant condition.
314// - @pre: specifies method pre-condition.
315// - @post: specifies method post-condition.
316// - @concurrency: specifies the method concurrency.
317//
318// Jcontract and Jtest also support the following tags:
319//
320// - @throws/@exception: to document exceptions.
321// - @assert: to put assertions in the method bodies.
322// - @verbose: to add verbose statements to the method bodies.
323//
324//    Please see "appendix10.txt" for information about those tags.
325//
326//
327// 2. Contract Syntax.
328// ------------------
329//
330//    The general syntax for a contract is:
331//
332//         DbcContract:
333//               DbcTag DbcCode
334//             | @concurrency { concurrent | guarded | sequential }
335//
336// where
337//
338//         DbcTag:
339//               @invariant
340//             | @pre
341//             | @post
342//
343//         DbcCode:
344//               BooleanExpression
345//             | '(' BooleanExpression ')'
346//             | '(' BooleanExpression ',' MessageExpression ')'
347//             | CodeBlock
348//             | $none
349//
350//         MessageExpression:
351//               Expression
352//
353//    Any Java code can be used in the DbcCode with the following restrictions:
354//
355// - The code should not have side effects, i.e. it should not have assignments
356//   or invocation of methods with side-effects.
357//
358//    Also the following extensions to Java (DbC keywords) are allowed in the
359// contract code:
360//
361// - $result: used in a @post contract, evaluates to the return value of the
362//   method.
363//
364// - $pre: used in a @post contract to refer to the value of an expression at
365//   @pre-time. The syntax to use it is: $pre (ExpressionType, Expression).
366//   Note: the full "$pre (...)" expression should not extend over multiple lines.
367//
368// - $assert: can be used in DbcCode CodeBlocks to specify the contract conditions.
369//   The syntax to use it is:
370//     $assert (BooleanExpression) or
371//     $assert (BooleanExpression , MessageExpression)
372//
373// - $none: to specify there is no contract.
374//
375//
376//    Notes:
377//
378// - The @pre, @post and @concurrent tags apply to the method that follows in
379//   the source file.
380//
381// - The MessageExpression is optional and will be used to identify the contract
382//   in the error messages or contract violation exceptions thrown.
383//   The MessageExpression can be of any type. If it is a reference type it will
384//   be converted to a String using the "toString ()" method. If it is of
385//   primitive type it will first be wrapped into an object.
386//
387// - There can be multiple conditions of the same kind for a given method.
388//   If there are multiple conditions, all conditions are checked. The
389//   conditions are ANDed together into one virtual condition.
390//   For example it is equivalent (and encouraged for clarity) to have multiple
391//   @pre conditions instead of a single big @pre condition.
392//
393//
394// Examples:
395//
396//
397//     /**
398//       * @pre {
399//       *     for (int i = 0; i < array.length; i++)
400//       *         $assert (array [i] != null, "array elements are non-null");
401//       * }
402//       */
403//
404//     public void set (int[] array) {...}
405//
406//
407//     /** @post $result == ($pre (int, arg) + 1) */
408//
409//     public int inc (arg) {...}
410//
411//
412//     /** @invariant size () >= 0 */
413//
414//     class Stack {...}
415//
416//
417//     /**
418//      * @concurrency sequential
419//      * @pre (value > 0, "value positive:" + value)
420//      */
421//
422//     void update (int value) {...}
423//
424//
425// 3. Contract Semantics.
426// ---------------------
427//
428//    The contracts are specified in comments and will not have any effect if
429// compiling or executing in a non DbC enhanced environment.
430//
431//    In a DbC enhanced environment the contracts are executed/checked when
432// methods of a class with DbC contracts are invoked.  Contracts placed inside
433// a method are treated like normal statements and can thus alter the flow of
434// poorly structured conditional statements.  See section 6: Coding Conventions
435// for details.
436//
437//    A contract fails if any of these conditions occur:
438//
439// a) The "BooleanExpression" evaluates to "false"
440//
441// b) An "$assert (BooleanExpression)" is called in a "CodeBlock" with an argument
442//    that evaluates to "false".
443//
444// c) The method is called in a way that violates its @concurrency contract.
445//
446//    If a contract fails the Runtime Handler for the class is notified of
447// the contract violation. Jcontract provides several Runtime Handlers, the
448// default one uses a GUI Monitor that shows the program progress and what
449// contracts are violated. The user can also write its own RuntimeHandlers,
450// for more information see {isntalldir}\docs\runtime.txt.
451//
452//    With the Moninor Runtime Handlers provided by Jcontract program execution
453// continues as if nothing had happened when a contract is violated, i.e. even
454// if a @pre contract is violated, the method will still be executed.
455//    This option makes the DbC enabled and non DbC enabled versions of the
456// program to work in exactly the same way. The only difference is that in the
457// DbC enabled version the contract violations are reported to the current
458// Jcontract Monitor.
459//
460//
461//    Note that contract evaluation is not nested, when a contract calls
462// another methods, the contracts in the other method are not executed.
463//
464//
465// 4. DbC Contracts.
466// ----------------
467//
468//    This section gives details about each specific DbC tag.
469//
470//
471// 4.1 @pre.
472// - - - - -
473//
474//    Pre-conditions check that the client calls the method correctly.
475//
476// - Point of execution: right before calling the method.
477//
478// - Scope: can access anything accessible from the method scope except local
479//   variables. I.e. can access method arguments, and methods/fields of the
480//   class.
481//
482//
483// 4.2 @post.
484// - - - - -
485//
486//    Post-conditions check that the method doesn't work incorrectly.
487//    Sometimes when a post-condition fails it means that the method was not
488// actually supposed to accept the arguments that were passed to it. The fix in
489// this case is to strengthen the @pre-condition.
490//
491// - Point of execution: right after the method returns successfully. Note that
492//   if the method throws an exception the @post contract is not executed.
493//
494// - Scope: same as @pre plus can access "$result" and "$pre (type, expression)".
495//
496//
497// 4.3 @invariant.
498// - - - - - - - -
499//
500//    Class invariants are contracts that the objects of the class should
501// always satisfy.
502//
503// - Point of execution: same as @pre/@post: invariant checked before checking
504//   the pre-condition and after checking the post-condition.
505//   Done for every non-static, non-private method entry and exit and for
506//   every non-private constructor exit.
507//   - Note that if a constructor throws an exception its @post contract is not
508//     executed.
509//   - Not done for "finalize ()".
510//   - When inner class methods are executed, the invariants of the outer classes
511//     are not checked.
512//
513// - Scope: class scope, can access anything a method in the class can access,
514//   except local variables.
515//
516//
517// 4.4 @concurrency.
518// - - - - - - - - -
519//
520//    The @concurrency tag specifies how the method can be called by multiple
521// threads. Its possible values are:
522//
523// a) concurrent: the method can be called simultaneously by different threads.
524//    I.e. the method is multi-thread safe. Note that this is the default
525//    mode for Java methods.
526//
527// b) guarded: the method can be called simultaneously by different threads,
528//    but only one will execute it in turn, while the other threads will
529//    wait for the executing one to finish.
530//    I.e. it specifies that the method is synchronized.
531//    Jcontract will just give a compile-time error if a method is declared
532//    as "guarded" but is not declared as "synchronized".
533//
534// c) sequential: the method can only by executed by one thread at once and it
535//    is not declared synchronized. It is thus the responsibility of the callers
536//    to ensure that no simultaneous calls to that method occur. For methods with
537//    this concurrency contract Jcontract will generate code to check if they are
538//    being executed by more than one thread at once. An error will be reported
539//    at runtime if the contract is violated.
540//
541// - Point of execution: right before calling the method.
542//
543//
544// 5. Inheritance.
545// --------------
546//
547//    Contracts are inherited. If the derived class or overriding method doesn't
548// define a contract, it inherits that of the super class or interface.
549//    Note that a contract of $none implies that the super contract is applied.
550//
551//    If an overriding method does define a contract then it can only:
552//
553// - Weaken the precondition: because it should at least accept the same input as
554//   the parent, but it can also accept more.
555//
556// - Strengthen the postcondition: because it should at least do as much as the
557//   parent one, but it can also do more.
558//
559//    To enforce this:
560//
561// - When checking the @pre condition, the pre-condition contract is assumed to
562//   succeed if any of the @pre conditions of the chain of overridded methods
563//   succeeds. I.e. the preconditions are ORed.
564//
565// - When checking the @post condition, the post-condition contract is assumed
566//   to succeed if all the @post conditions of the chain of overridded methods
567//   succeed. I.e. the postconditions are ANDed.
568//
569//    Note that if there are multiple @pre conditions for a given method, the
570// preconditions are ANDed together into one virtual @pre condition and then
571// ORed with the virtual @pre conditions for the other methods in the chain
572// of overridden methods.
573//
574//    For @invariant conditions the same logic as for @post applies.
575//
576//    @concurrency contracts are also inherited. If the overriding method doesn't
577// have a @concurrency contract it inherits that of the parent. If it has an
578// inheritance contract it can only weaken it, like for @pre conditions. For
579// example if the parent has a "sequential" @concurrency, the overriding method
580// can have a "guarded" or "concurrent" @concurrency.
581//
582// 6. Special Keyword and Functions
583//
584// 6.1 Keywords
585//
586// $implies
587//
588//     You can use the '$implies' keyword to compare two boolean expressions,
589//     ensuring that when the first expression is true, the second one also
590//     is true.
591//
592//     Example: a $implies b
593//     This is equivalent to '!a || b'.
594//
595// 6.2 Collection functions
596//
597// Class and method contracts can contain a few special functions that are
598// internally mapped to equivalent Java code.
599//
600// These are:
601//
602// public boolean Collection.$forall(Type t; <boolean expression>)
603//
604//     <boolean expression> is an expression that will be evaluated for all
605//     elements in the collection, with each element of type "Type", using
606//     an element named "t". The value of $forall is true when for all elements
607//     the expression evaluates to true.
608//
609//     Think of this as a modified 'for' statement.
610//
611//     Example:
612//     /** @pre names.$forall (String s; s.length() != 0) */
613//     void method (List names) { }
614//
615//     Since $forall generates a boolean value, you can use it with assert
616//     in a block precondition:
617//
618//     /** @pre {
619//             $assert (names != null);
620//             $assert (names.$forall (String e; e.length() > 1));
621//         }
622//      */
623//
624// public boolean Collection.$exists(Type t; <boolean expression>)
625//
626//     This is almost identical in structure to $forall, but it succeeds if any
627//     of the elements in the collection cause the expression to evaluate to true.
628//
629//     Example:
630//     /** @pre names.$exists (String s; s != null && "seth".equals (s)) */
631//     void method (List names) { }
632//
633// 7. Coding Conventions.
634// ---------------------
635//
636//    When using Design by Contract in Java, the following coding conventions
637// are recommended:
638//
639// 1) Place all the @invariant conditions in the class javadoc comment, the
640//    javadoc comment appearing immediately before the class definition.
641//
642// 2) It is recommended that the Javadoc comments with the @invariant tag appear
643//    before the class definition.
644//
645// 3) All public and protected methods should have a contract. It is recommended
646//    that all package-private and private methods also have a contract.
647//
648// 4) If a method has a DbC tag it should have a complete contract. This means
649//    that is should have both a precondition and a postcondition.
650//    One should use "DbcTag $none" to specify that a method doesn't have any
651//    condition for that tag.
652//
653// 5) No public class field should participate in an @invariant clause. Since any
654//    client can modify such a field arbitrarily, there is no way for the class to
655//    ensure any invariant on it.
656//
657// 6) The code contracts should only access members visible from the interface.
658//    For example the code in the @pre condition of a method should only access
659//    members that are accessible from any client that could use the method. I.e.
660//    the contract of a public method should only use public members from the
661//    method's class.
662//
663// 7) Contracts that appear in the body of a method should be thought of as
664//    normal statements.  Thus the following example would have undesireable
665//    behavior:
666//
667//        if (a) /** @assert (a) */
668//            System.out.println("poor use of contract code");
669//
670//    Because the assert is a normal statement when your code is compiled by
671//    dbc_javac, this example is equivalent to the following pseudocode:
672//
673//        if (a)
674//            assert (a);
675//        System.out.println("poor use of contract code");
676//
677//    which makes it clear that the assertion is the body of the "if".
678//
679//    The current version of Jcontract doesn't enforce these coding conventions.

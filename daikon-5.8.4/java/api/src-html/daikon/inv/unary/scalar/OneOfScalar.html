538   * Checks a single sample to the invariant. Returns
539   * the appropriate InvariantStatus from the result
540   * of adding the sample to this.
541   */
542  public InvariantStatus check_mod_elem(long v, int count) {
543
544    // Look for v in our list of previously seen values.  If it's
545    // found, we're all set.
546    for (int i = 0; i < num_elts; i++) {
547      // if (logDetail())
548      //  log ("add_modified (" + v + ")");
549      if (((elts[i]) == ( v))) {
550        return InvariantStatus.NO_CHANGE;
551      }
552    }
553
554    if (num_elts == dkconfig_size) {
555      return InvariantStatus.FALSIFIED;
556    }
557
558    if ((is_boolean() && (num_elts == 1)) || (is_hashcode() && (num_elts == 2))) {
559      return InvariantStatus.FALSIFIED;
560    }
561
562    if (is_hashcode() && (num_elts == 1)) {
563      // Permit two object values only if one of them is null
564      if ((elts[0] != 0) && (v != 0)) {
565        return InvariantStatus.FALSIFIED;
566      }
567    }
568
569    return InvariantStatus.WEAKENED;
570  }

651    // derivation of a base (underived) array.  Its quite possible this should
652    // just apply to all arrays.
653    if (v.isDerived() && (v.derived instanceof SequenceLength)
654      && (num_elts == 1) && (elts[0] == 0)) {
655      @NonNull VarInfo b = v.derived.getBase(0);
656      if (!b.isDerived()) {
657        if (logOn()) {
658          dlog.log("isObviousDynamically '" + v.name()
659                    + " == 0' ==> '" + b.name() + " == []'");
660        }
661        return new DiscardInfo(this, DiscardCode.obvious, "size(array) == 0 is implied by array == []");
662      }
663    }
664
665    if (v.isDerived() && (v.derived instanceof SequenceLength)) {
666      @NonNull SequenceLength sl = (SequenceLength) v.derived;
667      if (sl.shift != 0) {
668        String discardString = v.name() + " is derived with shift!=0 (shift==" + sl.shift + ")";
669        return new DiscardInfo(this, DiscardCode.obvious, discardString);
670      }
671    }
672
673    // For every EltOneOf at this program point, see if this variable is
674    // an obvious member of that sequence.
675    PptTopLevel parent = ppt.parent;
676    for (Iterator<Invariant> itor = parent.invariants_iterator(); itor.hasNext(); ) {
677      Invariant inv = itor.next();
678      if ((inv instanceof EltOneOf) && inv.enoughSamples()) {
679        VarInfo v1 = var();
680        VarInfo v2 = inv.ppt.var_infos[0];
681        // System.out.println("isObviousImplied: calling  Member.isObviousMember(" + v1.name + ", " + v2.name + ")");
682        // Don't use isEqualToObviousMember:  that is too subtle
683        // and eliminates desirable invariants such as "return == null".
684        if (Member.isObviousMember(v1, v2)) {
685          EltOneOf other = (EltOneOf) inv;
686          if (num_elts == other.num_elts()) {
687            sort_rep();
688            if (other.compare_rep(num_elts, elts)) {
689              // System.out.println("isObviousImplied true");
690              String discardString = v1.name()+" is a member of "+v2.name()+" for which this Invariant also holds";
691              return new DiscardInfo(this, DiscardCode.obvious, discardString);
692            }
693          }
694        }
695      }
696    }
697
698    return null;
699  }

710  @Pure
711  @Override
712  public boolean isSameFormula(Invariant o) {
713    OneOfScalar other = (OneOfScalar) o;
714    if (num_elts != other.num_elts) {
715      return false;
716    }
717    if (num_elts == 0 && other.num_elts == 0) {
718      return true;
719    }
720
721    sort_rep();
722    other.sort_rep();
723
724    // All nonzero hashcode values should be considered equal to each other
725    //
726    // Examples:
727    // inv1  inv2  result
728    // ----  ----  ------
729    // 19    0     false
730    // 19    22    true
731    // 0     0     true
732
733    if (is_hashcode() && other.is_hashcode()) {
734      if (num_elts == 1 && other.num_elts == 1) {
735        return ((elts[0] == 0 && other.elts[0] == 0)
736                || (elts[0] != 0 && other.elts[0] != 0));
737      } else if (num_elts == 2 && other.num_elts == 2) {
738        // add_modified allows two elements iff one is null
739        assert elts[0] == 0;
740        assert other.elts[0] == 0;
741        assert elts[1] != 0;
742        assert other.elts[1] != 0;
743
744        // Since we know the first elements of each invariant are
745        // zero, and the second elements are nonzero, we can immediately
746        // return true
747        return true;
748      } else {
749        return false;
750      }
751    }
752
753    for (int i = 0; i < num_elts; i++) {
754      if (!((elts[i]) == (other.elts[i]))) {
755        return false;
756      }
757    }
758
759    return true;
760  }

762  @Pure
763  @Override
764  public boolean isExclusiveFormula(Invariant o) {
765    if (o instanceof OneOfScalar) {
766      OneOfScalar other = (OneOfScalar) o;
767
768      if (num_elts == 0 || other.num_elts == 0) {
769        return false;
770      }
771      for (int i = 0; i < num_elts; i++) {
772        for (int j = 0; j < other.num_elts; j++) {
773          if (((elts[i]) == (other.elts[j]))) // elements are interned
774            return false;
775        }
776      }
777
778      // Don't consider two instances of "non-null" as exclusive.
779      if (is_hashcode() && num_elts == 1
780          && other.is_hashcode() && other.num_elts == 1
781          && elts[0] != 0 && other.elts[0] != 0) {
782        return false;
783      }
784
785      // Be even more aggressive about rejecting these for use in
786      // implications in this case, since, we'd be printing them as
787      // "true"
788      /*
789      if (dkconfig_omit_hashcode_values_Simplify
790          && (is_hashcode() || other.is_hashcode())) {
791        return false;
792      }
793      */
794
795      return true;
796    }
797
798    // Many more checks can be added here:  against nonzero, modulus, etc.
799    if ((o instanceof NonZero) && (num_elts == 1) && (elts[0] == 0)) {
800      return true;
801    }
802    long elts_min = Long.MAX_VALUE;
803    long elts_max = Long.MIN_VALUE;
804    for (int i = 0; i < num_elts; i++) {
805      elts_min = Math.min(elts_min, elts[i]);
806      elts_max = Math.max(elts_max, elts[i]);
807    }
808    if ((o instanceof LowerBound) && (elts_max < ((LowerBound)o).min())) {
809      return true;
810    }
811    if ((o instanceof UpperBound) && (elts_min > ((UpperBound)o).max())) {
812      return true;
813    }
814
815    return false;
816  }

858  /**
859   * Merge the invariants in invs to form a new invariant. Each must be a OneOfScalar invariant.
860   * This code finds all of the oneof values from each of the invariants and returns the merged
861   * invariant (if any).
862   *
863   * @param invs list of invariants to merge. The invariants must all be of the same type and should
864   *     come from the children of parent_ppt. They should also all be permuted to match the
865   *     variable order in parent_ppt.
866   * @param parent_ppt slice that will contain the new invariant
867   */
868  @Override
869  public @Nullable Invariant merge(List<Invariant> invs, PptSlice parent_ppt) {
870
871    // Create the initial parent invariant from the first child
872    OneOfScalar  first = (OneOfScalar) invs.get(0);
873    OneOfScalar result = first.clone();
874    result.ppt = parent_ppt;
875
876    // Loop through the rest of the child invariants
877    for (int i = 1; i < invs.size(); i++ ) {
878
879      // Get this invariant
880      OneOfScalar inv = (OneOfScalar) invs.get(i);
881
882      // Loop through each distinct value found in this child and add
883      // it to the parent.  If the invariant is falsified, there is no parent
884      // invariant
885      for (int j = 0; j < inv.num_elts; j++) {
886        long val = inv.elts[j];
887
888        InvariantStatus status = result.add_mod_elem(val, 1);
889        if (status == InvariantStatus.FALSIFIED) {
890
891          result.log("%s", "child value '" + val + "' destroyed oneof");
892
893          return null;
894        }
895      }
896    }
897
898    result.log("Merged '%s' from %s child invariants", result.format(), invs.size());
899    return result;
900  }

914  /**
915   * Returns true if every element in this invariant is contained in the specified state. For
916   * example if x = 1 and the state contains 1 and 2, true will be returned.
917   */
918  @Override
919  public boolean state_match(Object state) {
920
921    if (num_elts == 0) {
922      return false;
923    }
924
925    if (!(state instanceof long[])) {
926      System.out.println("state is of class '" + state.getClass().getName()
927                          + "'");
928    }
929    long[] e = (long[]) state;
930    for (int i = 0; i < num_elts; i++) {
931      boolean match = false;
932      for (int j = 0; j < e.length; j++) {
933        if (elts[i] == e[j]) {
934          match = true;
935          break;
936        }
937      }
938      if (!match) {
939        return false;
940      }
941    }
942    return true;
943  }

945  /**
946   * Map that holds dummy hashcode values for hashcodes. Each hashcode seen is assigned a small
947   * integer in the order they are seen. These values will be consistent as long as new hashcodes do
948   * not appear in the output. Not a perfect fix for regressions consistency, but workable.
949   */
950  private static Map<Long,Long> dummy_hashcode_vals = new LinkedHashMap<>();
951  private static long next_dummy_hashcode = 1001;
952
953  private long get_hashcode_val(@GuardSatisfied OneOfScalar this, long hashcode) {
954    if (!dkconfig_omit_hashcode_values_Simplify) {
955      return hashcode;
956    }
957
958    Long val = dummy_hashcode_vals.get(hashcode);
959    if (val != null) {
960      return val;
961    }
962    dummy_hashcode_vals.put(hashcode, next_dummy_hashcode);
963    return (next_dummy_hashcode++);
964  }

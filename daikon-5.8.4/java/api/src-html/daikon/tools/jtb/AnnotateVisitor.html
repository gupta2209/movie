032// For each class:  (UnmodifiedClassDeclaration)
033//  * insert object invariants
034//  * insert owner assertions for fields
035// For each field:
036//  * add spec_public
037// For each method:
038//  * add preconditions
039//  * add postconditions (including exsures)
040//  * add "modifies"
041//  * all of these may be prefixed by "also_"
042// For each field assignment:
043//  * add owner annotations
044
045// The overall strategy is to read the Java file into a list of strings
046// (javaFileLines) and then to manipulate that list directly.  It doesn't
047// insert the comments into the JTB tree.
048
049// TODO:  Unlike InsertCommentFormatter, this does not keep track of column
050// shifts.  (It does handle line shifts.)  This needs to be fixed.
051
052public class AnnotateVisitor extends DepthFirstVisitor {

091  public AnnotateVisitor(
092      String javafilename,
093      Node root,
094      PptMap ppts,
095      boolean slashslash,
096      boolean insert_inexpressible,
097      boolean lightweight,
098      boolean useReflection,
099      int maxInvariantsPP) {
100
101    this.pptMatcher = new PptNameMatcher(root);
102
103    // Read the Java file into a list of Strings.
104    this.javaFileLines = new ArrayList<String>();
105    try {
106      for (String line : new EntryReader(javafilename)) {
107        this.javaFileLines.add(line);
108      }
109    } catch (IOException e) {
110      throw new Error(e);
111    }
112
113    this.ppts = ppts;
114    this.slashslash = slashslash;
115    this.insert_inexpressible = insert_inexpressible;
116    this.lightweight = lightweight;
117    this.useReflection = useReflection;
118    this.maxInvariantsPP = maxInvariantsPP;
119  }

121  // Like Ast.addComment, but also keeps a list of what comments were added.
122  void addComment(Node n, String comment, boolean first) {
123    NodeToken nt = new NodeToken(comment);
124    Ast.findLineAndCol(n, nt, first);
125    addedComments.add(nt);
126    if (debug) {
127      System.out.printf("addComment at %d:%d : %s%n", nt.beginLine, nt.beginColumn, comment.trim());
128    }
129    int linenumber = nt.beginLine - 1; /* because in jtb lines start at 1 */
130    String lineString = javaFileLines.get(linenumber);
131    int column =
132        getTabbedIndex(nt.beginColumn - 1, /* because in jtb cols start at 1 */ lineString);
133    StringBuilder sb = new StringBuilder();
134    sb.append(lineString.substring(0, column));
135    sb.append(comment);
136    if (comment.endsWith("\n")) {
137      // Insert the whitespace preceding the line
138      sb.append(precedingWhitespace(lineString.substring(0, column)));
139    }
140    sb.append(lineString.substring(column));
141    javaFileLines.set(linenumber, sb.toString());
142    if (debug) {
143      System.out.printf("addComment result: <<<%s>>>%n", sb.toString());
144    }
145  }

217  // Insert object invariants for this class.
218  // Insert owner assertions for fields.
219  // Grammar production:
220  // f0 -> ( "class" | "interface" )
221  // f1 -> <IDENTIFIER>
222  // f2 -> [ TypeParameters() ]
223  // f3 -> [ ExtendsList(isInterface) ]
224  // f4 -> [ ImplementsList(isInterface) ]
225  // f5 -> ClassOrInterfaceBody(isInterface)
226  @Override
227  public void visit(ClassOrInterfaceDeclaration n) {
228    String classname = Ast.getClassName(n);
229    String pptname = classname + ":::OBJECT";
230    PptTopLevel object_ppt = ppts.get(pptname);
231    if (object_ppt == null) {
232      pptname = classname + ":::CLASS";
233      object_ppt = ppts.get(pptname); // might *still* be null; we'll check later
234    }
235
236    ClassFieldInfo cfi;
237    { // set fieldNames slots
238      CollectFieldsVisitor cfv = new CollectFieldsVisitor(n, false);
239      if (object_ppt == null) {
240        cfi =
241            new ClassFieldInfo(
242                cfv.ownedFieldNames(),
243                cfv.finalFieldNames(),
244                new ArrayList<String>(),
245                new HashMap<>());
246      } else {
247        cfi =
248            new ClassFieldInfo(
249                cfv.ownedFieldNames(),
250                cfv.finalFieldNames(),
251                not_contains_null_fields(object_ppt, cfv.allFieldNames()),
252                element_type_fields(object_ppt, cfv.allFieldNames()));
253      }
254    }
255    cfis.push(cfi);
256
257    super.visit(n); // call "accept(this)" on each field
258
259    if (lightweight && (Daikon.output_format != OutputFormat.DBCJAVA)) {
260      for (int i = cfi.ownedFieldNames.size() - 1; i >= 0; i--) {
261        addComment(
262            n.f5.f1,
263            javaLineComment("@ invariant " + cfi.ownedFieldNames.get(i) + ".owner == this;"),
264            true);
265      }
266    }
267    if (object_ppt == null) {
268      if (debug) {
269        System.out.println("No object program point found for " + classname);
270      }
271    } else {
272      InvariantsAndModifiedVars obj_invs = invariants_for(object_ppt, ppts);
273      String inv_tag = (Daikon.output_format == OutputFormat.DBCJAVA ? "@invariant" : "invariant");
274      insertInvariants(n.f5.f1, inv_tag, obj_invs);
275    }
276
277    cfis.pop();
278  }

302  // Node n is a MethodDeclaration or a ConstructorDeclaration
303  @Nullable InvariantsAndModifiedVars[] get_requires_and_ensures(PptMap ppts, Node n) {
304    InvariantsAndModifiedVars requires_invs = null;
305    InvariantsAndModifiedVars ensures_invs = null;
306
307    List<PptTopLevel> matching_ppts = null;
308    if (n instanceof MethodDeclaration) {
309      matching_ppts = pptMatcher.getMatches(ppts, (MethodDeclaration) n);
310    } else if (n instanceof ConstructorDeclaration) {
311      ConstructorDeclaration cd = (ConstructorDeclaration) n;
312      matching_ppts = pptMatcher.getMatches(ppts, cd);
313    } else {
314      throw new Error("Node must be MethodDeclaration or ConstructorDeclaration");
315    }
316
317    for (PptTopLevel ppt : matching_ppts) {
318      if (ppt.ppt_name.isEnterPoint()) {
319        requires_invs = invariants_for(ppt, ppts);
320      } else if (ppt.ppt_name.isExitPoint()) {
321        if (!ppt.ppt_name.isCombinedExitPoint()) {
322          continue;
323        }
324        ensures_invs = invariants_for(ppt, ppts);
325      }
326    }
327
328    // if (debug) {
329    //   System.out.printf("get_requires_and_ensures(%s):%n  => requires=%s%n  => ensures=%s%n", n,
330    //                     requires_invs, ensures_invs);
331    // }
332
333    return new @Nullable InvariantsAndModifiedVars[] {requires_invs, ensures_invs};
334  }

340  // n must be a MethodDeclaration or ConstructorDeclaration
341  public void insertBehavior(Node n) {
342    class InsertBehaviorVisitor extends DepthFirstVisitor {
343      Node n;
344      boolean behaviorInserted;
345
346      public InsertBehaviorVisitor(Node n) {
347        super();
348        this.n = n;
349        behaviorInserted = false;
350      }
351
352      private String getBehaviorString() {
353        return "normal_behavior";
354      }
355
356      @Override
357      public void visit(NodeChoice nc) {
358        // Since we know we are in a Modifiers() parse tree, the only
359        // thing a NodeChoice can hold is a NodeToken for the modifer.
360        Annotate.debug.fine("InsertBehavior visitor visiting a NodeChoice");
361        String modifier = (nc != null && nc.choice != null ? nc.choice.toString() : "");
362        Annotate.debug.fine("A node choice here: " + modifier);
363
364        if (Ast.isAccessModifier(modifier)) {
365          Annotate.debug.fine("addComment from nc");
366          addComment(
367              n.getParent().getParent(),
368              "@ "
369                  + modifier
370                  + " "
371                  + getBehaviorString()
372                  + (Daikon.output_format == OutputFormat.JML ? " // Generated by Daikon" : "")
373                  + lineSep,
374              true);
375          Annotate.debug.fine("addComment complete");
376          behaviorInserted = true;
377        }
378      }
379
380      @Override
381      @SuppressWarnings("JdkObsolete") // JTB uses Vector and Enumeration
382      public void visit(NodeListOptional nlo) {
383        Annotate.debug.fine("InsertBehavior visitor visiting a NodeListOptional");
384        Annotate.debug.fine("With " + nlo.nodes.size() + " nodes");
385
386        if (nlo.present()) {
387          for (Enumeration<Node> e = nlo.elements(); e.hasMoreElements(); ) {
388            e.nextElement().accept(this);
389          }
390        }
391
392        if (!behaviorInserted) {
393          Annotate.debug.fine("addComment from nlo");
394          addComment(
395              n.getParent().getParent(),
396              "@ "
397                  + "private "
398                  + getBehaviorString()
399                  + (Daikon.output_format == OutputFormat.JML ? " // Generated by Daikon" : "")
400                  + lineSep,
401              true);
402          Annotate.debug.fine("addComment complete");
403        }
404      }
405    }
406
407    InsertBehaviorVisitor v = new InsertBehaviorVisitor(n);
408
409    // We are going to move back up the parse tree so we can look
410    // at the Modifiers for this method (or constructor).
411
412    NodeChoice nc = (NodeChoice) n.getParent();
413    NodeSequence ns = (NodeSequence) nc.getParent();
414    Modifiers m = (Modifiers) ns.elementAt(0);
415    if (Annotate.debug.isLoggable(Level.FINE)) {
416      System.out.println(Ast.formatEntireTree(m));
417    }
418    m.accept(v);
419    Annotate.debug.fine("InsertBehavior visitor complete");
420  }

422  @Override
423  public void visit(MethodDeclaration n) {
424
425    // Grammar production for ClassOrInterfaceBodyDeclaration:
426    // f0 -> Initializer()
427    //       | Modifiers() ( ClassOrInterfaceDeclaration(modifiers) | EnumDeclaration(modifiers) |
428    // ConstructorDeclaration() | FieldDeclaration(modifiers) | MethodDeclaration(modifiers) )
429    //       | ";"
430
431    super.visit(n); // call "accept(this)" on each field
432
433    @Nullable InvariantsAndModifiedVars[] requires_and_ensures = get_requires_and_ensures(ppts, n);
434
435    InvariantsAndModifiedVars requires_invs = requires_and_ensures[0];
436    InvariantsAndModifiedVars ensures_invs = requires_and_ensures[1];
437
438    String ensures_tag = Daikon.output_format.ensures_tag();
439    String requires_tag = Daikon.output_format.requires_tag();
440
441    boolean isOverride = false;
442    boolean isImplementation = false;
443    if (useReflection) {
444      isOverride = Ast.isOverride(n); // of a superclass
445      isImplementation = Ast.isImplementation(n); // of an interface
446    }
447
448    if (lightweight) {
449      if (isImplementation) {
450        requires_tag = "also_" + requires_tag;
451        ensures_tag = "also_" + ensures_tag;
452      }
453      if (isOverride) {
454        requires_invs = null; // no requires permitted on overridden methods
455        ensures_tag = "also_" + ensures_tag;
456      }
457    }
458
459    if (!lightweight) {
460      addComment(
461          n.getParent().getParent() /* see  ClassOrInterfaceBodyDeclaration */,
462          JML_END_COMMENT,
463          true);
464    }
465
466    boolean invariantInserted =
467        insertInvariants(
468            n.getParent().getParent() /* see  ClassOrInterfaceBodyDeclaration */,
469            ensures_tag,
470            ensures_invs,
471            lightweight);
472
473    if (ensures_invs != null) {
474      insertModifies(
475          n.getParent().getParent() /* see  ClassOrInterfaceBodyDeclaration */,
476          ensures_invs.modifiedVars,
477          ensures_tag,
478          lightweight);
479    }
480
481    invariantInserted =
482        insertInvariants(
483                n.getParent().getParent() /* see  ClassOrInterfaceBodyDeclaration */,
484                requires_tag,
485                requires_invs,
486                lightweight)
487            || invariantInserted;
488
489    if (!lightweight) {
490      if (!invariantInserted) {
491        insertJMLWorkaround(n.getParent().getParent() /* see  ClassOrInterfaceBodyDeclaration */);
492      }
493      insertBehavior(n);
494      if (isImplementation
495          || isOverride
496          // temporary fix: not processed correctly by Ast.java
497          || n.f2.f0.toString().equals("clone")) {
498        insertAlso(n.getParent().getParent());
499      }
500      addComment(
501          n.getParent().getParent() /* see  ClassOrInterfaceBodyDeclaration */,
502          JML_START_COMMENT,
503          true);
504    }
505  }

567          true);

571  // This is a hack that indicates whether an "assignable \everything"
572  // clause would be legal for the given method.
573  public boolean pureInJML(Node n) {
574    if (n instanceof MethodDeclaration) {
575      String name = ((MethodDeclaration) n).f2.f0.toString();
576      if (name.equals("clone")
577          || name.equals("equals")
578          || name.equals("hashCode")
579          || name.equals("hasMoreElements")
580          || name.equals("hasNext")) {
581        return true;
582      }
583    }
584    return false;
585  }

597  public boolean insertModifies(
598      Node n, String modifiesString, String prefix, boolean useJavaComment) {
599
600    modifiesString = modifiesString.trim();
601
602    if (modifiesString.equals("modifies") || modifiesString.equals("assignable")) {
603      // No variables found.
604      return false;
605    }
606
607    if (!(modifiesString.startsWith("modifies") || modifiesString.startsWith("assignable"))) {
608      // Doesn't look ilke a modifies clause.
609      return false;
610    }
611
612    String inv = null;
613    boolean doInsert = true;
614
615    if (Daikon.output_format == OutputFormat.JML) {
616      // Currently, AnnotateVisitor currently does not produce sensible
617      // "assignable" clauses for JML.
618      //  * One possible default is "assignable \everything".  However, that
619      //    is incorrect when the method is pure (that is, it overrides a
620      //    method that is annotated as pure, such as hashCode).
621      //  * Another possible default is to omit the assignable clause
622      //    entirely.  Since "assignable \everything" is the JML default, this
623      //    omission strategy works both when the method is pure and when it
624      //    is not.  Thus, we use this strategy.  Thanks to Christoph
625      //    Csnallner for this idea.
626      // In the future, AnnotateVisitor should produce "assignable" clauses.
627      doInsert = false;
628    } else if (Daikon.output_format == OutputFormat.DBCJAVA) {
629      // Modifies/assignable has no translation in Jtest DBC
630      doInsert = false;
631    } else if (lightweight && prefix.startsWith("also_")) {
632      inv = "also_" + modifiesString;
633    } else {
634      inv = modifiesString;
635    }
636
637    if (!doInsert) {
638      return false;
639    }
640
641    String commentContents = "@ " + inv + ";";
642    if (useJavaComment) {
643      commentContents = javaLineComment(commentContents);
644    } else {
645      commentContents += lineSep;
646    }
647    addComment(n, commentContents, true);
648    return true;
649  }

651  // The "invs" argument may be null, in which case no work is done.
652  @EnsuresNonNullIf(result = true, expression = "#3")
653  public boolean insertInvariants(
654      Node n, String prefix, @Nullable InvariantsAndModifiedVars invs, boolean useJavaComment) {
655    if (invs == null) {
656      return false;
657    }
658
659    boolean invariantInserted = false;
660
661    int maxIndex = invs.invariants.size();
662    if (maxInvariantsPP > 0) {
663      maxIndex = Math.min(maxIndex, maxInvariantsPP);
664    }
665
666    for (int i = maxIndex - 1; i >= 0; i--) {
667      Invariant inv = invs.invariants.get(i);
668      if (!inv.isValidExpression(Daikon.output_format)) {
669        // inexpressible invariant
670        if (insert_inexpressible) {
671          addComment(n, javaLineComment("! " + inv + ";"), true);
672        }
673        continue;
674      } else {
675        String commentContents =
676            (Daikon.output_format == OutputFormat.DBCJAVA ? "  " : "@ ")
677                + prefix
678                + " "
679                + inv.format_using(Daikon.output_format)
680                + (Daikon.output_format == OutputFormat.DBCJAVA ? "  " : ";");
681        if (useJavaComment) {
682          commentContents = javaLineComment(commentContents);
683        } else {
684          commentContents += lineSep;
685        }
686
687        addComment(n, commentContents, true);
688        invariantInserted = true;
689      }
690    }
691
692    return invariantInserted;
693  }

695  // Set .owner and/or .containsnull for ++, --, etc expressions within a
696  // statement.
697  // f0 -> PreIncrementExpression()
698  //       | PreDecrementExpression()
699  //       | PrimaryExpression() [ "++" | "--" | AssignmentOperator() Expression() ]
700  @Override
701  public void visit(StatementExpression n) {
702    super.visit(n); // call "accept(this)" on each field
703
704    Annotate.debug.fine("Found a statement expression: " + n.f0.choice);
705
706    // Nothing to do for Jtest DBC format
707    if (Daikon.output_format == OutputFormat.DBCJAVA) {
708      return;
709    }
710
711    if (n.f0.choice instanceof NodeSequence) {
712      NodeSequence ns = (NodeSequence) n.f0.choice;
713      PrimaryExpression pe = (PrimaryExpression) ns.elementAt(0);
714      // System.out.println("pe:" + Ast.format(pe));
715      // for (int i = 0; i<ns.size(); i++) {
716      //   System.out.println("ns #" + i + ": " + ns.get(i));
717      // }
718      if (ns.size() == 2) {
719        NodeOptional no = (NodeOptional) ns.elementAt(1);
720        NodeChoice nc = (NodeChoice) no.node;
721        if ((nc != null) && (nc.choice instanceof NodeSequence)) {
722          // It's an assignment.
723
724          // Don't take action unless the PrimaryExpression is a simple
725          // Name (that's effectively checked below) and has no
726          // PrimarySuffix or else its prefix is "this" (check that here).
727          String fieldname = null;
728          // System.out.println("pe.f1.size:" + pe.f1.size());
729          if (pe.f1.size() == 0) {
730            fieldname = Ast.fieldName(pe);
731          } else if (pe.f1.size() == 1) {
732            if (pe.f0.f0.which == 1) { // prefix is "this"
733              PrimarySuffix ps = (PrimarySuffix) pe.f1.elementAt(0);
734              if (ps.f0.which == 3) { // suffix is an identifier
735                NodeSequence ns2 = (NodeSequence) ps.f0.choice;
736                fieldname = ((NodeToken) ns2.elementAt(1)).tokenImage;
737              }
738            }
739          }
740
741          // System.out.printf("In statement, fieldname = %s%n", fieldname);
742          if ((fieldname != null)
743              && (isOwned(fieldname) || isNotContainsNull(fieldname) || isElementType(fieldname))) {
744            ConstructorDeclaration cd =
745                (ConstructorDeclaration) Ast.getParent(ConstructorDeclaration.class, n);
746            MethodDeclaration md = (MethodDeclaration) Ast.getParent(MethodDeclaration.class, n);
747            if ((cd != null) || ((md != null) && !Ast.contains(md.f0, "static"))) {
748              @SuppressWarnings("nullness")
749              @NonNull Node parent = Ast.getParent(Statement.class, n);
750              // If parent isn't in a block (eg, if parent
751              // is sole element in then or else clause), then this is wrong.
752              // It's safe, however.  But does it cause syntax errors if an
753              // else clause follows a then clause without braces?
754              if (isOwned(fieldname)) {
755                if (lightweight) {
756                  addCommentAfter(parent, javaLineComment("@ set " + fieldname + ".owner = this;"));
757                }
758              }
759              if (isNotContainsNull(fieldname)) {
760                addCommentAfter(
761                    parent, javaLineComment("@ set " + fieldname + ".containsNull = false;"));
762              }
763              if (isElementType(fieldname)) {
764                addCommentAfter(
765                    parent,
766                    javaLineComment(
767                        "@ set " + fieldname + ".elementType = " + elementType(fieldname) + ";"));
768              }
769            }
770          }
771        }
772      }
773    }
774  }

776  // This is an assignment exactly if field f1 is present.
777  // f0 -> ConditionalExpression()
778  // f1 -> [ AssignmentOperator() Expression() ]
779  @Override
780  public void visit(Expression n) {
781    super.visit(n); // call "accept(this)" on each field
782
783    if (n.f1.present()) {
784      Annotate.debug.fine("found an assignment expression: " + n);
785      PrimaryExpression pe = Ast.assignment2primaryexpression(n);
786      String fieldname = Ast.fieldName(pe);
787      Annotate.debug.fine("In expression, fieldname = " + fieldname);
788      @SuppressWarnings("nullness") // every expression is within a statement
789      @NonNull Node stmt = Ast.getParent(Statement.class, n);
790      if ((fieldname != null) && isOwned(fieldname)) {
791        if (lightweight) {
792          addCommentAfter(stmt, javaLineComment("@ set " + fieldname + ".owner = this;"));
793        }
794      }
795    }
796  }

816  /**
817   * Find a Daikon variable for the given field. The variable is either "this.field" (for instance
818   * variables) or "ClassName.field" (for static variables).
819   */
820  private VarInfo findVar(String field, PptTopLevel ppt) {
821    String varname = "this." + field;
822    VarInfo vi = ppt.find_var_by_name(varname);
823    if (vi == null) {
824      // No "this.field" variable, so try (static) "ClassName.field".
825      varname = ppt.ppt_name.getFullClassName() + "." + field;
826      vi = ppt.find_var_by_name(varname);
827    }
828    if (vi == null) {
829      // We found a variable in the source code that is not computed by
830      // Daikon.  (This can happen if Daikon was run omitting the
831      // variable, for instance due to --std-visibility.  But I want to
832      // throw an error instead, since I expect bugs to be more common
833      // than such problems.)
834      debug_field_problem(field, ppt);
835      throw new Error(
836          "Warning: Annotate: Daikon knows nothing about variable " + varname + " at " + ppt);
837    }
838    return vi;
839  }

850  // Returns a list of fields with ".containsNull == false" invariants.
851  // ppt is an :::OBJECT or :::CLASS program point.
852  List<String> not_contains_null_fields(PptTopLevel ppt, List<String> allFieldNames) {
853    // System.out.println("not_contains_null_fields(" + ppt + ")");
854    List<String> result = new ArrayList<>();
855    for (String field : allFieldNames) {
856      // System.out.println("field: " + field);
857      VarInfo vi = findVar(field, ppt);
858      PptSlice1 slice = ppt.findSlice(vi);
859      if (slice == null) {
860        // There might be no slice if there are no invariants over this var.
861        continue;
862      }
863      EltNonZero enz = EltNonZero.find(slice);
864      if (enz != null) {
865        String enz_format = format((Invariant) enz);
866        if (enz_format.endsWith(".containsNull == false")) {
867          result.add(field);
868        }
869      }
870    }
871    return result;
872  }

874  // Returns a HashMap for fields with ".elementType == \type(...)" invariants,
875  // mapping the field to the type.
876  // ppt is an :::OBJECT or :::CLASS program point.
877  HashMap<String, String> element_type_fields(PptTopLevel ppt, List<String> allFieldNames) {
878    // System.out.println("element_type_fields(" + ppt + ")");
879    HashMap<String, String> result = new HashMap<>();
880    // FieldDeclaration[] fdecls = cfv.fieldDeclarations();
881    // System.out.println("fields: " + Arrays.toString(fields));
882    for (String field : allFieldNames) {
883      // System.out.printf("element_type_fields (%s) field: %s%n", ppt, field);
884      VarInfo vi = findVar(field, ppt);
885      if (!vi.is_array()) {
886        continue;
887      }
888      if (vi.type.elementType().isPrimitive()) {
889        continue;
890      }
891      String varname = vi.name();
892      String elt_varname = varname + "[]";
893      VarInfo elt_vi = ppt.find_var_by_name(elt_varname);
894      if (elt_vi == null) {
895        debug_field_problem(elt_varname, ppt);
896        throw new Daikon.UserError(
897            "Annotate: Daikon knows nothing about variable " + elt_varname + " at " + ppt);
898      }
899      // et_varname variable represents the types of the elements.
900      String et_varname = elt_varname + DaikonVariableInfo.class_suffix;
901      // System.out.printf("Found %s, seeking %s%n", elt_varname, et_varname);
902      // We found variable "a[]".  Now find "a[].getClass().getName()".
903      // (Why might it not be present?)
904      // It might not be present if DaikonVariableInfo.shouldAddRuntimeClass would return false.
905      assert elt_vi.rep_type.isArray();
906      if (elt_vi.rep_type.dimensions() == 1 && elt_vi.rep_type.baseIsPrimitive()) {
907        continue;
908      }
909      VarInfo et_vi = ppt.find_var_by_name(et_varname);
910      if (et_vi == null) {
911        debug_field_problem(et_varname, ppt);
912        throw new Daikon.UserError(
913            String.format(
914                "Annotate: Daikon knows nothing about variable %s at %s%n  with allFieldNames = %s",
915                et_varname, ppt, allFieldNames));
916      }
917
918      // et_vi != null
919      PptSlice1 slice = ppt.findSlice(et_vi);
920      if (slice == null) {
921        // There might be no slice if there are no invariants over this var.
922        continue;
923      }
924      // System.out.println("Found slice for " + et_vi.name.name());
925      {
926        EltOneOfString eoos = EltOneOfString.find(slice);
927        // System.out.println("eoos: " + (eoos == null ? "null" : format((Invariant)eoos)));
928        if ((eoos != null) && (eoos.num_elts() == 1)) {
929          String eoos_format = format((Invariant) eoos);
930          int et_pos = eoos_format.indexOf(".elementType == \\type(");
931          if (et_pos != -1) {
932            String type = eoos_format.substring(et_pos + ".elementType == ".length());
933            result.put(field, type);
934          }
935        }
936      }
937      {
938        OneOfStringSequence eooss = OneOfStringSequence.find(slice);
939        // System.out.println("eooss: " + (eooss == null ? "null" : format((Invariant)eooss)));
940        if (eooss != null) {
941          String eooss_format = format((Invariant) eooss);
942          int et_pos = eooss_format.indexOf(".elementType == \\type(");
943          if (et_pos != -1) {
944            String type = eooss_format.substring(et_pos + ".elementType == ".length());
945            result.put(field, type);
946          }
947        }
948      }
949    }
950    return result;
951  }

967  public static InvariantsAndModifiedVars invariants_for(PptTopLevel ppt, PptMap pptmap) {
968
969    StringWriter sw = new StringWriter();
970    PrintWriter pw = new PrintWriter(sw);
971    PrintInvariants.print_modified_vars(ppt, pw);
972
973    InvariantsAndModifiedVars retval =
974        new InvariantsAndModifiedVars(Ast.getInvariants(ppt, pptmap), sw.toString());
975
976    // PrintInvariants.print_modified_vars(ppt, pw) returns possibly
977    // several lines. In such a case, we're only interested in the second
978    // one, which contains the "modified" or "assignable" clause.
979    String[] splitModVars = UtilPlume.splitLines(retval.modifiedVars);
980    if (splitModVars.length > 1) {
981      for (int i = 0; i < splitModVars.length; i++) {
982        if (splitModVars[i].startsWith("modifies ") || splitModVars[i].startsWith("assignable ")) {
983          retval.modifiedVars = splitModVars[i];
984          break;
985        }
986      }
987    }
988
989    return retval;
990  }

1002  // Consider a line L1 of text that contains some tabs.
1003  //
1004  // untabbedIndex represents an index into a line L2, where L2 is L1
1005  // with all tabs substituted by their corresponding number of blank
1006  // spaces, where a tab's width is 8 characters.
1007  //
1008  // This method returns the index into L1 that corresponds to the same
1009  // position in L2.
1010  //
1011  // I assume that untabbedIndex will not be an index into one of the
1012  // blank spaces that replaced some tab.
1013  public static int getTabbedIndex(int untabbedIndex, String L1) {
1014    if (untabbedIndex == 0) {
1015      return 0;
1016    }
1017    int expanded = 0;
1018    int index = 0;
1019    for (int i = 0; i < L1.length(); i++) {
1020
1021      if (expanded > untabbedIndex) {
1022        throw new RuntimeException(
1023            String.join(
1024                System.lineSeparator(),
1025                "expanded:" + expanded,
1026                "untabbedIndex:" + untabbedIndex,
1027                " L1:" + L1));
1028      } else if (expanded == untabbedIndex) {
1029        index = i;
1030        break;
1031      }
1032
1033      char c = L1.charAt(i);
1034      if (c == '\t') {
1035        expanded += 8 - (expanded % 8);
1036      } else {
1037        expanded += 1;
1038      }
1039    }
1040
1041    assert expanded == untabbedIndex
1042        : String.join(
1043            System.lineSeparator(),
1044            "expanded:" + expanded,
1045            "untabbedIndex:" + untabbedIndex,
1046            "L1: " + L1);
1047    return index;
1048  }

1050  /** Return the whitespace at the front of the string. */
1051  public static String precedingWhitespace(String s) {
1052    for (int i = 0; i < s.length(); i++) {
1053      if (!Character.isWhitespace(s.charAt(i))) {
1054        return s.substring(0, i);
1055      }
1056    }
1057    return s;

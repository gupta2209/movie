003package daikon.diff;
004
005import java.io.FileOutputStream;
006import java.io.IOException;
007import java.io.PrintStream;
008import java.lang.reflect.InvocationTargetException;
009
010/**
011 * <B>MultiDiff</B> is an executable application that performs the same functionality as Diff with a
012 * few key change. First, it always outputs the histogram even when two files are called. Second, it
013 * allows the option of creating *.spinfo based on the invariants found.
014 */
015public class MultiDiff {
016  private MultiDiff() {
017    throw new Error("do not instantiate");
018  }
019
020  public static void main(String[] args)
021      throws IOException, ClassNotFoundException, InstantiationException, IllegalAccessException,
022          NoSuchMethodException, InvocationTargetException {
023    try {
024      mainHelper(args);
025    } catch (daikon.Daikon.DaikonTerminationException e) {
026      daikon.Daikon.handleDaikonTerminationException(e);
027    }
028  }
029
030  /**
031   * This does the work of {@link #main(String[])}, but it never calls System.exit, so it is
032   * appropriate to be called progrmmatically.
033   */
034  public static void mainHelper(final String[] args)
035      throws IOException, ClassNotFoundException, InstantiationException, IllegalAccessException,
036          InvocationTargetException, NoSuchMethodException {
037    PrintStream out = new PrintStream(new FileOutputStream("rand_sel.spinfo"));
038    /*
039      try {
040        if (args.length != 0) {
041            FileOutputStream file = new FileOutputStream (args[0]);
042            out = new PrintStream (file);
043        }
044    }
045
046    catch (IOException e) {e.printStackTrace(); }
047    */
048    MultiDiffVisitor.setForSpinfoOut(out);
049    Diff.main(args);
050  }
051}

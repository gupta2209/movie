057/**
058 * This is a standalone program that compares the invariants from two versions of (and/or runs of) a
059 * program, and determines using Simplify whether the invariants from one logically imply the
060 * invariants from the other. These are referred to below as the "test" and "application"
061 * invariants, and the conditions that are checked is that the each test precondition (ENTER point
062 * invariant) must be implied some combination of application preconditions, and that each
063 * application postcondition (EXIT point invariant) must be implied by some combination of test
064 * postconditions and application preconditions.
065 */
066public class LogicalCompare {
067  private LogicalCompare() {
068    throw new Error("do not instantiate");
069  }

139  private static List<Invariant> filterInvariants(List<Invariant> invs, boolean isPost) {
140    List<Invariant> new_invs = new ArrayList<>();
141    for (Invariant inv : invs) {
142      Invariant guarded_inv = inv;
143      //       System.err.println("Examining " + inv.format());
144      if (inv instanceof GuardingImplication) inv = ((Implication) inv).consequent();
145      if (inv instanceof LowerBound
146          || inv instanceof UpperBound
147          || inv instanceof EltLowerBound
148          || inv instanceof EltUpperBound
149          || inv instanceof LowerBoundFloat
150          || inv instanceof UpperBoundFloat
151          || inv instanceof EltLowerBoundFloat
152          || inv instanceof EltUpperBoundFloat) {
153        if (filters['b'] && inv.hasUninterestingConstant()) {
154          continue;
155        }
156        if (filters['B']) {
157          continue;
158        }
159      }
160      if (inv instanceof OneOf || inv instanceof OneOfString || inv instanceof OneOfFloat) {
161        if (filters['o'] && inv.hasUninterestingConstant()) {
162          continue;
163        }
164        if (filters['O']) {
165          continue;
166        }
167      }
168      // test used to be "(filters['m'] && inv.ppt.num_mod_samples() == 0)"
169      if (filters['m'] && inv.ppt.num_samples() == 0) {
170        continue;
171      }
172      if (filters['j'] && !inv.justified()) {
173        continue;
174      }
175      if (filters['p'] && isPost && shouldDiscardInvariant(inv)) {
176        continue;
177      }
178      if (filters['i'] && !isPost && inv instanceof Implication) {
179        continue;
180      }
181      String simp = inv.format_using(OutputFormat.SIMPLIFY);
182      if (simp.indexOf("format_simplify") != -1
183          || simp.indexOf("OutputFormat:Simplify") != -1
184          || simp.indexOf("Simplify not implemented") != -1) {
185        // Noisy, since we should be able to handle most invariants now
186        System.out.println("Bad Simplify formatting:%n  " + inv.format() + "%n  " + simp);
187        continue;
188      }
189      if (inv.format_using(OutputFormat.DAIKON).indexOf("warning: too few samples") != -1) {
190        continue;
191      }
192      if (inv.isGuardingPredicate) {
193        continue;
194      }
195      //       System.err.println("Keeping   " + inv.format());
196      new_invs.add(guarded_inv);
197    }
198    return new_invs;
199  }

201  // This is a modified version of the method with the same name from
202  // the DerivedParameterFilter, with an exception for invariants
203  // indicating that variables are unchanged (except that that
204  // exception is currently turned off)
205  private static boolean shouldDiscardInvariant(Invariant inv) {
206    for (int i = 0; i < inv.ppt.var_infos.length; i++) {
207      VarInfo vi = inv.ppt.var_infos[i];
208      // ppt has to be a PptSlice, not a PptTopLevel
209      if (vi.isDerivedParamAndUninteresting()) {
210        // Exception: let invariants like "orig(arg) == arg" through.
211        if (inv.isEqualityComparison()) {
212          EqualityComparison comp = (EqualityComparison) inv;
213          VarInfo var1 = comp.var1();
214          VarInfo var2 = comp.var2();
215          boolean vars_are_same =
216              var1.prestate_name().equals(var2.name()) || var2.prestate_name().equals(var1.name());
217          if (vars_are_same) {
218            return false;
219          }
220        }
221        //         if (inv instanceof OneOf || inv instanceof OneOfString ||
222        //             inv instanceof OneOfString)
223        //           return false;
224        //         System.err.println("Because of " + vi.name.name() + ", discarding "
225        //                            + inv.format());
226        return true;
227      }
228    }
229    return false;
230  }

232  // Translate a vector of Invariants into a vector of Lemmas, without
233  // changing the invariants.
234  private static List<Lemma> translateStraight(List<Invariant> invs) {
235    List<Lemma> lems = new ArrayList<>();
236    for (Invariant inv : invs) {
237      lems.add(new InvariantLemma(inv));
238    }
239    return lems;
240  }

242  // Translate a vector of Invariants into a vector of Lemmas,
243  // discarding any invariants that represent only facts about a
244  // routine's prestate.
245  private static List<Lemma> translateRemovePre(List<Invariant> invs) {
246    List<Lemma> lems = new ArrayList<>();
247    for (Invariant inv : invs) {
248      if (!inv.isAllPrestate()) lems.add(new InvariantLemma(inv));
249    }
250    return lems;
251  }

253  // Translate a vector of Invariants into a vector of Lemmas, adding
254  // orig(...) to each variable so that the invariant will be true
255  // about the precondition of a routine when it is examined in the
256  // poststate context.
257  // The arguments are invariants at the entry point, where no orig(...) variables exist.
258  private static List<Lemma> translateAddOrig(List<Invariant> invs) {
259    List<Lemma> lems = new ArrayList<>();
260    for (Invariant inv : invs) {
261      lems.add(InvariantLemma.makeLemmaAddOrig(inv));
262    }
263    return lems;
264  }

281  @RequiresNonNull("lemmas")
282  private static int checkConsequences(List<Lemma> assumptions, List<Lemma> consequences) {
283    Set<String> assumption_formulas = new HashSet<>();
284    for (Lemma lem : assumptions) {
285      assumption_formulas.add(lem.formula);
286    }
287
288    int invalidCount = 0;
289    for (Lemma inv : consequences) {
290      char result;
291      boolean identical = false;
292      if (assumption_formulas.contains(inv.formula)) {
293        result = 'T';
294        identical = true;
295      } else {
296        result = lemmas.checkLemma(inv);
297      }
298
299      if (opt_minimize_classes) {
300        if (result == 'T' && !identical) {
301          List<Set<Class<? extends Invariant>>> sets = lemmas.minimizeClasses(inv.formula);
302          for (Set<Class<? extends Invariant>> classes : sets) {
303            @SuppressWarnings(
304                "nullness") // application invariant: context; might be able to rewrite types to
305            // make consequences a List<InvariantLemma>"
306            @NonNull Class<? extends Invariant> inv_class = inv.invClass();
307            System.out.print(shortName(inv_class) + ":");
308            if (classes.contains(inv_class)) {
309              System.out.print(" " + shortName(inv_class));
310              classes.remove(inv_class);
311            }
312            for (Class<? extends Invariant> c : classes) {
313              System.out.print(" " + shortName(c));
314            }
315            System.out.println();
316          }
317          System.out.println(inv.summarize());
318          System.out.println();
319        }
320        if (true) {
321          continue;
322        }
323      }
324
325      if (result == 'T') {
326        if (opt_proofs) {
327          if (identical) {
328            System.out.println("Identical");
329          } else {
330            List<Lemma> assume = lemmas.minimizeProof(inv);
331            System.out.println();
332            for (Lemma lem : assume) {
333              System.out.println(lem.summarize());
334            }
335            System.out.println("----------------------------------");
336            System.out.println(inv.summarize());
337            if (opt_show_formulas) {
338              System.out.println();
339              for (Lemma lem : assume) {
340                System.out.println("    " + lem.formula);
341              }
342              System.out.println("    ------------------------------------------");
343              System.out.println("    " + inv.formula);
344            }
345          }
346        } else if (opt_show_valid) {
347          System.out.print("Valid: ");
348          System.out.println(inv.summary);
349          if (opt_show_formulas) System.out.println("    " + inv.formula);
350        }
351      } else if (result == 'F') {
352        invalidCount++;
353        if (opt_proofs) System.out.println();
354        System.out.print("Invalid: ");
355        System.out.println(inv.summary);
356        if (opt_show_formulas) System.out.println("    " + inv.formula);
357      } else {
358        assert result == '?';
359        if (opt_proofs) System.out.println();
360        System.out.print("Timeout: ");
361        System.out.println(inv.summary);
362        if (opt_show_formulas) System.out.println("    " + inv.formula);
363      }
364    }
365    return invalidCount;
366  }

368  // Check that each of the invariants in CONSEQUENCES follows from
369  // zero or more of the invariants in ASSUMPTIONS. Returns the number
370  // of invariants that can't be proven to follow.
371  @RequiresNonNull("lemmas")
372  private static int evaluateImplications(List<Lemma> assumptions, List<Lemma> consequences)
373      throws SimplifyError {
374    int mark = lemmas.markLevel();
375    lemmas.pushOrdering();
376    lemmas.pushLemmas(assumptions);
377    if (lemmas.checkForContradiction() == 'T') {
378      if (opt_post_after_pre) {
379        // Shouldn't be reached anymore
380        lemmas.removeContradiction();
381        System.out.println("Warning: had to remove contradiction(s)");
382      } else {
383        System.out.println("Contradictory assumptions:");
384        List<Lemma> min = lemmas.minimizeContradiction();
385        LemmaStack.printLemmas(System.out, min);
386        throw new Error("Aborting");
387      }
388    }
389
390    int invalidCount = checkConsequences(assumptions, consequences);
391
392    lemmas.popToMark(mark);
393    return invalidCount;
394  }

396  @RequiresNonNull("lemmas")
397  private static int evaluateImplicationsCarefully(
398      List<Lemma> safeAssumptions, List<Lemma> unsafeAssumptions, List<Lemma> consequences)
399      throws SimplifyError {
400    int mark = lemmas.markLevel();
401    List<Lemma> assumptions = new ArrayList<>();
402    lemmas.pushOrdering();
403    lemmas.pushLemmas(safeAssumptions);
404    if (lemmas.checkForContradiction() == 'T') {
405      System.out.println("Contradictory assumptions:");
406      List<Lemma> min = lemmas.minimizeContradiction();
407      LemmaStack.printLemmas(System.out, min);
408      throw new Error("Aborting");
409    }
410    assumptions.addAll(safeAssumptions);
411
412    int j = unsafeAssumptions.size();
413    for (int i = 0; i < unsafeAssumptions.size(); i++) {
414      List<Lemma> unsafe = unsafeAssumptions.subList(i, j);
415      boolean safe = false;
416      while (!safe && unsafe.size() > 0) {
417        int innerMark = lemmas.markLevel();
418        lemmas.pushLemmas(new ArrayList<Lemma>(unsafe));
419        if (lemmas.checkForContradiction() == 'T') {
420          lemmas.popToMark(innerMark);
421          j = i + unsafe.size();
422          unsafe = unsafe.subList(0, unsafe.size() / 2);
423        } else {
424          safe = true;
425          i += unsafe.size() - 1;
426          assumptions.addAll(unsafe);
427        }
428      }
429      if (!safe) {
430        assert unsafe.size() == 0;
431        j = unsafeAssumptions.size();
432      }
433    }
434
435    int invalidCount = checkConsequences(assumptions, consequences);
436
437    lemmas.popToMark(mark);
438    return invalidCount;
439  }

448  // Comparare the invariants for enter and exit points between two
449  // methods (usually two sets of invariants for methods of the same
450  // name).
451  // For historical reasons, one set of invariants is called the app
452  // invariants (from running the app in practice) and the other set of
453  // invariants is called the test invariants (from running the app on its
454  // test suite).
455  @RequiresNonNull({"extra_assumptions", "lemmas"})
456  private static void comparePpts(
457      PptTopLevel app_enter_ppt,
458      PptTopLevel test_enter_ppt,
459      PptTopLevel app_exit_ppt,
460      PptTopLevel test_exit_ppt) {
461    LemmaStack.clearInts();
462
463    //     app_enter_ppt.guardInvariants();
464    //     test_enter_ppt.guardInvariants();
465    //     app_exit_ppt.guardInvariants();
466    //     test_exit_ppt.guardInvariants();
467
468    List<Invariant> a_pre = app_enter_ppt.invariants_vector();
469    List<Invariant> t_pre = test_enter_ppt.invariants_vector();
470    List<Invariant> a_post = app_exit_ppt.invariants_vector();
471    List<Invariant> t_post = test_exit_ppt.invariants_vector();
472
473    if (opt_timing) System.out.println("Starting timer");
474    long processing_time_start = System.currentTimeMillis();
475
476    a_pre = filterInvariants(a_pre, false);
477    t_pre = filterInvariants(t_pre, false);
478    a_post = filterInvariants(a_post, true);
479    t_post = filterInvariants(t_post, true);
480
481    if (opt_show_sets) {
482      System.out.println("Background assumptions:");
483      LemmaStack.printLemmas(System.out, Lemma.lemmasList());
484      System.out.println("");
485
486      List<Lemma> v = new ArrayList<>();
487      v.addAll(translateAddOrig(a_pre));
488      System.out.println("Weak preconditions (Apre):");
489      LemmaStack.printLemmas(System.out, v);
490      System.out.println("");
491
492      v = new ArrayList<Lemma>();
493      v.addAll(translateAddOrig(t_pre));
494      System.out.println("Strong preconditions (Tpre):");
495      LemmaStack.printLemmas(System.out, v);
496      System.out.println("");
497
498      v = new ArrayList<Lemma>();
499      v.addAll(translateRemovePre(t_post));
500      System.out.println("Strong postconditions (Tpost):");
501      LemmaStack.printLemmas(System.out, v);
502      System.out.println("");
503
504      v = new ArrayList<Lemma>();
505      v.addAll(translateRemovePre(a_post));
506      System.out.println("Weak postconditions (Apost):");
507      LemmaStack.printLemmas(System.out, v);
508      return;
509    }
510
511    if (opt_show_count) {
512      System.out.println("Strong preconditions consist of " + t_pre.size() + " invariants.");
513    }
514    List<Lemma> pre_assumptions = new ArrayList<>();
515    pre_assumptions.addAll(translateStraight(a_pre));
516    List<Lemma> pre_conclusions = new ArrayList<>();
517    pre_conclusions.addAll(translateStraight(t_pre));
518    Collections.sort(pre_conclusions);
519
520    System.out.println("Testing preconditions:");
521    int bad_pre = evaluateImplications(pre_assumptions, pre_conclusions);
522    int num_checked = pre_conclusions.size();
523    if (bad_pre > 0 && !opt_post_after_pre) {
524      System.out.println("Precondition failure, skipping postconditions");
525      return;
526    }
527
528    System.out.println("Testing postconditions:");
529
530    List<Lemma> post_assumptions_safe = new ArrayList<>();
531    List<Lemma> post_assumptions_unsafe = new ArrayList<>();
532    List<Lemma> post_conclusions = new ArrayList<>();
533
534    post_assumptions_unsafe.addAll(translateAddOrig(a_pre));
535    post_assumptions_safe.addAll(translateStraight(t_post));
536    String ppt_name = test_enter_ppt.ppt_name.name();
537    if (extra_assumptions.containsKey(ppt_name)) {
538      List<Lemma> assumptions = extra_assumptions.get(ppt_name);
539      post_assumptions_unsafe.addAll(assumptions);
540    }
541
542    post_conclusions.addAll(translateRemovePre(a_post));
543    Collections.sort(post_conclusions);
544
545    if (opt_show_count) {
546      System.out.println(
547          "Weak postconditions consist of " + post_conclusions.size() + " invariants.");
548    }
549
550    evaluateImplicationsCarefully(post_assumptions_safe, post_assumptions_unsafe, post_conclusions);
551    long time_elapsed = System.currentTimeMillis() - processing_time_start;
552    num_checked += post_conclusions.size();
553    if (opt_show_count) System.out.println("Checked " + num_checked + " invariants total");
554    if (opt_timing) System.out.println("Total time " + time_elapsed + "ms");
555  }

557  @RequiresNonNull("extra_assumptions")
558  private static void readExtraAssumptions(String filename) {
559    File file = new File(filename);
560    try {
561      LineNumberReader reader = UtilPlume.lineNumberFileReader(file);
562      String line;
563      String ppt_name = null;
564      while ((line = reader.readLine()) != null) {
565        line = line.trim();
566        if (line.equals("") || line.startsWith("#")) {
567          continue;
568        } else if (line.startsWith("PPT_NAME")) {
569          ppt_name = line.substring("PPT_NAME".length()).trim();
570          if (!extra_assumptions.containsKey(ppt_name)) {
571            extra_assumptions.put(ppt_name, new ArrayList<Lemma>());
572          }
573        } else if (line.startsWith("(")) {
574          if (ppt_name == null) {
575            System.err.println("Must specify PPT_NAME before giving a formula");
576            throw new Error();
577          }
578          String formula, comment;
579          // XXX This should really read a balanced Simplify
580          // expression, then look for a comment after that. But that
581          // would involve counting parens and vertical bars and
582          // backslashes, which I'm too lazy to do right now.
583          if (line.indexOf("#") != -1) {
584            formula = line.substring(0, line.indexOf("#"));
585            comment = line.substring(line.indexOf("#") + 1);
586          } else {
587            formula = line;
588            comment = "User-supplied assumption";
589          }
590          formula = formula.trim();
591          comment = comment.trim();
592          @SuppressWarnings(
593              "nullness") // map: on previous loop iteration, this key was added to map
594          @NonNull List<Lemma> assumption_vec = extra_assumptions.get(ppt_name);
595          assumption_vec.add(new Lemma(comment, formula));
596        } else {
597          System.err.println("Can't parse " + line + " in assumptions file");
598          System.err.println(
599              "Should be `PPT_NAME <ppt_name>' or a Simplify formula (starting with `(')");
600          throw new Error();
601        }
602      }
603    } catch (FileNotFoundException e) {
604      System.err.println("File not found: " + filename);
605      throw new Error();
606    } catch (IOException e) {
607      System.err.println("IO error reading " + filename);
608      throw new Error();
609    }
610  }

622   * appropriate to be called progrmmatically.
623   *
624   * @param args command-line arguments, like those of {@link #main}
625   */
626  public static void mainHelper(final String[] args)
627      throws FileNotFoundException, IOException, SimplifyError {
628    LongOpt[] longopts =
629        new LongOpt[] {
630          new LongOpt("assume", LongOpt.REQUIRED_ARGUMENT, null, 0),
631          new LongOpt("config_option", LongOpt.REQUIRED_ARGUMENT, null, 0),
632          new LongOpt("config-file", LongOpt.REQUIRED_ARGUMENT, null, 0),
633          new LongOpt(Daikon.debugAll_SWITCH, LongOpt.NO_ARGUMENT, null, 0),
634          new LongOpt(Daikon.debug_SWITCH, LongOpt.REQUIRED_ARGUMENT, null, 0),
635          new LongOpt("filters", LongOpt.REQUIRED_ARGUMENT, null, 0),
636          new LongOpt("help", LongOpt.NO_ARGUMENT, null, 0),
637          new LongOpt("minimize-classes", LongOpt.NO_ARGUMENT, null, 0),
638          new LongOpt("no-post-after-pre-failure", LongOpt.NO_ARGUMENT, null, 0),
639          new LongOpt("post-after-pre-failure", LongOpt.NO_ARGUMENT, null, 0),
640          new LongOpt("proofs", LongOpt.NO_ARGUMENT, null, 0),
641          new LongOpt("show-count", LongOpt.NO_ARGUMENT, null, 0),
642          new LongOpt("show-formulas", LongOpt.NO_ARGUMENT, null, 0),
643          new LongOpt("show-sets", LongOpt.NO_ARGUMENT, null, 0),
644          new LongOpt("show-valid", LongOpt.NO_ARGUMENT, null, 0),
645          new LongOpt("timing", LongOpt.NO_ARGUMENT, null, 0),
646        };
647
648    Configuration.getInstance().apply("daikon.inv.Invariant.simplify_define_predicates=true");
649    Configuration.getInstance().apply("daikon.simplify.Session.simplify_max_iterations=2147483647");
650
651    extra_assumptions = new LinkedHashMap<>();
652
653    Getopt g = new Getopt("daikon.tools.compare.LogicalCompare", args, "h", longopts);
654    int c;
655    boolean user_filters = false;
656    while ((c = g.getopt()) != -1) {
657      switch (c) {
658        case 0:
659          // got a long option
660          String option_name = longopts[g.getLongind()].getName();
661          if (Daikon.help_SWITCH.equals(option_name)) {
662            System.out.println(usage);
663            throw new Daikon.NormalTermination();
664          } else if (option_name.equals("config-file")) {
665            String config_file = Daikon.getOptarg(g);
666            try {
667              InputStream stream = new FileInputStream(config_file);
668              Configuration.getInstance().apply(stream);
669            } catch (IOException e) {
670              throw new RuntimeException("Could not open config file " + config_file);
671            }
672          } else if (option_name.equals("config_option")) {
673            String item = Daikon.getOptarg(g);
674            Configuration.getInstance().apply(item);
675          } else if (Daikon.debugAll_SWITCH.equals(option_name)) {
676            Global.debugAll = true;
677          } else if (Daikon.debug_SWITCH.equals(option_name)) {
678            LogHelper.setLevel(Daikon.getOptarg(g), LogHelper.FINE);
679          } else if (option_name.equals("proofs")) {
680            opt_proofs = true;
681          } else if (option_name.equals("show-count")) {
682            opt_show_count = true;
683          } else if (option_name.equals("show-formulas")) {
684            opt_show_formulas = true;
685          } else if (option_name.equals("show-valid")) {
686            opt_show_valid = true;
687          } else if (option_name.equals("show-sets")) {
688            opt_show_sets = true;
689          } else if (option_name.equals("post-after-pre-failure")) {
690            opt_post_after_pre = true;
691          } else if (option_name.equals("no-post-after-pre-failure")) {
692            opt_post_after_pre = false;
693          } else if (option_name.equals("timing")) {
694            opt_timing = true;
695          } else if (option_name.equals("filters")) {
696            String f = Daikon.getOptarg(g);
697            for (int i = 0; i < f.length(); i++) {
698              filters[f.charAt(i)] = true;
699            }
700            user_filters = true;
701          } else if (option_name.equals("assume")) {
702            readExtraAssumptions(Daikon.getOptarg(g));
703          } else if (option_name.equals("minimize-classes")) {
704            opt_minimize_classes = true;
705          } else {
706            throw new Error();
707          }
708          break;
709        case 'h':
710          System.out.println(usage);
711          throw new Daikon.NormalTermination();
712        case '?':
713          // getopt() already printed an error
714          throw new Daikon.UserError("Bad argument");
715      }
716    }
717
718    if (!user_filters) {
719      filters['i'] = filters['j'] = filters['m'] = filters['p'] = true;
720    }
721
722    // Set up debug traces; note this comes after reading command line options.
723    LogHelper.setupLogs(Global.debugAll ? LogHelper.FINE : LogHelper.INFO);
724
725    int num_args = args.length - g.getOptind();
726
727    if (num_args < 2) {
728      throw new Daikon.UserError(
729          "Must have at least two non-option arguments" + Global.lineSep + usage);
730    }
731
732    String app_filename = args[g.getOptind() + 0];
733    String test_filename = args[g.getOptind() + 1];
734
735    PptMap app_ppts =
736        FileIO.read_serialized_pptmap(
737            new File(app_filename), true // use saved config
738            );
739    PptMap test_ppts =
740        FileIO.read_serialized_pptmap(
741            new File(test_filename), true // use saved config
742            );
743    if (num_args == 4 || num_args == 3) {
744      String enter_ppt_name = args[g.getOptind() + 2];
745      String exit_ppt_name;
746      if (num_args == 4) {
747        exit_ppt_name = args[g.getOptind() + 3];
748      } else {
749        exit_ppt_name = ""; // s/b nonexistent
750      }
751      startProver();
752
753      System.out.println(
754          "Comparing "
755              + enter_ppt_name
756              + " and "
757              + exit_ppt_name
758              + " in "
759              + app_filename
760              + " and "
761              + test_filename);
762
763      PptTopLevel app_enter_ppt = app_ppts.get(enter_ppt_name);
764      PptTopLevel test_enter_ppt = test_ppts.get(enter_ppt_name);
765      if (app_enter_ppt == null) {
766        throw new Daikon.UserError(
767            String.format("Ppt %s not found in %s", enter_ppt_name, app_filename));
768      }
769      if (test_enter_ppt == null) {
770        throw new Daikon.UserError(
771            String.format("Ppt %s not found in %s", enter_ppt_name, test_filename));
772      }
773
774      PptTopLevel app_exit_ppt = app_ppts.get(exit_ppt_name);
775      PptTopLevel test_exit_ppt = test_ppts.get(exit_ppt_name);
776      if (app_exit_ppt == null) {
777        app_exit_ppt = app_ppts.get(app_enter_ppt.ppt_name.makeExit());
778        if (app_exit_ppt == null) {
779          throw new Daikon.UserError(
780              String.format(
781                  "Neither ppt %s nor ppt %s found in %s",
782                  exit_ppt_name, app_enter_ppt.ppt_name.makeExit(), app_filename));
783        }
784      }
785      if (test_exit_ppt == null) {
786        test_exit_ppt = test_ppts.get(test_enter_ppt.ppt_name.makeExit());
787        if (test_exit_ppt == null) {
788          throw new Daikon.UserError(
789              String.format(
790                  "Neither ppt %s nor ppt %s found in %s",
791                  exit_ppt_name, test_enter_ppt.ppt_name.makeExit(), test_filename));
792        }
793      }
794
795      comparePpts(
796          app_enter_ppt, test_enter_ppt,
797          app_exit_ppt, test_exit_ppt);
798    } else if (num_args == 2) {
799      startProver();
800
801      Collection<@KeyFor("app_ppts.nameToPpt") String> app_ppt_names = app_ppts.nameStringSet();
802      Collection<@KeyFor("test_ppts.nameToPpt") String> test_ppt_names = test_ppts.nameStringSet();
803      // These are keys in both app_ppts and test_ppts.
804      Set<String> common_names = new TreeSet<>();
805
806      for (String name : app_ppt_names) {
807        @SuppressWarnings("nullness") // map: iterating over keyset
808        @NonNull PptTopLevel app_ppt = app_ppts.get(name);
809
810        if (!app_ppt.ppt_name.isEnterPoint()) {
811          // an exit point, and we only want entries
812          continue;
813        }
814        if (app_ppt.num_samples() > 0) {
815          if (test_ppt_names.contains(name)
816              && castNonNull(test_ppts.get(name)).num_samples()
817                  > 0) { // correlated maps: test_ppts.get(name) is non-null because
818            // test_ppt_names.contains(name) is true
819            common_names.add(name);
820          } else {
821            System.out.println(name + " was used but not tested");
822          }
823        }
824      }
825
826      for (String name : common_names) {
827        System.out.println();
828        System.out.println("Looking at " + name);
829        @SuppressWarnings("nullness") // map: iterating over subset of keySet
830        @NonNull PptTopLevel app_enter_ppt = app_ppts.get(name);
831        @SuppressWarnings("nullness") // map: iterating over subset of keySet
832        @NonNull PptTopLevel test_enter_ppt = test_ppts.get(name);
833        @SuppressWarnings("nullness") // map: exit should be in map if enter is
834        @NonNull PptTopLevel app_exit_ppt = app_ppts.get(app_enter_ppt.ppt_name.makeExit());
835        @SuppressWarnings("nullness") // map: exit should be in map if enter is
836        @NonNull PptTopLevel test_exit_ppt = test_ppts.get(test_enter_ppt.ppt_name.makeExit());
837
838        assert app_exit_ppt != null;
839        assert test_exit_ppt != null;
840        comparePpts(
841            app_enter_ppt, test_enter_ppt,
842            app_exit_ppt, test_exit_ppt);
843      }
844    } else {
845      throw new Daikon.UserError("Too many arguments" + Global.lineSep + usage);
846    }
847  }

093  public static void main(String[] args)
094      throws FileNotFoundException, StreamCorruptedException, OptionalDataException, IOException,
095          ClassNotFoundException {
096    try {
097      if (args.length == 0) {
098        throw new Daikon.UserError(usage);
099      }
100      mainHelper(args);
101    } catch (Daikon.DaikonTerminationException e) {
102      Daikon.handleDaikonTerminationException(e);
103    }
104  }

106  /**
107   * This does the work of {@link #main(String[])}, but it never calls System.exit, so it is
108   * appropriate to be called progrmmatically.
109   *
110   * @param args command-line arguments, like those of {@link #main}
111   */
112  public static void mainHelper(final String[] args)
113      throws FileNotFoundException, StreamCorruptedException, OptionalDataException, IOException,
114          ClassNotFoundException {
115    daikon.LogHelper.setupLogs(daikon.LogHelper.INFO);
116
117    LongOpt[] longopts =
118        new LongOpt[] {
119          new LongOpt(Daikon.config_option_SWITCH, LongOpt.REQUIRED_ARGUMENT, null, 0),
120          new LongOpt(output_SWITCH, LongOpt.REQUIRED_ARGUMENT, null, 0),
121          new LongOpt(dir_SWITCH, LongOpt.REQUIRED_ARGUMENT, null, 0),
122          new LongOpt(conf_SWITCH, LongOpt.NO_ARGUMENT, null, 0),
123          new LongOpt(filter_SWITCH, LongOpt.NO_ARGUMENT, null, 0),
124          new LongOpt(verbose_SWITCH, LongOpt.NO_ARGUMENT, null, 0),
125          new LongOpt(Daikon.debugAll_SWITCH, LongOpt.NO_ARGUMENT, null, 0),
126          new LongOpt(Daikon.debug_SWITCH, LongOpt.REQUIRED_ARGUMENT, null, 0),
127          new LongOpt(Daikon.ppt_regexp_SWITCH, LongOpt.REQUIRED_ARGUMENT, null, 0),
128          new LongOpt(Daikon.track_SWITCH, LongOpt.REQUIRED_ARGUMENT, null, 0),
129        };
130    Getopt g = new Getopt("daikon.tools.InvariantChecker", args, "h", longopts);
131    int c;
132    while ((c = g.getopt()) != -1) {
133      switch (c) {
134        case 0:
135          // got a long option
136          String option_name = longopts[g.getLongind()].getName();
137          if (Daikon.help_SWITCH.equals(option_name)) {
138            System.out.println(usage);
139            throw new Daikon.NormalTermination();
140          } else if (conf_SWITCH.equals(option_name)) {
141            doConf = true;
142          } else if (filter_SWITCH.equals(option_name)) {
143            doFilter = true;
144          } else if (verbose_SWITCH.equals(option_name)) {
145            quiet = false;
146          } else if (dir_SWITCH.equals(option_name)) {
147            dir_file = new File(Daikon.getOptarg(g));
148            if (!dir_file.exists() || !dir_file.isDirectory()) {
149              throw new Daikon.UserError("Error reading the directory " + dir_file);
150            }
151
152          } else if (output_SWITCH.equals(option_name)) {
153            File output_file = new File(Daikon.getOptarg(g));
154            output_stream = new PrintStream(new FileOutputStream(output_file));
155          } else if (Daikon.config_option_SWITCH.equals(option_name)) {
156            String item = Daikon.getOptarg(g);
157            daikon.config.Configuration.getInstance().apply(item);
158            break;
159          } else if (Daikon.debugAll_SWITCH.equals(option_name)) {
160            Global.debugAll = true;
161          } else if (Daikon.debug_SWITCH.equals(option_name)) {
162            LogHelper.setLevel(Daikon.getOptarg(g), LogHelper.FINE);
163          } else if (Daikon.track_SWITCH.equals(option_name)) {
164            LogHelper.setLevel("daikon.Debug", LogHelper.FINE);
165            String error = Debug.add_track(Daikon.getOptarg(g));
166            if (error != null) {
167              throw new Daikon.UserError(
168                  "Error parsing track argument '" + Daikon.getOptarg(g) + "' - " + error);
169            }
170          } else {
171            throw new RuntimeException("Unknown long option received: " + option_name);
172          }
173          break;
174        case 'h':
175          System.out.println(usage);
176          throw new Daikon.NormalTermination();
177        case '?':
178          break; // getopt() already printed an error
179        default:
180          System.out.println("getopt() returned " + c);
181          break;
182      }
183    }
184
185    {
186      File inv_file = null;
187
188      // Loop through each filename specified
189      for (int i = g.getOptind(); i < args.length; i++) {
190
191        // Get the file and make sure it exists
192        File file = new File(args[i]);
193        if (!file.exists()) {
194          throw new Error("File " + file + " not found.");
195        }
196
197        // These aren't "endsWith()" because there might be a suffix on the end
198        // (eg, a date).
199        String filename = file.toString();
200        if (filename.indexOf(".inv") != -1) {
201          if (inv_file != null) {
202            throw new Daikon.UserError("multiple inv files specified" + Global.lineSep + usage);
203          }
204          inv_file = file;
205        } else if (filename.indexOf(".dtrace") != -1) {
206          dtrace_files.add(filename);
207        } else {
208          throw new Error("Unrecognized argument: " + file);
209        }
210      }
211      if (dir_file == null) {
212        if (inv_file == null) {
213          throw new Daikon.UserError("No inv file specified" + Global.lineSep + usage);
214        }
215        checkInvariants(inv_file);
216        return;
217      }
218    }
219
220    // Yoav additions:
221    File[] filesInDir = dir_file.listFiles();
222    if (filesInDir == null || filesInDir.length == 0) {
223      throw new Daikon.UserError(
224          "The directory " + dir_file + " is empty" + Global.lineSep + usage);
225    }
226    ArrayList<File> invariants = new ArrayList<>();
227    for (File f : filesInDir) {
228      if (f.toString().indexOf(".inv") != -1) {
229        invariants.add(f);
230      }
231    }
232    if (invariants.size() == 0) {
233      throw new Daikon.UserError(
234          "Did not find any invariant files in the directory " + dir_file + Global.lineSep + usage);
235    }
236    ArrayList<File> dtraces = new ArrayList<>();
237    for (File f : filesInDir) {
238      if (f.toString().indexOf(".dtrace") != -1) {
239        dtraces.add(f);
240      }
241    }
242    if (dtraces.size() == 0) {
243      throw new Daikon.UserError(
244          "Did not find any dtrace files in the directory " + dir_file + Global.lineSep + usage);
245    }
246
247    System.out.println(
248        "Collecting data for invariants files " + invariants + " and dtrace files " + dtraces);
249
250    dtrace_files.clear();
251    for (File dtrace : dtraces) {
252      dtrace_files.add(dtrace.toString());
253    }
254
255    String commaLine = "";
256    for (File inFile : invariants) {
257      String name = inFile.getName().replace(".inv", "").replace(".gz", "");
258      commaLine += "," + name;
259    }
260    outputComma.add(commaLine);
261
262    commaLine = "";
263    for (File inFile : invariants) {
264      File inv_file = inFile;
265      failedInvariants.clear();
266      testedInvariants.clear();
267      error_cnt = 0;
268
269      output_stream =
270          new PrintStream(
271              new FileOutputStream(
272                  inFile.toString().replace(".inv", "").replace(".gz", "")
273                      + ".false-positives.txt"));
274      checkInvariants(inv_file);
275      output_stream.close();
276
277      int failedCount = failedInvariants.size();
278      int testedCount = testedInvariants.size();
279      String percent = toPercentage(failedCount, testedCount);
280      commaLine += "," + percent;
281    }
282    outputComma.add(commaLine);
283
284    System.out.println();
285    for (String output : outputComma) {
286      System.out.println(output);
287    }
288  }

290  private static String toPercentage(int portion, int total) {
291    double s = portion * 100;
292    return String.format("%.2f", s / total) + "%";
293  }

295  private static void checkInvariants(File inv_file) throws IOException {
296    // Read the invariant file
297    PptMap ppts = FileIO.read_serialized_pptmap(inv_file, true);
298
299    // Yoav: make sure we have unique invariants
300    InvariantFilters fi = InvariantFilters.defaultFilters();
301    // Set<String> allInvariantsStr = new HashSet<>();
302    // Set<Invariant> allInvariants = new HashSet<>();
303    for (PptTopLevel ppt : ppts.all_ppts()) {
304      for (PptSlice slice : ppt.views_iterable()) {
305        for (Invariant inv : slice.invs) {
306          if (doConf && inv.getConfidence() < Invariant.dkconfig_confidence_limit) {
307            // System.out.printf("inv ignored (conf): %s:%s%n", inv.ppt.name(),
308            //                   inv.format());
309            continue;
310          }
311
312          if (doFilter && fi.shouldKeep(inv) == null) {
313            // System.out.printf("inv ignored (filter): %s:%s%n",
314            //                     inv.ppt.name(), inv.format());
315            continue;
316          }
317          activeInvariants.add(inv);
318
319          // String n = invariant2str(ppt, inv);
320          // if (!allInvariants.contains(inv) && allInvariantsStr.contains(n)) {
321          //   throw new
322          //     Daikon.UserError("Two invariants have the same ppt.name+inv.rep:"+n);
323          // }
324          // allInvariants.add(inv);
325          // allInvariantsStr.add(n);
326        }
327      }
328    }
329
330    // Read and process the data trace files
331    FileIO.Processor processor = new InvariantCheckProcessor();
332
333    Daikon.FileIOProgress progress = new Daikon.FileIOProgress();
334    progress.start();
335    progress.clear();
336    FileIO.read_data_trace_files(dtrace_files, ppts, processor, false);
337    progress.shouldStop = true;
338    System.out.println();
339    System.out.printf(
340        "%s: %,d errors found in %,d samples (%s)%n",
341        inv_file, error_cnt, sample_cnt, toPercentage(error_cnt, sample_cnt));
342    int failedCount = failedInvariants.size();
343    int testedCount = testedInvariants.size();
344    String percent = toPercentage(failedCount, testedCount);
345    System.out.println(
346        inv_file
347            + ": "
348            + failedCount
349            + " false positives, out of "
350            + testedCount
351            + ", which is "
352            + percent
353            + ".");
354    if (false) {
355      for (Invariant inv : failedInvariants) {
356        System.out.printf("+%s:%s%n", inv.ppt.name(), inv.format());
357      }
358    }
359  }

361  /** Class to track matching ppt and its values. */
362  static final class EnterCall {
363
364    public PptTopLevel ppt;
365    public ValueTuple vt;
366
367    public EnterCall(PptTopLevel ppt, ValueTuple vt) {
368
369      this.ppt = ppt;
370      this.vt = vt;
371    }
372  }
373
374  public static class InvariantCheckProcessor extends FileIO.Processor {
375
376    Map<Integer, EnterCall> call_map = new LinkedHashMap<>();
377
378    /**
379     * process the sample by checking it against each existing invariant and issuing an error if any
380     * invariant is falsified or weakened.
381     */
382    @RequiresNonNull("FileIO.data_trace_state")
383    @Override
384    public void process_sample(
385        PptMap all_ppts, PptTopLevel ppt, ValueTuple vt, @Nullable Integer nonce) {
386
387      debug.fine("processing sample from: " + ppt.name);
388
389      // Add orig and derived variables
390      FileIO.compute_orig_variables(ppt, vt.vals, vt.mods, nonce);
391      FileIO.compute_derived_variables(ppt, vt.vals, vt.mods);
392
393      // Intern the sample
394      vt = new ValueTuple(vt.vals, vt.mods);
395
396      // If this is an enter point, just remember it for later
397      if (ppt.ppt_name.isEnterPoint()) {
398        assert nonce != null : "@AssumeAssertion(nullness): nonce exists for enter & exit points";
399        if (dir_file != null) {
400          // Yoav: I had to do a hack to handle the case that several dtrace files are concatenated
401          // together, and Sung's dtrace files have unterminated calls, and when concatenating two
402          // files you can have the same nonce.
403          // So I have to remove the nonce found from the call_map.
404          call_map.remove(nonce);
405        } else {
406          assert call_map.get(nonce) == null;
407        }
408        call_map.put(nonce, new EnterCall(ppt, vt));
409        debug.fine("Skipping enter sample");
410        return;
411      }
412
413      // If this is an exit point, process the saved enter point
414      if (ppt.ppt_name.isExitPoint()) {
415        assert nonce != null : "@AssumeAssertion(nullness): nonce exists for enter & exit points";
416        EnterCall ec = call_map.get(nonce);
417        if (ec != null) {
418          call_map.remove(nonce);
419          debug.fine("Processing enter sample from " + ec.ppt.name);
420          add(ec.ppt, ec.vt, all_ppts);
421        } else { // didn't find the enter
422          if (!quiet) {
423            System.out.printf("couldn't find enter for nonce %d at ppt %s%n", nonce, ppt.name());
424          }
425          return;
426        }
427      }
428
429      add(ppt, vt, all_ppts);
430    }
431
432    @RequiresNonNull("FileIO.data_trace_state")
433    private void add(PptTopLevel ppt, ValueTuple vt, PptMap all_ppts) {
434      // Add the sample to any splitters
435      if (ppt.has_splitters()) {
436        assert ppt.splitters != null; // because ppt.has_splitters() = true
437        for (PptSplitter ppt_split : ppt.splitters) {
438          PptConditional ppt_cond = ppt_split.choose_conditional(vt);
439          if (ppt_cond != null) {
440            add(ppt_cond, vt, all_ppts);
441          } else {
442            debug.fine(": sample doesn't pick conditional");
443          }
444        }
445      }
446
447      // if this is a numbered exit, apply to the combined exit as well
448      if (!(ppt instanceof PptConditional) && ppt.ppt_name.isNumberedExitPoint()) {
449        PptTopLevel parent = all_ppts.get(ppt.ppt_name.makeExit());
450        if (parent != null) {
451          parent.get_missingOutOfBounds(ppt, vt);
452          add(parent, vt, all_ppts);
453        }
454      }
455
456      // If the point has no variables, skip it
457      if (ppt.var_infos.length == 0) {
458        return;
459      }
460
461      // We should have received sample here before, or there is nothing to check.
462      // Yoav added: It can be that the different dtrace and inv files have different program points
463      // if (false && ppt.num_samples() <= 0) {
464      //   assert ppt.num_samples() > 0
465      //       : "ppt " + ppt.name + " has 0 samples and " + ppt.var_infos.length + " variables";
466      // }
467
468      // Loop through each slice
469      slice_loop:
470      for (PptSlice slice : ppt.views_iterable()) {
471        if (debug_detail.isLoggable(Level.FINE)) {
472          debug_detail.fine(
473              ": processing slice " + slice + "vars: " + Debug.toString(slice.var_infos, vt));
474        }
475
476        // If any variables are missing, skip this slice
477        for (int j = 0; j < slice.var_infos.length; j++) {
478          VarInfo v = slice.var_infos[j];
479          if (v.isMissing(vt)) {
480            if (debug_detail.isLoggable(Level.FINE)) {
481              debug_detail.fine(": : Skipping slice, " + v.name() + " missing");
482            }
483            continue slice_loop;
484          }
485          if (v.missingOutOfBounds()) {
486            if (debug_detail.isLoggable(Level.FINE)) {
487              debug.fine(": : Skipping slice, " + v.name() + " out of bounds");
488            }
489            continue slice_loop;
490          }
491        }
492
493        // Loop through each invariant
494        for (Invariant inv : slice.invs) {
495          if (debug_detail.isLoggable(Level.FINE)) {
496            debug_detail.fine(": : Processing invariant: " + inv);
497          }
498          if (!inv.isActive()) {
499            if (debug_detail.isLoggable(Level.FINE)) {
500              debug_detail.fine(": : skipped non-active " + inv);
501            }
502            continue;
503          }
504
505          // Yoav added
506          if (!activeInvariants.contains(inv)) {
507            // System.out.printf("skipping invariant %s:%s%n", inv.ppt.name(),
508            //                   inv.format());
509            continue;
510          }
511
512          // String invRep = invariant2str(ppt, inv);
513          testedInvariants.add(inv);
514
515          // Store string representation of original invariant for verbose mode
516          String invRep = quiet ? null : inv.format();
517
518          InvariantStatus status = inv.add_sample(vt, 1);
519          sample_cnt++;
520          if (status != InvariantStatus.NO_CHANGE) {
521            if (!quiet) {
522              output_stream.println(
523                  "At ppt "
524                      + ppt.name
525                      + ", Invariant '"
526                      + invRep
527                      + "' invalidated by sample "
528                      + Debug.toString(slice.var_infos, vt)
529                      + "at line "
530                      + FileIO.get_linenum()
531                      + " in file "
532                      + FileIO.data_trace_state.filename);
533            }
534            failedInvariants.add(inv);
535            activeInvariants.remove(inv);
536            error_cnt++;
537          }
538        }
539      }
540    }
541  }

543  @SuppressWarnings("UnusedMethod") // for debugging (which is currently commented out)
544  private static String invariant2str(PptTopLevel ppt, Invariant inv) {
545    return ppt.name + " == " + inv.repr() + inv.getClass() + inv.varNames() + ": " + inv.format();
